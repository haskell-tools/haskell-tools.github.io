   38 trfOperator :: forall n r . TransformName n r => Located (IdP n) -> Trf (Ann AST.UOperator (Dom r) RangeStage)
   39 trfOperator = trfLocNoSema (trfOperator' @n)
   40 
   41 trfOperator' :: forall n r . TransformName n r => IdP n -> Trf (AST.UOperator (Dom r) RangeStage)
   42 trfOperator' n
   43   | isSymOcc (occName @n n) = AST.UNormalOp <$> (trfQualifiedNameFocus @n) True n
   44   | otherwise = AST.UBacktickOp <$> (trfQualifiedNameFocus @n) True n
   45 
   46 trfName :: forall n r . TransformName n r => Located (IdP n) -> Trf (Ann AST.UName (Dom r) RangeStage)
   47 trfName = trfLocNoSema (trfName' @n)
   48 
   49 trfName' :: forall n r . TransformName n r => IdP n -> Trf (AST.UName (Dom r) RangeStage)
   50 trfName' n
   51   | isSymOcc (occName @n n) = (if isSpecKind then AST.UNormalName else AST.UParenName) <$> (trfQualifiedNameFocus @n) isSpecKind n
   52   | otherwise = AST.UNormalName <$> (trfQualifiedNameFocus @n) False n
   53   where -- special names that are operators, but appear in name context
   54     isSpecKind = occNameString (occName @n n) `elem` ["*", "#", "?", "??"]
   55 
   56 trfAmbiguousFieldName :: TransformName n r => Located (AmbiguousFieldOcc n) -> Trf (Ann AST.UName (Dom r) RangeStage)
   57 trfAmbiguousFieldName (L l af) = trfAmbiguousFieldName' l af
   58 
   59 trfAmbiguousFieldName' :: forall n r . TransformName n r => SrcSpan -> AmbiguousFieldOcc n -> Trf (Ann AST.UName (Dom r) RangeStage)
   60 trfAmbiguousFieldName' l (Unambiguous pr (L _ rdr)) = annLocNoSema (pure l) $ trfName' @n (fieldOccToId @n rdr pr)
   61 -- no Id transformation is done, so we can basically ignore the postTC value
   62 trfAmbiguousFieldName' _ (Ambiguous _ (L l rdr))
   63   = annLocNoSema (pure l)
   64       $ (if (isSymOcc (occName @GhcPs rdr)) then AST.UParenName else AST.UNormalName)
   65           <$> (annLoc (createAmbigousNameInfo rdr l) (pure l) $ AST.nameFromList <$> trfNameStr (isSymOcc (occName @GhcPs rdr)) (rdrNameStr rdr))
   66 
   67 trfAmbiguousOperator' :: forall n r . TransformName n r => SrcSpan -> AmbiguousFieldOcc n -> Trf (Ann AST.UOperator (Dom r) RangeStage)
   68 trfAmbiguousOperator' l (Unambiguous pr (L _ rdr)) = annLocNoSema (pure l) $ trfOperator' @n (fieldOccToId @n rdr pr)
   69 -- no Id transformation is done, so we can basically ignore the postTC value
   70 trfAmbiguousOperator' _ (Ambiguous _ (L l rdr))
   71   = annLocNoSema (pure l)
   72       $ (if (isSymOcc (occName @GhcPs rdr)) then AST.UNormalOp else AST.UBacktickOp)
   73           <$> (annLoc (createAmbigousNameInfo rdr l) (pure l) $ AST.nameFromList <$> trfOperatorStr (not $ isSymOcc (occName @GhcPs rdr)) (rdrNameStr rdr))

  110 -- | This class allows us to use the same transformation code for multiple variants of the GHC AST.
  111 -- GHC UName annotated with 'name' can be transformed to our representation with semantic annotations of 'res'.
  112 class (TransformableName name, HsHasName (IdP name), FromGHCName (IdP res), Eq (IdP name), GHCName res, NameOrRdrName (IdP name) ~ (IdP name), XUnambiguous name ~ XCFieldOcc name)
  113         => TransformName name res where
  114   -- | Demote a given name
  115   transformName :: IdP name -> IdP res
  116 
  117 instance TransformName GhcPs GhcPs where transformName = id

  119 instance {-# OVERLAPPABLE #-} (FromGHCName (IdP res), GHCName res) => TransformName GhcRn res where
  120   transformName = fromGHCName

  122 trfNameText :: String -> Trf (Ann AST.UName (Dom r) RangeStage)
  123 trfNameText str
  124   = annContNoSema (AST.UNormalName <$> annLoc (createImplicitNameInfo str) (asks contRange) (AST.nameFromList <$> trfNameStr (isOperatorStr str) str))
  125 
  126 trfImplicitName :: HsIPName -> Trf (Ann AST.UName (Dom r) RangeStage)
  127 trfImplicitName (HsIPName fs)
  128   = let nstr = unpackFS fs
  129      in do rng <- asks contRange
  130            let rng' = mkSrcSpan (updateCol (+1) (srcSpanStart rng)) (srcSpanEnd rng)
  131            annContNoSema (AST.UImplicitName <$> annLoc (createImplicitNameInfo nstr) (pure rng')
  132                                                   (AST.nameFromList <$> trfNameStr (isOperatorStr nstr) nstr))
  133 
  134 isOperatorStr :: String -> Bool
  135 isOperatorStr = any (not . isAlphaNum)

  137 trfQualifiedName :: forall n r . TransformName n r => Bool -> Located (IdP n) -> Trf (Ann AST.UQualifiedName (Dom r) RangeStage)
  138 trfQualifiedName isOperator (L l n) = focusOn l $ (trfQualifiedNameFocus @n) isOperator n
  139 
  140 trfQualifiedNameFocus :: forall n r . TransformName n r => Bool -> IdP n -> Trf (Ann AST.UQualifiedName (Dom r) RangeStage)
  141 trfQualifiedNameFocus isOperator n
  142   = do rng <- asks contRange
  143        let rng' = if isOperator == isSymOcc (occName @n n) then rng
  144                     else mkSrcSpan (updateCol (+1) (srcSpanStart rng)) (updateCol (subtract 1) (srcSpanEnd rng))
  145        annLoc (createNameInfo (transformName @n @r n)) (pure rng') (trfQualifiedName' @n n)
  146 
  147 trfQualifiedName' :: forall n r . TransformName n r => IdP n -> Trf (AST.UQualifiedName (Dom r) RangeStage)
  148 trfQualifiedName' n = AST.nameFromList <$> ((if isSymOcc (occName @n n) then trfOperatorStr else trfNameStr) False =<< (correctNameString @n) n)
  149 
  150 trfOperatorStr :: Bool -> String -> Trf (AnnListG AST.UNamePart (Dom r) RangeStage)
  151 trfOperatorStr isInParen str = do rng <- correctSpan <$> asks contRange
  152                                   makeList "." (pure $ srcSpanStart rng)
  153                                                (pure [Ann (noSemaInfo $ AST.NodeSpan rng) (AST.UNamePart str)])
  154   where correctSpan sp = if isInParen then mkSrcSpan (updateCol (+1) (srcSpanStart sp))
  155                                                      (updateCol (subtract 1) (srcSpanEnd sp))
  156                                       else sp
  157 
  158 -- | Creates a qualified name from a name string
  159 trfNameStr :: Bool -> String -> Trf (AnnListG AST.UNamePart (Dom r) RangeStage)
  160 trfNameStr isInBackticks str = makeList "." atTheStart (trfNameStr' str . correct <$> atTheStart)
  161   where correct = if isInBackticks then updateCol (+1) else id
  162 
  163 trfNameStr' :: String -> SrcLoc -> [Ann AST.UNamePart (Dom r) RangeStage]
  164 trfNameStr' str startLoc = fst $
  165   foldl (\(r,loc) np -> let nextLoc = advanceAllSrcLoc loc np
  166                          in ( r ++ [Ann (noSemaInfo $ AST.NodeSpan (mkSrcSpan loc nextLoc)) (AST.UNamePart np)], advanceAllSrcLoc nextLoc "." ) )
  167   ([], startLoc) (splitOn "." str)
  168   where -- | Move the source location according to a string
  169         advanceAllSrcLoc :: SrcLoc -> String -> SrcLoc
  170         advanceAllSrcLoc (RealSrcLoc rl) str = RealSrcLoc $ foldl advanceSrcLoc rl str
  171         advanceAllSrcLoc oth _ = oth
