   32 -- | Looks up the Type of an entity with an Id of any locality.
   33 -- If the entity being scrutinised is a type variable, it fails.
   34 lookupTypeFromId :: (HasIdInfo' id, GhcMonad m) => id -> MaybeT m GHC.Type
   35 lookupTypeFromId idn
   36   | GHC.isLocalId  . semanticsId $ idn = return . typeOrKindFromId $ idn
   37   | GHC.isGlobalId . semanticsId $ idn = lookupTypeFromGlobalName idn
   38   | otherwise = fail "Couldn't lookup name"
   39 
   40 -- | Looks up the Type or the Kind of an entity that has an Id.
   41 -- Note: In some cases we only get the Kind of the Id (e.g. for type constructors)
   42 typeOrKindFromId :: HasIdInfo' id => id -> GHC.Type
   43 typeOrKindFromId idn = GHC.varType . semanticsId $ idn

   45 -- | Extracts a Type from a TyThing when possible.
   46 typeFromTyThing :: GHC.TyThing -> Maybe GHC.Type
   47 typeFromTyThing (GHC.AnId     idn)  = Just . GHC.varType $ idn
   48 typeFromTyThing (GHC.ATyCon   tc)   = GHC.synTyConRhs_maybe tc
   49 typeFromTyThing (GHC.ACoAxiom _)    = fail "CoAxioms are not supported for type lookup"
   50 typeFromTyThing (GHC.AConLike con)  = handleCon con
   51   where handleCon (GHC.RealDataCon dc) = Just . GHC.dataConUserType $ dc
   52         handleCon (GHC.PatSynCon   pc) = do
   53           (idn,_) <- GHC.patSynBuilder pc
   54           return . GHC.varType $ idn
   55 
   56 -- | Looks up a GHC Type from a Haskell Tools Name (given the name is global)
   57 -- For an identifier, it returns its type.
   58 -- For a data constructor, it returns its type.
   59 -- For a pattern synonym, it returns its builder's type.
   60 -- For a type synonym constructor, it returns its right-hand side.
   61 -- For a coaxiom, it fails.
   62 lookupTypeFromGlobalName :: (HasNameInfo' n, GhcMonad m) => n -> MaybeT m GHC.Type
   63 lookupTypeFromGlobalName name = do
   64   sname <- liftMaybe . semanticsName $ name
   65   tt    <- MaybeT    . GHC.lookupName $ sname
   66   liftMaybe . typeFromTyThing $ tt

   69 -- | Looks up the right-hand side (GHC representation)
   70 -- of a Haskell Tools Name corresponding to a type synonym
   71 lookupTypeSynRhs :: (HasNameInfo' n, GhcMonad m) => n -> MaybeT m GHC.Type
   72 lookupTypeSynRhs name = do
   73   sname <- liftMaybe . semanticsName $ name
   74   tt    <- MaybeT    . GHC.lookupName $ sname
   75   tc    <- liftMaybe . tyconFromTyThing $ tt
   76   liftMaybe . GHC.synTyConRhs_maybe $ tc
   77 
   78 -- NOTE: Returns Nothing if it is not a type synonym
   79 lookupSynDef :: GHC.TyThing -> Maybe GHC.TyCon
   80 lookupSynDef syn = do
   81   tycon <- tyconFromTyThing syn
   82   rhs   <- GHC.synTyConRhs_maybe tycon
   83   tyconFromGHCType rhs
   84 
   85 tyconFromTyThing :: GHC.TyThing -> Maybe GHC.TyCon
   86 tyconFromTyThing (GHC.ATyCon tycon) = Just tycon
   87 tyconFromTyThing _ = Nothing
   88 
   89 -- won't bother
   90 tyconFromGHCType :: GHC.Type -> Maybe GHC.TyCon
   91 tyconFromGHCType (GHC.AppTy t1 _) = tyconFromGHCType t1
   92 tyconFromGHCType (GHC.TyConApp tycon _) = Just tycon
   93 tyconFromGHCType _ = Nothing

   96 -- NOTE: Returns false if the type is certainly not a newtype
   97 --       Returns true if it is a newtype or it could not have been looked up
   98 isNewtype :: GhcMonad m => AST.Type -> m Bool
   99 isNewtype t = do
  100   tycon <- runMaybeT . lookupType $ t
  101   return $! maybe True isNewtypeTyCon tycon

  105 lookupType :: GhcMonad m => AST.Type -> MaybeT m GHC.TyThing
  106 lookupType t = do
  107   name  <- liftMaybe . nameFromType $ t
  108   sname <- liftMaybe . semanticsName   $ name
  109   MaybeT . GHC.lookupName $ sname

  111 -- | Looks up a GHC.Class from something that has a type class constructor in it
  112 -- Fails if the argument does not contain a class type constructor
  113 lookupClassWith :: GhcMonad m => (a -> MaybeT m GHC.Name) -> a -> MaybeT m GHC.Class
  114 lookupClassWith getName x = do
  115   sname   <- getName x
  116   tything <- MaybeT . GHC.lookupName $ sname
  117   case tything of
  118     GHC.ATyCon tc | GHC.isClassTyCon tc -> liftMaybe . GHC.tyConClass_maybe $ tc
  119     _ -> fail "TypeLookup.lookupClassWith: Argument does not contain a class type constructor"
  120 
  121 lookupClass :: (GhcMonad m, HasNameInfo' n) => n -> MaybeT m GHC.Class
  122 lookupClass = lookupClassWith (liftMaybe . semanticsName)
  123 
  124 lookupClassFromInstance :: GhcMonad m => InstanceHead -> MaybeT m GHC.Class
  125 lookupClassFromInstance = lookupClassWith (liftMaybe . instHeadSemName)
  126 
  127 lookupClassFromDeclHead :: GhcMonad m => DeclHead -> MaybeT m GHC.Class
  128 lookupClassFromDeclHead = lookupClassWith (liftMaybe . declHeadSemName)
  129 
  130 -- | Looks up the right-hand side (GHC representation)
  131 -- of a Haskell Tools Type corresponding to a type synonym
  132 semanticsTypeSynRhs :: GhcMonad m => AST.Type -> MaybeT m GHC.Type
  133 semanticsTypeSynRhs ty = (liftMaybe . nameFromType $ ty) >>= lookupTypeSynRhs
  134 
  135 -- | Converts a global Haskell Tools type to a GHC type
  136 semanticsType :: GhcMonad m => AST.Type -> MaybeT m GHC.Type
  137 semanticsType ty = (liftMaybe . nameFromType $ ty) >>= lookupTypeFromGlobalName

  139 isNewtypeTyCon :: GHC.TyThing -> Bool
  140 isNewtypeTyCon (GHC.ATyCon tycon) = GHC.isNewTyCon tycon
  141 isNewtypeTyCon _ = False

  143 -- | Looks up the given name, extracts something out of it.
  144 -- If the extraction is not succesful, it returns False,
  145 -- if it is successful, then checks the result against the predicate.
  146 -- The reasoning behind this, is that the predicate can only be
  147 -- satisfied by a proper name.
  148 satisfies :: (HasNameInfo' n, GhcMonad m) =>
  149              (GHC.TyThing -> Maybe a) -> (a -> Bool) -> n -> MaybeT m Bool
  150 satisfies extract pred name = do
  151   sname <- liftMaybe . semanticsName  $ name
  152   tt    <- MaybeT    . GHC.lookupName $ sname
  153   return $ maybe False pred (extract tt)
  154 
  155 -- | Decides whether a given name is a type family constructor.
  156 -- Fails if the lookup is not successful.
  157 isClassTyConNameM :: (HasNameInfo' n, GhcMonad m) => n -> MaybeT m Bool
  158 isClassTyConNameM = satisfies extractTyCon GHC.isClassTyCon
  159   where extractTyCon (GHC.ATyCon tc) = Just tc
  160         extractTyCon  _              = Nothing
  161 
  162 -- | Decides whether a given name is a standard Haskell98 data constructor.
  163 -- Fails if the lookup is not successful.
  164 isVanillaDataConNameM :: (HasNameInfo' n, GhcMonad m) => n -> MaybeT m Bool
  165 isVanillaDataConNameM = satisfies extractDataCon GHC.isVanillaDataCon
  166   where extractDataCon (GHC.AConLike (GHC.RealDataCon dc)) = Just dc
  167         extractDataCon  _                                  = Nothing
  168 
  169 -- | Looks up a closed type family from a name.
  170 lookupClosedTyFam :: (HasNameInfo' n, GhcMonad m) => n -> MaybeT m ClosedTyFam
  171 lookupClosedTyFam name = do
  172   sname <- liftMaybe . semanticsName $ name
  173   tt    <- MaybeT    . GHC.lookupName $ sname
  174   liftMaybe . coAxiomFromTyThing $ tt
  175 
  176 -- | Extract the CoAxioms from a TyThing representing a closed type family.
  177 coAxiomFromTyThing :: GHC.TyThing -> Maybe (GHC.CoAxiom GHC.Branched)
  178 coAxiomFromTyThing (GHC.ATyCon tc)   = GHC.isClosedSynFamilyTyConWithAxiom_maybe tc
  179 coAxiomFromTyThing (GHC.ACoAxiom ax) = Just ax
  180 coAxiomFromTyThing _                 = Nothing

  182 -- | Determines whether a Type itself has a type variable head.
  183 hasTyVarHead :: Type -> Bool
  184 hasTyVarHead (ForallType _ t) = hasTyVarHead t
  185 hasTyVarHead (CtxType _ t) = hasTyVarHead t
  186 hasTyVarHead FunctionType{} = False
  187 hasTyVarHead TupleType{} = False
  188 hasTyVarHead UnboxedTupleType{} = False
  189 hasTyVarHead ListType{} = False
  190 hasTyVarHead ParArrayType{} = False
  191 hasTyVarHead (TypeApp f _) = hasTyVarHead f
  192 hasTyVarHead InfixTypeApp{} = False
  193 hasTyVarHead (ParenType t) = hasTyVarHead t
  194 hasTyVarHead (VarType n) = maybe False GHC.isTyVarName (semanticsName n)
  195 hasTyVarHead (KindedType t _) = hasTyVarHead t
  196 hasTyVarHead (BangType t) = hasTyVarHead t
  197 hasTyVarHead (LazyType t) = hasTyVarHead t
  198 hasTyVarHead (UnpackType t) = hasTyVarHead t
  199 hasTyVarHead (NoUnpackType t) = hasTyVarHead t
  200 hasTyVarHead WildcardType{} = False
  201 hasTyVarHead NamedWildcardType{} = False
  202 hasTyVarHead SpliceType{} = False
  203 hasTyVarHead QuasiQuoteType{} = False
  204 hasTyVarHead PromotedIntType{} = False
