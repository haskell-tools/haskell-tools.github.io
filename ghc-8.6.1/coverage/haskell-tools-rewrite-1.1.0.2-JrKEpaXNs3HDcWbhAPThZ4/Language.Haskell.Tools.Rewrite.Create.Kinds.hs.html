    1 -- | Generation of statement-level AST fragments for refactorings.
    2 -- The bindings defined here are the AST constructor names with an "mk" prefix.
    3 {-# LANGUAGE MonoLocalBinds #-}
    4 {-# LANGUAGE OverloadedStrings #-}
    5 
    6 module Language.Haskell.Tools.Rewrite.Create.Kinds where
    7 
    8 import Data.String (IsString(..), String)
    9 import Language.Haskell.Tools.AST (UPromoted(..), UKind(..), UKindConstraint(..))
   10 import Language.Haskell.Tools.PrettyPrint.Prepare
   11 import Language.Haskell.Tools.Rewrite.Create.Utils (mkAnn, mkAnnList)
   12 import Language.Haskell.Tools.Rewrite.ElementTypes (Name, Kind, KindConstraint)
   13 
   14 -- | Kind constraint (@ :: * -> * @)
   15 mkKindConstraint :: Kind -> KindConstraint
   16 mkKindConstraint = mkAnn (" :: " <> child) . UKindConstraint
   17 
   18 -- | @*@, the kind of types
   19 mkKindStar :: Kind
   20 mkKindStar = mkAnn "*" UStarKind
   21 
   22 -- | @#@, the kind of unboxed types
   23 mkKindUnbox :: Kind
   24 mkKindUnbox = mkAnn "#" UUnboxKind
   25 
   26 -- | @->@, the kind of type constructor
   27 mkKindFun :: Kind -> Kind -> Kind
   28 mkKindFun lhs rhs = mkAnn (child <> " -> " <> child) $ UFunKind lhs rhs
   29 
   30 -- | A parenthesised kind
   31 mkKindParen :: Kind -> Kind
   32 mkKindParen = mkAnn ("(" <> child <> ")") . UParenKind
   33 
   34 -- | Kind variable (using @PolyKinds@ extension)
   35 mkKindVar :: Name -> Kind
   36 mkKindVar = mkAnn child . UVarKind
   37 
   38 -- | Kind application (@ k1 k2 @)
   39 mkKindApp :: Kind -> Kind -> Kind
   40 mkKindApp lhs rhs = mkAnn (child <> " " <> child) $ UAppKind lhs rhs
   41 
   42 -- | A list kind (@ [k] @)
   43 mkKindList :: Kind -> Kind
   44 mkKindList = mkAnn ("[" <> child <> "]") . UListKind
   45 
   46 -- | Numeric value promoted to the kind level.
   47 mkIntKind :: Integer -> Kind
   48 mkIntKind i = mkAnn child $ UPromotedKind $ mkAnn (fromString $ show i) (UPromotedInt i)
   49 
   50 -- | String value promoted to the kind level.
   51 mkStringKind :: String -> Kind
   52 mkStringKind i = mkAnn child $ UPromotedKind $ mkAnn (fromString $ show i) (UPromotedString i)
   53 
   54 -- | A data constructor value promoted to the kind level.
   55 mkConKind :: Name -> Kind
   56 mkConKind = mkAnn child . UPromotedKind . mkAnn child . UPromotedCon
   57 
   58 -- | A list of elements as a kind.
   59 mkListKind :: [Kind] -> Kind
   60 mkListKind = mkAnn child . UPromotedKind . mkAnn ("[" <> child <> "]") . UPromotedList . mkAnnList (separatedBy ", " list)
   61 
   62 -- | A tuple of elements as a kind.
   63 mkTupleKind :: [Kind] -> Kind
   64 mkTupleKind = mkAnn child . UPromotedKind . mkAnn ("(" <> child <> ")") . UPromotedTuple . mkAnnList (separatedBy ", " list)
   65 
   66 -- | Kind of the unit value @()@. 
   67 mkUnitKind :: Kind
   68 mkUnitKind = mkAnn child $ UPromotedKind $ mkAnn "()" UPromotedUnit
