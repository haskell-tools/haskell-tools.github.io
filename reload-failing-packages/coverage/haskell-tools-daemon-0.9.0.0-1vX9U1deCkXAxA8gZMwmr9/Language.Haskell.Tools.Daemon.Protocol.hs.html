   19 -- | The messages expected from the client.
   20 data ClientMessage
   21   = KeepAlive -- ^ A simple ping message to check that the server is running.
   22   | Reset -- ^ A message that instructs the server to reset its internal state and re-load loaded packages.
   23   | Handshake { clientVersion :: [Int] }
   24     -- ^ Tells the client version and asks the servers version.
   25   | SetPackageDB { pkgDB :: PackageDB }
   26     -- ^ Sets the package database for the engine to use.
   27   | AddPackages { addedPathes :: [FilePath] }
   28     -- ^ Registers packages to the engine. They will be subject to subsequent
   29     -- refactorings. Will cause the packages to be loaded, resulting in
   30     -- LoadingModules, LoadedModules or CompilationProblem responses.
   31   | RemovePackages { removedPathes :: [FilePath] }
   32     -- ^ Deregisters the given packages from the engine. They will not be
   33     -- subjects of further refactorings.
   34   | SetWorkingDir { newWorkingDir :: FilePath }
   35     -- ^ Sets the working directory for the compilation. Important when
   36     -- compiling code that loads resources based on relative pathes.
   37   | SetGHCFlags { ghcFlags :: [String] }
   38     -- ^ Sets the compilation flags. The unused flags are returned via the
   39     -- UnusedFlags response.
   40   | PerformRefactoring { refactoring :: String
   41                        , modulePath :: FilePath
   42                        , editorSelection :: String
   43                        , details :: [String] -- ^ Additional details for the refactoring like the
   44                                              -- names of generated definitions.
   45                        , shutdownAfter :: Bool -- ^ Stop the daemon after performing the refactoring.
   46                        , diffMode :: Bool -- ^ Don't change the files, send back the result as
   47                                           -- a unified diff.
   48                        }
   49     -- ^ Orders the engine to perform the refactoring on the module given
   50     -- with the selection and details. Successful refactorings will cause re-loading of modules.
   51     -- If 'shutdownAfter' or 'diffMode' is not set, after the refactoring,
   52     -- modules are re-loaded, LoadingModules, LoadedModules responses are sent.
   53   | UndoLast
   54     -- ^ Asks the daemon to undo the last refactoring.
   55   | Disconnect
   56     -- ^ Stops the engine. It replies with Disconnected.
   57   | ReLoad { addedModules :: [FilePath]
   58            , changedModules :: [FilePath]
   59            , removedModules :: [FilePath]
   60            }
   61     -- ^ Instructs the engine to re-load a changed module.
   62     -- LoadingModules, LoadedModules responses may be sent.
   63   | Stop -- TODO: remove
   64     -- ^ Stops the server. OBSOLATE
   65   deriving (Show, Generic)
   66 
   67 instance FromJSON ClientMessage
   68 
   69 -- | The possible responses that the server can give.
   70 data ResponseMsg
   71   = KeepAliveResponse -- ^ A response to KeepAlive
   72   | HandshakeResponse { serverVersion :: [Int] }
   73     -- ^ Tells the version of the server.
   74   | ErrorMessage { errorMsg :: String }
   75     -- ^ An error message marking internal problems or user mistakes.
   76     -- TODO: separate internal problems and user mistakes.
   77   | CompilationProblem { errorMarkers :: [(SrcSpan, String)]
   78                        , errorHints :: [String]
   79                        }
   80     -- ^ A response that tells there are errors in the source code given.
   81   | DiffInfo { diffInfo :: String }
   82     -- ^ Information about changes that would be caused by the refactoring.
   83   | LoadingModules { modulesToLoad :: [FilePath] }
   84     -- ^ The traversal of the project is done, now the engine is loading the
   85     -- given modules.
   86   | LoadedModules { loadedModules :: [(FilePath, String)] }
   87     -- ^ The engine has loaded the given module.
   88   | UnusedFlags { unusedFlags :: [String] }
   89     -- ^ Returns the flags that are not used by the engine.
   90   | Disconnected
   91     -- ^ The engine has closed the connection.
   92   deriving (Show, Generic)
   93 
   94 instance ToJSON ResponseMsg

   96 instance ToJSON SrcSpan where
   97   toJSON (RealSrcSpan sp) = object [ "file" A..= unpackFS (srcSpanFile sp)
   98                                    , "startRow" A..= srcLocLine (realSrcSpanStart sp)
   99                                    , "startCol" A..= srcLocCol (realSrcSpanStart sp)
  100                                    , "endRow" A..= srcLocLine (realSrcSpanEnd sp)
  101                                    , "endCol" A..= srcLocCol (realSrcSpanEnd sp)
  102                                    ]
  103   toJSON _ = Null

  105 data UndoRefactor = RemoveAdded { undoRemovePath :: FilePath }
