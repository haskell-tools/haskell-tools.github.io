   43 trfDecls :: TransformName n r => [LHsDecl n] -> Trf (AnnListG AST.UDecl (Dom r) RangeStage)
   44 trfDecls decls = addToCurrentScope decls $ makeIndentedListNewlineBefore atTheEnd (mapM trfDecl decls)

   46 trfDeclsGroup :: forall n r . TransformName n r => HsGroup n -> Trf (AnnListG AST.UDecl (Dom r) RangeStage)
   47 trfDeclsGroup g@(HsGroup vals splices tycls derivs fixities defaults foreigns warns anns rules vects _)
   48   = do rdrSpls <- asks declSplices -- now we don't want to rename the splices, just interested in their locations to
   49                                    -- filter out the declarations that are generated from them
   50        let (sigs, bagToList -> binds) = getBindsAndSigs vals
   51            -- collect the declarations from the group
   52            alldecls :: [Located (HsDecl n)]
   53            alldecls = (map (fmap SpliceD) splices)
   54                         ++ (map (fmap ValD) binds)
   55                         ++ (map (fmap SigD) sigs)
   56                         ++ (map (fmap TyClD) (concat $ map group_tyclds tycls))
   57                         ++ (map (fmap DerivD) derivs)
   58                         ++ (map (fmap (SigD . FixSig)) (mergeFixityDefs fixities))
   59                         ++ (map (fmap DefD) defaults)
   60                         ++ (map (fmap ForD) foreigns)
   61                         ++ (map (fmap WarningD) warns)
   62                         ++ (map (fmap AnnD) anns)
   63                         ++ (map (fmap RuleD) rules)
   64                         ++ (map (fmap VectD) vects)
   65                         ++ (map (fmap InstD) (hsGroupInstDecls g))
   66        -- Declarations generated from TH should only be in scope after the splice.
   67        let (genNames, sourceNames) = partition (\d -> any (\spl -> getLoc spl `containsRealSpan` getLoc d) rdrSpls) alldecls
   68        addToCurrentScope sourceNames $ do
   69          -- use the definitions generated by previous splices when renaming one
   70          spls <- asks declSplices >>= mapM (\(L l e) -> let namesGeneratedBefore = filter ((srcSpanStart l >) . srcSpanEnd . getLoc) genNames
   71                                                          in addToCurrentScope namesGeneratedBefore ((L l) <$> transformSplice e))
   72          let actualDefinitions = removeContained $ orderElems $ replaceSpliceDecls spls alldecls
   73            in makeIndentedListNewlineBefore atTheEnd
   74                 (orderDefs <$> ((++) <$> getDeclsToInsert <*> (mapM (trfDeclsWithScope genNames) actualDefinitions)))
   75   where
   76     -- use the definitions generated by previous splices when transforming a definition
   77     trfDeclsWithScope genNames d = local (\s -> s {declSplices = []})
   78                                      $ addToCurrentScope namesGeneratedBefore (trfDecl d)
   79       where namesGeneratedBefore = filter ((srcSpanStart (getLoc d) >) . srcSpanEnd . getLoc) genNames
   80 
   81     replaceSpliceDecls :: [Located (HsSplice n)] -> [Located (HsDecl n)] -> [Located (HsDecl n)]
   82     replaceSpliceDecls splices decls = foldl mergeSplice decls splices
   83 
   84     orderElems :: [Located a] -> [Located a]
   85     orderElems = sortOn (srcSpanStart . getLoc)
   86 
   87     removeContained :: [Located (HsDecl n)] -> [Located (HsDecl n)]
   88     removeContained (fst:snd:rest) | getLoc fst `containsRealSpan` getLoc snd
   89       = removeContained (fst:rest)
   90     removeContained (fst:rest) = fst : removeContained rest
   91     removeContained [] = []
   92 
   93     (RealSrcSpan sp1) `containsRealSpan` (RealSrcSpan sp2) = sp1 `containsSpan` sp2
   94     _ `containsRealSpan` _ = False
   95 
   96     mergeSplice :: [Located (HsDecl n)] -> Located (HsSplice n) -> [Located (HsDecl n)]
   97     mergeSplice decls spl@(L spLoc@(RealSrcSpan rss) _)
   98       = L spLoc (SpliceD (SpliceDecl spl ExplicitSplice)) : filter (\(L (RealSrcSpan rdsp) _) -> not (rss `containsSpan` rdsp)) decls
   99     mergeSplice _ (L (UnhelpfulSpan {}) _) = convProblem "mergeSplice: no real span"
  100 
  101     getDeclsToInsert :: Trf [Ann AST.UDecl (Dom r) RangeStage]
  102     getDeclsToInsert = do decls <- asks declsToInsert
  103                           allLocals <- asks localsInScope
  104                           case allLocals of locals:_ -> liftGhc $ mapM (loadIdsForDecls (map (^. _1) locals)) decls
  105                                             [] -> convertionProblem "getDeclsToInsert: empty scope"
  106        where loadIdsForDecls :: [GHC.Name] -> Ann AST.UDecl (Dom RdrName) RangeStage -> GHC.Ghc (Ann AST.UDecl (Dom r) RangeStage)
  107              loadIdsForDecls locals = AST.semaTraverse $
  108                 AST.SemaTrf (AST.nameInfo !~ findName) pure (traverse findName) pure pure pure
  109                where findName rdr = pure $ fromGHCName $ fromMaybe (convProblem $ "Data definition name not found: " ++ showSDocUnsafe (ppr rdr)
  110                                                                                     ++ ", locals: " ++ (concat $ intersperse ", " $ map (showSDocUnsafe . ppr) locals))
  111                                                        $ find ((occNameString (rdrNameOcc rdr) ==) . occNameString . nameOccName) locals

  113 trfDecl :: TransformName n r => Located (HsDecl n) -> Trf (Ann AST.UDecl (Dom r) RangeStage)
  114 trfDecl = trfLocNoSema $ \case
  115   TyClD (FamDecl (FamilyDecl (ClosedTypeFamily typeEqs) name tyVars _ kindSig inj))
  116     -> AST.UClosedTypeFamilyDecl <$> focusAfter AnnType (createDeclHead name tyVars)
  117                                 <*> trfFamilyResultSig kindSig inj
  118                                 <*> trfTypeEqs typeEqs
  119   TyClD (FamDecl fd) -> AST.UTypeFamilyDecl <$> annContNoSema (trfTypeFam' fd)
  120   TyClD (SynDecl name vars _ rhs _)
  121     -> AST.UTypeDecl <$> between AnnType AnnEqual (createDeclHead name vars) <*> trfType rhs
  122   TyClD (DataDecl name vars _ (HsDataDefn nd ctx _ kind cons derivs) _ _)
  123     -> do let ctxTok = case nd of DataType -> AnnData
  124                                   NewType -> AnnNewtype
  125               consLoc = focusBeforeIfPresent AnnDeriving atTheEnd
  126           whereLoc <- tokenLoc AnnWhere
  127           if isGoodSrcSpan whereLoc then trfGADT nd name vars ctx kind cons derivs ctxTok consLoc
  128                                     else trfDataDef nd name vars ctx cons derivs ctxTok consLoc
  129   TyClD (ClassDecl ctx name vars _ funDeps sigs defs typeFuns typeFunDefs _ _)
  130     -> AST.UClassDecl <$> trfCtx (after AnnClass) ctx
  131                      <*> betweenIfPresent AnnClass AnnWhere (createDeclHead name vars)
  132                      <*> trfFunDeps funDeps
  133                      <*> createClassBody sigs defs typeFuns typeFunDefs
  134   InstD (ClsInstD (ClsInstDecl typ binds sigs typefam datafam overlap))
  135     -> AST.UInstDecl <$> trfMaybeDefault " " "" trfOverlap (after AnnInstance) overlap
  136                     <*> trfInstanceRule (hsib_body typ)
  137                     <*> trfInstBody binds sigs typefam datafam
  138   InstD (DataFamInstD (DataFamInstDecl con pats _ (HsDataDefn nd _ _ _ cons derivs) _))
  139     | all ((\case ConDeclH98{} -> True; _ -> False) . unLoc) cons
  140     -> AST.UDataInstDecl <$> trfDataKeyword nd
  141                         <*> (focusAfter AnnInstance . focusBeforeIfPresent AnnEqual . focusBeforeIfPresent AnnDeriving)
  142                               (makeInstanceRuleTyVars con pats)
  143                                                        -- the location is needed when there is no = sign
  144                         <*> makeListBefore " = " " | " (pure $ srcSpanStart $ foldLocs $ getLoc con : map getLoc (hsib_body pats)) (mapM trfConDecl cons)
  145                         <*> makeIndentedList atTheEnd (mapM trfDerivings (unLoc derivs))
  146   InstD (DataFamInstD (DataFamInstDecl con pats _ (HsDataDefn nd _ _ kind cons _) _))
  147     -> AST.UGDataInstDecl <$> trfDataKeyword nd
  148                         <*> (focusAfter AnnInstance . focusBeforeIfPresent AnnWhere)
  149                               (makeInstanceRuleTyVars con pats)
  150                         <*> focusBefore AnnWhere (trfKindSig kind)
  151                         <*> makeIndentedListBefore " where " atTheEnd (mapM trfGADTConDecl cons)
  152   InstD (TyFamInstD (TyFamInstDecl (L _ (TyFamEqn con pats _ rhs)) _))
  153     -> AST.UTypeInstDecl <$> between AnnInstance AnnEqual (makeInstanceRuleTyVars con pats) <*> trfType rhs
  154   ValD bind -> trfVal bind
  155   SigD sig -> trfSig sig
  156   DerivD (DerivDecl t _ overlap) -> AST.UDerivDecl <$> trfMaybeDefault " " "" trfOverlap (after AnnInstance) overlap <*> trfInstanceRule (hsib_body t)
  157   RuleD (HsRules _ rules) -> AST.UPragmaDecl <$> annContNoSema (AST.URulePragma <$> makeIndentedList (before AnnClose) (mapM trfRewriteRule rules))
  158   RoleAnnotD (RoleAnnotDecl name roles) -> AST.URoleDecl <$> trfQualifiedName False name <*> makeList " " atTheEnd (mapM trfRole roles)
  159   DefD (DefaultDecl types) -> AST.UDefaultDecl <$> makeList "," (after AnnOpenP) (mapM trfType types)
  160   ForD (ForeignImport name (hsib_body -> typ) _ (CImport ccall safe _ _ _))
  161     -> AST.UForeignImport <$> trfCallConv ccall <*> trfSafety (getLoc ccall) safe <*> define (trfName name) <*> trfType typ
  162   ForD (ForeignExport name (hsib_body -> typ) _ (CExport (L l (CExportStatic _ _ ccall)) _))
  163     -> AST.UForeignExport <$> annLocNoSema (pure l) (trfCallConv' ccall) <*> trfName name <*> trfType typ
  164   SpliceD (SpliceDecl (unLoc -> spl) _) -> AST.USpliceDecl <$> trfSplice spl
  165   WarningD (Warnings _ [])
  166     -> AST.UPragmaDecl <$> annContNoSema (AST.UDeprPragma <$> (makeList " " (after AnnOpen) (pure []))
  167                                                           <*> makeList ", " (before AnnClose) (pure []))
  168   WarningD (Warnings _ [L _ (Warning names (DeprecatedTxt _ stringLits))])
  169     -> AST.UPragmaDecl <$> annContNoSema (AST.UDeprPragma <$> (makeList " " (after AnnOpen) $ mapM trfName names)
  170                                                           <*> makeList ", " (before AnnClose) (mapM (\(L l (StringLiteral _ fs)) -> trfFastString (L l fs)) stringLits))
  171   WarningD (Warnings _ [L _ (Warning names (WarningTxt _ stringLits))])
  172     -> AST.UPragmaDecl <$> annContNoSema (AST.UWarningPragma <$> (makeNonemptyList " " $ mapM trfName names)
  173                                                              <*> makeList ", " (before AnnClose) (mapM (\(L l (StringLiteral _ fs)) -> trfFastString (L l fs)) stringLits))
  174   AnnD (HsAnnotation stxt subject expr)
  175     -> AST.UPragmaDecl <$> annContNoSema (AST.UAnnPragma <$> trfAnnotationSubject stxt subject (srcSpanStart $ getLoc expr) <*> trfExpr expr)
  176   d -> unhandledElement "declaration" d

  178 trfGADT :: TransformName n r => NewOrData -> Located n -> LHsQTyVars n -> Located (HsContext n)
  179                                  -> Maybe (Located (HsKind n)) -> [Located (ConDecl n)]
  180                                  -> Located [LHsDerivingClause n] -> AnnKeywordId -> Trf SrcLoc -> Trf (AST.UDecl (Dom r) RangeStage)
  181 trfGADT nd name vars ctx kind cons derivs ctxTok consLoc
  182   = AST.UGDataDecl <$> trfDataKeyword nd
  183                    <*> trfCtx (after ctxTok) ctx
  184                    <*> betweenIfPresent ctxTok AnnEqual (createDeclHead name vars)
  185                    <*> focusBefore AnnWhere (trfKindSig kind)
  186                    <*> makeIndentedListBefore " where " consLoc (mapM trfGADTConDecl cons)
  187                    <*> makeIndentedList atTheEnd (mapM trfDerivings (unLoc derivs))
  188 
  189 trfDataDef :: TransformName n r => NewOrData -> Located n -> LHsQTyVars n -> Located (HsContext n)
  190                                      -> [Located (ConDecl n)] -> Located [LHsDerivingClause n]
  191                                      -> AnnKeywordId -> Trf SrcLoc -> Trf (AST.UDecl (Dom r) RangeStage)
  192 trfDataDef nd name vars ctx cons derivs ctxTok consLoc
  193   = AST.UDataDecl <$> trfDataKeyword nd
  194                   <*> trfCtx (after ctxTok) ctx
  195                   <*> betweenIfPresent ctxTok AnnEqual (createDeclHead name vars)
  196                   <*> makeListBefore "=" " | " consLoc (mapM trfConDecl cons)
  197                   <*> makeIndentedList atTheEnd (mapM trfDerivings (unLoc derivs))

  199 trfVal :: TransformName n r => HsBindLR n n -> Trf (AST.UDecl (Dom r) RangeStage)
  200 trfVal (PatSynBind psb) = AST.UPatternSynonymDecl <$> annContNoSema (trfPatternSynonym psb)
  201 trfVal bind = AST.UValueBinding <$> (annContNoSema $ trfBind' bind)

  203 trfSig :: TransformName n r => Sig n -> Trf (AST.UDecl (Dom r) RangeStage)
  204 trfSig (ts @ (TypeSig {})) = AST.UTypeSigDecl <$> defineTypeVars (annContNoSema $ trfTypeSig' ts)
  205 trfSig (FixSig fs) = AST.UFixityDecl <$> (annContNoSema $ trfFixitySig fs)
  206 trfSig (PatSynSig ids typ)
  207   = AST.UPatTypeSigDecl <$> annContNoSema (AST.UPatternTypeSignature <$> trfAnnList ", " trfName' ids <*> trfType (hsib_body typ))
  208 trfSig (InlineSig name prag)
  209   = AST.UPragmaDecl <$> annContNoSema (AST.UInlinePragmaDecl <$> trfInlinePragma name prag)
  210 trfSig (SpecSig name (map hsib_body -> types) (inl_act -> phase))
  211   = AST.UPragmaDecl <$> annContNoSema (AST.USpecializeDecl <$> trfSpecializePragma name types phase)
  212 trfSig s = unhandledElement "signature" s

  214 trfSpecializePragma :: TransformName n r
  215                     => Located n -> [Located (HsType n)] -> Activation -> Trf (Ann AST.USpecializePragma (Dom r) RangeStage)
  216 trfSpecializePragma name types phase
  217   = annContNoSema $ AST.USpecializePragma <$> trfPhase (pure $ srcSpanStart (getLoc name)) phase
  218                                           <*> trfName name
  219                                           <*> (orderAnnList <$> trfAnnList ", " trfType' types)

  224 trfConDecl' :: TransformName n r => ConDecl n -> Trf (AST.UConDecl (Dom r) RangeStage)
  225 trfConDecl' (ConDeclH98 { con_name = name, con_qvars = tyVars, con_cxt = ctx, con_details = PrefixCon args })
  226   = AST.UConDecl <$> trfConTyVars tyVars <*> trfConCtx ctx <*> define (trfName name) <*> makeList " " atTheEnd (mapM trfType args)
  227 trfConDecl' (ConDeclH98 { con_name = name, con_qvars = tyVars, con_cxt = ctx, con_details = RecCon (unLoc -> flds) })
  228   = AST.URecordDecl <$> trfConTyVars tyVars <*> trfConCtx ctx <*> define (trfName name) <*> (between AnnOpenC AnnCloseC $ trfAnnList ", " trfFieldDecl' flds)
  229 trfConDecl' (ConDeclH98 { con_name = name, con_qvars = tyVars, con_cxt = ctx, con_details = InfixCon t1 t2 })
  230   = AST.UInfixConDecl <$> trfConTyVars tyVars <*> trfConCtx ctx <*> trfType t1 <*> define (trfOperator name) <*> trfType t2
  231 trfConDecl' gadt@(ConDeclGADT {}) = unhandledElement "normal constructor declaration" gadt

  241 trfGADTConDecl :: TransformName n r => Located (ConDecl n) -> Trf (Ann AST.UGadtConDecl (Dom r) RangeStage)

  244 trfGADTConDecl' :: TransformName n r => ConDecl n -> Trf (AST.UGadtConDecl (Dom r) RangeStage)
  245 trfGADTConDecl' (ConDeclGADT { con_names = names, con_type = hsib_body -> typ })
  246   = let nameLoc = collectLocs names
  247         typLoc = getLoc typ
  248         (vars, ctx, t) = getTypeVarsAndCtx typ
  249      in AST.UGadtConDecl <$> define (trfAnnList ", " trfName' names)
  250                          <*> focusOn (mkSrcSpan (srcSpanEnd nameLoc) (srcSpanStart typLoc)) (trfBindings vars)
  251                          <*> updateFocus (return . updateEnd (\_ -> srcSpanStart typLoc)) (focusAfterIfPresent AnnDot (trfCtx atTheStart ctx))
  252                          <*> trfGadtConType t
  253   where getTypeVarsAndCtx :: LHsType n -> ([LHsTyVarBndr n], LHsContext n, LHsType n)
  254         getTypeVarsAndCtx (L _ (HsForAllTy [] typ)) = getTypeVarsAndCtx typ
  255         getTypeVarsAndCtx (L _ (HsForAllTy bndrs typ)) = let (_,ctx,t) = getTypeVarsAndCtx typ in (bndrs, ctx, t)
  256         getTypeVarsAndCtx (L _ (HsQualTy ctx typ)) = let (vars,_,t) = getTypeVarsAndCtx typ in (vars, ctx, t)
  257         getTypeVarsAndCtx t = ([], L noSrcSpan [], t)

  289 trfInstanceRule' :: TransformName n r => HsType n -> Trf (AST.UInstanceRule (Dom r) RangeStage)
  290 trfInstanceRule' (HsForAllTy bndrs (unLoc -> HsQualTy ctx typ))
  291   = AST.UInstanceRule <$> (makeJust <$> annLocNoSema (pure $ collectLocs bndrs) (trfBindings bndrs))
  292                       <*> trfCtx (after AnnDot) ctx
  293                       <*> trfInstanceHead typ
  294 trfInstanceRule' (HsQualTy ctx typ) = AST.UInstanceRule <$> nothing "" " . " atTheStart
  295                                                         <*> trfCtx atTheStart ctx
  296                                                         <*> trfInstanceHead typ
  297 trfInstanceRule' (HsParTy typ) = instanceHead $ annContNoSema (AST.UInstanceHeadParen <$> trfInstanceHead typ)
  298 trfInstanceRule' (HsTyVar _ tv) = instanceHead $ annContNoSema (AST.UInstanceHeadCon <$> trfName tv)
  299 trfInstanceRule' (HsAppTy t1 t2) = instanceHead $ annContNoSema (AST.UInstanceHeadApp <$> trfInstanceHead t1 <*> trfType t2)
  300 trfInstanceRule' (HsOpTy t1 op t2) = instanceHead $ annContNoSema (AST.UInstanceHeadApp <$> annLocNoSema (pure $ getLoc t1 `combineSrcSpans` getLoc op) (AST.UInstanceHeadInfix <$> trfType t1 <*> trfOperator op) <*> trfType t2)
  301 trfInstanceRule' t = unhandledElement "instance rule" t

  303 instanceHead :: Trf (Ann AST.UInstanceHead (Dom r) RangeStage) -> Trf (AST.UInstanceRule (Dom r) RangeStage)
  304 instanceHead hd = AST.UInstanceRule <$> (nothing "" " . " atTheStart) <*> (nothing " " "" atTheStart) <*> hd

  306 makeInstanceRuleTyVars :: TransformName n r => Located n -> HsImplicitBndrs n [LHsType n] -> Trf (Ann AST.UInstanceRule (Dom r) RangeStage)
  307 makeInstanceRuleTyVars n vars
  308   | isSymOcc (occName (unLoc n))
  309   , leftOp:rest <- hsib_body vars
  310   , srcSpanStart (getLoc n) > srcSpanEnd (getLoc leftOp)
  311   = annContNoSema
  312       $ AST.UInstanceRule <$> nothing "" " . " atTheStart
  313                           <*> nothing " " "" atTheStart
  314                           <*> foldl foldTypeArgs
  315                                     (annLocNoSema (pure $ combineSrcSpans (getLoc leftOp) (getLoc n))
  316                                       (AST.UInstanceHeadInfix <$> trfType leftOp <*> trfOperator n)) rest
  317   | otherwise
  318   = annContNoSema
  319       $ AST.UInstanceRule <$> nothing "" " . " atTheStart
  320                           <*> nothing " " "" atTheStart
  321                           <*> foldl foldTypeArgs (copyAnnot AST.UInstanceHeadCon (trfName n)) (hsib_body vars)
  322   where foldTypeArgs base typ = annLocNoSema (pure $ combineSrcSpans (getLoc n) (getLoc typ)) $ AST.UInstanceHeadApp <$> base <*> (trfType typ)

  328 trfInstanceHead' :: TransformName n r => HsType n -> Trf (AST.UInstanceHead (Dom r) RangeStage)
  329 trfInstanceHead' = trfInstanceHead'' . cleanHsType where
  330   trfInstanceHead'' (HsForAllTy [] (unLoc -> t)) = trfInstanceHead' t
  331   trfInstanceHead'' (HsTyVar _ tv) = AST.UInstanceHeadCon <$> trfName tv
  332   trfInstanceHead'' (HsAppTy t1 t2) = AST.UInstanceHeadApp <$> trfInstanceHead t1 <*> trfType t2
  333   trfInstanceHead'' (HsParTy typ) = AST.UInstanceHeadParen <$> trfInstanceHead typ
  334   trfInstanceHead'' (HsOpTy t1 op t2)
  335     = AST.UInstanceHeadApp <$> (annLocNoSema (pure $ combineSrcSpans (getLoc t1) (getLoc op))
  336                                              (AST.UInstanceHeadInfix <$> trfType t1 <*> trfOperator op))
  337                           <*> trfType t2
  338   trfInstanceHead'' t = unhandledElement "instance head" t

  346 trfTypeEq :: TransformName n r => Located (TyFamInstEqn n) -> Trf (Ann AST.UTypeEqn (Dom r) RangeStage)
  347 trfTypeEq = trfLocNoSema $ \(TyFamEqn name pats _ rhs)
  348   -> AST.UTypeEqn <$> defineTypeVars (focusBefore AnnEqual (combineTypes name (hsib_body pats))) <*> trfType rhs

  369 createDeclHead :: TransformName n r => Located n -> LHsQTyVars n -> Trf (Ann AST.UDeclHead (Dom r) RangeStage)
  370 createDeclHead name (hsq_explicit -> lhs : rhs : rest)
  371   | srcSpanStart (getLoc name) > srcSpanEnd (getLoc lhs)
  372   -- infix declaration
  373   = wrapDeclHead rest
  374       $ annLocNoSema (addParenLocs $ getLoc lhs `combineSrcSpans` getLoc rhs)
  375                      (AST.UDHInfix <$> defineTypeVars (trfTyVar lhs) <*> define (trfOperator name) <*> defineTypeVars (trfTyVar rhs))
  376 createDeclHead name vars = defineTypeVars $ wrapDeclHead (hsq_explicit vars) (define $ copyAnnot AST.UDeclHead (trfName name))

  378 wrapDeclHead :: TransformName n r => [LHsTyVarBndr n] -> Trf (Ann AST.UDeclHead (Dom r) RangeStage) -> Trf (Ann AST.UDeclHead (Dom r) RangeStage)
  379 wrapDeclHead vars base
  380   = foldl (\t p -> do typ <- t
  381                       annLocNoSema (addParenLocs $ combineSrcSpans (getRange typ) (getLoc p))
  382                              (AST.UDHApp typ <$> trfTyVar p)
  383           ) base vars

  385 -- | Get the parentheses directly before and after (for parenthesized application)
  386 addParenLocs :: SrcSpan -> Trf SrcSpan
  387 addParenLocs sp
  388   = let possibleSpan = mkSrcSpan (updateCol (subtract 1) (srcSpanStart sp)) (updateCol (+1) (srcSpanEnd sp))
  389      in local (\s -> s { contRange = possibleSpan })
  390               (combineSrcSpans <$> (combineSrcSpans sp <$> tokenLoc AnnOpenP) <*> tokenLocBack AnnCloseP)

  393 createClassBody :: TransformName n r => [LSig n] -> LHsBinds n -> [LFamilyDecl n]
  394                                -> [LTyFamDefltEqn n] -> Trf (AnnMaybeG AST.UClassBody (Dom r) RangeStage)
  395 createClassBody sigs binds typeFams typeFamDefs
  396   = do isThereWhere <- isGoodSrcSpan <$> (tokenLoc AnnWhere)
  397        if isThereWhere
  398          then makeJust <$> annLocNoSema (combinedLoc <$> tokenLoc AnnWhere)
  399                                         (AST.UClassBody <$> makeList "" (after AnnWhere)
  400                                                                        (orderDefs . concat <$> sequenceA allDefs))
  401          else nothing " where " "" atTheEnd
  402   where combinedLoc wh = foldl combineSrcSpans wh allLocs
  403         allLocs = map getLoc sigs ++ map getLoc (bagToList binds) ++ map getLoc typeFams ++ map getLoc typeFamDefs
  404         allDefs = [getSigs, getBinds, getFams, getFamDefs]
  405         getSigs = mapM trfClassElemSig sigs
  406         getBinds = mapM (copyAnnot AST.UClsDef . trfBind) (bagToList binds)
  407         getFams = mapM (copyAnnot AST.UClsTypeFam . trfTypeFam) typeFams
  408         getFamDefs = mapM trfTypeFamDef typeFamDefs

  410 trfClassElemSig :: TransformName n r => Located (Sig n) -> Trf (Ann AST.UClassElement (Dom r) RangeStage)

  413                                   <*> trfType (hsib_body $ hswc_body typ))
  414   ClassOpSig True [name] typ -> AST.UClsDefSig <$> trfName name <*> trfType (hsib_body typ)
  415   ClassOpSig False names typ -> AST.UClsSig <$> (annContNoSema $ AST.UTypeSignature <$> define (makeNonemptyList ", " (mapM trfName names))
  416                                            <*> trfType (hsib_body typ))

  418   InlineSig name prag -> AST.UClsInline <$> trfInlinePragma name prag
  419   FixSig fixity -> AST.UClsFixity <$> annContNoSema (trfFixitySig fixity)

  438 trfInstBody :: TransformName n r => LHsBinds n -> [LSig n] -> [LTyFamInstDecl n] -> [LDataFamInstDecl n] -> Trf (AnnMaybeG AST.UInstBody (Dom r) RangeStage)

  468 trfInstDataFam :: TransformName n r => Located (DataFamInstDecl n) -> Trf (Ann AST.UInstBodyDecl (Dom r) RangeStage)
  469 trfInstDataFam = trfLocNoSema $ \case
  470   (DataFamInstDecl tc (hsib_body -> pats) _ (HsDataDefn dn ctx _ ks cons derivs) _)
  471     | all ((\case ConDeclH98{} -> True; _ -> False) . unLoc) cons
  472     -> AST.UInstBodyDataDecl
  473          <$> trfDataKeyword dn
  474          <*> annLocNoSema (pure $ collectLocs pats `combineSrcSpans` getLoc tc `combineSrcSpans` getLoc ctx)
  475                           (AST.UInstanceRule <$> nothing "" " . " atTheStart
  476                                              <*> trfCtx atTheStart ctx
  477                                              <*> transformNameAndPats tc pats)
  478          <*> trfAnnList "" trfConDecl' cons
  479          <*> makeIndentedList atTheEnd (mapM trfDerivings (unLoc derivs))
  480     | otherwise
  481     -> AST.UInstBodyGadtDataDecl
  482         <$> trfDataKeyword dn
  483         <*> annLocNoSema (pure $ collectLocs pats `combineSrcSpans` getLoc tc `combineSrcSpans` getLoc ctx)
  484                          (AST.UInstanceRule <$> nothing "" " . " atTheStart
  485                                             <*> trfCtx atTheStart ctx
  486                                             <*> transformNameAndPats tc pats)
  487         <*> trfKindSig ks
  488         <*> trfAnnList "" trfGADTConDecl' cons
  489         <*> makeIndentedList atTheEnd (mapM trfDerivings (unLoc derivs))
  490   where transformNameAndPats tc pats -- TODO: this is simpler with lexical fixity
  491           | all (\p -> srcSpanEnd (getLoc tc) < srcSpanStart (getLoc p)) pats -- prefix instance head application
  492           = foldl (\r t -> annLocNoSema (combineSrcSpans (getLoc t) . getRange <$> r)
  493                                           (AST.UInstanceHeadApp <$> r <*> (trfType t)))
  494                   (copyAnnot AST.UInstanceHeadCon (trfName tc)) pats
  495         transformNameAndPats tc (p:rest)
  496           | otherwise -- infix instance head application
  497           = foldl (\r t -> annLocNoSema (combineSrcSpans (getLoc t) . getRange <$> r)
  498                                           (AST.UInstanceHeadApp <$> r <*> (trfType t)))
  499                   (annLocNoSema (pure $ getLoc p `combineSrcSpans` getLoc tc)
  500                           (AST.UInstanceHeadInfix <$> trfType p <*> trfOperator tc)) rest

  502 trfPatternSynonym :: forall n r . TransformName n r => PatSynBind n n -> Trf (AST.UPatternSynonym (Dom r) RangeStage)
  503 trfPatternSynonym (PSB id _ lhs def dir)
  504   = let sep = case dir of ImplicitBidirectional -> AnnEqual
  505                           _                     -> AnnLarrow
  506         rhsLoc = combineSrcSpans (getLoc def) <$> tokenLoc sep
  507         -- we use the selector name instead of the pattern variable name
  508         rewrites = case lhs of RecordPatSyn flds -> map (\r -> (unLoc (recordPatSynPatVar r), unLoc (recordPatSynSelectorId r))) flds
  509                                _                 -> []
  510         changedRhs = biplateRef .- (\n -> case lookup n rewrites of Just x -> x; Nothing -> n) $ def
  511      in AST.UPatternSynonym <$> trfPatSynLhs id lhs sep
  512                             <*> annLocNoSema rhsLoc (trfPatSynRhs dir changedRhs)
  513 
  514   where trfPatSynLhs :: Located n -> HsPatSynDetails (Located n) -> AnnKeywordId -> Trf (Ann AST.UPatSynLhs (Dom r) RangeStage)
  515         trfPatSynLhs id (PrefixPatSyn args) _
  516           = annLocNoSema (pure $ foldLocs (getLoc id : map getLoc args)) $ AST.UNormalPatSyn <$> define (trfName id) <*> trfAnnList " " trfName' args
  517         trfPatSynLhs op (InfixPatSyn lhs rhs) _
  518           = annLocNoSema (pure $ getLoc lhs `combineSrcSpans` getLoc rhs) $ AST.UInfixPatSyn <$> define (trfName lhs) <*> trfOperator op <*> trfName rhs
  519         trfPatSynLhs id (RecordPatSyn flds) kw
  520           = annLocNoSema (mkSrcSpan (srcSpanStart (getLoc id)) <$> before kw)
  521               $ AST.URecordPatSyn <$> define (trfName id) <*> trfAnnList ", " trfName' (map recordPatSynSelectorId flds)
  522 
  523         trfPatSynRhs :: HsPatSynDir n -> Located (Pat n) -> Trf (AST.UPatSynRhs (Dom r) RangeStage)
  524         trfPatSynRhs ImplicitBidirectional pat = AST.UBidirectionalPatSyn <$> trfPattern pat <*> nothing " where " "" atTheEnd
  525         trfPatSynRhs (ExplicitBidirectional mg) pat = AST.UBidirectionalPatSyn <$> trfPattern pat <*> (makeJust <$> trfPatSynWhere mg)
  526         trfPatSynRhs Unidirectional pat = AST.UOneDirectionalPatSyn <$> trfPattern pat
  527 
  528         trfPatSynWhere :: MatchGroup n (LHsExpr n) -> Trf (Ann AST.UPatSynWhere (Dom r) RangeStage)
  529         trfPatSynWhere (MG { mg_alts = alts }) = annLocNoSema (pure $ getLoc alts) (AST.UPatSynWhere <$> makeIndentedList (after AnnWhere) (mapM (trfMatch (unLoc id)) (unLoc alts)))
