   81 type AssocSema dom = ( AssocData (SemanticInfo' dom SameInfoModuleCls), AssocData (SemanticInfo' dom SameInfoImportCls)
   82                      , AssocData (SemanticInfo' dom SameInfoNameCls), AssocData (SemanticInfo' dom SameInfoExprCls)
   83                      , AssocData (SemanticInfo' dom SameInfoWildcardCls) )

   85 astDebug :: (ASTDebug e dom st, AssocSema dom) => e dom st -> String
   86 astDebug ast = toList (astDebugToJson (astDebug' ast))
   87 
   88 astDebugToJson :: AssocSema dom => [DebugNode dom] -> Seq Char
   89 astDebugToJson nodes = fromList "[ " >< childrenJson >< fromList " ]"
   90     where treeNodes = List.filter (\case TreeNode {} -> True; _ -> False) nodes
   91           childrenJson = case map debugTreeNode treeNodes of
   92                            first:rest -> first >< foldl (><) Seq.empty (fmap (fromList ", " ><) (fromList rest))
   93                            []         -> Seq.empty
   94           debugTreeNode (TreeNode "" s) = astDebugElemJson s
   95           debugTreeNode (TreeNode (dropWhile (=='_') -> l) s) = astDebugElemJson (nodeName .- (("<span class='astlab'>" ++ l ++ "</span>: ") ++) $ s)
   96           debugTreeNode (SimpleNode {}) = error "debugTreeNode: simple SimpleNode not allowed"
   97 
   98 astDebugElemJson :: AssocSema dom => TreeDebugNode dom -> Seq Char
   99 astDebugElemJson (TreeDebugNode name info children)
  100   = fromList "{ \"text\" : \"" >< fromList name
  101      >< fromList "\", \"state\" : { \"opened\" : true }, \"a_attr\" : { \"data-range\" : \""
  102      >< fromList (shortShowSpan (semaInfoTypeRng info))
  103      >< fromList "\", \"data-elems\" : \""
  104      >< foldl (><) Seq.empty dataElems
  105      >< fromList "\", \"data-sema\" : \""
  106      >< fromList (showSema info)
  107      >< fromList "\" }, \"children\" : "
  108      >< astDebugToJson children >< fromList " }"
  109   where dataElems = catMaybes (map (\case SimpleNode l v -> Just (fromList (formatScalarElem l v)); _ -> Nothing) children)
  110         formatScalarElem l v = "<div class='scalarelem'><span class='astlab'>" ++ l ++ "</span>: " ++ tail (init (show v)) ++ "</div>"
  111         showSema info = "<div class='semaname'>" ++ assocName info ++ "</div>"
  112                           ++ concatMap (\(l,i) -> "<div class='scalarelem'><span class='astlab'>" ++ l ++ "</span>: " ++ i ++ "</div>") (toAssoc info)

  114 class AssocData a where
  115   assocName :: a -> String
  116   toAssoc :: a -> [(String, String)]
  117 
  118 instance AssocSema dom => AssocData (SemanticInfoType dom) where
  119   assocName (DefaultInfoType {}) = "NoSemanticInfo"
  120   assocName s@(NameInfoType {}) = assocName (semaInfoTypeName s)
  121   assocName s@(ExprInfoType {}) = assocName (semaInfoTypeExpr s)
  122   assocName s@(ImportInfoType {}) = assocName (semaInfoTypeImport s)
  123   assocName s@(ModuleInfoType {}) = assocName (semaInfoTypeModule s)
  124   assocName s@(ImplicitFieldInfoType {}) = assocName (semaInfoTypeImplicitFld s)
  125 
  126   toAssoc (DefaultInfoType {}) = []
  127   toAssoc s@(NameInfoType {}) = toAssoc (semaInfoTypeName s)
  128   toAssoc s@(ExprInfoType {}) = toAssoc (semaInfoTypeExpr s)
  129   toAssoc s@(ImportInfoType {}) = toAssoc (semaInfoTypeImport s)
  130   toAssoc s@(ModuleInfoType {}) = toAssoc (semaInfoTypeModule s)
  131   toAssoc s@(ImplicitFieldInfoType {}) = toAssoc (semaInfoTypeImplicitFld s)

  134 instance AssocData ScopeInfo where
  135   assocName _ = "ScopeInfo"
  136   toAssoc si = [ ("namesInScope", inspectScope (semanticsScope si)) ]
  137 
  138 instance HasNameInfo' (NameInfo n) => AssocData (NameInfo n) where
  139   assocName _ = "NameInfo"
  140 
  141   toAssoc ni = [ ("name", maybe "<ambiguous>" inspect (semanticsName ni))
  142                , ("isDefined", show (semanticsDefining ni))
  143                , ("namesInScope", inspectScope (semanticsScope ni))
  144                ]
  145 
  146 instance AssocData CNameInfo where
  147   assocName _ = "CNameInfo"
  148   toAssoc ni = [ ("name", inspect (semanticsId ni))
  149                , ("isDefined", show (semanticsDefining ni))
  150                , ("fixity", maybe "" (showSDocUnsafe . ppr) (semanticsFixity ni))
  151                , ("namesInScope", inspectScope (semanticsScope ni))
  152                ]
  153 
  154 instance (HasModuleInfo' (ModuleInfo n)) => AssocData (ModuleInfo n) where
  155   assocName _ = "ModuleInfo"
  156   toAssoc mi = [ ("moduleName", showSDocUnsafe (ppr (semanticsModule mi)))
  157                , ("isBoot", show (isBootModule mi))
  158                , ("implicitImports", concat (intersperse ", " (map inspect (semanticsImplicitImports mi))))
  159                ]
  160 
  161 instance (HasImportInfo' (ImportInfo n)) => AssocData (ImportInfo n) where
  162   assocName _ = "ImportInfo"
  163   toAssoc ii = [ ("moduleName", showSDocUnsafe (ppr (semanticsImportedModule ii)))
  164                , ("availableNames", concat (intersperse ", " (map inspect (semanticsAvailable ii))))
  165                , ("importedNames", concat (intersperse ", " (map inspect (semanticsImported ii))))
  166                ]
  167 
  168 instance AssocData ImplicitFieldInfo where
  169   assocName _ = "ImplicitFieldInfo"
  170   toAssoc ifi = [ ("bindings", concat (intersperse ", " (map (\(from,to) -> "(" ++ inspect from ++ " -> " ++ inspect to ++ ")") (semanticsImplicitFlds ifi))))
  171                 ]

  173 inspectScope :: InspectableName n => [[(n, Maybe [UsageSpec], Maybe n)]] -> String
  174 inspectScope = concat . intersperse " | " . map (concat . intersperse ", " . map inspect)

  176 class Outputable n => InspectableName n where
  177   inspect :: n -> String
  178 
  179 instance InspectableName n => InspectableName (n, Maybe [UsageSpec], Maybe n) where
  180   inspect (n,usage,parent) = inspect n ++ showSDocUnsafe (ppr usage) ++ maybe "" (\p -> "( in " ++ showSDocUnsafe (ppr p) ++ ")") parent
  181 
  182 instance InspectableName GHC.Name where
  183   inspect name = showSDocUnsafe (ppr name) ++ "[" ++ show (getUnique name) ++ "]"
  184 
  185 instance InspectableName GHC.RdrName where
  186   inspect name = showSDocUnsafe (ppr name)
  187 
  188 instance InspectableName GHC.Id where
  189   inspect name = showSDocUnsafe (ppr name) ++ "[" ++ show (getUnique name) ++ "] :: " ++ showSDocOneLine unsafeGlobalDynFlags (ppr (idType name)) ++ " (" ++ addendum ++ ")"
  190     where addendum = concat $ intersperse "," $ map (\v -> showSDocUnsafe (ppr v) ++ "[" ++ show (getUnique v) ++ "]") (getTVs (idType name))
  191 
  192 getTVs :: GHC.Type -> [GHC.Var]
  193 getTVs t
  194   | Just tv <- getTyVar_maybe t = [tv]
  195   | Just (op, arg) <- splitAppTy_maybe t = getTVs op `union` getTVs arg
  196   | Just (_, t') <- splitForAllTy_maybe t = getTVs t'
  197   | otherwise = []

  205 class GAstDebug f dom where
  206   gAstDebug :: f p -> [DebugNode dom]
  207 
  208 instance GAstDebug V1 dom where
  209   gAstDebug _ = error "GAstDebug V1"
  210 
  211 instance GAstDebug U1 dom where
  212   gAstDebug U1 = []
  213 
  214 instance (GAstDebug f dom, GAstDebug g dom) => GAstDebug (f :+: g) dom where
  215   gAstDebug (L1 x) = gAstDebug x
  216   gAstDebug (R1 x) = gAstDebug x
  217 
  218 instance (GAstDebug f dom, GAstDebug g dom) => GAstDebug (f :*: g) dom where
  219   gAstDebug (x :*: y)
  220     = gAstDebug x ++ gAstDebug y
  221 
  222 instance {-# OVERLAPPING #-} ASTDebug e dom st => GAstDebug (K1 i (e dom st)) dom where
  223   gAstDebug (K1 x) = astDebug' x
  224 
  225 instance {-# OVERLAPPABLE #-} Show x => GAstDebug (K1 i x) dom where
  226   gAstDebug (K1 x) = [SimpleNode "" (show x)]
  227 
  228 instance (GAstDebug f dom, Constructor c) => GAstDebug (M1 C c f) dom where
  229   gAstDebug c@(M1 x) = [TreeNode "" (TreeDebugNode (conName c) undefined (gAstDebug x))]
  230 
  231 instance (GAstDebug f dom, Selector s) => GAstDebug (M1 S s f) dom where
  232   gAstDebug s@(M1 x) = traversal&nodeLabel .= selName s $ gAstDebug x
  233 
  234 -- don't have to do anything with datatype metainfo
  235 instance GAstDebug f dom => GAstDebug (M1 D t f) dom where
  236   gAstDebug (M1 x) = gAstDebug x
