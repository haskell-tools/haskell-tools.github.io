   18 -- | The modules of a library, executable, test or benchmark. A package contains one or more module collection.
   19 data ModuleCollection k
   20   = ModuleCollection { _mcId :: ModuleCollectionId
   21                      , _mcRoot :: FilePath
   22                      , _mcSourceDirs :: [FilePath]
   23                      , _mcModuleFiles :: [(ModuleNameStr, FilePath)]
   24                      , _mcModules :: (Map.Map k ModuleRecord)
   25                      , _mcFlagSetup :: (DynFlags -> IO DynFlags) -- ^ Sets up the ghc environment for compiling the modules of this collection
   26                      , _mcLoadFlagSetup :: (DynFlags -> IO DynFlags) -- ^ Sets up the ghc environment for dependency analysis
   27                      , _mcDependencies :: [ModuleCollectionId]
   28                      }
   29 
   30 modCollToSfk :: ModuleCollection ModuleNameStr -> ModuleCollection SourceFileKey
   31 modCollToSfk ModuleCollection{..} = ModuleCollection{ _mcModules = Map.mapKeys (SourceFileKey "") _mcModules, ..}
   32 
   33 -- | The state of a module.
   34 data ModuleRecord
   35       = ModuleNotLoaded { _recModuleWillNeedCode :: Bool
   36                         , _recModuleExposed :: Bool
   37                         }
   38       | ModuleParsed { _parsedRecModule :: UnnamedModule (Dom RdrName)
   39                      , _modRecMS :: ModSummary
   40                      }
   41       | ModuleRenamed { _renamedRecModule :: UnnamedModule (Dom GHC.Name)
   42                       , _modRecMS :: ModSummary
   43                       }
   44       | ModuleTypeChecked { _typedRecModule :: UnnamedModule IdDom
   45                           , _modRecMS :: ModSummary
   46                           }
   47       | ModuleCodeGenerated { _typedRecModule :: UnnamedModule IdDom
   48                             , _modRecMS :: ModSummary
   49                             }
   50 
   51 -- | An alias for module names
   52 type ModuleNameStr = String
   53 
   54 -- | This data structure identifies a module collection.
   55 data ModuleCollectionId = DirectoryMC FilePath
   56                         | LibraryMC String
   57                         | ExecutableMC String String
   58                         | TestSuiteMC String String
   59                         | BenchmarkMC String String
   60  deriving (Eq, Ord, Show)
   61 
   62 instance Eq (ModuleCollection k) where
   63   (==) = (==) `on` _mcId
   64 
   65 instance Show k => Show (ModuleCollection k) where
   66   show (ModuleCollection id root srcDirs mapping mods _ _ deps)
   67     = "ModuleCollection (" ++ show id ++ ") " ++ root ++ " " ++ show srcDirs ++ " " ++ show mapping
   68         ++ " (" ++ show mods ++ ") " ++ show deps
   69 
   70 makeReferences ''ModuleCollection
   71 makeReferences ''ModuleRecord
   72 
   73 instance Show ModuleRecord where
   74   show (ModuleNotLoaded code exposed) = "ModuleNotLoaded " ++ show code ++ " " ++ show exposed
   75   show mr@(ModuleParsed {}) = "ModuleParsed (" ++ (GHC.moduleNameString $ GHC.moduleName $ GHC.ms_mod $ fromJust $ mr ^? modRecMS) ++ ")"
   76   show mr@(ModuleRenamed {}) = "ModuleRenamed (" ++ (GHC.moduleNameString $ GHC.moduleName $ GHC.ms_mod $ fromJust $ mr ^? modRecMS) ++ ")"
   77   show mr@(ModuleTypeChecked {}) = "ModuleTypeChecked (" ++ (GHC.moduleNameString $ GHC.moduleName $ GHC.ms_mod $ fromJust $ mr ^? modRecMS) ++ ")"
   78   show mr@(ModuleCodeGenerated {}) = "ModuleCodeGenerated (" ++ (GHC.moduleNameString $ GHC.moduleName $ GHC.ms_mod $ fromJust $ mr ^? modRecMS) ++ ")"
