    1 -- | Creating a dependency graph of the modules loaded into a session.
    2 -- Code copied from GHC because it is not public in GhcMake module
    3 module Language.Haskell.Tools.Daemon.ModuleGraph (moduleGraphNodes, getModFromNode) where
    4 
    5 import qualified Data.Map as Map (fromList, Map, lookup)
    6 import Data.Maybe (Maybe(..), mapMaybe)
    7 
    8 import Digraph as GHC (Graph, graphFromEdgedVerticesOrd)
    9 import FastString as GHC (FastString, fsLit)
   10 import GHC
   11 import HscTypes as GHC
   12 
   13 type NodeKey   = (ModuleName, HscSource)
   14 type NodeMap a = Map.Map NodeKey a
   15 type SummaryNode = (ModSummary, Int, [Int])
   16 
   17 getModFromNode :: SummaryNode -> ModSummary
   18 getModFromNode (ms, _, _) = ms
   19 
   20 -- Creates the dependency graph of modules currently loaded. Used for checking which modules need
   21 -- to be reloaded after a recompilation.
   22 moduleGraphNodes :: Bool -> [ModSummary]
   23   -> (Graph SummaryNode, HscSource -> ModuleName -> Maybe SummaryNode)
   24 moduleGraphNodes drop_hs_boot_nodes summaries = (graphFromEdgedVerticesOrd nodes, lookup_node)
   25   where
   26     numbered_summaries = zip summaries [1..]
   27 
   28     lookup_node :: HscSource -> ModuleName -> Maybe SummaryNode
   29     lookup_node hs_src mod = Map.lookup (mod, hs_src) node_map
   30 
   31     lookup_key :: HscSource -> ModuleName -> Maybe Int
   32     lookup_key hs_src mod = fmap summaryNodeKey (lookup_node hs_src mod)
   33 
   34     node_map :: NodeMap SummaryNode
   35     node_map = Map.fromList [ ((moduleName (ms_mod s), (ms_hsc_src s)), node)
   36                             | node@(s, _, _) <- nodes ]
   37 
   38     nodes :: [SummaryNode]
   39     nodes = [ (s, key, out_keys)
   40             | (s, key) <- numbered_summaries
   41             , not (isBootSummary s && drop_hs_boot_nodes)
   42             , let out_keys = out_edge_keys hs_boot_key (map unLoc (ms_home_srcimps s)) ++
   43                              out_edge_keys HsSrcFile   (map unLoc (ms_home_imps s)) ++
   44                              (-- see [boot-edges] below
   45                               if drop_hs_boot_nodes || ms_hsc_src s == HsBootFile
   46                               then []
   47                               else case lookup_key HsBootFile (ms_mod_name s) of
   48                                     Nothing -> []
   49                                     Just k  -> [k]) ]
   50 
   51     hs_boot_key | drop_hs_boot_nodes = HsSrcFile
   52                 | otherwise          = HsBootFile
   53 
   54     out_edge_keys :: HscSource -> [ModuleName] -> [Int]
   55     out_edge_keys hi_boot ms = mapMaybe (lookup_key hi_boot) ms
   56 
   57 summaryNodeKey :: SummaryNode -> Int
   58 summaryNodeKey (_, k, _) = k
   59 
   60 ms_home_imps :: ModSummary -> [Located ModuleName]
   61 ms_home_imps = home_imps . ms_imps
   62 
   63 ms_home_srcimps :: ModSummary -> [Located ModuleName]
   64 ms_home_srcimps = home_imps . ms_srcimps
   65 
   66 home_imps :: [(Maybe FastString, Located ModuleName)] -> [Located ModuleName]
   67 home_imps imps = [ lmodname |  (mb_pkg, lmodname) <- imps,
   68                                   isLocal mb_pkg ]
   69   where isLocal Nothing = True
   70         isLocal (Just pkg) | pkg == fsLit "this" = True -- "this" is special
   71         isLocal _ = False
