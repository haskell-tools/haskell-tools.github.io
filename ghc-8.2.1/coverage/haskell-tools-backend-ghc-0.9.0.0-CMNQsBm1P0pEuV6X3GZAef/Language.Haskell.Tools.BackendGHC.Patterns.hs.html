   30 trfPattern :: TransformName n r => Located (Pat n) -> Trf (Ann AST.UPattern (Dom r) RangeStage)
   31 -- field wildcards are not directly represented in GHC AST
   32 trfPattern (L l (ConPatIn name (RecCon (HsRecFields flds _)))) | any ((l ==) . getLoc) flds
   33   = focusOn l $ do
   34       let (fromWC, notWC) = partition ((l ==) . getLoc) flds
   35       normalFields <- mapM (trfLocNoSema trfPatternField') notWC
   36       wildc <- annLocNoSema (tokenLocBack AnnDotdot) (AST.UFieldWildcardPattern <$> annCont (createImplicitFldInfo (unLoc . (\(VarPat n) -> n) . unLoc) (map unLoc fromWC)) (pure AST.FldWildcard))
   37       annLocNoSema (pure l) (AST.URecPat <$> trfName name <*> makeNonemptyList ", " (pure (normalFields ++ [wildc])))
   38 trfPattern p = trfLocNoSema trfPattern' (correctPatternLoc p)
   39 
   40 -- | Locations for right-associative infix patterns are incorrect in GHC AST
   41 correctPatternLoc :: Located (Pat n) -> Located (Pat n)
   42 correctPatternLoc (L _ p@(ConPatIn _ (InfixCon left right)))
   43   = L (getLoc (correctPatternLoc left) `combineSrcSpans` getLoc (correctPatternLoc right)) p
   44 correctPatternLoc p = p
   45 
   46 trfPattern' :: TransformName n r => Pat n -> Trf (AST.UPattern (Dom r) RangeStage)
   47 trfPattern' (WildPat _) = pure AST.UWildPat
   48 trfPattern' (VarPat name) = define $ AST.UVarPat <$> trfName name
   49 trfPattern' (LazyPat pat) = AST.UIrrefutablePat <$> trfPattern pat
   50 trfPattern' (AsPat name pat) = AST.UAsPat <$> define (trfName name) <*> trfPattern pat
   51 trfPattern' (ParPat pat) = AST.UParenPat <$> trfPattern pat
   52 trfPattern' (BangPat pat) = AST.UBangPat <$> trfPattern pat
   53 trfPattern' (ListPat pats _ _) = AST.UListPat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   54 trfPattern' (TuplePat pats Boxed _) = AST.UTuplePat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   55 trfPattern' (TuplePat pats Unboxed _) = AST.UUnboxTuplePat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   56 trfPattern' (PArrPat pats _) = AST.UParArrPat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   57 trfPattern' (ConPatIn name (PrefixCon args)) = AST.UAppPat <$> trfName name <*> makeList " " atTheEnd (mapM trfPattern args)
   58 trfPattern' (ConPatIn name (RecCon (HsRecFields flds _))) = AST.URecPat <$> trfName name <*> trfAnnList ", " trfPatternField' flds
   59 trfPattern' (ConPatIn name (InfixCon left right)) = AST.UInfixAppPat <$> trfPattern left <*> trfOperator name <*> trfPattern right
   60 trfPattern' (ViewPat expr pat _) = AST.UViewPat <$> trfExpr expr <*> trfPattern pat
   61 trfPattern' (SplicePat qq@(HsQuasiQuote {})) = AST.UQuasiQuotePat <$> annContNoSema (trfQuasiQuotation' qq)
   62 trfPattern' (SplicePat splice) = AST.USplicePat <$> trfSplice splice
   63 trfPattern' (LitPat lit) = AST.ULitPat <$> annContNoSema (trfLiteral' lit)
   64 trfPattern' (SigPatIn pat (hsib_body . hswc_body -> typ)) = AST.UTypeSigPat <$> trfPattern pat <*> trfType typ
   65 trfPattern' (NPat (ol_val . unLoc -> lit) _ _ _) = AST.ULitPat <$> annContNoSema (trfOverloadedLit lit)
   66 trfPattern' (NPlusKPat id (L l lit) _ _ _ _) = AST.UNPlusKPat <$> define (trfName id) <*> annLocNoSema (pure l) (trfOverloadedLit (ol_val lit))
   67 trfPattern' (CoPat _ pat _) = trfPattern' pat -- coercion pattern introduced by GHC
   68 trfPattern' p = unhandledElement "pattern" p
   69 
   70 trfPatternField' :: TransformName n r => HsRecField n (LPat n) -> Trf (AST.UPatternField (Dom r) RangeStage)
   71 trfPatternField' (HsRecField id arg False) = AST.UNormalFieldPattern <$> trfName (getFieldOccName id) <*> trfPattern arg
   72 trfPatternField' (HsRecField id _ True) = AST.UFieldPunPattern <$> trfName (getFieldOccName id)
