    4 -- | Setting the package database to use when compiling modules. The daemon must have one single
    5 -- package database that cannot be changed after a package is loaded using that package database.
    6 -- Available package databases are the cabal global, the cabal sandbox, the stack or one that had
    7 -- been explicitely set by a file path.
    8 module Language.Haskell.Tools.Daemon.PackageDB (PackageDB(..), packageDBLoc, detectAutogen) where
    9 
   10 import Control.Applicative (Applicative(..), (<$>), Alternative(..))
   11 import Control.Exception (SomeException, try)
   12 import Control.Monad (Monad(..), when)
   13 import Data.Aeson (FromJSON(..))
   14 import Data.Char (isSpace)
   15 import Data.List
   16 import GHC.Generics (Generic(..))
   17 import System.Directory (withCurrentDirectory, doesFileExist, doesDirectoryExist)
   18 import System.FilePath (FilePath, (</>))
   19 import System.Process (readProcessWithExitCode)
   20 
   21 -- | Possible package database configurations.
   22 data PackageDB = AutoDB -- ^ Decide the package database automatically.
   23                | DefaultDB -- ^ Use the global cabal package database (like when using ghc).
   24                | CabalSandboxDB -- ^ Use the sandboxed cabal package database.
   25                | StackDB -- ^ Use the stack package databases (local and snapshot).
   26                | ExplicitDB { packageDBPath :: FilePath } -- ^ Set the package database explicitely.
   27   deriving (Show, Generic)
   28 
   29 instance FromJSON PackageDB
   30 
   31 -- | Finds the location of the package database based on the configuration.
   32 packageDBLoc :: PackageDB -> FilePath -> IO [FilePath]
   33 packageDBLoc AutoDB path = (++) <$> packageDBLoc StackDB path <*> packageDBLoc CabalSandboxDB path
   34 packageDBLoc DefaultDB _ = return []
   35 packageDBLoc CabalSandboxDB path = do
   36   hasConfigFile <- doesFileExist (path </> "cabal.config")
   37   hasSandboxFile <- doesFileExist (path </> "cabal.sandbox.config")
   38   config <- if hasConfigFile then readFile (path </> "cabal.config")
   39               else if hasSandboxFile then readFile (path </> "cabal.sandbox.config")
   40                                      else return ""
   41   return $ map (drop (length "package-db: ")) $ filter ("package-db: " `isPrefixOf`) $ lines config
   42 packageDBLoc StackDB path = withCurrentDirectory path $ (fmap $ either (\(_ :: SomeException) -> []) id) $ try $ do
   43      (_, globalDB, globalDBErrs) <- readProcessWithExitCode "stack" ["path", "--allow-different-user", "--global-pkg-db"] ""
   44      (_, snapshotDB, snapshotDBErrs) <- readProcessWithExitCode "stack" ["path", "--allow-different-user", "--snapshot-pkg-db"] ""
   45      (_, localDB, localDBErrs) <- readProcessWithExitCode "stack" ["path", "--allow-different-user", "--local-pkg-db"] ""
   46      return $ [trim localDB | null localDBErrs] ++ [trim snapshotDB | null snapshotDBErrs] ++ [trim globalDB | null globalDBErrs]
   47 packageDBLoc (ExplicitDB dir) path = do
   48   hasDir <- doesDirectoryExist (path </> dir)
   49   if hasDir then return [path </> dir]
   50             else return []
   51 
   52 -- | Gets the (probable) location of autogen folder depending on which type of
   53 -- build we are using.
   54 detectAutogen :: FilePath -> PackageDB -> IO (Maybe FilePath)
   55 detectAutogen root AutoDB = do
   56   defDB <- detectAutogen root DefaultDB
   57   sandboxDB <- detectAutogen root CabalSandboxDB
   58   stackDB <- detectAutogen root StackDB
   59   return $ choose [ defDB, sandboxDB, stackDB ]
   60 detectAutogen root DefaultDB = ifExists (root </> "dist" </> "build" </> "autogen")
   61 detectAutogen root (ExplicitDB _) = ifExists (root </> "dist" </> "build" </> "autogen")
   62 detectAutogen root CabalSandboxDB = ifExists (root </> "dist" </> "build" </> "autogen")
   63 detectAutogen root StackDB = (fmap $ either (\(_ :: SomeException) -> Nothing) id) $ try $ do
   64   dir <- withCurrentDirectory root $ do
   65     (_, distDir, distDirErrs) <- readProcessWithExitCode "stack" ["path", "--allow-different-user", "--dist-dir"] ""
   66     when (not $ null distDirErrs)  -- print errors if they occurred
   67       $ putStrLn $ "Errors while checking dist directory with stack: " ++ distDirErrs
   68     return $ trim distDir
   69   ifExists $ root </> dir </> "build" </> "autogen"
   70 
   71 trim :: String -> String
   72 trim = f . f
   73    where f = reverse . dropWhile isSpace
   74 
   75 choose :: Alternative f => [f a] -> f a
   76 choose = foldl (<|>) empty
   77 
   78 ifExists :: FilePath -> IO (Maybe FilePath)
   79 ifExists fp = do exists <- doesDirectoryExist fp
   80                  if exists then return (Just fp)
   81                            else return Nothing
