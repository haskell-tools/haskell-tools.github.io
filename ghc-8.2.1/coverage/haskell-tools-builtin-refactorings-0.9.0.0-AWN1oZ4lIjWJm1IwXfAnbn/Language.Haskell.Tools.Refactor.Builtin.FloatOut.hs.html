   21 floatOutRefactoring :: (FloatOutDefinition dom, HasModuleInfo dom) => RefactoringChoice dom
   22 floatOutRefactoring = SelectionRefactoring "FloatOut" (localRefactoring . floatOut)
   23 
   24 type FloatOutDefinition dom = (HasNameInfo dom, HasScopeInfo dom)
   25 
   26 floatOut :: FloatOutDefinition dom => RealSrcSpan -> LocalRefactoring dom
   27 floatOut sp mod
   28   = do (mod', st) <- runStateT (nodesContaining sp !~ extractAndInsert sp $ mod) NotEncountered
   29        case st of NotEncountered -> refactError "No definition is selected. The selection range must be inside the definition."
   30                   Extracted bnds -> -- insert it to the global definition list
   31                                     return $ modDecl & annListElems .- (++ map toTopLevel bnds) $ removeEmpties mod'
   32                   Inserted -> -- already inserted to a local scope
   33                                return (removeEmpties mod')
   34   where toTopLevel :: LocalBind dom -> Decl dom
   35         toTopLevel (LocalValBind vb) = mkValueBinding vb
   36         toTopLevel (LocalTypeSig sg) = mkTypeSigDecl sg
   37         toTopLevel (LocalFixity fx) = mkFixityDecl fx
   38 
   39         removeEmpties = removeEmptyBnds (nodesContaining sp) (nodesContaining sp)
   40 
   41 data FloatState dom = NotEncountered | Extracted [LocalBind dom] | Inserted
   42 
   43 extractAndInsert :: FloatOutDefinition dom => RealSrcSpan -> LocalBindList dom -> StateT (FloatState dom) (LocalRefactor dom) (LocalBindList dom)
   44 extractAndInsert sp locs
   45   | hasSharedSig = refactError "Cannot float out a definition, since it has a signature shared with other bindings that stay in the scope."
   46   | not (null nameConflicts) = refactError $ "Cannot float out a definition, since it would cause a name conflicts in the target scope: "
   47                                                 ++ concat (intersperse ", " nameConflicts)
   48   | not (null implicitConflicts) = refactError $ "Cannot float out a definition, since it uses the implicit parameters: "
   49                                                    ++ concat (intersperse ", " implicitConflicts)
   50   | otherwise = get >>= \case NotEncountered -> put (Extracted floated) >> return filteredLocs
   51                               Extracted binds -> put Inserted >> (return $ annListElems .- (++ binds) $ locs)
   52                               Inserted -> return locs
   53   where selected = locs ^? annList & filtered (isInside sp)
   54         floated = normalizeElements $ selected ++ (locs ^? annList & filtered (nameIsSelected . (^? elementName)))
   55           where nameIsSelected [n] = n `elem` concatMap (^? elementName) selected
   56                 nameIsSelected _ = False
   57 
   58         filteredLocs = filterList (\e -> not (getRange e `elem` floatedElemRanges)) locs
   59           where floatedElemRanges = map getRange floated
   60         hasSharedSig = any (\e -> not $ null ((filteredLocs ^? annList & elementName) `intersect` (e ^? elementName))) selected
   61         conflicts = map checkConflict selected
   62 
   63         nameConflicts = concat $ map fst conflicts
   64         implicitConflicts = concat $ map snd conflicts
   65 
   66 checkConflict :: forall dom . FloatOutDefinition dom => LocalBind dom -> ([String], [String])
   67 checkConflict bnd = (concatMap @[] getConflict bndNames, implicits)
   68   where bndNames = bnd ^? elementName
   69         getConflict bndName = filter ((== nameStr) . Just) $ map (occNameString . getOccName) outerScope
   70           where outerScope = map (^. _1) $ concat $ take 1 $ drop 2 $ semanticsScope bndName
   71                 nameStr = fmap (occNameString . getOccName) $ semanticsName bndName
   72         implicits = map (occNameString . getOccName)
   73                         (concatMap getPossibleImplicits bndNames `intersect` getQNames (bnd ^? biplateRef))
   74 
   75         getQNames :: [QualifiedName dom] -> [GHC.Name]
   76         getQNames = catMaybes . map semanticsName
   77 
   78         getPossibleImplicits :: QualifiedName dom -> [GHC.Name]
   79         getPossibleImplicits qn = concat (map (map (^. _1)) $ take 2 $ semanticsScope qn) \\ catMaybes (map semanticsName bndNames)
