   29 extractBindingRefactoring :: RefactoringChoice
   30 extractBindingRefactoring = NamingRefactoringIndent "ExtractBinding" (\loc s i -> localRefactoring (extractBinding' s i loc ))

   32 extractBinding' :: String -> Maybe String -> RealSrcSpan -> LocalRefactoring
   33 extractBinding' name indent sp mod
   34   = if isNothing (isValidBindingName name)
   35       then extractBinding sp (read $ fromMaybe "4" indent) (nodesContaining sp) (nodesContaining sp) name mod
   36       else refactError $ "The given name is not a valid for the extracted binding: " ++ fromJust (isValidBindingName name)

   38 -- | Safely performs the transformation to introduce the local binding and replace the expression with the call.
   39 -- Checks if the introduction of the name causes a name conflict.
   40 extractBinding :: RealSrcSpan -> Int -> Simple Traversal Module ValueBind
   41                    -> Simple Traversal ValueBind Expr
   42                    -> String -> LocalRefactoring
   43 extractBinding sp indent selectDecl selectExpr name mod
   44   = let conflicting = filter (isConflicting name) ((take 1 $ reverse $ mod ^? selectDecl) ^? biplateRef :: [QualifiedName])
   45         exprRanges = map getRange (mod ^? selectDecl & selectExpr)
   46         decl = last (mod ^? selectDecl)
   47         declPats = decl ^? valBindPat &+& funBindMatches & annList & matchLhs
   48                                             & (matchLhsArgs & annList &+& matchLhsLhs &+& matchLhsRhs &+& matchLhsArgs & annList)
   49      in case exprRanges of
   50           (reverse -> exprRange:_) ->
   51             if | not (null conflicting)
   52                -> refactError $ "The given name causes name conflict with the definition(s) at: " ++ concat (intersperse "," (map (shortShowSpanWithFile . getRange) conflicting))
   53                | any (`containsRange` exprRange) $ map getRange declPats
   54                -> refactError "Extract binding cannot be applied to view pattern expressions."
   55                | otherwise
   56                -> case decl ^? actualContainingExpr exprRange of
   57                     expr:_ -> do (res, st) <- runStateT (selectDecl&selectExpr !~ extractThatBind sp name expr $ mod) Nothing
   58                                  case st of Just def -> return $ evalState (selectDecl !~ addLocalBinding exprRange indent def $ res) False
   59                                             Nothing -> refactError "There is no applicable expression to extract."
   60                     [] -> refactError $ "There is no applicable expression to extract."
   61           [] -> refactError "There is no applicable expression to extract."
   62   where RealSrcSpan sp1 `containsRange` RealSrcSpan sp2 = sp1 `containsSpan` sp2
   63         _ `containsRange` _ = False

   65 -- | Decides if a new name defined to be the given string will conflict with the given AST element
   66 isConflicting :: String -> QualifiedName -> Bool
   67 isConflicting name used
   68   = semanticsDefining used
   69       && (GHC.occNameString . GHC.getOccName <$> semanticsName used) == Just name

   71 -- Replaces the selected expression with a call and generates the called binding.
   72 extractThatBind :: RealSrcSpan -> String -> Expr -> Expr -> StateT (Maybe ValueBind) LocalRefactor Expr
   73 extractThatBind sp name cont e
   74   = do ret <- get -- being in a state monad to only apply the
   75        if (isJust ret) then return e
   76           else case e of
   77             -- only the expression inside the parameters should be extracted
   78             Paren {} | hasParameter -> exprInner !~ doExtract name cont $ e
   79                      | otherwise    -> doExtract name cont (fromJust $ e ^? exprInner)
   80             -- a single variable cannot be extracted (would lead to precedence problems)
   81             Var {} -> lift $ refactError "The selected expression is too simple to be extracted."
   82             -- extract operator sections
   83             InfixApp lhs op rhs
   84                | (lhs `outside` sp) && (sp `encloses` op) && (sp `encloses` rhs)
   85                -> do let params = getExternalBinds cont rhs ++ opName op
   86                      put (Just (generateBind name (map mkVarPat params) (mkRightSection op (parenIfInfix rhs))))
   87                      return (mkApp (generateCall name params) (parenIfInfix lhs))
   88                | (sp `encloses` lhs) && (sp `encloses` op) && (rhs `outside` sp)
   89                -> do let params = getExternalBinds cont lhs ++ opName op
   90                      put (Just (generateBind name (map mkVarPat params) (mkLeftSection (parenIfInfix lhs) op)))
   91                      return (mkApp (generateCall name params) (parenIfInfix rhs))
   92               where parenIfInfix e@(InfixApp {}) = mkParen e
   93                     parenIfInfix e = e
   94             -- extract parts of known associative infix operators
   95             InfixApp (InfixApp lhs lop mid) rop rhs -- correction for left-associative operators
   96               | (Just lName, Just rName) <- (semanticsName (lop ^. operatorName), semanticsName (rop ^. operatorName))
   97               , (lop `outside` sp) && (sp `encloses` mid) && (sp `encloses` rhs)
   98                   && lName == rName && isKnownCommutativeOp lName
   99               -> do let params = getExternalBinds cont mid ++ opName rop ++ getExternalBinds cont rhs
  100                     put (Just (generateBind name (map mkVarPat params) (mkInfixApp mid rop rhs)))
  101                     return (mkInfixApp lhs lop (generateCall name params))
  102             InfixApp lhs lop (InfixApp mid rop rhs) -- correction for right-associative operators
  103               | (Just lName, Just rName) <- (semanticsName (lop ^. operatorName), semanticsName (rop ^. operatorName))
  104               , (sp `encloses` lhs) && (sp `encloses` mid) && (rop `outside` sp)
  105                   && lName == rName && isKnownCommutativeOp lName
  106               -> do let params = getExternalBinds cont lhs ++ opName lop ++ getExternalBinds cont mid
  107                     put (Just (generateBind name (map mkVarPat params) (mkInfixApp lhs lop mid)))
  108                     return (mkInfixApp (generateCall name params) rop rhs)
  109             -- normal case
  110             el | isParenLikeExpr el && hasParameter -> mkParen <$> doExtract name cont e
  111                | otherwise -> doExtract name cont e
  112   where hasParameter = not (null (getExternalBinds cont e))
  113         -- True if the elem is completely inside the given source range
  114         sp `encloses` elem = case getRange elem of RealSrcSpan enc -> sp `containsSpan` enc
  115                                                    _               -> False
  116         -- True if the elem is completely outside the given range (no overlapping)
  117         elem `outside` sp = case getRange elem of RealSrcSpan out -> realSrcSpanStart sp > realSrcSpanEnd out
  118                                                                        || realSrcSpanEnd sp < realSrcSpanStart out
  119                                                   _ -> False
  120         opName op = case semanticsName (op ^. operatorName) of
  121                       Nothing -> []
  122                       Just n -> [mkUnqualName' n | not $ n `inScope` semanticsScope cont]
  123         isKnownCommutativeOp :: GHC.Name -> Bool
  124         isKnownCommutativeOp n = isJust $ find (maybe False (\(mn, occ) -> (nameModule_maybe n) == Just mn && occName n == occ) . isOrig_maybe) ops
  125           where ops = [plus_RDR, times_RDR, append_RDR, and_RDR, {- or_RDR, -} compose_RDR] -- somehow or is missing... WHY?

  127 -- | Adds a local binding to the where clause of the enclosing binding
  128 addLocalBinding :: SrcSpan -> Int -> ValueBind -> ValueBind -> State Bool ValueBind
  129 -- this uses the state monad to only add the local binding to the first selected element
  130 addLocalBinding exprRange indent local bind
  131   = do done <- get
  132        if not done then do put True
  133                            return $ indentBody $ doAddBinding exprRange local bind
  134                    else return bind
  135   where
  136     doAddBinding _ local sb@(SimpleBind {}) = valBindLocals .- insertLocalBind local $ sb
  137     doAddBinding (RealSrcSpan rng) local fb@(FunctionBind {})
  138       = funBindMatches & annList & filtered (isInside rng) & matchBinds
  139           .- insertLocalBind local $ fb
  140     doAddBinding _ _ _ = error "doAddBinding: invalid expression range"
  141 
  142     indentBody = (valBindRhs .- updIndent) . (funBindMatches & annList & matchLhs .- updIndent) . (funBindMatches & annList & matchRhs .- updIndent)
  143 
  144     updIndent :: SourceInfoTraversal elem => elem dom SrcTemplateStage -> elem dom SrcTemplateStage
  145     updIndent = setMinimalIndent indent

  147 -- | Puts a value definition into a list of local binds
  148 insertLocalBind :: ValueBind -> MaybeLocalBinds -> MaybeLocalBinds
  149 insertLocalBind toInsert locals
  150   | isAnnNothing locals = mkLocalBinds [mkLocalValBind toInsert]
  151   | otherwise = annJust & localBinds .- insertWhere True (mkLocalValBind toInsert) (const True) isNothing $ locals

  153 -- | All expressions that are bound stronger than function application.
  154 isParenLikeExpr :: Expr -> Bool
  155 isParenLikeExpr (If {}) = True
  156 isParenLikeExpr (Paren {}) = True
  157 isParenLikeExpr (List {}) = True
  158 isParenLikeExpr (ParArray {}) = True
  159 isParenLikeExpr (LeftSection {}) = True
  160 isParenLikeExpr (RightSection {}) = True
  161 isParenLikeExpr (RecCon {}) = True
  162 isParenLikeExpr (RecUpdate {}) = True
  163 isParenLikeExpr (Enum {}) = True
  164 isParenLikeExpr (ParArrayEnum {}) = True
  165 isParenLikeExpr (ListComp {}) = True
  166 isParenLikeExpr (ParArrayComp {}) = True
  167 isParenLikeExpr (BracketExpr {}) = True
  168 isParenLikeExpr (SpliceExpr {}) = True
  169 isParenLikeExpr (QuasiQuoteExpr {}) = True
  170 isParenLikeExpr _ = False

  172 -- | Replaces the expression with the call and stores the binding of the call in its state
  173 doExtract :: String -> Expr -> Expr -> StateT (Maybe ValueBind) LocalRefactor Expr
  174 doExtract name cont e@(Lambda (AnnList bindings) inner)
  175   = do let params = getExternalBinds cont e
  176        put (Just (generateBind name (map mkVarPat params ++ bindings) inner))
  177        return (generateCall name params)
  178 doExtract name cont e
  179   = do let params = getExternalBinds cont e
  180        put (Just (generateBind name (map mkVarPat params) e))
  181        return (generateCall name params)

  183 -- | Gets the values that have to be passed to the extracted definition
  184 getExternalBinds :: Expr -> Expr -> [Name]
  185 getExternalBinds cont expr = map exprToName $ keepFirsts $ filter isApplicableName (expr ^? uniplateRef)
  186   where isApplicableName (getExprNameInfo -> Just nm) = inScopeForOriginal nm && notInScopeForExtracted nm
  187         isApplicableName _ = False
  188 
  189         getExprNameInfo :: Expr -> Maybe GHC.Name
  190         getExprNameInfo expr = semanticsName =<< (listToMaybe $ expr ^? (exprName&simpleName &+& exprOperator&operatorName))
  191 
  192         -- | Creates the parameter value to pass the name (operators are passed in parentheses)
  193         exprToName :: Expr -> Name
  194         exprToName e | Just n <- e ^? exprName                     = n
  195                      | Just op <- e ^? exprOperator & operatorName = mkParenName op
  196                      | otherwise                                   = error "exprToName: name not found"
  197 
  198         notInScopeForExtracted :: GHC.Name -> Bool
  199         notInScopeForExtracted n = not $ n `inScope` semanticsScope cont
  200 
  201         inScopeForOriginal :: GHC.Name -> Bool
  202         inScopeForOriginal n = n `inScope` semanticsScope expr
  203 
  204         keepFirsts (e:rest) = e : keepFirsts (filter (/= e) rest)
  205         keepFirsts [] = []

  207 actualContainingExpr :: SrcSpan -> Simple Traversal ValueBind Expr
  208 actualContainingExpr (RealSrcSpan rng) = accessRhs & accessExpr
  209   where accessRhs :: Simple Traversal ValueBind Rhs
  210         accessRhs = valBindRhs &+& funBindMatches & annList & filtered (rng `isInside`) & matchRhs
  211         accessExpr :: Simple Traversal Rhs Expr
  212         accessExpr = rhsExpr &+& rhsGuards & annList & filtered (rng `isInside`) & guardExpr
  213 actualContainingExpr _ = error "actualContainingExpr: not a real range"
  214 
  215 -- | Generates the expression that calls the local binding
  216 generateCall :: String -> [Name] -> Expr
  217 generateCall name args = foldl (\e a -> mkApp e (mkVar a)) (mkVar $ mkNormalName $ mkSimpleName name) args
  218 
  219 -- | Generates the local binding for the selected expression
  220 generateBind :: String -> [Pattern] -> Expr -> ValueBind
  221 generateBind name [] e = mkSimpleBind (mkVarPat $ mkNormalName $ mkSimpleName name) (mkUnguardedRhs e) Nothing
  222 generateBind name args e = mkFunctionBind [mkMatch (mkMatchLhs (mkNormalName $ mkSimpleName name) args) (mkUnguardedRhs e) Nothing]
