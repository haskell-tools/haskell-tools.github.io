   23 trfDoStmt :: TransformName n r => Located (Stmt n (LHsExpr n)) -> Trf (Ann AST.UStmt (Dom r) RangeStage)
   24 trfDoStmt = trfLocNoSema trfDoStmt'
   25 
   26 trfDoStmt' :: TransformName n r => Stmt n (Located (HsExpr n)) -> Trf (AST.UStmt' AST.UExpr (Dom r) RangeStage)
   27 trfDoStmt' = gTrfDoStmt' trfExpr
   28 
   29 gTrfDoStmt' :: (TransformName n r, Data (ge n), Outputable (ge n))
   30             => (Located (ge n) -> Trf (Ann ae (Dom r) RangeStage)) -> Stmt n (Located (ge n)) -> Trf (AST.UStmt' ae (Dom r) RangeStage)
   31 gTrfDoStmt' et (BindStmt pat expr _ _ _) = AST.UBindStmt <$> trfPattern pat <*> et expr
   32 gTrfDoStmt' et (BodyStmt expr _ _ _) = AST.UExprStmt <$> et expr
   33 gTrfDoStmt' _ (LetStmt (unLoc -> binds)) = AST.ULetStmt . orderAnnList <$> addToScope binds (trfLocalBinds AnnLet binds)
   34 gTrfDoStmt' et (LastStmt body _ _) = AST.UExprStmt <$> et body
   35 gTrfDoStmt' et (RecStmt { recS_stmts = stmts }) = AST.URecStmt <$> trfAnnList "," (gTrfDoStmt' et) stmts
   36 gTrfDoStmt' _ stmt = unhandledElement "simple statement" stmt

   38 trfListCompStmts :: TransformName n r => [Located (Stmt n (LHsExpr n))] -> Trf (AnnListG AST.UListCompBody (Dom r) RangeStage)
   39 trfListCompStmts [unLoc -> ParStmt blocks _ _ _, unLoc -> (LastStmt {})]
   40   = nonemptyAnnList
   41       <$> trfScopedSequence (\(ParStmtBlock stmts _ _) ->
   42                                 let ann = collectLocs $ getNormalStmts stmts
   43                                  in annLocNoSema (pure ann) (AST.UListCompBody <$> makeList "," (pure $ srcSpanStart ann) (concat <$> trfScopedSequence trfListCompStmt stmts))
   44                             ) blocks
   45 trfListCompStmts others
   46   = let ann = (collectLocs $ getNormalStmts others)
   47      in makeList "|" (pure $ srcSpanStart ann)
   48           ((:[]) <$> annLocNoSema (pure ann)
   49                                   (AST.UListCompBody <$> makeList "," (pure $ srcSpanStart ann) (concat <$> trfScopedSequence trfListCompStmt others)))
   50 
   51 trfListCompStmt :: TransformName n r => Located (Stmt n (LHsExpr n)) -> Trf [Ann AST.UCompStmt (Dom r) RangeStage]
   52 trfListCompStmt (L _ trst@(TransStmt { trS_stmts = stmts }))
   53   = (++) <$> (concat <$> local (\s -> s { contRange = mkSrcSpan (srcSpanStart (contRange s)) (srcSpanEnd (getLoc (last stmts))) }) (trfScopedSequence trfListCompStmt stmts))
   54          <*> ((:[]) <$> extractActualStmt trst)
   55 -- last statement is extracted
   56 trfListCompStmt (unLoc -> LastStmt _ _ _) = pure []
   57 trfListCompStmt other = (:[]) <$> copyAnnot AST.UCompStmt (trfDoStmt other)
   58 
   59 extractActualStmt :: TransformName n r => Stmt n (LHsExpr n) -> Trf (Ann AST.UCompStmt (Dom r) RangeStage)
   60 extractActualStmt = \case
   61   TransStmt { trS_form = ThenForm, trS_using = using, trS_by = by }
   62     -> addAnnotation by using (AST.UThenStmt <$> trfExpr using <*> trfMaybe "," "" trfExpr by)
   63   TransStmt { trS_form = GroupForm, trS_using = using, trS_by = by }
   64     -> addAnnotation by using (AST.UGroupStmt <$> trfMaybe "," "" trfExpr by <*> (makeJust <$> trfExpr using))
   65   _ -> convertionProblem "extractActualStmt: called on a statement that is not then or group"
   66   where addAnnotation by using
   67           = annLocNoSema (combineSrcSpans (getLoc using) . combineSrcSpans (maybe noSrcSpan getLoc by)
   68                             <$> tokenLocBack AnnThen)
   69 
   70 getNormalStmts :: [Located (Stmt n (LHsExpr n))] -> [Located (Stmt n (LHsExpr n))]
   71 getNormalStmts (L _ (LastStmt _ _ _) : rest) = getNormalStmts rest
   72 getNormalStmts (stmt : rest) = stmt : getNormalStmts rest
   73 getNormalStmts [] = []
   74 
   75 getLastStmt :: [Located (Stmt n (LHsExpr n))] -> Located (HsExpr n)
   76 getLastStmt (L _ (LastStmt body _ _) : _) = body
   77 getLastStmt (_ : rest) = getLastStmt rest
   78 getLastStmt [] = convProblem "getLastStmt: empty"
