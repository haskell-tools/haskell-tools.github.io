   22 chkMagicHashLiteral' :: CheckNode Literal
   23 chkMagicHashLiteral' l@(PrimIntLit _)    = addOccurence MagicHash l
   24 chkMagicHashLiteral' l@(PrimWordLit _)   = addOccurence MagicHash l
   25 chkMagicHashLiteral' l@(PrimFloatLit _)  = addOccurence MagicHash l
   26 chkMagicHashLiteral' l@(PrimDoubleLit _) = addOccurence MagicHash l
   27 chkMagicHashLiteral' l@(PrimCharLit _)   = addOccurence MagicHash l
   28 chkMagicHashLiteral' l@(PrimStringLit _) = addOccurence MagicHash l
   29 chkMagicHashLiteral' l = return l

   32 chkMagicHashNamePart' :: CheckNode NamePart
   33 chkMagicHashNamePart' n@(NamePart name) =
   34   if (last name == '#') then addOccurence MagicHash n
   35                         else return n
