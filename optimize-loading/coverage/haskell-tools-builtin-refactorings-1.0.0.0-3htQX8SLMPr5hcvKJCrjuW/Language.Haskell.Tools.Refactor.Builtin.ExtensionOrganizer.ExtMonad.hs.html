   52 isTurnedOn :: Extension -> ExtMonad Bool
   53 isTurnedOn ext = do
   54   defaults <- ask
   55   return $! ext `elem` defaults

   57 conditional :: (node -> ExtMonad node) ->
   58                Extension ->
   59                node ->
   60                ExtMonad node
   61 conditional checker ext = conditionalAny checker [ext]
   62 
   63 conditionalNot :: (node -> ExtMonad node) ->
   64                   Extension ->
   65                   node ->
   66                   ExtMonad node
   67 conditionalNot checker ext node = do
   68   b <-isTurnedOn ext
   69   if b then return node else checker node
   70 
   71 conditionalAny :: (node -> ExtMonad node) ->
   72                    [Extension] ->
   73                    node ->
   74                    ExtMonad node
   75 conditionalAny checker exts node = do
   76   bs <- mapM isTurnedOn exts
   77   if or bs then checker node else return node
