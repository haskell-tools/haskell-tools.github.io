   23 -- | Remove the container (where or let) when the last binding is removed.
   24 removeEmptyBnds :: Simple Traversal Module ValueBind
   25                      -> Simple Traversal Module Expr
   26                      -> AST.Module -> AST.Module
   27 removeEmptyBnds binds exprs = (binds .- removeEmptyBindsAndGuards) . (exprs .- removeEmptyLetsAndStmts)
   28   where removeEmptyBindsAndGuards sb@(SimpleBind _ _ _)
   29           = (valBindLocals .- removeIfEmpty) . (valBindRhs .- removeEmptyGuards) $ sb
   30         removeEmptyBindsAndGuards fb@(FunctionBind _)
   31           = (funBindMatches & annList & matchBinds .- removeIfEmpty) . (funBindMatches & annList & matchRhs .- removeEmptyGuards) $ fb
   32 
   33         removeEmptyGuards rhs = rhsGuards & annList & guardStmts .- filterList (\case GuardLet (AnnList []) -> False; _ -> True) $ rhs
   34 
   35         removeIfEmpty mb@(AnnJust (LocalBinds (AnnList []))) = annMaybe .= Nothing $ mb
   36         removeIfEmpty mb = mb
   37 
   38         removeEmptyLetsAndStmts (Let (AnnList []) e) = e
   39         removeEmptyLetsAndStmts e = exprStmts .- removeEmptyStmts $ e
   40 
   41         removeEmptyStmts ls = (annList & cmdStmtBinds .- removeEmptyStmts)
   42                                 . filterList (\case LetStmt (AnnList []) -> False; _ -> True) $ ls
   43 
   44 -- | Puts the elements in the orginal order and remove duplicates (elements with the same source range)
   45 normalizeElements :: [Ann e dom SrcTemplateStage] -> [Ann e dom SrcTemplateStage]
   46 normalizeElements elems = nubBy ((==) `on` getRange) $ sortBy (compare `on` srcSpanStart . getRange) elems
