   25 -- | Starts the watch process and a thread that receives notifications from it. The notification
   26 -- thread will invoke updates on the daemon state to re-load files.
   27 createWatchProcess' :: Maybe FilePath -> Session -> MVar DaemonSessionState -> MVar [Marker] -> (ResponseMsg -> IO ())
   28                         -> IO (Maybe WatchProcess, [ThreadId])
   29 createWatchProcess' watchExePath ghcSess daemonSess warnMVars upClient = do
   30     exePath <- case watchExePath of Just exe -> return exe
   31                                     Nothing -> guessExePath
   32     process <- createWatchProcess exePath 500
   33     initProcess process
   34   where
   35     initProcess process = do
   36       reloaderThread <- forkIO $ forever $ void $ do
   37         changes <- waitForChanges process
   38         putStrLn $ "changes: " ++ show changes
   39         let changedFiles = catMaybes $ map getModifiedFile changes
   40             addedFiles = catMaybes $ map getAddedFile changes
   41             removedFiles = catMaybes $ map getRemovedFile changes
   42             reloadAction = updateForFileChanges upClient warnMVars addedFiles changedFiles removedFiles
   43             handlers = userExceptionHandlers
   44                            (upClient . ErrorMessage)
   45                            (\err hint -> upClient (CompilationProblem err hint))
   46                          ++ exceptionHandlers (return ()) (upClient . ErrorMessage)
   47         when (length changedFiles + length addedFiles + length removedFiles > 0)
   48           (void (modifyMVar daemonSess (\st -> swap <$> reflectGhc (runStateT reloadAction st) ghcSess))
   49              `catches` handlers)
   50       return $ (Just process, [reloaderThread])
   51 
   52     waitForChanges process = do
   53       changes <- waitNotifies process
   54       refactoring <- isNothing <$> tryReadMVar daemonSess
   55       -- if a refactoring is in progress, we should wait for all the changes to appear
   56       if refactoring then (changes ++) <$> waitForChanges process
   57                      else return changes
   58 
   59     getModifiedFile (Mod file) | takeExtension file `elem` sourceExtensions = Just file
   60     getModifiedFile _ = Nothing
   61 
   62     getAddedFile (Add file) | takeExtension file `elem` sourceExtensions = Just file
   63     getAddedFile _ = Nothing
   64 
   65     getRemovedFile (Rem file) | takeExtension file `elem` sourceExtensions = Just file
   66     getRemovedFile _ = Nothing
   67 
   68     sourceExtensions = [ ".hs", ".hs-boot", ".cabal" ]
   69 
   70     guessExePath = do exePath <- getExecutablePath
   71                       return $ takeDirectory exePath </> "hfswatch"

   74 stopWatch :: WatchProcess -> [ThreadId] -> IO ()
   75 stopWatch WatchProcess{..} threads
