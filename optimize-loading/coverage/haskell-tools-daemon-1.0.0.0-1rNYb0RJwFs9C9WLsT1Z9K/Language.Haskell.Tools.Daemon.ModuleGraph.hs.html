    1 -- | Creating a dependency graph of the modules loaded into a session.
    2 -- Code copied from GHC because it is not public in GhcMake module
    3 module Language.Haskell.Tools.Daemon.ModuleGraph
    4   (moduleGraphNodes, getModFromNode, dependentModules, supportingModules) where
    5 
    6 import Control.Monad (Monad(..), Functor(..), filterM)
    7 import qualified Data.Map as Map (fromList, Map, lookup)
    8 import Data.Maybe (Maybe(..), mapMaybe, catMaybes)
    9 
   10 import Digraph as GHC
   11 import FastString as GHC (FastString, fsLit)
   12 import GHC
   13 import HscTypes as GHC
   14 
   15 type NodeKey   = (ModuleName, HscSource)
   16 type NodeMap a = Map.Map NodeKey a
   17 type SummaryNode = (ModSummary, Int, [Int])
   18 
   19 getModFromNode :: SummaryNode -> ModSummary
   20 getModFromNode (ms, _, _) = ms
   21 
   22 -- Creates the dependency graph of modules currently loaded. Used for checking which modules need
   23 -- to be reloaded after a recompilation.
   24 moduleGraphNodes :: Bool -> [ModSummary]
   25   -> (Graph SummaryNode, HscSource -> ModuleName -> Maybe SummaryNode)
   26 moduleGraphNodes drop_hs_boot_nodes summaries = (graphFromEdgedVerticesOrd nodes, lookup_node)
   27   where
   28     numbered_summaries = zip summaries [1..]
   29 
   30     lookup_node :: HscSource -> ModuleName -> Maybe SummaryNode
   31     lookup_node hs_src mod = Map.lookup (mod, hs_src) node_map
   32 
   33     lookup_key :: HscSource -> ModuleName -> Maybe Int
   34     lookup_key hs_src mod = fmap summaryNodeKey (lookup_node hs_src mod)
   35 
   36     node_map :: NodeMap SummaryNode
   37     node_map = Map.fromList [ ((moduleName (ms_mod s), (ms_hsc_src s)), node)
   38                             | node@(s, _, _) <- nodes ]
   39 
   40     nodes :: [SummaryNode]
   41     nodes = [ (s, key, out_keys)
   42             | (s, key) <- numbered_summaries
   43             , not (isBootSummary s && drop_hs_boot_nodes)
   44             , let out_keys = out_edge_keys hs_boot_key (map unLoc (ms_home_srcimps s)) ++
   45                              out_edge_keys HsSrcFile   (map unLoc (ms_home_imps s)) ++
   46                              (-- see [boot-edges] below
   47                               if drop_hs_boot_nodes || ms_hsc_src s == HsBootFile
   48                               then []
   49                               else case lookup_key HsBootFile (ms_mod_name s) of
   50                                     Nothing -> []
   51                                     Just k  -> [k]) ]
   52 
   53     hs_boot_key | drop_hs_boot_nodes = HsSrcFile
   54                 | otherwise          = HsBootFile
   55 
   56     out_edge_keys :: HscSource -> [ModuleName] -> [Int]
   57     out_edge_keys hi_boot ms = mapMaybe (lookup_key hi_boot) ms
   58 
   59 summaryNodeKey :: SummaryNode -> Int
   60 summaryNodeKey (_, k, _) = k
   61 
   62 ms_home_imps :: ModSummary -> [Located ModuleName]
   63 ms_home_imps = home_imps . ms_imps
   64 
   65 ms_home_srcimps :: ModSummary -> [Located ModuleName]
   66 ms_home_srcimps = home_imps . ms_srcimps
   67 
   68 home_imps :: [(Maybe FastString, Located ModuleName)] -> [Located ModuleName]
   69 home_imps imps = [ lmodname |  (mb_pkg, lmodname) <- imps,
   70                                   isLocal mb_pkg ]
   71   where isLocal Nothing = True
   72         isLocal (Just pkg) | pkg == fsLit "this" = True -- "this" is special
   73         isLocal _ = False
   74 
   75 supportingModules :: (ModSummary -> Ghc Bool) -> Ghc [ModSummary]
   76 supportingModules = reachedModules False
   77 
   78 dependentModules :: (ModSummary -> Ghc Bool) -> Ghc [ModSummary]
   79 dependentModules = reachedModules True
   80 
   81 reachedModules :: Bool -> (ModSummary -> Ghc Bool) -> Ghc [ModSummary]
   82 reachedModules dependent pred = do
   83   let op = if dependent then transposeG else id
   84   allMods <- getModuleGraph
   85   selected <- filterM pred allMods
   86   let (allModsGraph, lookup) = moduleGraphNodes False allMods
   87       selectedMods = catMaybes $ map (\ms -> lookup (ms_hsc_src ms) (moduleName $ ms_mod ms)) selected
   88       recompMods = map (moduleName . ms_mod . getModFromNode) $ reachablesG (op allModsGraph) selectedMods -- TODO: compare on file name
   89       sortedMods = map getModFromNode $ reverse $ topologicalSortG allModsGraph
   90       sortedSelectedMods = filter ((`elem` recompMods) . moduleName . ms_mod) sortedMods -- TODO: compare on file name
   91   return sortedSelectedMods
