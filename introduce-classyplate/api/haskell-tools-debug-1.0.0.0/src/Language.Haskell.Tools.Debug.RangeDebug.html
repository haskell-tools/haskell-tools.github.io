-- | A module for displaying debug info about the source annotations of the syntax tree in different phases.
module Language.Haskell.Tools.Debug.RangeDebug where

import Control.Reference
import Control.Monad.State
import Control.Monad.Reader
import Data.Generics.ClassyPlate
import GHC.Generics
import Data.Data
import Language.Haskell.Tools.AST
import Language.Haskell.Tools.BackendGHC ()
import Language.Haskell.Tools.PrettyPrint.Prepare ()
import Language.Haskell.Tools.Debug.Show ()

type ShowSrcInfo st = (Show (SpanInfo st), Show (ListInfo st), Show (OptionalInfo st)) 

srcInfoDebug :: (Domain dom, SourceInfo st, ShowSrcInfo st) => Ann UModule dom st -> String
srcInfoDebug elem = snd $ execState (topDownM @TreeDebug treeDebug elem) ([0], "")

type instance AppSelector TreeDebug elem = TreeDebugChecker elem

type family TreeDebugChecker t where
  TreeDebugChecker (Ann e dom st)       = 'True
  TreeDebugChecker (AnnListG e dom st)  = 'True
  TreeDebugChecker (AnnMaybeG e dom st) = 'True
  TreeDebugChecker (e dom st)           = 'True
  TreeDebugChecker t                    = 'False

class (Data t) => TreeDebug t where
  treeDebug :: t -> State ([Int], String) t
  treeDebug e = do i <- gets (length . fst)
                   modify $ _2 .- (++ treeDebug' i e)
                   return e
  treeDebug' :: Int -> t -> String

instance {-# OVERLAPPABLE #-} 
         (Typeable e, Data (e dom st)) 
      => TreeDebug (e dom st) where
  treeDebug e = do d:desc <- gets fst
                   let numChildren = length $ constrFields (toConstr e)
                   if d == numChildren 
                     then modify $ _1 .= desc 
                     else modify $ _1 .= 0 : d+1 : desc
                   return e

instance {-# OVERLAPPING #-} 
         (Typeable e, Data (e dom st), DomainWith e dom, SourceInfo st, Show (e dom st), Show (SpanInfo st)) 
      => TreeDebug (Ann e dom st) where
  treeDebug' i (Ann a e) = identLine i ++ show (a ^. sourceInfo) ++ " " ++ take 40 (show e) ++ "..."

instance {-# OVERLAPPING #-} 
         (Typeable e, Data (e dom st), DomainWith e dom, SourceInfo st, Show (ListInfo st)) 
      => TreeDebug (AnnListG e dom st) where
  treeDebug' i (AnnListG a ls) = identLine i ++ show (a ^. sourceInfo) ++ " <*>"

instance {-# OVERLAPPING #-} 
         (Typeable e, Data (e dom st), DomainWith e dom, SourceInfo st, Show (OptionalInfo st)) 
      => TreeDebug (AnnMaybeG e dom st) where
  treeDebug' i (AnnMaybeG a e) = identLine i ++ show (a ^. sourceInfo) ++ " <?>"

identLine :: Int -> String
identLine i = "\n" ++ replicate ((i-1) * 2) ' '
