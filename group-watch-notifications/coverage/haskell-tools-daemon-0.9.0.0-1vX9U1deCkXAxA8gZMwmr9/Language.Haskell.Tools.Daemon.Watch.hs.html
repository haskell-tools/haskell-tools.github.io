   27 -- | Starts the watch process and a thread that receives notifications from it. The notification
   28 -- thread will invoke updates on the daemon state to re-load files.
   29 createWatchProcess' :: Maybe FilePath -> Session -> MVar DaemonSessionState -> (ResponseMsg -> IO ())
   30                         -> IO (Maybe WatchProcess, [ThreadId])
   31 createWatchProcess' watchExePath ghcSess daemonSess upClient = do
   32     exePath <- case watchExePath of Just exe -> return exe
   33                                     Nothing -> guessExePath
   34     process <- createWatchProcess exePath 500
   35     initProcess process
   36   where
   37     initProcess process = do
   38       reloaderThread <- forkIO $ forever $ void $ do
   39         changes <- waitForChanges process
   40         let changedFiles = catMaybes $ map getModifiedFile changes
   41             addedFiles = catMaybes $ map getAddedFile changes
   42             removedFiles = catMaybes $ map getRemovedFile changes
   43             reloadAction = updateForFileChanges upClient addedFiles changedFiles removedFiles
   44             handlers = userExceptionHandlers
   45                            (upClient . ErrorMessage)
   46                            (\err hint -> upClient (CompilationProblem err hint))
   47                          ++ exceptionHandlers (return ()) (upClient . ErrorMessage)
   48         when (length changedFiles + length addedFiles + length removedFiles > 0)
   49           (void (modifyMVar daemonSess (\st -> swap <$> reflectGhc (runStateT reloadAction st) ghcSess))
   50              `catches` handlers)
   51       return $ (Just process, [reloaderThread])
   52 
   53     waitForChanges process = do
   54       changes <- waitNotifies process
   55       refactoring <- isNothing <$> tryReadMVar daemonSess
   56       -- if a refactoring is in progress, we should wait for all the changes to appear
   57       if refactoring then (changes ++) <$> waitForChanges process
   58                      else return changes
   59 
   60     getModifiedFile (Mod file) | takeExtension file `elem` sourceExtensions = Just file
   61     getModifiedFile _ = Nothing
   62 
   63     getAddedFile (Add file) | takeExtension file `elem` sourceExtensions = Just file
   64     getAddedFile _ = Nothing
   65 
   66     getRemovedFile (Rem file) | takeExtension file `elem` sourceExtensions = Just file
   67     getRemovedFile _ = Nothing
   68 
   69     sourceExtensions = [ ".hs", ".hs-boot", ".cabal" ]
   70 
   71     guessExePath = do exePath <- getExecutablePath
   72                       return $ takeDirectory exePath </> "hfswatch"
   73 
   74 -- | Stops the watch process and all threads associated with it.
   75 stopWatch :: WatchProcess -> [ThreadId] -> IO ()
   76 stopWatch WatchProcess{..} threads
   77   = do forM threads killThread
