    1 -- | UPattern matching on pattern-level AST fragments for refactorings.
    2 {-# LANGUAGE PatternSynonyms #-}

    4 module Language.Haskell.Tools.Rewrite.Match.Patterns where

    9 -- | Pattern name binding
   10 pattern VarPat :: Name -> Pattern
   11 pattern VarPat var <- Ann _ (UVarPat var)
   12 
   13 -- | Literal pattern
   14 pattern LitPat :: Literal -> Pattern
   15 pattern LitPat lit <- Ann _ (ULitPat lit)
   16 
   17 -- | Infix constructor application pattern (@ a :+: b @)
   18 pattern InfixAppPat :: Pattern -> Operator -> Pattern -> Pattern
   19 pattern InfixAppPat lhs op rhs <- Ann _ (UInfixAppPat lhs op rhs)
   20 
   21 -- | Constructor application pattern (@ Point x y @)
   22 pattern AppPat :: Name -> PatternList -> Pattern

   24 
   25 -- | Tuple pattern (@ (x,y) @)
   26 pattern TuplePat :: PatternList -> Pattern

   28 
   29 -- | Unboxed tuple pattern (@ (\# x, y \#) @)
   30 pattern UnboxTuplePat :: PatternList -> Pattern

   32 
   33 -- | List pattern (@ [1,2,a,x] @)
   34 pattern ListPat :: PatternList -> Pattern
   35 pattern ListPat pats <- Ann _ (UListPat pats)
   36 
   37 -- | Parallel array pattern (@ [:1,2,a,x:] @)
   38 pattern ParArrayPat :: PatternList -> Pattern
   39 pattern ParArrayPat pats <- Ann _ (UParArrPat pats)
   40 
   41 -- | Parenthesised patterns
   42 pattern ParenPat :: Pattern -> Pattern
   43 pattern ParenPat pat <- Ann _ (UParenPat pat)
   44 
   45 -- | Record pattern (@ Point { x = 3, y } @)
   46 pattern RecPat :: Name -> PatternFieldList -> Pattern

   49 -- | As-pattern (explicit name binding) (@ ls\@(hd:_) @)
   50 pattern AsPat :: Name -> Pattern -> Pattern

   57 -- | Irrefutable pattern (@ ~(x:_) @)

   59 pattern IrrefutablePat pat <- Ann _ (UIrrefutablePat pat)

   61 -- | Bang pattern (@ !x @)
   62 pattern BangPat :: Pattern -> Pattern

   66 pattern TypeSigPat :: Pattern -> Type -> Pattern

   70 pattern ViewPat :: Expr -> Pattern -> Pattern

   81 pattern NPlusKPat :: Name -> Literal -> Pattern

   89 pattern FieldPunPattern :: Name -> PatternField
   90 pattern FieldPunPattern name <- Ann _ (UFieldPunPattern name)

   93 pattern FieldWildcardPattern :: FieldWildcard -> PatternField
