    1 -- | Representation of Haskell modules, imports and exports. Also contains file-level pragmas.
    2 module Language.Haskell.Tools.AST.Representation.Modules where
    3 
    4 import Language.Haskell.Tools.AST.Ann (Ann, AnnListG, AnnMaybeG)
    5 import Language.Haskell.Tools.AST.Representation.Decls (UDecl)
    6 import Language.Haskell.Tools.AST.Representation.Names (UStringNode, UName)
    7 
    8 -- | The representation of a haskell module, that is a separate compilation unit.
    9 -- It may or may not have a header.
   10 data UModule dom stage
   11   = UModule { _filePragmas :: AnnListG UFilePragma dom stage
   12             , _modHead :: AnnMaybeG UModuleHead dom stage
   13             , _modImports :: AnnListG UImportDecl dom stage
   14             , _modDecl :: AnnListG UDecl dom stage
   15             }
   16 
   17 -- | Module declaration with name and (optional) exports
   18 data UModuleHead dom stage
   19   = UModuleHead { _mhName :: Ann UModuleName dom stage
   20                 , _mhPragma :: AnnMaybeG UModulePragma dom stage
   21                 , _mhExports :: AnnMaybeG UExportSpecs dom stage
   22                 }
   23 
   24 -- | A list of export specifications surrounded by parentheses
   25 data UExportSpecs dom stage
   26   = UExportSpecs { _espExports :: AnnListG UExportSpec dom stage }
   27 
   28 -- | Export specifier
   29 data UExportSpec dom stage
   30   = UDeclExport { _exportDecl :: Ann UIESpec dom stage
   31                 } -- ^ Export a name and related names
   32   | UModuleExport { _exportModuleName :: Ann UModuleName dom stage
   33                   } -- ^ The export of an imported module (@ module A @)
   34 
   35 -- | Marks a name to be imported or exported with related names (subspecifier)
   36 data UIESpec dom stage
   37   = UIESpec { _ieModifier :: AnnMaybeG UImportModifier dom stage
   38             , _ieName :: Ann UName dom stage
   39             , _ieSubspec :: AnnMaybeG USubSpec dom stage
   40             }
   41 
   42 -- | Specifies the imported element
   43 data UImportModifier dom stage
   44   = UImportPattern -- ^ @pattern@: modifier for importing pattern synonyms
   45   | UImportType -- ^ @type@: modifier for importing types
   46 
   47 -- | Marks how related names will be imported or exported with a given name
   48 data USubSpec dom stage
   49   = USubSpecAll -- ^ @(..)@: a class exported with all of its methods, or a datatype exported with all of its constructors.
   50   | USubSpecList { _essList :: AnnListG UName dom stage
   51                  } -- ^ @(a,b,c)@: a class exported with some of its methods, or a datatype exported with some of its constructors.
   52 
   53 -- | Pragmas that must be used before defining the module
   54 data UFilePragma dom stage
   55   = ULanguagePragma { _lpPragmas :: AnnListG ULanguageExtension dom stage
   56                     } -- ^ @LANGUAGE@ pragma, listing the enabled language extensions in that file
   57   | UOptionsPragma {  _opStr :: Ann UStringNode dom stage
   58                    } -- ^ @OPTIONS@ pragma, possibly qualified with a tool, e.g. OPTIONS_GHC
   59 
   60 -- | Pragmas that must be used after the module head
   61 data UModulePragma dom stage
   62   = UModuleWarningPragma { _modWarningStr :: AnnListG UStringNode dom stage
   63                          }  -- ^ A warning pragma attached to the module
   64   | UModuleDeprecatedPragma {  _modDeprecatedPragma :: AnnListG UStringNode dom stage
   65                             } -- ^ A deprecated pragma attached to the module
   66 
   67 -- | The name of the enabled language extension, for example (@ LambdaCase @)
   68 data ULanguageExtension dom stage = ULanguageExtension { _langExt :: String }
   69 
   70 -- | An import declaration: @import Module.Name@
   71 data UImportDecl dom stage
   72   = UImportDecl { _importSource :: AnnMaybeG UImportSource dom stage
   73                 , _importQualified :: AnnMaybeG UImportQualified dom stage
   74                 , _importSafe :: AnnMaybeG UImportSafe dom stage
   75                 , _importPkg :: AnnMaybeG UStringNode dom stage
   76                 , _importModule :: Ann UModuleName dom stage
   77                 , _importAs :: AnnMaybeG UImportRenaming dom stage
   78                 , _importSpec :: AnnMaybeG UImportSpec dom stage
   79                 }
   80 
   81 -- | Restriction on the imported names
   82 data UImportSpec dom stage
   83   = UImportSpecList { _importSpecList :: AnnListG UIESpec dom stage
   84                     } -- ^ Restrict the import definition to ONLY import the listed names
   85   | UImportSpecHiding { _importSpecHiding :: AnnListG UIESpec dom stage
   86                       } -- ^ Restrict the import definition to DONT import the listed names
   87 
   88 -- | Marks the import as qualified: @qualified@
   89 data UImportQualified dom stage = UImportQualified
   90 
   91 -- | Marks the import as source: @{-# SOURCE #-}@
   92 data UImportSource dom stage = UImportSource
   93 
   94 -- | Marks the import as safe: @safe@
   95 data UImportSafe dom stage = UImportSafe
   96 
   97 -- | Marks an imported name to belong to the type namespace: @type@
   98 data UTypeNamespace dom stage = UTypeNamespace
   99 
  100 -- | Renaming imports (@ as A @)
  101 data UImportRenaming dom stage = UImportRenaming { _importRename :: Ann UModuleName dom stage }
  102 
  103 -- | The name of a module
  104 data UModuleName dom stage = UModuleName { _moduleNameString :: String }
  105 
  106 -- | The @type@ keyword used to qualify that the type and not the constructor of the same name is referred
  107 data TypeKeyword dom stage = TypeKeyword
