   18 -- | The actual state of the daemon process. Contains loaded modules and user settings.
   19 -- The GHC state is handled separately.
   20 data DaemonSessionState
   21   = DaemonSessionState { _refSessMCs :: [ModuleCollection SourceFileKey]
   22                            -- ^ The package components loaded into the session.
   23                        , _packageDB :: Maybe (PackageDB, Bool)
   24                            -- ^ The package database that is selected and a flag to decide if it is forced.
   25                        , _ghcFlagsSet :: DynFlags -> DynFlags
   26                            -- ^ GHC flags for compiling modules. Overrides settings in cabal files.
   27                        , _pkgDbFlags :: DynFlags -> DynFlags
   28                        , _packageDBLocs :: [FilePath]
   29                            -- ^ The pathes where the package databases are located.
   30                        , _exiting :: Bool
   31                            -- ^ True if in the process of shutting down the session.
   32                        , _undoStack :: [[UndoRefactor]]
   33                            -- ^ True if in the process of shutting down the session.
   34                        , _watchProc :: Maybe WatchProcess
   35                            -- ^ Information about the file system watch process.
   36                        , _watchThreads :: [ThreadId]
   37                            -- ^ Extra threads started for handling the
   38                            -- information from the watch process.
   39                        , _touchedFiles :: Set FilePath
   40                            -- ^ Marks files as being modified by this process
   41                            -- Changes detected on marked files will not invalidate refactoring history.
   42                        }
   43 
   44 -- | An initial state of a daemon session.
   45 initSession :: DaemonSessionState
   46 initSession = DaemonSessionState [] Nothing id id [] False [] Nothing [] empty
   47 
   48 resetSession :: DaemonSessionState -> DaemonSessionState
   49 resetSession DaemonSessionState{..} = initSession { _packageDB, _ghcFlagsSet }
