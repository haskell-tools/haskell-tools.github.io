   33 chkTypeOperatorsType' :: CheckNode Type
   34 chkTypeOperatorsType' t@(InfixTypeApp _ op _)
   35   | Just name <- semanticsName op
   36   , name == eqTyConName
   37   = return t
   38   | otherwise = addEvidence TypeOperators t
   39 chkTypeOperatorsType' t = return t

   49 chkTypeOperatorsDecl' :: CheckNode Decl
   50 chkTypeOperatorsDecl' d = do
   51   let dhs = universeBi d :: [DeclHead]
   52   anyNeedsTO <- liftM or $ mapM isOperatorM dhs
   53   if anyNeedsTO then addEvidence TypeOperators d
   54                 else return d

   56 -- OccName: [Type and class operator definitions]
   57   -- We are not just looking for a *syntactically-infix* declaration,
   58   -- but one that uses an operator OccName.
   59 isOperatorM :: DeclHead -> ExtMonad Bool
   60 isOperatorM dh = do
   61   let mSemName = declHeadSemName dh
   62   case mSemName of
   63     Just semName
   64       | occ <- GHC.nameOccName semName
   65       , GHC.isTcOcc occ && GHC.isSymOcc occ
   66       -> return True
   67       | otherwise -> return False
   68     Nothing -> return True
