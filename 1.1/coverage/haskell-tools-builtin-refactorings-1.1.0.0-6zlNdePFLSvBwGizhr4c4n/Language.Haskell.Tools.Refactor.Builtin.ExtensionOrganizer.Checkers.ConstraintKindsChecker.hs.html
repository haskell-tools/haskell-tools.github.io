   15 chkConstraintKindsDecl :: CheckNode Decl
   16 chkConstraintKindsDecl = conditional chkConstraintKindsDecl' ConstraintKinds
   17 
   18 chkConstraintKindsDecl' :: CheckNode Decl
   19 chkConstraintKindsDecl' d@(TypeDecl dh rhs)
   20   -- Has any constraints of form (x t1 t2)
   21   | ctxts <- universeBi rhs :: [Context]
   22   , any hasTyVarHeadAsserts ctxts
   23   = addEvidence ConstraintKinds d
   24   -- Right-hand side has kind Constraint
   25   | otherwise = do
   26   let ty = typeOrKindFromId . declHeadQName $ dh
   27   if hasConstraintKind ty || returnsConstraintKind ty
   28      then addEvidence ConstraintKinds d
   29      else return d
   30 chkConstraintKindsDecl' d = return d
   31 
   32 hasTyVarHeadAsserts :: Context -> Bool
   33 hasTyVarHeadAsserts = hasAnyTyVarHeads . (^. contextAssertion)
   34 
   35 hasAnyTyVarHeads :: Assertion -> Bool
   36 hasAnyTyVarHeads (ClassAssert n _)
   37   | Just n' <- semanticsName n = isTyVarName n'
   38   | otherwise               = False
   39 hasAnyTyVarHeads ta@TupleAssert{}
   40   | Just assertions <- ta ^? innerAsserts & annListElems
   41   = any hasAnyTyVarHeads assertions
   42 hasAnyTyVarHeads _ = False
