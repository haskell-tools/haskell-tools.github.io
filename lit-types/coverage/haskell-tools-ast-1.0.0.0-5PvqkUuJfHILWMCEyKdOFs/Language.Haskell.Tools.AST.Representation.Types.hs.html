    9 -- | Type variable declaration
   10 data UTyVar dom stage
   11   = UTyVarDecl { _tyVarName :: Ann UName dom stage
   12                , _tyVarKind :: AnnMaybeG UKindConstraint dom stage
   13                }
   14 
   15 -- | Haskell types
   16 data UType dom stage
   17   = UTyForall     { _typeBounded :: AnnListG UTyVar dom stage
   18                   , _typeType :: Ann UType dom stage
   19                   } -- ^ Forall types (@ forall x y . type @)
   20   | UTyCtx        { _typeCtx :: Ann UContext dom stage
   21                   , _typeType :: Ann UType dom stage
   22                   } -- ^ Type with a context (@ forall x y . type @)
   23   | UTyFun        { _typeParam :: Ann UType dom stage
   24                   , _typeResult :: Ann UType dom stage
   25                   } -- ^ Function types (@ a -> b @)
   26   | UTyTuple      { _typeElements :: AnnListG UType dom stage
   27                   } -- ^ Tuple types (@ (a,b) @)
   28   | UTyUnbTuple   { _typeElements :: AnnListG UType dom stage
   29                   } -- ^ Unboxed tuple types (@ (#a,b#) @)
   30   | UTyList       { _typeElement :: Ann UType dom stage
   31                   } -- ^ List type with special syntax (@ [a] @)
   32   | UTyParArray   { _typeElement :: Ann UType dom stage
   33                   } -- ^ Parallel array type (@ [:a:] @)
   34   | UTyApp        { _typeCon :: Ann UType dom stage
   35                   , _typeArg :: Ann UType dom stage
   36                   } -- ^ Type application (@ F a @)
   37   | UTyVar        { _typeName :: Ann UName dom stage
   38                   } -- ^ Type variable or constructor (@ a @)
   39   | UTyParen      { _typeInner :: Ann UType dom stage
   40                   } -- ^ Type surrounded by parentheses (@ (T a) @)
   41   | UTyInfix      { _typeLeft :: Ann UType dom stage
   42                   , _typeOperator :: Ann UOperator dom stage
   43                   , _typeRight :: Ann UType dom stage
   44                   } -- ^ Infix type constructor (@ (a <: b) @)
   45   | UTyKinded     { _typeInner :: Ann UType dom stage
   46                   , _typeKind :: Ann UKind dom stage
   47                   } -- ^ Type with explicit kind signature (@ a :: * @)
   48   | UTyPromoted   { _tpPromoted :: Ann (UPromoted UType) dom stage
   49                   } -- A promoted data type with @-XDataKinds@ (@ 3 @, @ Left @, @ 'Left @).
   50   | UTySplice     { _tsSplice :: Ann USplice dom stage
   51                   } -- ^ A Template Haskell splice type (@ $(genType) @).
   52   | UTyQuasiQuote { _typeQQ :: Ann UQuasiQuote dom stage
   53                   } -- ^ A Template Haskell quasi-quote type (@ [quoter| ... ] @).
   54   | UTyBang       { _typeInner :: Ann UType dom stage
   55                   } -- ^ Strict type marked with @!@.
   56   | UTyLazy       { _typeInner :: Ann UType dom stage
   57                   } -- ^ Lazy type marked with @~@. (Should only be used if @Strict@ or @StrictData@ language extension is used)
   58   | UTyUnpack     { _typeInner :: Ann UType dom stage
   59                   } -- ^ Strict type marked with UNPACK pragma. (Usually contains the bang mark.)
   60   | UTyNoUnpack   { _typeInner :: Ann UType dom stage
   61                   } -- ^ Strict type marked with NOUNPACK pragma. (Usually contains the bang mark.)
   62   | UTyWildcard   -- ^ A wildcard type (@ _ @) with @-XPartialTypeSignatures@
   63   | UTyNamedWildc { _typeWildcardName :: Ann UName dom stage
   64                   } -- ^ A named wildcard type (@ _t @) with @-XPartialTypeSignatures@
   65   | UUnbSumType   { _typeElements :: AnnListG UType dom stage
   66                   } -- ^ An unboxed sum type (@ (# Bool | Int | String #) @)
   67 
   68 -- One or more assertions
   69 data UContext dom stage
   70   = UContext { _contextAssertion :: Ann UAssertion dom stage
   71              } -- ^ Assertions with the fat arrow (@ C a => ... @)
   72 
   73 -- | A single assertion in the context
   74 data UAssertion dom stage
   75   = UClassAssert { _assertClsName :: Ann UName dom stage
   76                  , _assertTypes :: AnnListG UType dom stage
   77                  } -- ^ Class assertion (@Cls x@)
   78   | UInfixAssert { _assertLhs :: Ann UType dom stage
   79                  , _assertOp :: Ann UOperator dom stage
   80                  , _assertRhs :: Ann UType dom stage
   81                  } -- ^ Infix class assertion, also contains type equations (@ a ~ X y @)
   82   | UImplicitAssert { _assertImplVar :: Ann UName dom stage
   83                     , _assertImplType :: Ann UType dom stage
   84                     } -- ^ Assertion for implicit parameter binding (@ ?cmp :: a -> a -> Bool @)
   85   | UTupleAssert { _innerAsserts :: AnnListG UAssertion dom stage
   86                  } -- ^ Multiple assertions in one (@ (Ord a, Show a) @)
   87   | UWildcardAssert -- ^ Wildcard assertion (@ _ @), enabled by @PartialTypeSignatures@
