   21 rangeToSource :: SourceInfoTraversal node => StringBuffer -> Ann node dom RngTemplateStage
   22                                                           -> Ann node dom SrcTemplateStage
   23 rangeToSource srcInput tree = let locIndices = getLocIndices tree
   24                                   srcMap = mapLocIndices srcInput locIndices
   25                                in applyFragments (Map.elems srcMap) tree

   29 -- | Assigns an index (in the order they are used) for each range
   30 getLocIndices :: SourceInfoTraversal e => Ann e dom RngTemplateStage -> Set (RealSrcLoc, Int)
   31 getLocIndices = snd . flip execState (0, Set.empty) .
   32   sourceInfoTraverseDown (SourceInfoTrf
   33       (\ni -> do { mapM_ (\el -> case getRangeElemSpan el of Just sp -> modify (insertElem sp); _ -> return ()) (ni ^. rngTemplateNodeElems); return ni })
   34       (\ni -> do { mapM_ (modify . insertElem) (ni ^. rngTmpSeparators); return ni })
   35       pure )
   36     (return ()) (return ())
   37   where insertElem sp (i,m) = (i+1, Set.insert (realSrcSpanEnd sp, i) m)
   38 
   39 -- | Partitions the source file in the order where the parts are used in the AST
   40 mapLocIndices :: Ord k => StringBuffer -> Set (RealSrcLoc, k) -> Map k String
   41 mapLocIndices inp = (^. _1) . Set.foldl (\(new, str, pos) (sp, k) -> let (rem, val, newPos) = takeSpan str pos sp
   42                                                                       in (Map.insert k (reverse val) new, rem, newPos))
   43                                         (Map.empty, inp, mkRealSrcLoc (mkFastString "") 1 1)
   44   where takeSpan :: StringBuffer -> RealSrcLoc -> RealSrcLoc -> (StringBuffer, String, RealSrcLoc)
   45         takeSpan str pos end = takeSpan' end (str,"", pos)
   46 
   47         takeSpan' :: RealSrcLoc -> (StringBuffer, String, RealSrcLoc) -> (StringBuffer, String, RealSrcLoc)
   48         takeSpan' end (sb, taken, pos) | (srcLocLine pos `compare` srcLocLine end) `thenCmp` (srcLocCol pos `compare` srcLocCol end) == LT && not (atEnd sb)
   49           = let (c,rem) = nextChar sb in takeSpan' end (rem, c:taken, advanceSrcLoc pos c)
   50         takeSpan' _ (rem, taken, pos) = (rem, taken, pos)
   51 
   52         thenCmp EQ o2 = o2
   53         thenCmp o1 _  = o1
   54 
   55 -- | Replaces the ranges in the AST with the source file parts
   56 applyFragments :: SourceInfoTraversal node => [String] -> Ann node dom RngTemplateStage
   57                                                        -> Ann node dom SrcTemplateStage
   58 applyFragments srcs = flip evalState srcs
   59   . sourceInfoTraverseDown (SourceInfoTrf
   60      (\ni -> do template <- mapM getTextFor (ni ^. rngTemplateNodeElems)
   61                 return $ SourceTemplateNode (RealSrcSpan $ ni ^. rngTemplateNodeRange) (concat template) 0 Nothing)
   62      (\(RangeTemplateList rng bef aft sep indented seps)
   63          -> do (own, rest) <- List.splitAt (length seps) <$> get
   64                put rest
   65                return (SourceTemplateList (RealSrcSpan rng) bef aft sep indented (Prelude.zip (Prelude.map ((:[]) . NormalText) own) (Prelude.map RealSrcSpan seps)) 0 Nothing))
   66      (\(RangeTemplateOpt rng bef aft) -> return (SourceTemplateOpt (RealSrcSpan rng) bef aft 0 Nothing)))
   67      (return ()) (return ())
   68   where getTextFor RangeChildElem = return [ChildElem]
   69         getTextFor (RangeElem rng) = do (src:rest) <- get
   70                                         put rest
   71                                         return [TextElem [NormalText src] (RealSrcSpan rng)]
   72 
   73 -- | Marks template elements in the AST that should always be present in the source code, regardless of their
   74 -- containing elements being deleted.
   75 -- Currently it recognizes CPP pragmas (lines starting with #)
   76 -- This function should only be applied to an AST if CPP is enabled.
   77 extractStayingElems :: SourceInfoTraversal node => Ann node dom SrcTemplateStage -> Ann node dom SrcTemplateStage
   78 extractStayingElems = runIdentity . sourceInfoTraverse (SourceInfoTrf
   79     (sourceTemplateNodeElems & traversal & sourceTemplateTextElem !- breakStaying)
   80     (srcTmpSeparators & traversal & _1 !- breakStaying)
   81     pure)
   82 
   83     where -- splits the elements into separate lines and then recombines them
   84           breakStaying :: [SourceTemplateTextElem] -> [SourceTemplateTextElem]
   85           breakStaying = concat . Prelude.map (\(NormalText s) -> toTxtElems s)
   86 
   87           toTxtElems :: String -> [SourceTemplateTextElem]
   88           toTxtElems str = extractStaying $ splitOn "\n" $ str
   89             where
   90               extractStaying lines | not (any ("#" `isPrefixOf`) lines) = [NormalText str]
   91               extractStaying lines = Prelude.foldr appendTxt []
   92                                        $ Prelude.map (\ln -> if "#" `isPrefixOf` ln then StayingText ln "\n" else NormalText ln) lines
   93           -- recombines the lines if they are both normal text
   94           -- otherwise it moves the windows '\r' characters to the correct position
   95           appendTxt (NormalText n1) (NormalText n2 : rest) = NormalText (n1 ++ '\n':n2) : rest
   96           appendTxt e (next@NormalText{} : ls) = case reverse (e ^. sourceTemplateText) of
   97                                               -- fix '\r' characters that are separated from '\n'
   98                                     '\r':_ -> ((sourceTemplateText .- init) . (lineEndings .= "\r\n") $ e) : (sourceTemplateText .- ("\r\n" ++) $ next) : ls
   99                                     _      -> e : (sourceTemplateText .- ('\n':) $ next) : ls
  100           appendTxt e (next : ls) = case reverse (e ^. sourceTemplateText) of
  101                                               -- fix '\r' characters that are separated from '\n'
  102                                     '\r':_ -> ((sourceTemplateText .- init) . (lineEndings .= "\r\n") $ e) : NormalText "\r\n" : next : ls
  103                                     _      -> e : NormalText "\n" : next : ls
  104           appendTxt e [] = [e]
