   20 -- | Pretty prints an AST by using source templates stored as node info
   21 prettyPrint :: (SourceInfoTraversal node) => node dom SrcTemplateStage -> String
   22 prettyPrint = toList . printRose . toRoseTree

   24 printRose :: RoseTree SrcTemplateStage -> Seq Char
   25 printRose rt = evalState (printRose' startLoc rt) startLoc
   26   where startLoc = mkRealSrcLoc (fsLit "") 1 1

   30 -- | Pretty prints a rose tree according to the source templates remaining from the original AST
   31 printRose' :: RealSrcLoc -> RoseTree SrcTemplateStage -> PPState (Seq Char)
   32 -- simple implementation could be optimized a bit
   33 -- warning: the length of the file should not exceed maxbound::Int
   34 printRose' parent (RoseTree (RoseSpan (SourceTemplateNode rng elems minInd relInd)) children)
   35   = do slide <- calculateSlide rng
   36        let printTemplateElems :: [SourceTemplateElem] -> [RoseTree SrcTemplateStage] -> PPState (Seq Char)
   37            printTemplateElems (TextElem txtElems _ : rest) children = putString slide min txt >+< printTemplateElems rest children
   38              where txt = concatMap (^. sourceTemplateText) txtElems
   39            printTemplateElems (ChildElem : rest) (child : children) = printRose' parent child >+< printTemplateElems rest children
   40            printTemplateElems [] [] = return empty
   41            printTemplateElems _ []
   42              = pprProblem $ "More child elem in template than actual children in: " 
   43                               ++ shortShowSpanWithFile (srcLocSpan $ RealSrcLoc parent)
   44            printTemplateElems [] _
   45              = pprProblem $ "Not all children are used to pretty printing in: " 
   46                               ++ shortShowSpanWithFile (srcLocSpan $ RealSrcLoc parent)
   47 
   48            min = minInd `max` getPosByRelative parent relInd
   49 
   50        printTemplateElems elems children
   51 
   52 printRose' _ (RoseTree (RoseList (SourceTemplateList {})) []) = return empty
   53 printRose' parent (RoseTree (RoseList (SourceTemplateList rng bef aft defSep indented seps minInd relInd)) children)
   54     = do slide <- calculateSlide rng
   55          actRng <- get
   56          let min = minInd `max` getPosByRelative parent relInd
   57          putString slide min bef
   58            >+< (maybe printListWithSeps printListWithSepsIndented indented) actRng slide min actualSeps children
   59            >+< putString slide min aft
   60   where stringSeps :: [String]
   61         stringSeps = map (concatMap (^. sourceTemplateText)) (map fst seps)
   62         actualSeps = case stringSeps of [] -> repeat defSep
   63                                         _  -> stringSeps ++ repeat (last stringSeps)
   64 
   65 printRose' _ (RoseTree (RoseOptional (SourceTemplateOpt {})) []) = return empty
   66 printRose' parent (RoseTree (RoseOptional (SourceTemplateOpt rng bef aft minInd relInd)) [child])
   67   = do slide <- calculateSlide rng
   68        actRng <- get
   69        let min = minInd `max` getPosByRelative parent relInd
   70        putString slide min bef >+< printRose' actRng child >+< putString slide min aft
   71 printRose' _ (RoseTree (RoseOptional _) _) = pprProblem "More than one child element in an optional node."
   72 
   73 getPosByRelative :: RealSrcLoc -> Maybe Int -> Int
   74 getPosByRelative sp (Just i) = srcLocCol sp + i - 1
   75 getPosByRelative _ _ = 0
   76 
   77 calculateSlide :: SrcSpan -> PPState Int
   78 calculateSlide (RealSrcSpan originalSpan) = do
   79   actualSpan <- get
   80   return $ srcLocCol actualSpan - srcLocCol (realSrcSpanStart originalSpan)
   81 calculateSlide _ = return 0
   82 
   83 putString :: Int -> Int -> String -> PPState (Seq Char)
   84 putString slide minInd s
   85   = do modify $ advanceStr newStr
   86        return (fromList newStr)
   87   where start:rest = splitOn "\n" s
   88         newStr = concat $ List.intersperse ("\n" ++ replicate slide ' ') (start : map (extendToNSpaces minInd) rest)
   89         extendToNSpaces n str = replicate n ' ' ++ (List.dropWhile (== ' ') $ List.take n str) ++ List.drop n str
   90 
   91 advanceStr :: String -> RealSrcLoc -> RealSrcLoc
   92 advanceStr s loc = foldl advanceSrcLoc loc s
   93 
   94 untilReaches :: String -> RealSrcLoc -> RealSrcLoc -> (String, Int)
   95 untilReaches s start end
   96   = let ls = splitOn "\n" s
   97      in case ls of _:_:_ -> (unlines (init ls) ++)
   98                               `mapFst` untilReaches' (last ls) (advanceSrcLoc start '\n') end
   99                    _ -> (s, srcLocCol $ foldl advanceSrcLoc start s)
  100   where
  101     untilReaches' [] curr _ = ([], srcLocCol curr)
  102     untilReaches' (c:rest) curr until | srcLocCol advancedLoc <= srcLocCol until
  103       = (c:) `mapFst` untilReaches' rest advancedLoc until
  104       where advancedLoc = advanceSrcLoc curr c
  105     untilReaches' _ curr _ = ([], srcLocCol curr)
  106 
  107 mapFst :: (a -> b) -> (a, x) -> (b, x)
  108 mapFst f (a, x) = (f a, x)
  109 
  110 (>+<) :: PPState (Seq Char) -> PPState (Seq Char) -> PPState (Seq Char)
  111 (>+<) = liftM2 (><)
  112 
  113 printListWithSeps :: RealSrcLoc -> Int -> Int -> [String] -> [RoseTree SrcTemplateStage] -> PPState (Seq Char)
  114 printListWithSeps = printListWithSeps' (const putString) 0

  116 -- | Prints the elements of a list where the elements must be printed in the same line (do stmts, case alts, let binds, ...)
  117 printListWithSepsIndented :: [Bool] -> RealSrcLoc -> Int -> Int -> [String] -> [RoseTree SrcTemplateStage] -> PPState (Seq Char)
  118 printListWithSepsIndented indentedChildren parent slide minInd seps children
  119   = do base <- get
  120        let putCorrectSep i _ min s | isIndented i
  121               = do curr <- get
  122                    let (shortened, currCol) = untilReaches s curr base
  123                    putString 0 min $ shortened ++ replicate (srcLocCol base - currCol) ' '
  124            putCorrectSep _ slide minInd s = putString slide minInd s
  125        printListWithSeps' putCorrectSep 0 parent slide minInd seps children
  126   where -- the ith separator is before the ith element
  127         isIndented i = case List.drop i indentedChildren of False:_ -> False; _ -> True
  128 
  129 printListWithSeps' :: (Int -> Int -> Int -> String -> PPState (Seq Char)) -> Int -> RealSrcLoc -> Int -> Int -> [String] -> [RoseTree SrcTemplateStage] -> PPState (Seq Char)
  130 printListWithSeps' _ _ _ _ _ _ [] = return empty
  131 printListWithSeps' _ _ parent _ _ _ [child] = printRose' parent child
  132 printListWithSeps' putCorrectSep i parent slide minInd (sep:seps) (child:children)
  133   = printRose' parent child >+< putCorrectSep i slide minInd sep >+< printListWithSeps' putCorrectSep (i+1) parent slide minInd seps children
  134 printListWithSeps' _ _ _ _ _ [] _ = pprProblem "printListWithSeps': the number of elements and separators does not match"
