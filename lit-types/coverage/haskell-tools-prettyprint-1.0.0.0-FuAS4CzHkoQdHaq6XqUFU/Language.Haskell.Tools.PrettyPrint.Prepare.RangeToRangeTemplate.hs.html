   19 -- | Creates a source template from the ranges and the input file.
   20 -- All source ranges must be good ranges.
   21 cutUpRanges :: forall node dom . SourceInfoTraversal node
   22                  => Ann node dom NormRangeStage
   23                  -> Ann node dom RngTemplateStage
   24 cutUpRanges n = evalState (cutUpRanges' n) [[],[]]
   25   where cutUpRanges' :: Ann node dom NormRangeStage -> State [[SrcSpan]] (Ann node dom RngTemplateStage)
   26         cutUpRanges' = sourceInfoTraverseUp (SourceInfoTrf (trf cutOutElemSpan) (trf cutOutElemList) (trf cutOutElemOpt)) desc asc
   27 
   28         -- keep the stack to contain the children elements on the place of the parent element
   29         desc = modify ([]:)
   30         asc  = modify tail
   31 
   32         -- combine the current node with its children, and add it to the list of current nodes
   33         trf :: HasRange (x RngTemplateStage)
   34             => ([SrcSpan] -> x NormRangeStage -> x RngTemplateStage) -> x NormRangeStage -> State [[SrcSpan]] (x RngTemplateStage)
   35         trf f ni = do (below : top : xs) <- get
   36                       let res = f below ni
   37                       put ([] : (top ++ [ getRange res ]) : xs)
   38                       return res
   39 
   40 -- | Cuts out a list of source ranges from a given range
   41 cutOutElemSpan :: [SrcSpan] -> SpanInfo NormRangeStage -> SpanInfo RngTemplateStage
   42 cutOutElemSpan sps (NormNodeInfo (RealSrcSpan sp))
   43   = RangeTemplateNode sp $ foldl breakFirstHit (foldl breakFirstHit [RangeElem sp] loc) span
   44   where (loc,span) = partition (\sp -> srcSpanStart sp == srcSpanEnd sp) sps
   45         breakFirstHit (elem:rest) sp
   46           = case breakUpRangeElem elem sp of
   47              -- only continue if the correct place for the child range is not found
   48               Just pieces -> pieces ++ rest
   49               Nothing -> elem : breakFirstHit rest sp
   50         breakFirstHit [] inner = throw $ BreakUpProblem sp inner sps
   51 cutOutElemSpan _ (NormNodeInfo (UnhelpfulSpan {}))
   52   = trfProblem "cutOutElemSpan: no real span"
   53 
   54 data BreakUpProblem = BreakUpProblem { bupOuter :: RealSrcSpan
   55                                      , bupInner :: SrcSpan
   56                                      , bupSiblings :: [SrcSpan]
   57                                      }
   58 
   59 instance Show BreakUpProblem where
   60  show (BreakUpProblem _ (RealSrcSpan inner) _)
   61    = unpackFS (srcSpanFile inner) ++ ": didn't find correct place for AST element at " ++ shortShowSpan (RealSrcSpan inner)
   62  show (BreakUpProblem outer _ _)
   63    = unpackFS (srcSpanFile outer) ++ ": didn't find correct place for AST element in " ++ shortShowSpan (RealSrcSpan outer)
   64 
   65 instance Exception BreakUpProblem

   67 cutOutElemList :: [SrcSpan] -> ListInfo NormRangeStage -> ListInfo RngTemplateStage
   68 cutOutElemList sps (NormListInfo bef aft sep indented sp)
   69   = let RealSrcSpan wholeRange = foldl1 combineSrcSpans $ sp : sps
   70      in RangeTemplateList wholeRange bef aft sep indented (getSeparators wholeRange sps)
   71 
   72 -- | Cuts out all elements from a list, the rest is the list of separators
   73 getSeparators :: RealSrcSpan -> [SrcSpan] -> [RealSrcSpan]
   74 getSeparators sp infos@(_:_:_)
   75   = mapMaybe getRangeElemSpan (cutOutElemSpan infos (NormNodeInfo (RealSrcSpan sp)) ^. rngTemplateNodeElems)
   76 -- at least two elements needed or there can be no separators
   77 getSeparators _ _ = []
   78 
   79 cutOutElemOpt :: [SrcSpan] -> OptionalInfo NormRangeStage -> OptionalInfo RngTemplateStage
   80 cutOutElemOpt sps (NormOptInfo bef aft sp)
   81   = let RealSrcSpan wholeRange = foldl1 combineSrcSpans $ sp : sps
   82      in RangeTemplateOpt wholeRange bef aft
   83 
   84 -- | Breaks the given template element into possibly 2 or 3 parts by cutting out the given part
   85 -- if it is inside the range of the template element. Returns Nothing if the second argument is not inside.
   86 breakUpRangeElem :: RangeTemplateElem -> SrcSpan -> Maybe [RangeTemplateElem]
   87 breakUpRangeElem (RangeElem outer) (RealSrcSpan inner)
   88   | outer `containsSpan` inner
   89   = Just $ (if (realSrcSpanStart outer) < (realSrcSpanStart inner)
   90               then [ RangeElem (mkRealSrcSpan (realSrcSpanStart outer) (realSrcSpanStart inner)) ]
   91               else []) ++
   92            [ RangeChildElem ] ++
   93            (if (realSrcSpanEnd inner) < (realSrcSpanEnd outer)
   94               then [ RangeElem (mkRealSrcSpan (realSrcSpanEnd inner) (realSrcSpanEnd outer)) ]
   95               else [])
   96 breakUpRangeElem _ _ = Nothing

   99 -- | Modifies ranges to contain their children
  100 fixRanges :: SourceInfoTraversal node
  101           => Ann node dom RangeStage
  102           -> Ann node dom NormRangeStage
  103 fixRanges node = evalState (sourceInfoTraverseUp (SourceInfoTrf (trf expandToContain) (trf expandListToContain) (trf expandOptToContain)) desc asc node) [[],[]]
  104   where -- keep the stack to contain the children elements on the place of the parent element
  105         desc = modify ([]:)
  106         asc  = modify tail
  107 
  108         trf :: HasRange (x NormRangeStage)
  109             => ([SrcSpan] -> x RangeStage -> x NormRangeStage) -> x RangeStage -> State [[SrcSpan]] (x NormRangeStage)
  110         trf f ni = do (below : top : xs) <- get
  111                       let res = f below ni
  112                           resRange = getRange res
  113                           endOfSiblings = srcSpanEnd (collectSpanRanges (srcSpanStart resRange) top)
  114                           correctedRange = if endOfSiblings > srcSpanStart resRange
  115                                              then mkSrcSpan endOfSiblings (max endOfSiblings (srcSpanEnd resRange))
  116                                              else resRange
  117                       put ([] : (top ++ [ correctedRange ]) : xs)
  118                       return $ setRange correctedRange res
  119 
  120 -- | Expand a simple node to contain its children
  121 expandToContain :: [SrcSpan] -> SpanInfo RangeStage -> SpanInfo NormRangeStage
  122 expandToContain cont (NodeSpan sp)
  123   = NormNodeInfo (checkSpans cont $ foldl1 combineSrcSpans $ sp : cont)
  124 
  125 expandListToContain :: [SrcSpan] -> ListInfo RangeStage -> ListInfo NormRangeStage
  126 expandListToContain cont (ListPos bef aft def ind sp)
  127   = NormListInfo bef aft def ind (checkSpans cont $ collectSpanRanges sp cont)
  128 
  129 expandOptToContain :: [SrcSpan] -> OptionalInfo RangeStage -> OptionalInfo NormRangeStage
  130 expandOptToContain cont (OptionalPos bef aft sp)
  131   = NormOptInfo bef aft (checkSpans cont $ collectSpanRanges sp cont)
  132 
  133 collectSpanRanges :: SrcLoc -> [SrcSpan] -> SrcSpan
  134 collectSpanRanges loc@(RealSrcLoc _) [] = srcLocSpan loc
  135 collectSpanRanges _ ls = foldl combineSrcSpans noSrcSpan ls
  136 
  137 -- | Checks the contained source ranges to detect the convertion problems where we can see their location.
  138 checkSpans :: [SrcSpan] -> SrcSpan -> SrcSpan
  139 checkSpans spans res
  140   = if any (not . isGoodSrcSpan) spans && isGoodSrcSpan res
  141       then trfProblem $ "Wrong src spans in " ++ show res
  142       else res
