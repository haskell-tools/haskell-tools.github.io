   78 type AssocSema dom = ( AssocData (SemanticInfo' dom SemaInfoModuleCls), AssocData (SemanticInfo' dom SemaInfoImportCls)

   82 astDebug :: (ASTDebug e dom st, AssocSema dom) => e dom st -> String
   83 astDebug ast = toList (astDebugToJson (astDebug' ast))
   84 
   85 astDebugToJson :: AssocSema dom => [DebugNode dom] -> Seq Char
   86 astDebugToJson nodes = fromList "[ " >< childrenJson >< fromList " ]"
   87     where treeNodes = List.filter (\case TreeNode {} -> True; _ -> False) nodes
   88           childrenJson = case map debugTreeNode treeNodes of
   89                            first:rest -> first >< foldl (><) Seq.empty (fmap (fromList ", " ><) (fromList rest))
   90                            []         -> Seq.empty
   91           debugTreeNode (TreeNode "" s) = astDebugElemJson s
   92           debugTreeNode (TreeNode (dropWhile (=='_') -> l) s) = astDebugElemJson (nodeName .- (("<span class='astlab'>" ++ l ++ "</span>: ") ++) $ s)
   93           debugTreeNode (SimpleNode {}) = error "debugTreeNode: simple SimpleNode not allowed"
   94 
   95 astDebugElemJson :: AssocSema dom => TreeDebugNode dom -> Seq Char
   96 astDebugElemJson (TreeDebugNode name info children)
   97   = fromList "{ \"text\" : \"" >< fromList name
   98      >< fromList "\", \"state\" : { \"opened\" : true }, \"a_attr\" : { \"data-range\" : \""
   99      >< fromList (shortShowSpan (semaInfoTypeRng info))
  100      >< fromList "\", \"data-elems\" : \""
  101      >< foldl (><) Seq.empty dataElems
  102      >< fromList "\", \"data-sema\" : \""
  103      >< fromList (showSema info)
  104      >< fromList "\" }, \"children\" : "
  105      >< astDebugToJson children >< fromList " }"
  106   where dataElems = catMaybes (map (\case SimpleNode l v -> Just (fromList (formatScalarElem l v)); _ -> Nothing) children)
  107         formatScalarElem l v = "<div class='scalarelem'><span class='astlab'>" ++ l ++ "</span>: " ++ tail (init (show v)) ++ "</div>"
  108         showSema info = "<div class='semaname'>" ++ assocName info ++ "</div>"
  109                           ++ concatMap (\(l,i) -> "<div class='scalarelem'><span class='astlab'>" ++ l ++ "</span>: " ++ i ++ "</div>") (toAssoc info)

  115 instance AssocSema dom => AssocData (SemanticInfoType dom) where

  127   toAssoc s@(ModuleInfoType {}) = toAssoc (semaInfoTypeModule s)

  132   assocName _ = "ScopeInfo"
  133   toAssoc si = [ ("namesInScope", inspectScope (semanticsScope si)) ]

  135 instance HasNameInfo' (NameInfo n) => AssocData (NameInfo n) where
  136   assocName _ = "NameInfo"
  137 
  138   toAssoc ni = [ ("name", maybe "<ambiguous>" inspect (semanticsName ni))
  139                , ("isDefined", show (semanticsDefining ni))
  140                , ("namesInScope", inspectScope (semanticsScope ni))
  141                ]
  142 
  143 instance AssocData CNameInfo where
  144   assocName _ = "CNameInfo"
  145   toAssoc ni = [ ("name", inspect (semanticsId ni))
  146                , ("isDefined", show (semanticsDefining ni))
  147                , ("fixity", maybe "" (showSDocUnsafe . ppr) (semanticsFixity ni))
  148                , ("namesInScope", inspectScope (semanticsScope ni))
  149                ]
  150 
  151 instance (HasModuleInfo' (ModuleInfo n)) => AssocData (ModuleInfo n) where
  152   assocName _ = "ModuleInfo"
  153   toAssoc mi = [ ("moduleName", showSDocUnsafe (ppr (semanticsModule mi)))
  154                , ("isBoot", show (isBootModule mi))
  155                , ("implicitImports", concat (intersperse ", " (map inspect (semanticsImplicitImports mi))))
  156                ]
  157 
  158 instance (HasImportInfo' (ImportInfo n)) => AssocData (ImportInfo n) where
  159   assocName _ = "ImportInfo"
  160   toAssoc ii = [ ("moduleName", showSDocUnsafe (ppr (semanticsImportedModule ii)))
  161                , ("availableNames", concat (intersperse ", " (map inspect (semanticsAvailable ii))))
  162                , ("importedNames", concat (intersperse ", " (map inspect (semanticsImported ii))))
  163                ]
  164 
  165 instance AssocData ImplicitFieldInfo where
  166   assocName _ = "ImplicitFieldInfo"
  167   toAssoc ifi = [ ("bindings", concat (intersperse ", " (map (\(from,to) -> "(" ++ inspect from ++ " -> " ++ inspect to ++ ")") (semanticsImplicitFlds ifi))))
  168                 ]

  170 inspectScope :: InspectableName n => [[(n, Maybe [UsageSpec], Maybe n)]] -> String
  171 inspectScope = concat . intersperse " | " . map (concat . intersperse ", " . map showUsage)
  172 
  173 class Outputable n => InspectableName n where
  174   inspect :: n -> String
  175 
  176 showUsage :: InspectableName n => (n, Maybe [UsageSpec], Maybe n) -> String
  177 showUsage (n,usage,parent) = inspect n ++ show usage ++ maybe "" (\p -> "( in " ++ showSDocUnsafe (ppr p) ++ ")") parent
  178 
  179 instance Show UsageSpec where
  180   show (UsageSpec q useQ asQ)
  181     = (if q then "qualified " else "") ++ "as " ++ (if useQ == asQ || q then asQ else asQ ++ " or " ++ useQ)
  182 
  183 instance InspectableName GHC.Name where
  184   inspect name = showSDocUnsafe (ppr name) ++ "[" ++ show (getUnique name) ++ "]"
  185 
  186 instance InspectableName GHC.RdrName where
  187   inspect name = showSDocUnsafe (ppr name)
  188 
  189 instance InspectableName GHC.Id where
  190   inspect name = showSDocUnsafe (ppr name) ++ "[" ++ show (getUnique name) ++ "] :: " ++ showSDocOneLine unsafeGlobalDynFlags (ppr (idType name)) ++ " (" ++ addendum ++ ")"
  191     where addendum = concat $ intersperse "," $ map (\v -> showSDocUnsafe (ppr v) ++ "[" ++ show (getUnique v) ++ "]") (getTVs (idType name))
  192 
  193 getTVs :: GHC.Type -> [GHC.Var]
  194 getTVs t
  195   | Just tv <- getTyVar_maybe t = [tv]
  196   | Just (op, arg) <- splitAppTy_maybe t = getTVs op `union` getTVs arg
  197   | Just (_, t') <- splitForAllTy_maybe t = getTVs t'
  198   | otherwise = []

  215 instance (GAstDebug f dom, GAstDebug g dom) => GAstDebug (f :+: g) dom where
  216   gAstDebug (L1 x) = gAstDebug x
  217   gAstDebug (R1 x) = gAstDebug x

  219 instance (GAstDebug f dom, GAstDebug g dom) => GAstDebug (f :*: g) dom where
  220   gAstDebug (x :*: y)
  221     = gAstDebug x ++ gAstDebug y

  223 instance {-# OVERLAPPING #-} ASTDebug e dom st => GAstDebug (K1 i (e dom st)) dom where
  224   gAstDebug (K1 x) = astDebug' x
  225 
  226 instance {-# OVERLAPPABLE #-} Show x => GAstDebug (K1 i x) dom where
  227   gAstDebug (K1 x) = [SimpleNode "" (show x)]
  228 
  229 instance (GAstDebug f dom, Constructor c) => GAstDebug (M1 C c f) dom where
  230   gAstDebug c@(M1 x) = [TreeNode "" (TreeDebugNode (conName c) undefined (gAstDebug x))]
  231 
  232 instance (GAstDebug f dom, Selector s) => GAstDebug (M1 S s f) dom where
  233   gAstDebug s@(M1 x) = traversal&nodeLabel .= selName s $ gAstDebug x
