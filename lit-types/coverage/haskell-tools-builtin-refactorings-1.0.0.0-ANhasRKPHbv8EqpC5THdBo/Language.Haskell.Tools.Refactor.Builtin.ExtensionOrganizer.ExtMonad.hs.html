   55 isTurnedOn :: Extension -> ExtMonad Bool
   56 isTurnedOn ext = do
   57   defaults <- ask
   58   return $! ext `elem` defaults

   60 conditional :: (node -> ExtMonad node) ->
   61                Extension ->
   62                node ->
   63                ExtMonad node
   64 conditional checker ext = conditionalAny checker [ext]
   65 
   66 conditionalNot :: (node -> ExtMonad node) ->
   67                   Extension ->
   68                   node ->
   69                   ExtMonad node
   70 conditionalNot checker ext node = do
   71   b <-isTurnedOn ext
   72   if b then return node else checker node
   73 
   74 conditionalAny :: (node -> ExtMonad node) ->
   75                    [Extension] ->
   76                    node ->
   77                    ExtMonad node
   78 conditionalAny checker exts node = do
   79   bs <- mapM isTurnedOn exts
   80   if or bs then checker node else return node

   82 conditionalAdd :: HasRange node => Extension -> node -> ExtMonad node
   83 conditionalAdd ext = conditional (addOccurence ext) ext
