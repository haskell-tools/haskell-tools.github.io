   28 chkFlexibleInstances :: CheckNode Decl
   29 chkFlexibleInstances = conditional chkFlexibleInstances' FlexibleInstances

   31 chkFlexibleInstances' :: CheckNode Decl
   32 chkFlexibleInstances' d@(Refact.StandaloneDeriving _ _ rule) = checkedReturn rule d
   33 chkFlexibleInstances' d@(InstanceDecl rule _)                = checkedReturn rule d
   34 chkFlexibleInstances' d = return d

   36 checkedReturn :: InstanceRule -> a -> ExtMonad a
   37 checkedReturn rule x = chkInstanceRule rule >> return x

   39 -- this check DOES transform the AST for its internal computations
   40 -- but returns the original one in the end
   41 -- NOTE: There are two traversals:
   42 --       First one on the class level, and the second one one on the type level.
   43 --       Since biplateRef is lazy, it won't go down to the type level in the first traversal
   44 chkInstanceRule :: CheckNode InstanceRule
   45 chkInstanceRule r@(InstanceRule _ _ ihead) = do
   46   chkInstanceHead ihead
   47   return $! r
   48 chkInstanceRule r = return r

   50 refact ::
   51      (Data.Data.Data (node dom stage), Data.Data.Data (inner dom stage),
   52       Monad m) =>
   53      (inner dom stage -> m (inner dom stage))
   54      -> node dom stage -> m (node dom stage)
   55 refact op = biplateRef !~ op

   58 -- one IHApp will only check its own tyvars (their structure and uniqueness)
   59 -- thus with MultiParamTypeclasses each param will be checked independently
   60 -- (so the same type variable can appear in multiple params)
   61 chkInstanceHead :: CheckNode InstanceHead
   62 chkInstanceHead x@(InfixInstanceHead tyvars _) = do
   63   tyvars' <- refact rmTypeMisc tyvars
   64   chkTyVars tyvars'
   65   addOccurence_ MultiParamTypeClasses x
   66   addOccurence_ TypeOperators x
   67   return x
   68 chkInstanceHead app@(AppInstanceHead f tyvars) = do
   69   tyvars' <- refact rmTypeMisc tyvars
   70   chkTyVars tyvars'
   71   case f of
   72     AppInstanceHead _ _ -> addOccurence_ MultiParamTypeClasses app
   73     _ -> return ()
   74   chkInstanceHead f
   75   return app
   76 chkInstanceHead x@(ParenInstanceHead h) = do
   77   chkInstanceHead h
   78   return x
   79 chkInstanceHead app = return app

   87 -- synonym expansion (runMaybeT . lookupSynDefM $ vars) (now: if synonym, keep FC)
   88 chkTyVars :: CheckNode Type
   89 chkTyVars vars = do
   90   msyn <- runMaybeT . lookupSynDefM $ vars
   91   maybe (performCheck vars) (const $ addOccurence FlexibleInstances vars) msyn
   92 
   93   where performCheck vars = do
   94           (isOk, (_, vs)) <- runStateT (runMaybeT (chkAll vars)) ([],[])
   95           case isOk of
   96             Just isOk ->
   97               unless (isOk && length vs == (length . nubBy ((==) `on` (semanticsName . (^. simpleName))) $ vs)) --tyvars are different
   98                 (addOccurence_ FlexibleInstances vars)
   99             Nothing   -> error "chkTyVars: Couldn't look up something"
  100           return vars
  101 
  102         chkAll x =
  103           ifM (chkTopLevel x) $
  104             chkOnlyApp x
  105 
  106         chkTopLevel x = -- NOTE: this resembles a monadic bind ... (Cont?)
  107           ifM (chkListType x) .
  108             ifM (chkTupleType x) .
  109               ifM (chkUnitTyCon x) $
  110                 return False
  111 
  112         ifM cond f = do b <- cond; if b then (return b) else f
  113 
  114         chkUnitTyCon (VarType x) = do
  115           sname <- tyVarSemNameM x
  116           -- standalone top-level type variables are not accepted
  117           -- NOTE: -XHaskell98 operator type variables??
  118           -- NOTE VarType is either TyCon or TyVar
  119           --      if it is a TyCon, it cannot be wired in (Int, Char, etc)
  120           if | isTyVarName   sname -> addTyVarM x >> return False
  121              | isWiredInName sname -> addTyConM x >> return False
  122              | isTyConName   sname -> addTyConM x >> return True
  123              | otherwise           -> return True -- NEVER
  124         chkUnitTyCon _ = return False
  125 
  126 
  127         chkSingleTyVar (VarType x) = do
  128           sname <- tyVarSemNameM x
  129           if (isTyVarName sname)
  130             then addTyVarM x >> return True
  131             else addTyConM x >> return False
  132         chkSingleTyVar _ = return False
  133 
  134 
  135         chkTupleType (TupleType args) = do
  136           let xs  = args ^. annListElems
  137           bs <- mapM chkSingleTyVar xs
  138           return $! and bs
  139         chkTupleType _ = return False
  140 
  141         chkListType (ListType v) = chkSingleTyVar v
  142         chkListType _            = return False
  143 
  144         chkOnlyApp :: (MonadState ([Name],[Name]) (m1 m2),
  145                        MonadTrans m1,
  146                        MonadState ExtMap m2) =>
  147                        Type -> MaybeT (m1 m2) Bool
  148         chkOnlyApp (TypeApp f v@(VarType _)) = do
  149           isTyVar <- chkSingleTyVar v
  150           if isTyVar
  151             then case f of
  152               (VarType c) -> addTyConM c >> return True
  153               _           -> chkOnlyApp f
  154             else return False
  155         chkOnlyApp x@(InfixTypeApp lhs op rhs) = do
  156           lift . lift $ addOccurence_ TypeOperators x
  157           addTyConM . mkNormalName $ (op ^. operatorName)
  158           lOK <- chkSingleTyVar lhs
  159           rOK <- chkSingleTyVar rhs
  160           return $! lOK && rOK
  161         chkOnlyApp _ = return False
  162 
  163         addTyCon  n (ctors, vars) = (n:ctors, vars)
  164         addTyVar  n (ctors, vars) = (ctors, n:vars)
  165         addTyConM n               = modify $ addTyCon n
  166         addTyVarM n               = modify $ addTyVar  n
  167 
  168         tyVarSemNameM x = MaybeT . return . semanticsName $ x ^. simpleName

  170 rmTypeMisc :: CheckNode Type
  171 rmTypeMisc = rmTParens >=> rmTKinded
  172 
  173 rmTKinded :: CheckNode Type
  174 rmTKinded kt@(KindedType t _) = addOccurence_ KindSignatures kt >> return t
  175 rmTKinded x                   = return x

  177 -- removes Parentheses from the AST
  178 -- the structure is reserved
  179 rmTParens :: CheckNode Type
  180 rmTParens (ParenType x) = return x
  181 rmTParens x             = return x
