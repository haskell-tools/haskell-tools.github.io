    8 -- | Kind constraint (@ :: * -> * @)
    9 pattern KindConstraint :: Kind -> KindConstraint
   10 pattern KindConstraint k <- Ann _ (UKindConstraint k)
   11 
   12 -- | @*@, the kind of types
   13 pattern StarKind :: Kind
   14 pattern StarKind <- Ann _ UStarKind
   15 
   16 -- | @#@, the kind of unboxed types
   17 pattern UnboxKind :: Kind
   18 pattern UnboxKind <- Ann _ UUnboxKind
   19 
   20 -- | @->@, the kind of type constructor
   21 pattern FunKind :: Kind -> Kind -> Kind
   22 pattern FunKind a r <- Ann _ (UFunKind a r)
   23 
   24 -- | A parenthesised kind
   25 pattern ParenKind :: Kind -> Kind
   26 pattern ParenKind k <- Ann _ (UParenKind k)
   27 
   28 -- | Kind variable (using @PolyKinds@ extension)
   29 pattern VarKind :: Name -> Kind
   30 pattern VarKind v <- Ann _ (UVarKind v)
   31 
   32 -- | Kind application (@ k1 k2 @)
   33 pattern AppKind :: Kind -> Kind -> Kind
   34 pattern AppKind f a <- Ann _ (UAppKind f a)
   35 
   36 -- | A list kind (@ [k] @)
   37 pattern ListKind :: Kind -> Kind
   38 pattern ListKind k <- Ann _ (UListKind k)
   39 
   40 -- | Numeric value promoted to the kind level.
   41 pattern IntKind :: Integer -> Kind
   42 pattern IntKind i <- Ann _ (UPromotedKind (Ann _ (UPromotedInt i)))
   43 
   44 -- | String value promoted to the kind level.
   45 pattern StringKind :: String -> Kind
   46 pattern StringKind s <- Ann _ (UPromotedKind (Ann _ (UPromotedString s)))
   47 
   48 -- | A data constructor value promoted to the kind level.
   49 pattern ConKind :: Name -> Kind
   50 pattern ConKind s <- Ann _ (UPromotedKind (Ann _ (UPromotedCon s)))

   52 -- | A list of elements as a kind.
   53 pattern ListKindPromoted :: KindList -> Kind

   55 
   56 -- | A tuple of elements as a kind.
   57 pattern TupleKind :: KindList -> Kind
   58 pattern TupleKind elems <- Ann _ (UPromotedKind (Ann _ (UPromotedTuple elems)))
   59 
   60 -- | Kind of the unit value @()@. 
   61 pattern UnitKind :: Kind
