    1 -- | Generation of type-level AST fragments for refactorings.
    2 -- The bindings defined here create a the annotated version of the AST constructor with the same name.
    3 -- For example, @mkTyForall@ creates the annotated version of the @TyForall@ AST constructor.
    4 {-# LANGUAGE OverloadedStrings, TypeFamilies #-}
    5 module Language.Haskell.Tools.Rewrite.Create.Types where
    6 
    7 import Data.String (IsString(..), String)
    8 import Language.Haskell.Tools.AST
    9 import Language.Haskell.Tools.PrettyPrint.Prepare
   10 import Language.Haskell.Tools.Rewrite.Create.Kinds (mkKindConstraint)
   11 import Language.Haskell.Tools.Rewrite.Create.Names (mkUnqualName')
   12 import Language.Haskell.Tools.Rewrite.Create.Utils
   13 import Language.Haskell.Tools.Rewrite.ElementTypes
   14 import qualified Name as GHC (Name)
   15 
   16 -- * Generation of types
   17 
   18 -- | Forall types (@ forall x y . type @)
   19 mkForallType :: [TyVar] -> Type -> Type
   20 mkForallType vars t = mkAnn ("forall " <> child <> " . " <> child) (UTyForall (mkAnnList (separatedBy " " list) vars) t)
   21 
   22 -- | Simplified creation of type variables
   23 mkTypeVar' :: GHC.Name -> TyVar
   24 mkTypeVar' = mkTypeVar . mkUnqualName'
   25 
   26 -- | Type with a context (@ forall x y . type @)
   27 mkCtxType :: Context -> Type -> Type
   28 mkCtxType ctx t = mkAnn (child <> " " <> child) (UTyCtx ctx t)
   29 
   30 -- | Function types (@ a -> b @)
   31 mkFunctionType :: Type -> Type -> Type
   32 mkFunctionType at rt = mkAnn (child <> " -> " <> child) (UTyFun at rt)
   33 
   34 -- | Tuple types (@ (a,b) @)
   35 mkTupleType :: [Type] -> Type
   36 mkTupleType args = mkAnn ("(" <> child <> ")") (UTyTuple (mkAnnList (separatedBy ", " list) args))
   37 
   38 -- | Unboxed tuple types (@ (\#a,b\#) @)
   39 mkUnboxedTupleType :: [Type] -> Type
   40 mkUnboxedTupleType args = mkAnn ("(#" <> child <> "#)") (UTyUnbTuple (mkAnnList (separatedBy ", " list) args))
   41 
   42 -- | List type with special syntax (@ [a] @)
   43 mkListType :: Type -> Type
   44 mkListType = mkAnn ("[" <> child <> "]") . UTyList
   45 
   46 -- | Parallel array type (@ [:a:] @)
   47 mkParArrayType :: Type -> Type
   48 mkParArrayType = mkAnn ("[:" <> child <> ":]") . UTyParArray
   49 
   50 -- | Type application (@ F a @)
   51 mkTypeApp :: Type -> Type -> Type
   52 mkTypeApp ft at = mkAnn (child <> " " <> child) (UTyApp ft at)
   53 
   54 -- | Infix type constructor (@ (a <: b) @)
   55 mkInfixTypeApp :: Type -> Operator -> Type -> Type
   56 mkInfixTypeApp left op right = mkAnn (child <> " " <> child <> " " <> child) (UTyInfix left op right)
   57 
   58 -- | Type surrounded by parentheses (@ (T a) @)
   59 mkParenType :: Type -> Type
   60 mkParenType = mkAnn ("(" <> child <> ")") . UTyParen
   61 
   62 -- | Creates a simple type variable
   63 mkTypeVar :: Name -> TyVar
   64 mkTypeVar n = mkAnn (child <> child) (UTyVarDecl n noth)
   65 
   66 -- | Creates a type variable with kind specification (@ t :: * @)
   67 mkKindedTypeVar :: Name -> Kind -> TyVar
   68 mkKindedTypeVar n k = mkAnn (child <> child) (UTyVarDecl n (justVal (mkKindConstraint k)))
   69 
   70 -- | Type variable or constructor (@ a @)
   71 mkVarType :: Name -> Type
   72 mkVarType = wrapperAnn . UTyVar
   73 
   74 -- | Type with explicit kind signature (@ a :: * @)
   75 mkKindedType :: Type -> Kind -> Type
   76 mkKindedType t k = mkAnn (child <> " :: " <> child) (UTyKinded t k)
   77 
   78 -- | Strict type marked with @!@.
   79 mkBangType :: Type -> Type
   80 mkBangType = mkAnn ("!" <> child) . UTyBang
   81 
   82 -- | Lazy type marked with @~@. (Should only be used if @Strict@ or @StrictData@ language extension is used)
   83 mkLazyType :: Type -> Type
   84 mkLazyType = mkAnn ("~" <> child) . UTyLazy
   85 
   86 -- | Strict type marked with UNPACK pragma. (Usually contains the bang mark.)
   87 mkUnpackType :: Type -> Type
   88 mkUnpackType = mkAnn ("{-# UNPACK #-} " <> child) . UTyUnpack
   89 
   90 -- | Strict type marked with UNPACK pragma. (Usually contains the bang mark.)
   91 mkNoUnpackType :: Type -> Type
   92 mkNoUnpackType = mkAnn ("{-# NOUNPACK #-} " <> child) . UTyNoUnpack
   93 
   94 -- | A wildcard type (@ _ @) with @-XPartialTypeSignatures@
   95 mkWildcardType :: Type
   96 mkWildcardType = mkAnn "_" UTyWildcard
   97 
   98 -- | A named wildcard type (@ _t @) with @-XPartialTypeSignatures@
   99 mkNamedWildcardType :: Name -> Type
  100 mkNamedWildcardType = mkAnn ("_" <> child) . UTyNamedWildc
  101 
  102 -- | A Template Haskell splice type (@ $(genType) @).
  103 mkSpliceType :: Splice -> Type
  104 mkSpliceType = mkAnn child . UTySplice
  105 
  106 -- | A Template Haskell quasi-quote type (@ [quoter| ... ] @).
  107 mkQuasiQuoteType :: QuasiQuote -> Type
  108 mkQuasiQuoteType = mkAnn child . UTyQuasiQuote
  109 
  110 
  111 -- | Numeric value promoted to the kind level.
  112 mkPromotedIntType :: Integer -> Type
  113 mkPromotedIntType i = mkAnn child $ UTyPromoted $ mkAnn (fromString $ show i) (UPromotedInt i)
  114 
  115 -- | String value promoted to the kind level.
  116 mkPromotedStringType :: String -> Type
  117 mkPromotedStringType i = mkAnn child $ UTyPromoted $ mkAnn (fromString $ show i) (UPromotedString i)
  118 
  119 -- | A data constructor value promoted to the kind level.
  120 mkPromotedConType :: Name -> Type
  121 mkPromotedConType = mkAnn child . UTyPromoted . mkAnn child . UPromotedCon
  122 
  123 -- | A list of elements as a kind.
  124 mkPromotedListType :: [Type] -> Type
  125 mkPromotedListType
  126   = mkAnn child . UTyPromoted . mkAnn ("[" <> child <> "]") . UPromotedList . mkAnnList (separatedBy ", " list)
  127 
  128 -- | A tuple of elements as a kind.
  129 mkPromotedTupleType :: [Type] -> Type
  130 mkPromotedTupleType
  131   = mkAnn child . UTyPromoted . mkAnn ("(" <> child <> ")") . UPromotedTuple . mkAnnList (separatedBy ", " list)
  132 
  133 -- | Kind of the unit value @()@.
  134 mkPromotedUnitType :: Type
  135 mkPromotedUnitType = mkAnn child $ UTyPromoted $ mkAnn "()" UPromotedUnit
  136 
  137 -- * Generation of contexts
  138 
  139 -- | Creates a context of assertions (@ C a => ... @)
  140 mkContext :: Assertion -> Context
  141 mkContext = mkAnn (child <> " =>") . UContext
  142 
  143 -- * Generation of assertions
  144 
  145 -- | Class assertion (@Cls x@)
  146 mkClassAssert :: Name -> [Type] -> Assertion
  147 -- fixme: class assertion without parameters should not have the last space
  148 mkClassAssert n args = mkAnn (child <> " " <> child) $ UClassAssert n (mkAnnList (separatedBy " " list) args)
  149 
  150 -- | Infix class assertion, also contains type equations (@ a ~ X y @)
  151 mkInfixAssert :: Type -> Operator -> Type -> Assertion
  152 mkInfixAssert left op right = mkAnn (child <> " " <> child <> " " <> child) $ UInfixAssert left op right
  153 
  154 -- | Creates an assertion for implicit parameter binding (@ ?cmp :: a -> a -> Bool @)
  155 mkImplicitAssert :: Name -> Type -> Assertion
  156 mkImplicitAssert n t = mkAnn (child <> " :: " <> child) $ UImplicitAssert n t
  157 
  158 -- | Creates a list of assertions (@ (Eq a, Show a) @)
  159 mkTupleAssertion :: [Assertion] -> Assertion
  160 mkTupleAssertion ass = mkAnn ("(" <> child <> ")") $ UTupleAssert $ mkAnnList (separatedBy ", " list) ass
