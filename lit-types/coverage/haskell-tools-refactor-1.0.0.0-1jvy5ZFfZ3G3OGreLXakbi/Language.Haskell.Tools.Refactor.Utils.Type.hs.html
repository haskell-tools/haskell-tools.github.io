   36 typeExpr :: Expr -> Ghc GHC.Type
   37 typeExpr e = do usupp <- liftIO $ mkSplitUniqSupply 'z'
   38                 evalStateT (typeExpr' e) (uniqsFromSupply usupp)
   39 
   40 typeExpr' :: Expr -> StateT [Unique] Ghc GHC.Type
   41 typeExpr' (Var n) = return $ idType $ semanticsId (n ^. simpleName)
   42 typeExpr' (Lit l) = literalType' l
   43 typeExpr' (InfixApp lhs op rhs) = do
   44   lhsType <- typeExpr' lhs
   45   rhsType <- typeExpr' rhs
   46   let opType = idType $ semanticsId (op ^. operatorName)
   47   let (opt, repack) = splitType' opType
   48   case splitFunTys opt of
   49     (lhsT:rhsT:rest, resTyp) -> do
   50       let subst = tcUnifyTys (\_ -> BindMe) [lhsType, rhsType] [lhsT, rhsT]
   51       return $ maybe id substTy subst $ repack (mkFunTys rest resTyp)
   52     _ -> resultType
   53 typeExpr' (PrefixApp op rhs) = do
   54   let opType = idType $ semanticsId (op ^. operatorName)
   55   argType <- typeExpr' rhs
   56   let (ft, repack) = splitType' opType
   57   case splitFunTy_maybe ft of
   58     Just (argTyp, resTyp) -> do
   59       let subst = tcUnifyTy argType argTyp
   60       return $ maybe id substTy subst $ repack resTyp
   61     _ -> resultType
   62 typeExpr' (App f arg) = do
   63   argType <- typeExpr' arg
   64   funType <- typeExpr' f
   65   let (ft, repack) = splitType' funType
   66   case splitFunTy_maybe ft of
   67     Just (argTyp, resTyp) -> do
   68       let subst = tcUnifyTy argType argTyp
   69       return $ maybe id substTy subst $ repack resTyp
   70     Nothing -> return funType
   71 typeExpr' (Lambda args e) = do
   72   (resType, recomp) <- splitType' <$> typeExpr' e
   73   (argTypes, recomps) <- unzip <$> mapM (\_ -> splitType' <$> resultType) (args ^? annList)
   74   return $ foldr (.) recomp recomps $ mkFunTys argTypes resType
   75 typeExpr' (Let _ e) = typeExpr' e
   76 typeExpr' (If _ then' _) = typeExpr' then'
   77 typeExpr' (MultiIf alts) =
   78   case alts ^? annList & caseGuardExpr of e:_ -> typeExpr' e
   79                                           _   -> resultType
   80 typeExpr' (Case _ alts) =
   81   case alts ^? annList & altRhs & rhsCaseExpr of e:_ -> typeExpr' e
   82                                                  _   -> resultType
   83 typeExpr' (Do stmts) =
   84   case stmts ^? annList & stmtExpr of e:_ -> typeExpr' e
   85                                       _   -> resultType
   86 typeExpr' (Tuple elems) = do
   87   (elemTypes, pack) <- unzip . map splitType' <$> mapM typeExpr' (elems ^? annList)
   88   return $ foldr (.) id pack $ mkTupleTy Boxed elemTypes
   89 typeExpr' (AST.UnboxedTuple elems) = do
   90   (elemTypes, pack) <- unzip . map splitType' <$> mapM typeExpr' (elems ^? annList)
   91   return $ foldr (.) id pack $ mkTupleTy Unboxed elemTypes

   94 --      in do args <- mapM resultType holes
   95 -- 
   96 -- typeExpr' (UnboxedTupSec elems) = 
   97 typeExpr' (List elems) = do
   98   case elems ^? annList of []  -> do (typ, pack) <- splitType' <$> resultType
   99                                      return $ pack $ mkListTy typ
  100                            e:_ -> do (typ, pack) <- splitType' <$> typeExpr' e
  101                                      return $ pack $ mkListTy typ
  102 typeExpr' (ParArray elems) = do
  103   case elems ^? annList of []  -> do (typ, pack) <- splitType' <$> resultType
  104                                      return $ pack $ mkPArrTy typ
  105                            e:_ -> do (typ, pack) <- splitType' <$> typeExpr' e
  106                                      return $ pack $ mkPArrTy typ
  107 typeExpr' (Paren inner) = typeExpr' inner
  108 typeExpr' (LeftSection lhs op) = do
  109   let opType = idType $ semanticsId (op ^. operatorName)
  110   argType <- typeExpr' lhs
  111   let (ft, repack) = splitType' opType
  112   case splitFunTy_maybe ft of 
  113     Just (argTyp, resTyp) -> do
  114       let subst = tcUnifyTy argType argTyp
  115       return $ maybe id substTy subst $ repack resTyp
  116     _ -> resultType
  117 typeExpr' (RightSection op rhs) = do
  118   let opType = idType $ semanticsId (op ^. operatorName)
  119   argType <- typeExpr' rhs
  120   let (ft, repack) = splitType' opType
  121   case splitFunTys ft of 
  122     (arg1:arg2:rest, resTyp) -> do 
  123       let subst = tcUnifyTy arg2 argType
  124       return $ maybe id substTy subst $ repack (mkFunTys (arg1:rest) resTyp)
  125     _ -> resultType
  126 typeExpr' (AST.RecCon name _) 
  127   = do def <- lift $ maybe (return Nothing) GHC.lookupName (semanticsName (name ^. simpleName))
  128        case def of 
  129          Just (AConLike (RealDataCon con)) -> return $ dataConSig con ^. _4
  130          Just (AConLike (PatSynCon patSyn)) -> return $ patSynSig patSyn ^. _6
  131          _ -> resultType
  132 typeExpr' (AST.RecUpdate e _) = typeExpr' e
  133 typeExpr' (AST.Enum from _ _) = mkListTy <$> typeExpr' from
  134 typeExpr' (AST.ParArrayEnum from _ _) = mkPArrTy <$> typeExpr' from
  135 typeExpr' (AST.ListComp e _) = mkListTy <$> typeExpr' e
  136 typeExpr' (AST.ParArrayComp e _) = mkPArrTy <$> typeExpr' e
  137 -- typeExpr' (AST.UTypeSig _ t) = -- TODO: evaluate type
  138 typeExpr' Hole = resultType
  139 typeExpr' _ = resultType

  141 literalType :: Literal -> Ghc GHC.Type
  142 literalType e = do usupp <- liftIO $ mkSplitUniqSupply 'z'
  143                    evalStateT (literalType' e) (uniqsFromSupply usupp)
  144 
  145 literalType' :: Literal -> StateT [Unique] Ghc GHC.Type
  146 literalType' (StringLit {}) = return stringTy
  147 literalType' (CharLit {}) = return charTy
  148 literalType' (IntLit {}) = litType numClassName
  149 literalType' (FracLit {}) = litType fractionalClassName
  150 literalType' (PrimIntLit {}) = return $ intPrimTy
  151 literalType' (PrimWordLit {}) = return $ word32PrimTy
  152 literalType' (PrimFloatLit {}) = return $ floatX4PrimTy
  153 literalType' (PrimDoubleLit {}) = return $ doubleX2PrimTy
  154 literalType' (PrimCharLit {}) = return $ charPrimTy
  155 literalType' (PrimStringLit {}) = return $ addrPrimTy

  157 appTypeMatches :: [ClsInst] -> GHC.Type -> [GHC.Type] -> Maybe (TCvSubst, GHC.Type)
  158 appTypeMatches insts functionT argTs -- TODO: check instances
  159   = let (funT, repackFun) = splitType functionT
  160         (args, resT) = splitFunTys funT
  161         argTypes = fst $ unzip $ map splitType argTs
  162      in if length args >= length argTypes
  163           then case tcUnifyTys (\_ -> BindMe) (take (length argTypes) args) argTypes of
  164                  Just st -> let (t', check) = repackFun st insts (substTy st $ mkFunTys (drop (length argTypes) args) resT)
  165                              in if check then Just (st, t')
  166                                          else Nothing
  167                  Nothing -> Nothing
  168           else Nothing

  170 splitType' :: GHC.Type -> (GHC.Type, GHC.Type -> GHC.Type)
  171 splitType' t = case splitType t of (t', repack) -> (t', fst . repack emptyTCvSubst [])
  172 
  173 splitType :: GHC.Type -> (GHC.Type, TCvSubst -> [ClsInst] -> GHC.Type -> (GHC.Type, Bool))
  174 splitType typ =
  175   let (foralls, typ') = splitForAllTys typ
  176       (args, coreType) = splitFunTys typ'
  177       (implicitArgs, realArgs) = partition isPredTy args
  178       repack subst insts t = (mkInvForAllTys (filter (`elem` tvs) foralls) . mkFunTys keptConstraints $ t, check)
  179         where
  180           check = and $ map checkInst implicitArgs
  181           checkInst t =
  182             case splitTyConApp_maybe (substTy subst t) of
  183               Just (tc, args) -> case tyConClass_maybe tc of 
  184                                    Just cls -> case lookupInstEnv False instEnvs cls args of 
  185                                                  ([],[],[]) -> False
  186                                                  _          -> True
  187                                    Nothing -> True
  188               Nothing -> True
  189           instEnvs = InstEnvs (extendInstEnvList emptyInstEnv insts) emptyInstEnv emptyModuleSet
  190           keptConstraints = filter hasCommonTv implicitArgs
  191           tvs = tyCoVarsOfTypeWellScoped t
  192           hasCommonTv = not . null . intersect tvs . tyCoVarsOfTypeWellScoped
  193    in (mkFunTys realArgs coreType, repack)

  195 resultType :: StateT [Unique] Ghc GHC.Type
  196 resultType = do 
  197   name <- newName
  198   let tv = mkTyVar name (mkTyConTy starKindTyCon)
  199   return $ mkInvForAllTys [tv] (mkTyVarTy tv)
  200 
  201 litType :: GHC.Name -> StateT [Unique] Ghc GHC.Type
  202 litType constraint = do 
  203   Just (ATyCon numTyCon) <- lift $ lookupName constraint
  204   name <- newName
  205   let tv = mkTyVar name (mkTyConTy starKindTyCon)
  206   return $ mkInvForAllTys [tv] $ mkFunTy (mkTyConApp numTyCon [mkTyVarTy tv]) (mkTyVarTy tv)
  207 
  208 newName :: Monad m => StateT [Unique] m GHC.Name
  209 newName = do uniq <- gets head
  210              modify tail
  211              let occn  = mkOccName tvName "a"
  212              return (mkInternalName uniq occn noSrcSpan)
