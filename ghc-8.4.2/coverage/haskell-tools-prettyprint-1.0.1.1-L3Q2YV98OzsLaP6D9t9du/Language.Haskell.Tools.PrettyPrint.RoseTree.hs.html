    3 -- | A simpler representation of the original AST.
    4 module Language.Haskell.Tools.PrettyPrint.RoseTree where
    5 
    6 import Control.Exception (Exception(..), throw)
    7 import Control.Monad.State (Monad(..), execState, modify)
    8 import Language.Haskell.Tools.AST
    9 
   10 -- | A rose tree containing additional node information
   11 data RoseTree st = RoseTree { roseInfo :: RoseSourceInfo st
   12                             , roseChildren :: [RoseTree st]
   13                             }
   14 
   15 -- | Heterogenous representation of source information, for pretty printing
   16 data RoseSourceInfo st
   17   = RoseSpan (SpanInfo st)
   18   | RoseList (ListInfo st)
   19   | RoseOptional (OptionalInfo st)
   20 
   21 -- | Transforms the heterogeneous AST into a homogeneous representation for pretty printing
   22 toRoseTree :: (SourceInfoTraversal n) => n dom st -> RoseTree st
   23 toRoseTree = (\case (root:_):_ -> root; _ -> pprProblem "toRoseTree: the result has no root")
   24                 . tail . flip execState [[],[]] . toSrcRoseSt
   25   where toSrcRoseSt = sourceInfoTraverseUp (SourceInfoTrf (trf RoseSpan) (trf RoseList) (trf RoseOptional)) desc asc
   26 
   27         desc  = modify ([]:)
   28         asc   = modify tail
   29         trf wrap inf = do modify (\(y : x : xs) -> [] : (RoseTree (wrap inf) (reverse y) : x) : xs)
   30                           return inf
   31 
   32 data PrettyPrintProblem = PrettyPrintProblem String
   33   deriving Show
   34 
   35 instance Exception PrettyPrintProblem
   36 
   37 pprProblem :: String -> a
   38 pprProblem = throw . PrettyPrintProblem
