    6 -- | The range template is an intermediate annotation level, where the children nodes of the tree
    7 -- had been cut from the parent nodes, but the annotations still contain ranges instead of text.
    8 module Language.Haskell.Tools.PrettyPrint.Prepare.RangeTemplate where

   10 import Control.Exception (Exception, throw)

   12 import Data.Data (Data)

   16 instance SourceInfo RngTemplateStage where
   17   data SpanInfo RngTemplateStage = RangeTemplateNode { _rngTemplateNodeRange :: RealSrcSpan
   18                                                      , _rngTemplateNodeElems :: [RangeTemplateElem]
   19                                                      }
   20     deriving Data
   21   data ListInfo RngTemplateStage = RangeTemplateList { _rngTemplateListRange :: RealSrcSpan
   22                                                      , _rngTmpListBefore :: String -- ^ Text that should be put before the first element if the list becomes populated
   23                                                      , _rngTmpListAfter :: String -- ^ Text that should be put after the last element if the list becomes populated
   24                                                      , _rngTmpDefaultSeparator :: String -- ^ The default separator if the list were empty
   25                                                      , _rngTmpIndented :: Maybe [Bool] -- ^ False for elements that should be not aligned
   26                                                      , _rngTmpSeparators :: [RealSrcSpan] -- ^ The actual separators that were found in the source code
   27                                                      }
   28     deriving Data
   29   data OptionalInfo RngTemplateStage = RangeTemplateOpt { _rngTemplateOptRange :: RealSrcSpan
   30                                                         , _rngTmpOptBefore :: String -- ^ Text that should be put before the element if it appears
   31                                                         , _rngTmpOptAfter :: String -- ^ Text that should be put after the element if it appears
   32                                                         }
   33     deriving Data

   69 -- | An element of a range template for a singleton AST node.
   70 data RangeTemplateElem = RangeElem RealSrcSpan -- ^ A range for the source code of the element
   71                        | RangeChildElem        -- ^ The place for a child element
   72                        deriving Data
   73 
   74 getRangeElemSpan :: RangeTemplateElem -> Maybe RealSrcSpan
   75 getRangeElemSpan (RangeElem sp) = Just sp
   76 getRangeElemSpan _ = Nothing
   77 
   78 instance HasRange (SpanInfo RngTemplateStage) where
   79   getRange = RealSrcSpan . (^. rngTemplateNodeRange)
   80   setRange (RealSrcSpan sp) = rngTemplateNodeRange .= sp
   81   setRange _ = id
   82 
   83 instance HasRange (ListInfo RngTemplateStage) where
   84   getRange = RealSrcSpan . (^. rngTemplateListRange)
   85   setRange (RealSrcSpan sp) = rngTemplateListRange .= sp
   86   setRange _ = id
   87 
   88 instance HasRange (OptionalInfo RngTemplateStage) where
   89   getRange = RealSrcSpan . (^. rngTemplateOptRange)
   90   setRange (RealSrcSpan sp) = rngTemplateOptRange .= sp
   91   setRange _ = id
   92 
   93 instance Show (SpanInfo RngTemplateStage) where
   94   show rngNode = concatMap show $ rngNode ^. rngTemplateNodeElems
   95 instance Show (ListInfo RngTemplateStage) where
   96   show RangeTemplateList{..} = "<*" ++ shortShowSpan (RealSrcSpan _rngTemplateListRange) ++ " " ++ show _rngTmpListBefore ++ " " ++ show _rngTmpDefaultSeparator ++ " " ++ show _rngTmpListAfter ++ "*>"
   97 instance Show (OptionalInfo RngTemplateStage) where
   98   show RangeTemplateOpt{..} = "<?" ++ shortShowSpan (RealSrcSpan _rngTemplateOptRange) ++ " " ++ show _rngTmpOptBefore ++ " " ++ show _rngTmpOptAfter ++ "?>"
   99 
  100 instance Show RangeTemplateElem where
  101   show (RangeElem sp) = shortShowSpan (RealSrcSpan sp)
  102   show RangeChildElem = "<.>"
