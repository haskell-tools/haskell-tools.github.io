   23 getNormalComments :: Map SrcSpan [Located AnnotationComment] -> Map.Map SrcSpan [Located AnnotationComment]
   24 getNormalComments = Map.map (filter (not . isPragma . unLoc))
   25 
   26 getPragmaComments :: Map SrcSpan [Located AnnotationComment] -> Map.Map String [Located String]
   27 getPragmaComments comms = Map.fromListWith (++) $ map (\(L l (AnnBlockComment str)) -> (getPragmaCommand str, [L l str]))
   28                                                 $ filter (isPragma . unLoc) $ concatMap snd $ Map.toList comms
   29   where getPragmaCommand = takeWhile (\c -> isAlphaNum c || c == '_') . dropWhile isSpace . drop 3
   30 
   31 isPragma :: AnnotationComment -> Bool
   32 isPragma (AnnBlockComment str) = take 3 str == "{-#" && take 3 (reverse str) == "}-#"
   33 isPragma _ = False

   35 -- | Puts comments in the nodes they should be attached to. Watches for lexical tokens
   36 -- that may divide the comment and the supposed element.
   37 -- Leaves the AST in a state where parent nodes does not contain all of their children.
   38 placeComments :: RangeInfo stage => Map ApiAnnKey [SrcSpan] -> Map.Map SrcSpan [Located AnnotationComment]
   39               -> Ann UModule dom stage -> Ann UModule dom stage
   40 placeComments tokens comms mod
   41   = resizeAnnots (Set.filter (\rng -> srcSpanStart rng /= srcSpanEnd rng) $ Set.fromList $ concat (Map.elems tokens))
   42       (concatMap (map nextSrcLoc . snd) (Map.toList cleanedComments)) mod
   43   where spans = allElemSpans mod
   44         sortedElemStarts = Set.fromList $ map srcSpanStart spans
   45         sortedElemEnds = Set.fromList $ map srcSpanEnd spans
   46         nextSrcLoc comm@(L sp _)
   47           = let after = fromMaybe noSrcLoc (Set.lookupLE (srcSpanStart sp) sortedElemEnds)
   48                 before = fromMaybe noSrcLoc (Set.lookupGE (srcSpanEnd sp) sortedElemStarts)
   49              in ((after,before),comm)
   50         cleanedComments = Map.map (map cleanComment) comms
   51         cleanComment (L loc (AnnLineComment txt))
   52           | last txt `elem` "\n\r" = L (mkSrcSpan (srcSpanStart loc) (decreaseCol (srcSpanEnd loc))) (AnnLineComment (init txt))
   53         cleanComment c = c
   54         decreaseCol (RealSrcLoc l) = mkSrcLoc (srcLocFile l) (srcLocLine l) (srcLocCol l - 1)
   55         decreaseCol l = l
   56 
   57 allElemSpans :: (SourceInfoTraversal node, RangeInfo stage) => Ann node dom stage -> [SrcSpan]
   58 allElemSpans = execWriter . sourceInfoTraverse (SourceInfoTrf (\ni -> tell [ni ^. nodeSpan] >> pure ni) pure pure)
   59 
   60 resizeAnnots :: RangeInfo stage => Set SrcSpan -> [((SrcLoc, SrcLoc), Located AnnotationComment)]
   61               -> Ann UModule dom stage
   62               -> Ann UModule dom stage
   63 resizeAnnots tokens comments elem
   64   = flip evalState comments $ flip runReaderT tokens $
   65         -- if a comment that could be attached to more than one documentable element (possibly nested)
   66         -- the order of different documentable elements here decide which will be chosen
   67 
   68         modImports&annList !~ expandAnnot -- expand imports to cover their comments
   69           >=> modDecl&annList !~ expandTopLevelDecl -- expand declarations to cover their comments
   70           >=> expandAnnot -- expand the module itself to cover its comments
   71       $ elem
   72 
   73 type ExpandType elem dom stage = Ann elem dom stage -> ReaderT (Set SrcSpan) (State [((SrcLoc, SrcLoc), Located AnnotationComment)]) (Ann elem dom stage)
   74 
   75 expandTopLevelDecl :: RangeInfo stage => ExpandType UDecl dom stage
   76 expandTopLevelDecl
   77   = declBody & annJust & cbElements & annList !~ expandClsElement
   78       >=> declCons & annList !~ expandConDecl
   79       >=> declGadt & annList !~ expandGadtConDecl
   80       >=> declTypeSig !~ expandTypeSig
   81       >=> expandAnnot
   82 
   83 expandTypeSig :: RangeInfo stage => ExpandType UTypeSignature dom stage
   84 expandTypeSig
   85   = tsType & typeParams !~ expandAnnot >=> expandAnnot
   86 
   87 expandClsElement :: RangeInfo stage => ExpandType UClassElement dom stage
   88 expandClsElement
   89   = ceTypeSig !~ expandTypeSig
   90       >=> ceBind !~ expandValueBind
   91       >=> expandAnnot
   92 
   93 expandValueBind :: RangeInfo stage => ExpandType UValueBind dom stage
   94 expandValueBind
   95   = valBindLocals & annJust & localBinds & annList !~ expandLocalBind
   96       >=> funBindMatches & annList & matchBinds & annJust & localBinds & annList !~ expandLocalBind
   97       >=> expandAnnot
   98 
   99 expandLocalBind :: RangeInfo stage => ExpandType ULocalBind dom stage
  100 expandLocalBind
  101   = localVal !~ expandValueBind
  102       >=> localSig !~ expandTypeSig
  103       >=> expandAnnot
  104 
  105 expandConDecl :: RangeInfo stage => ExpandType UConDecl dom stage
  106 expandConDecl
  107   = conDeclFields & annList !~ expandAnnot >=> expandAnnot
  108 
  109 expandGadtConDecl :: RangeInfo stage => ExpandType UGadtConDecl dom stage
  110 expandGadtConDecl
  111   = gadtConType & gadtConRecordFields & annList !~ expandAnnot >=> expandAnnot

  113 -- | Expands tree elements to contain the comments that should be attached to them.
  114 expandAnnot :: forall elem dom stage . RangeInfo stage => ExpandType elem dom stage
  115 expandAnnot elem
  116   = do let Just sp = elem ^? annotation&sourceInfo&nodeSpan
  117        tokens <- ask
  118        applicable <- lift $ gets (applicableComments tokens (srcSpanStart sp) (srcSpanEnd sp))
  119 
  120        -- this check is just for performance (quick return if no modification is needed)
  121        if not (null applicable) then do
  122          -- the new span is the original plus all the covered spans
  123          let newSp@(RealSrcSpan newSpan)
  124                = foldl combineSrcSpans (fromJust $ elem ^? nodeSp) (map (getLoc . snd) applicable)
  125          -- take out all comments that are now covered
  126          lift $ modify (filter (not . (\case RealSrcSpan s -> newSpan `containsSpan` s; _ -> True) . getLoc . snd))
  127          return $ nodeSp .= newSp $ elem
  128        else return elem
  129   where nodeSp :: Simple Partial (Ann elem dom stage) SrcSpan
  130         nodeSp = annotation&sourceInfo&nodeSpan
  131 
  132 -- This classification does not prefer inline comments to previous line comments, this is implicitly done
  133 -- by the order in which the elements are traversed.
  134 applicableComments :: Set SrcSpan -> SrcLoc -> SrcLoc
  135                    -> [((SrcLoc, SrcLoc), Located AnnotationComment)]
  136                    -> [((SrcLoc, SrcLoc), Located AnnotationComment)]
  137 applicableComments tokens start end = filter applicableComment
  138   where -- A comment that starts with | binds to the next documented element
  139         applicableComment ((_, before), L sp comm)
  140           | isCommentOnNext comm = before == start && noTokenBetween (srcSpanEnd sp) start
  141         -- A comment that starts with ^ binds to the previous documented element
  142         applicableComment ((after, _), L sp comm)
  143           | isCommentOnPrev comm = after == end && noTokenBetween end (srcSpanStart sp)
  144         -- All other comment binds to the previous definition if it is on the same line
  145         applicableComment ((after, _), L sp@(RealSrcSpan loc) _)
  146           | after == end && srcLocLine (realSrcSpanStart loc) == getLineLocDefault end = True
  147                          && noTokenBetween end (srcSpanStart sp)
  148         -- or the next one if that is on the next line and the columns line up
  149         applicableComment ((_, before), L sp@(RealSrcSpan loc) _)
  150           | before == start && srcLocLine (realSrcSpanEnd loc) + 1 == getLineLocDefault start
  151                             && srcLocCol (realSrcSpanStart loc) == getLineColDefault start
  152                             && noTokenBetween (srcSpanEnd sp) start
  153           = True
  154         applicableComment _ = False
  155 
  156         getLineLocDefault (RealSrcLoc l) = srcLocLine l
  157         getLineLocDefault _              = -1
  158 
  159         getLineColDefault (RealSrcLoc l) = srcLocCol l
  160         getLineColDefault _              = -1
  161 
  162         noTokenBetween start end
  163           = case Set.lookupGE (srcLocSpan start) tokens of
  164               Just tok -> srcSpanStart tok >= end
  165               Nothing -> True

  167 -- * GHC mistakenly parses -- ^ and -- | comments as simple line comments.
  168 -- These functions check if a given comment is attached to the previous or next comment.
  169 
  170 -- | Checks if a doc comment belongs to the next definition.
  171 isCommentOnNext :: AnnotationComment -> Bool
  172 isCommentOnNext (AnnDocCommentNext _) = True
  173 isCommentOnNext (AnnLineComment s) = firstNonspaceCharIs '|' s
  174 isCommentOnNext (AnnBlockComment s) = firstNonspaceCharIs '|' s
  175 isCommentOnNext _ = False
  176 
  177 -- | Checks if a doc comment belongs to the previous definition.
  178 isCommentOnPrev :: AnnotationComment -> Bool
  179 isCommentOnPrev (AnnDocCommentPrev _) = True
  180 isCommentOnPrev (AnnLineComment s) = firstNonspaceCharIs '^' s
  181 isCommentOnPrev (AnnBlockComment s) = firstNonspaceCharIs '^' s
  182 isCommentOnPrev _ = False
  183 
  184 -- the comment string contains the -- or {- characters
  185 firstNonspaceCharIs :: Char -> String -> Bool
  186 firstNonspaceCharIs c s = Just c == listToMaybe (dropWhile isSpace (drop 2 s))
