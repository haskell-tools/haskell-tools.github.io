   23 trfQuasiQuotation' :: forall n r . TransformName n r => HsSplice n -> Trf (AST.UQuasiQuote (Dom r) RangeStage)
   24  -- the lexer does not provide us with tokens '[', '|' and '|]'
   25 trfQuasiQuotation' (HsQuasiQuote _ id l str)
   26   = AST.UQuasiQuote <$> annLocNoSema quoterLoc (trfName' @n id)
   27                     <*> annLocNoSema (pure strLoc) (pure $ AST.QQString (unpackFS str))
   28   where -- assume that there are no white spaces ain the head and the end of the quasi quote
   29         quoterLoc = do rng <- asks contRange
   30                        return $ mkSrcSpan (updateCol (+1) (srcSpanStart rng)) (updateCol (subtract 1) (srcSpanStart l))
   31         strLoc = mkSrcSpan (srcSpanStart l) (updateCol (subtract 2) (srcSpanEnd l))
   32 trfQuasiQuotation' qq = unhandledElement "quasi quotation" qq

   34 trfSplice :: TransformName n r => HsSplice n -> Trf (Ann AST.USplice (Dom r) RangeStage)
   35 trfSplice spls = do rng <- asks contRange
   36                     annLocNoSema (pure $ getSpliceLoc spls `mappend` rng) (trfSplice' spls)
   37 
   38 getSpliceLoc :: HsSplice a -> SrcSpan
   39 getSpliceLoc (HsTypedSplice _ _ e) = getLoc e
   40 getSpliceLoc (HsUntypedSplice _ _ e) = getLoc e
   41 getSpliceLoc (HsQuasiQuote _ _ sp _) = sp
   42 getSpliceLoc (HsSpliced _ _) = noSrcSpan
   43 
   44 trfSplice' :: TransformName n r => HsSplice n -> Trf (AST.USplice (Dom r) RangeStage)
   45 trfSplice' (HsTypedSplice _ _ expr) = trfSpliceExpr expr
   46 trfSplice' (HsUntypedSplice _ _ expr) = trfSpliceExpr expr
   47 trfSplice' s = unhandledElement "splice" s
   48 
   49 -- | TODO: easier with splice decoration
   50 trfSpliceExpr :: forall n r . TransformName n r => Located (HsExpr n) -> Trf (AST.USplice (Dom r) RangeStage)
   51 trfSpliceExpr expr =
   52   do hasDollar <- allTokenLoc AnnThIdSplice
   53      hasDoubleDollar <- allTokenLoc AnnThIdTySplice
   54      let newSp = case (hasDollar, hasDoubleDollar) of
   55                    ([], []) -> getLoc expr
   56                    (_, []) -> updateStart (updateCol (+1)) (getLoc expr)
   57                    ([], _) -> updateStart (updateCol (+2)) (getLoc expr)
   58      case expr of L _ (HsVar (L _ varName)) -> AST.UIdSplice <$> trfName @n (L newSp varName)
   59                   L _ (HsRecFld fldName) -> AST.UIdSplice <$> trfAmbiguousFieldName' newSp fldName
   60                   expr -> AST.UParenSplice <$> trfExpr expr
   61 
   62 trfBracket' :: forall n r . TransformName n r => HsBracket n -> Trf (AST.UBracket (Dom r) RangeStage)
   63 trfBracket' (ExpBr expr) = AST.UExprBracket <$> trfExpr expr
   64 trfBracket' (TExpBr expr) = AST.UExprBracket <$> trfExpr expr
   65 trfBracket' (VarBr isSingle expr)
   66   = AST.UExprBracket <$> annLoc createScopeInfo (updateStart (updateCol (if isSingle then (+1) else (+2))) <$> asks contRange)
   67       (AST.UVar <$> (annContNoSema (trfName' @n expr)))
   68 trfBracket' (PatBr pat) = AST.UPatternBracket <$> trfPattern pat
   69 trfBracket' (DecBrL decls) = AST.UDeclsBracket <$> trfDecls decls
   70 trfBracket' (DecBrG decls) = AST.UDeclsBracket <$> trfDeclsGroup decls
   71 trfBracket' (TypBr typ) = AST.UTypeBracket <$> trfType typ
