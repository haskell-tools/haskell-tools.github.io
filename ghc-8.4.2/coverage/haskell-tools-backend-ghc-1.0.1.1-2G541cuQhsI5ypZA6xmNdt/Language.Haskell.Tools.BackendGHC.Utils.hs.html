   45 createModuleInfo :: ModSummary -> SrcSpan -> [LImportDecl n] -> Trf (Sema.ModuleInfo GhcRn)
   46 createModuleInfo mod nameLoc (filter (not . ideclImplicit . unLoc) -> imports) = do
   47   let prelude = (xopt ImplicitPrelude $ ms_hspp_opts mod)
   48                   && all (\idecl -> ("Prelude" /= (GHC.moduleNameString $ unLoc $ ideclName $ unLoc idecl))
   49                                       || nameLoc == getLoc idecl) imports
   50   (_, preludeImports) <- if prelude then getImportedNames "Prelude" Nothing else return (ms_mod mod, [])
   51   deps <- if prelude then lift $ getDeps (Module baseUnitId (GHC.mkModuleName "Prelude"))
   52                      else return []
   53   -- This function (via getInstances) refers the ghc environment,
   54   -- we must evaluate the result or the reference may be kept preventing garbage collection.
   55   return $ mkModuleInfo (ms_mod mod) (ms_hspp_opts mod) (case ms_hsc_src mod of HsSrcFile -> False; _ -> True)
   56                         (forceElements preludeImports) deps

   84 -- | Adds semantic information to an impord declaration. See ImportInfo.
   85 createImportData :: forall r n . (GHCName r, HsHasName (IdP n)) => GHC.ImportDecl n -> Trf (ImportInfo r)
   86 createImportData (GHC.ImportDecl _ name pkg _ _ _ _ _ declHiding) =
   87   do (mod,importedNames) <- getImportedNames (GHC.moduleNameString $ unLoc name) (fmap (unpackFS . sl_fs) pkg)
   88      names <- liftGhc $ filterM (checkImportVisible declHiding . (^. pName)) importedNames
   89      -- TODO: only use getFromNameUsing once
   90      lookedUpNames <- liftGhc $ mapM translatePName $ names
   91      lookedUpImported <- liftGhc $ mapM ((getFromNameUsing @r) getTopLevelId . (^. pName)) $ importedNames
   92      deps <- lift $ getDeps mod
   93      -- This function (via getInstances) refers the ghc environment,
   94      -- we must evaluate the result or the reference may be kept preventing garbage collection.
   95      return $ mkImportInfo mod (forceElements $ catMaybes lookedUpImported)
   96                                (forceElements $ catMaybes lookedUpNames)
   97                                deps
   98   where translatePName :: PName GhcRn -> Ghc (Maybe (PName r))
   99         translatePName (PName n p) = do n' <- (getFromNameUsing @r) getTopLevelId n
  100                                         p' <- maybe (return Nothing) ((getFromNameUsing @r) getTopLevelId) p
  101                                         return (PName <$> n' <*> Just p')

  103 getDeps :: Module -> Ghc [Module]
  104 getDeps mod = do
  105   env <- GHC.getSession
  106   eps <- liftIO $ hscEPS env
  107   case lookupIfaceByModule (hsc_dflags env) (hsc_HPT env) (eps_PIT eps) mod of
  108     Just ifc -> (mod :) <$> mapM (liftIO . getModule env . fst) (dep_mods (mi_deps ifc))
  109     Nothing -> return [mod]
  110   where getModule env modName = do
  111           res <- findHomeModule env modName
  112           case res of Found _ m -> return m
  113                       _ -> case lookupPluginModuleWithSuggestions (hsc_dflags env) modName Nothing of
  114                              LookupFound m _ -> return m
  115                              LookupHidden hiddenPack hiddenMod -> return (head $ map fst hiddenMod ++ map fst hiddenPack)
  116                              _ -> error $ "getDeps: module not found: " ++ GHC.moduleNameString modName

  118 -- | Get names that are imported from a given import
  119 getImportedNames :: String -> Maybe String -> Trf (GHC.Module, [PName GhcRn])
  120 getImportedNames name pkg = liftGhc $ do
  121   hpt <- hsc_HPT <$> getSession
  122   eps <- getSession >>= liftIO . readIORef . hsc_EPS
  123   mod <- findModule (mkModuleName name) (fmap mkFastString pkg)
  124   -- load exported names from interface file
  125   let ifaceNames = maybe [] mi_exports $ flip lookupModuleEnv mod
  126                                        $ eps_PIT eps
  127   let homeExports = maybe [] (md_exports . hm_details) (lookupHptByModule hpt mod)
  128   -- TODO: Why selectors are added in one case and not added in the other?
  129   return (mod, concatMap (availToPName availNames) ifaceNames ++ concatMap (availToPName availNamesWithSelectors) homeExports)
  130     where availToPName f a = map (\n -> if n == availName a then PName n Nothing else PName n (Just (availName a))) (f a)

  132 -- | Check is a given name is imported from an import with given import specification.
  133 checkImportVisible :: (HsHasName (IdP name), GhcMonad m)
  134                    => Maybe (Bool, Located [LIE name]) -> GHC.Name -> m Bool
  135 checkImportVisible (Just (isHiding, specs)) name
  136   | isHiding  = not . or @[] <$> mapM (`ieSpecMatches` name) (map unLoc (unLoc specs))
  137   | otherwise = or @[] <$> mapM (`ieSpecMatches` name) (map unLoc (unLoc specs))
  138 checkImportVisible _ _ = return True

  140 -- | Forces strict evaluation of a list. Forces elements into WHNF.
  141 forceElements :: [a] -> [a]
  142 forceElements [] = []
  143 forceElements (a : ls) = let res = forceElements ls
  144                           in a `seq` res `seq` (a : ls)
  145 
  146 ieSpecMatches :: (HsHasName (IdP name), GhcMonad m) => IE name -> GHC.Name -> m Bool
  147 ieSpecMatches (concatMap hsGetNames' . HsSyn.ieNames -> ls) name
  148   | name `elem` ls = return True
  149 -- ieNames does not consider field names
  150 ieSpecMatches (IEThingWith thing _ with flds) name
  151   | name `elem` concatMap hsGetNames' (map (ieWrappedName . unLoc) (thing : with) ++ map (flSelector . unLoc) flds)
  152   = return True
  153 ieSpecMatches ie@(IEThingAll _) name | [n] <- hsGetNames' (HsSyn.ieName ie), isTyConName n
  154   = do entity <- lookupName n
  155        return $ case entity of Just (ATyCon tc)
  156                                  | Just cls <- tyConClass_maybe tc
  157                                      -> name `elem` map getName (classMethods cls)
  158                                  | otherwise -> name `elem` concatMap (\dc -> getName dc : map flSelector (dataConFieldLabels dc))
  159                                                                       (tyConDataCons tc)
  160                                _             -> False
  161 ieSpecMatches _ _ = return False

  210 -- | Get the elements where there is no ; before
  211 elementsWithoutSemi :: [Ann e (Dom n) RangeStage] -> Trf [Bool]
  212 elementsWithoutSemi [] = return []
  213 elementsWithoutSemi (fst:rest) = indentedElements' (srcSpanEnd $ getRange fst) rest
  214   where indentedElements' lastEnd (elem:rest)
  215           = let sepRange = mkSrcSpan lastEnd (srcSpanStart $ getRange elem)
  216              in (:) <$> (not . (\l -> isGoodSrcSpan l && srcSpanStart l < srcSpanEnd l) <$> focusOn sepRange (tokenLoc AnnSemi))
  217                     <*> indentedElements' (srcSpanEnd $ getRange elem) rest
  218         indentedElements' _ [] = return []

  293 -- | Focuses the transformation to go between tokens. The tokens must be found inside the current range.
  294 between :: AnnKeywordId -> AnnKeywordId -> Trf a -> Trf a
  295 between firstTok lastTok = focusAfter firstTok . focusBefore lastTok

  297 -- | Focuses the transformation to go between tokens if they are present
  298 betweenIfPresent :: AnnKeywordId -> AnnKeywordId -> Trf a -> Trf a
  299 betweenIfPresent firstTok lastTok = focusAfterIfPresent firstTok . focusBeforeIfPresent lastTok

  301 -- | Focuses the transformation to be performed after the given token. The token must be found inside the current range.
  302 focusAfter :: AnnKeywordId -> Trf a -> Trf a
  303 focusAfter firstTok trf
  304   = do firstToken <- tokenLoc firstTok
  305        if (isGoodSrcSpan firstToken)
  306           then local (\s -> s { contRange = mkSrcSpan (srcSpanEnd firstToken) (srcSpanEnd (contRange s))}) trf
  307           else do rng <- asks contRange
  308                   convertionProblem $ "focusAfter: token not found in " ++ show rng ++ ": " ++ show firstTok
  309 
  310 focusAfterIfPresent :: AnnKeywordId -> Trf a -> Trf a
  311 focusAfterIfPresent firstTok trf
  312   = do firstToken <- tokenLoc firstTok
  313        if (isGoodSrcSpan firstToken)
  314           then local (\s -> s { contRange = mkSrcSpan (srcSpanEnd firstToken) (srcSpanEnd (contRange s))}) trf
  315           else trf
  316 
  317 -- | Focuses the transformation to be performed before the given token. The token must be found inside the current range.
  318 focusBefore :: AnnKeywordId -> Trf a -> Trf a
  319 focusBefore lastTok trf
  320   = do lastToken <- tokenLocBack lastTok
  321        if (isGoodSrcSpan lastToken)
  322           then local (\s -> s { contRange = mkSrcSpan (srcSpanStart (contRange s)) (srcSpanStart lastToken)}) trf
  323           else do rng <- asks contRange
  324                   convertionProblem $ "focusBefore: token not found in " ++ show rng ++ ": " ++ show lastTok
  325 
  326 focusBeforeIfPresent :: AnnKeywordId -> Trf a -> Trf a
  327 focusBeforeIfPresent lastTok trf
  328   = do lastToken <- tokenLocBack lastTok
  329        if (isGoodSrcSpan lastToken)
  330           then local (\s -> s { contRange = mkSrcSpan (srcSpanStart (contRange s)) (srcSpanStart lastToken)}) trf
  331           else trf

  383 -- | Searches for tokens in the given order inside the parent element and returns their location
  384 eachTokenLoc :: [AnnKeywordId] -> Trf [SrcSpan]
  385 eachTokenLoc (keyw:rest)
  386   = do spanFirst <- tokenLoc keyw
  387        spanRest <- focusAfterLoc (srcSpanEnd spanFirst) (eachTokenLoc rest)
  388        return (spanFirst : spanRest)
  389 eachTokenLoc [] = pure []

  419 updateCol f (RealSrcLoc loc) = mkSrcLoc (srcLocFile loc) (srcLocLine loc) (f $ srcLocCol loc)

  441 -- | Only keeps one of the elements that are on the same source location
  442 removeDuplicates :: [Located e] -> [Located e]

  450 -- | Transform a list of definitions where the defined names are in scope for subsequent definitions
  451 trfScopedSequence :: HsHasName d => (d -> Trf e) -> [d] -> Trf [e]
  452 trfScopedSequence f (def:rest) = (:) <$> f def <*> addToScope def (trfScopedSequence f rest)

  455 -- | Splits a given string at whitespaces while calculating the source location of the fragments
  456 splitLocated :: Located String -> [Located String]
  457 splitLocated = splitLocatedOn isSpace
  458 
  459 -- | Splits a given string while calculating the source location of the fragments
  460 splitLocatedOn :: (Char -> Bool) -> Located String -> [Located String]
  461 splitLocatedOn pred (L (RealSrcSpan l) str) = splitLocated' str (realSrcSpanStart l) Nothing
  462   where splitLocated' :: String -> RealSrcLoc -> Maybe (RealSrcLoc, String) -> [Located String]
  463         splitLocated' (c:rest) currLoc (Just (startLoc, str)) | pred c
  464           = L (RealSrcSpan $ mkRealSrcSpan startLoc currLoc) (reverse str) : splitLocated' rest (advanceSrcLoc currLoc c) Nothing
  465         splitLocated' (c:rest) currLoc Nothing | pred c = splitLocated' rest (advanceSrcLoc currLoc c) Nothing
  466         splitLocated' (c:rest) currLoc (Just (startLoc, str)) = splitLocated' rest (advanceSrcLoc currLoc c) (Just (startLoc, c:str))
  467         splitLocated' (c:rest) currLoc Nothing = splitLocated' rest (advanceSrcLoc currLoc c) (Just (currLoc, [c]))
  468         splitLocated' [] currLoc (Just (startLoc, str)) = [L (RealSrcSpan $ mkRealSrcSpan startLoc currLoc) (reverse str)]
  469         splitLocated' [] _ Nothing = []
  470 splitLocatedOn _ _ = convProblem "splitLocated: unhelpful span given"

  472 compareSpans :: SrcSpan -> SrcSpan -> Ordering
  473 compareSpans (RealSrcSpan a) (RealSrcSpan b)
  474   | a `containsSpan` b = GT
  475   | b `containsSpan` a = LT
  476 compareSpans _ _ = EQ
