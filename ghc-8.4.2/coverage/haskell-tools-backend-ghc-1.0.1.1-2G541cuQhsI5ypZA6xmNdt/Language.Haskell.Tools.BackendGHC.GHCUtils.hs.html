   22 class OutputableBndrId name => GHCName name where

   27   unpackPostRn :: RdrName -> PostRn name (IdP name) -> IdP name
   28   gunpackPostRn :: a -> (IdP name -> a) -> PostRn name (IdP name) -> a

   30 instance GHCName GhcPs where
   31   rdrName = id
   32   getFromNameUsing _ n = return $ pure (nameRdrName n)
   33   getBindsAndSigs (ValBindsIn binds sigs) = (sigs, binds)
   34   getBindsAndSigs _ = error "ValBindsOut: ValBindsOut in parsed source"
   35   nameFromId = nameRdrName . getName
   36   unpackPostRn rdr _ = rdr
   37   gunpackPostRn a _ _ = a

   58 -- | Loading ids for top-level ghc names
   59 getTopLevelId :: GHC.Name -> Ghc (Maybe GHC.Id)
   60 getTopLevelId name =
   61     lookupName name >>= \case
   62       Just (AnId id) -> return (Just id)
   63       Just (AConLike (RealDataCon dc)) -> return $ Just $ mkVanillaGlobal name (dataConUserType dc)
   64       Just (AConLike (PatSynCon ps)) -> return $ Just $ mkVanillaGlobal name (createPatSynType ps)
   65       Just (ATyCon tc) -> return $ Just $ mkVanillaGlobal name (tyConKind tc)
   66       _ -> return Nothing
   67   where createPatSynType patSyn = case patSynSig patSyn of (_, _, _, _, args, res) -> mkFunTys args res

  228 -- | Tries to simplify the type that has HsAppsTy before renaming. Does not always provide the correct form.
  229 -- Treats each operator as if they are of equivalent precedence and always left-associative.
  230 cleanHsType :: forall n . (OutputableBndrId n, SourceTextX n) => HsType n -> HsType n
  231 -- for some reason * is considered infix
  232 cleanHsType (HsAppsTy apps) = unLoc $ guessType apps
  233   where guessType :: OutputableBndrId n => [LHsAppType n] -> LHsType n
  234         guessType (L l (HsAppInfix n) : rest) -- must be a prefix actually
  235           = guessType' (L l (HsTyVar NotPromoted n)) rest
  236         guessType (L _ (HsAppPrefix t) : rest) = guessType' t rest
  237         guessType [] = error $ "guessType: empty" ++ showSDocUnsafe (ppr apps)
  238 
  239         guessType' :: LHsType n -> [LHsAppType n] -> LHsType n
  240         guessType' fun (L _ (HsAppPrefix t) : rest) = guessType' (hsAppTy fun t) rest
  241         guessType' fun (L l (HsAppInfix n) : rest)
  242           -- TODO: find a better check
  243           | showSDocUnsafe (ppr n) == "*" = guessType' (hsAppTy fun (L l (HsTyVar NotPromoted n))) rest
  244         guessType' left (L _ (HsAppInfix n) : right) = hsOpTy left n (guessType right)
  245         guessType' t [] = t
  246 
  247         hsAppTy :: LHsType n -> LHsType n -> LHsType n
  248         hsAppTy t1 t2 = L (getLoc t1 `combineSrcSpans` getLoc t2) $ HsAppTy t1 t2
  249 
  250         hsOpTy :: LHsType n -> Located (IdP n) -> LHsType n -> LHsType n
  251         hsOpTy t1 n t2 = L (getLoc t1 `combineSrcSpans` getLoc t2) $ HsOpTy t1 n t2
  252 cleanHsType t = t

  254 mergeFixityDefs :: [Located (FixitySig n)] -> [Located (FixitySig n)]
  255 mergeFixityDefs (s@(L l _) : rest)
  256   = let (same, different) = partition ((== l) . getLoc) rest
  257      in foldl mergeWith s (map unLoc same) : mergeFixityDefs different
  258   where mergeWith (L l (FixitySig names fixity)) (FixitySig otherNames _) = L l (FixitySig (names ++ otherNames) fixity)
  259 mergeFixityDefs [] = []

  261 getGroupRange :: HsGroup n -> SrcSpan
  262 getGroupRange (HsGroup {..})
  263   = foldr combineSrcSpans noSrcSpan locs
  264   where locs = [getHsValRange hs_valds] ++ map getLoc hs_splcds ++ map getLoc (concatMap group_tyclds hs_tyclds) ++ map getLoc (concatMap group_roles hs_tyclds)
  265                  ++ map getLoc hs_derivds ++ map getLoc hs_fixds ++ map getLoc hs_defds
  266                  ++ map getLoc hs_fords ++ map getLoc hs_warnds ++ map getLoc hs_annds ++ map getLoc hs_ruleds ++ map getLoc hs_vects
  267                  ++ map getLoc hs_docs

  269 getHsValRange :: HsValBinds n -> SrcSpan
  270 getHsValRange (ValBindsIn vals sig) = foldr combineSrcSpans noSrcSpan $ map getLoc (bagToList vals) ++ map getLoc sig
  271 getHsValRange (ValBindsOut vals sig) = foldr combineSrcSpans noSrcSpan $ concatMap (map getLoc . bagToList . snd) vals ++ map getLoc sig
