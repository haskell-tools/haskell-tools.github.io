   29 trfPattern :: forall n r . TransformName n r => Located (Pat n) -> Trf (Ann AST.UPattern (Dom r) RangeStage)
   30 -- field wildcards are not directly represented in GHC AST
   31 trfPattern (L l (ConPatIn name (RecCon (HsRecFields flds _)))) | any ((l ==) . getLoc) flds
   32   = focusOn l $ do
   33       let (fromWC, notWC) = partition ((l ==) . getLoc) flds
   34       normalFields <- mapM (trfLocNoSema trfPatternField') notWC
   35       wildc <- annLocNoSema (tokenLocBack AnnDotdot) (AST.UFieldWildcardPattern <$> annCont (createImplicitFldInfo (unLoc . (\(VarPat n) -> n) . unLoc) (map unLoc fromWC)) (pure AST.FldWildcard))
   36       annLocNoSema (pure l) (AST.URecPat <$> trfName @n name <*> makeNonemptyList ", " (pure (normalFields ++ [wildc])))
   37 trfPattern p = trfLocNoSema trfPattern' (correctPatternLoc p)
   38 
   39 -- | Locations for right-associative infix patterns are incorrect in GHC AST
   40 correctPatternLoc :: Located (Pat n) -> Located (Pat n)
   41 correctPatternLoc (L _ p@(ConPatIn _ (InfixCon left right)))
   42   = L (getLoc (correctPatternLoc left) `combineSrcSpans` getLoc (correctPatternLoc right)) p
   43 correctPatternLoc p = p

   45 trfPattern' :: forall n r . TransformName n r => Pat n -> Trf (AST.UPattern (Dom r) RangeStage)
   46 trfPattern' (WildPat _) = pure AST.UWildPat
   47 trfPattern' (VarPat name) = define $ AST.UVarPat <$> trfName @n name
   48 trfPattern' (LazyPat pat) = AST.UIrrefutablePat <$> trfPattern pat
   49 trfPattern' (AsPat name pat) = AST.UAsPat <$> define (trfName @n name) <*> trfPattern pat
   50 trfPattern' (ParPat pat) = AST.UParenPat <$> trfPattern pat
   51 trfPattern' (BangPat pat) = AST.UBangPat <$> trfPattern pat
   52 trfPattern' (ListPat pats _ _) = AST.UListPat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   53 trfPattern' (TuplePat pats Boxed _) = AST.UTuplePat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   54 trfPattern' (TuplePat pats Unboxed _) = AST.UUnboxTuplePat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   55 trfPattern' (PArrPat pats _) = AST.UParArrPat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   56 trfPattern' (ConPatIn name (PrefixCon args)) = AST.UAppPat <$> trfName @n name <*> makeList " " atTheEnd (mapM trfPattern args)
   57 trfPattern' (ConPatIn name (RecCon (HsRecFields flds _))) = AST.URecPat <$> trfName @n name <*> trfAnnList ", " trfPatternField' flds
   58 trfPattern' (ConPatIn name (InfixCon left right)) = AST.UInfixAppPat <$> trfPattern left <*> trfOperator @n name <*> trfPattern right
   59 trfPattern' (ViewPat expr pat _) = AST.UViewPat <$> trfExpr expr <*> trfPattern pat
   60 trfPattern' (SplicePat qq@(HsQuasiQuote {})) = AST.UQuasiQuotePat <$> annContNoSema (trfQuasiQuotation' qq)
   61 trfPattern' (SplicePat splice) = AST.USplicePat <$> trfSplice splice
   62 trfPattern' (LitPat lit) = AST.ULitPat <$> annCont (pure $ RealLiteralInfo (monoLiteralType lit)) (trfLiteral' lit)
   63 trfPattern' (SigPatIn pat (hsib_body . hswc_body -> typ)) = AST.UTypeSigPat <$> trfPattern pat <*> trfType typ
   64 trfPattern' (NPat (ol_val . unLoc -> lit) _ _ _) = AST.ULitPat <$> annCont (asks contRange >>= pure . PreLiteralInfo) (trfOverloadedLit lit)
   65 trfPattern' (NPlusKPat id (L l lit) _ _ _ _) = AST.UNPlusKPat <$> define (trfName @n id) <*> annLoc (asks contRange >>= pure . PreLiteralInfo) (pure l) (trfOverloadedLit (ol_val lit))
   66 trfPattern' (CoPat _ pat _) = trfPattern' pat -- coercion pattern introduced by GHC
   67 trfPattern' (SumPat pat tag arity _)
   68   = do sepsBefore <- focusBeforeLoc (srcSpanStart (getLoc pat)) (eachTokenLoc (AnnOpen : replicate (tag - 1) AnnVbar))
   69        sepsAfter <- focusAfterLoc (srcSpanEnd (getLoc pat)) (eachTokenLoc (replicate (arity - tag) AnnVbar))
   70        let locsBefore = map srcSpanEnd $ init sepsBefore
   71            locsAfter = map srcSpanEnd sepsAfter
   72        AST.UUnboxedSumPat <$> makeList " | " (after AnnOpen) (mapM makePlaceholder locsBefore)
   73                           <*> trfPattern pat
   74                           <*> makeList " | " (before AnnClose) (mapM makePlaceholder locsAfter)
   75   where makePlaceholder l = annLocNoSema (pure (srcLocSpan l)) (pure AST.UUnboxedSumPlaceHolder)
   76 trfPattern' p = unhandledElement "pattern" p

   78 trfPatternField' :: forall n r . TransformName n r => HsRecField n (LPat n) -> Trf (AST.UPatternField (Dom r) RangeStage)
   79 trfPatternField' (HsRecField id arg False) = AST.UNormalFieldPattern <$> trfName @n (getFieldOccName id) <*> trfPattern arg
   80 trfPatternField' (HsRecField id _ True) = AST.UFieldPunPattern <$> trfName @n (getFieldOccName id)
