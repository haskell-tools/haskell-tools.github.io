   19 trfLiteral' :: TransformName n r => HsLit n -> Trf (AST.ULiteral (Dom r) RangeStage)
   20 trfLiteral' (HsChar _ ch) = pure $ AST.UCharLit ch
   21 trfLiteral' (HsCharPrim _ ch) = pure $ AST.UPrimCharLit ch
   22 trfLiteral' (HsString _ str) = pure $ AST.UStringLit (unpackFS str)
   23 trfLiteral' (HsStringPrim _ str) = pure $ AST.UPrimStringLit (BS.foldr (:) "" str)
   24 trfLiteral' (HsIntPrim _ i) = pure $ AST.UPrimIntLit i
   25 trfLiteral' (HsWordPrim _ i) = pure $ AST.UPrimWordLit i
   26 trfLiteral' (HsInt64Prim _ i) = pure $ AST.UPrimIntLit i
   27 trfLiteral' (HsWord64Prim _ i) = pure $ AST.UPrimWordLit i
   28 trfLiteral' (HsFloatPrim _ frac) = pure $ AST.UPrimFloatLit (fl_value frac)
   29 trfLiteral' (HsDoublePrim _ frac) = pure $ AST.UPrimDoubleLit (fl_value frac)
   30 trfLiteral' l = unhandledElement "literal" l

   32 monoLiteralType :: HsLit n -> Type
   33 monoLiteralType (HsChar _ _) = charTy
   34 monoLiteralType (HsCharPrim _ _) = charPrimTy
   35 monoLiteralType (HsString _ _) = stringTy

   44 trfOverloadedLit :: OverLitVal -> Trf (AST.ULiteral (Dom r) RangeStage)
   45 trfOverloadedLit (HsIntegral (IL _ _ i)) = pure $ AST.UIntLit i
   46 trfOverloadedLit (HsFractional frac) = pure $ AST.UFracLit (fl_value frac)
   47 trfOverloadedLit (HsIsString _ str) = pure $ AST.UStringLit (unpackFS str)
