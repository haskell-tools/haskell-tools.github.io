    3 -- | A representation of the tokens that build up the source file.
    4 module Language.Haskell.Tools.BackendGHC.SourceMap where
    5 
    6 import ApiAnnotation (AnnKeywordId, ApiAnnKey)
    7 import Data.List as List
    8 import Data.Map as Map
    9 import Data.Maybe (Maybe(..), maybe, fromMaybe)
   10 import Safe (headMay)
   11 
   12 import SrcLoc as GHC
   13 
   14 -- We store tokens in the source map so it is not a problem that they cannot overlap
   15 type SourceMap = (Map AnnKeywordId (Map SrcLoc SrcLoc), Map SrcLoc (SrcSpan, AnnKeywordId))
   16 
   17 -- | Returns the first occurrence of the keyword in the whole source file
   18 getKeywordAnywhere :: AnnKeywordId -> SourceMap -> Maybe SrcSpan
   19 getKeywordAnywhere keyw srcmap = return . uncurry mkSrcSpan =<< headMay . assocs =<< (Map.lookup keyw (fst srcmap))
   20 
   21 -- | Get the source location of a token restricted to a certain source span
   22 getKeywordInside :: AnnKeywordId -> SrcSpan -> SourceMap -> Maybe SrcSpan
   23 getKeywordInside keyw sr srcmap = getSourceElementInside True sr =<< Map.lookup keyw (fst srcmap)
   24 
   25 getKeywordsInside :: AnnKeywordId -> SrcSpan -> SourceMap -> [SrcSpan]
   26 getKeywordsInside keyw sr srcmap
   27   = let tokensOfType = Map.lookup keyw (fst srcmap)
   28         (_, startsAtBegin, startAfterBegin) = Map.splitLookup (srcSpanStart sr) $ fromMaybe empty tokensOfType
   29         (startsBeforeEnd, _) = Map.split (srcSpanEnd sr) $ maybe id (Map.insert (srcSpanStart sr)) startsAtBegin startAfterBegin -- tokens are minimum 1 char long
   30      in List.map (uncurry mkSrcSpan) $ List.filter (\(_, end) -> end <= srcSpanEnd sr) $ assocs startsBeforeEnd
   31 
   32 getKeywordInsideBack :: AnnKeywordId -> SrcSpan -> SourceMap -> Maybe SrcSpan
   33 getKeywordInsideBack keyw sr srcmap = getSourceElementInside False sr =<< Map.lookup keyw (fst srcmap)
   34 
   35 getSourceElementInside :: Bool -> SrcSpan -> Map SrcLoc SrcLoc -> Maybe SrcSpan
   36 getSourceElementInside b sr srcmap =
   37   case (if b then lookupGE (srcSpanStart sr) else lookupLT (srcSpanEnd sr)) srcmap of
   38     Just (k, v) -> let sp = mkSrcSpan k v in if sp `isSubspanOf` sr then Just sp else Nothing
   39     Nothing -> Nothing
   40 
   41 -- | Returns the next token on the token stream (including the token that starts on the given location)
   42 getNextToken :: SrcLoc -> SourceMap -> Maybe (SrcSpan, AnnKeywordId)
   43 getNextToken loc srcmap = fmap snd $ Map.lookupGE loc $ snd srcmap
   44 
   45 -- | Returns all subsequent tokens (including the token that starts on the given location)
   46 getTokensAfter :: SrcLoc -> SourceMap -> [(SrcSpan, AnnKeywordId)]
   47 getTokensAfter loc srcmap = case Map.splitLookup loc $ snd srcmap of
   48     (_, Just elem, after) -> elem : elems after
   49     (_, Nothing, after) -> elems after
   50 
   51 -- | Converts GHC Annotations into a convenient format for looking up tokens
   52 annotationsToSrcMap :: Map ApiAnnKey [SrcSpan] -> SourceMap
   53 annotationsToSrcMap anns = (Map.map (List.foldr addToSrcRanges Map.empty) $ mapKeysWith (++) snd anns, tokenMap)
   54   where
   55     addToSrcRanges :: SrcSpan -> Map SrcLoc SrcLoc -> Map SrcLoc SrcLoc
   56     addToSrcRanges span srcmap = Map.insert (srcSpanStart span) (srcSpanEnd span) srcmap
   57 
   58     tokenMap = Map.fromList $ List.map (\(k,v) -> (srcSpanStart k, (k, v))) $ concatMap (\(key,vals) -> List.map ((, snd key)) vals) $ Map.assocs anns
