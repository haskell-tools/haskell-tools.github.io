   24 trfDoStmt :: TransformName n r => Located (Stmt n (LHsExpr n)) -> Trf (Ann AST.UStmt (Dom r) RangeStage)
   25 trfDoStmt = trfLocNoSema trfDoStmt'
   26 
   27 trfDoStmt' :: TransformName n r => Stmt n (Located (HsExpr n)) -> Trf (AST.UStmt' AST.UExpr (Dom r) RangeStage)
   28 trfDoStmt' = gTrfDoStmt' trfExpr
   29 
   30 gTrfDoStmt' :: (TransformName n r, Data (ge n), Outputable (ge n))
   31             => (Located (ge n) -> Trf (Ann ae (Dom r) RangeStage)) -> Stmt n (Located (ge n)) -> Trf (AST.UStmt' ae (Dom r) RangeStage)
   32 gTrfDoStmt' et (BindStmt pat expr _ _ _) = AST.UBindStmt <$> trfPattern pat <*> et expr
   33 gTrfDoStmt' et (BodyStmt expr _ _ _) = AST.UExprStmt <$> et expr
   34 gTrfDoStmt' _ (LetStmt (unLoc -> binds)) = AST.ULetStmt . orderAnnList <$> addToScope binds (trfLocalBinds AnnLet binds)
   35 gTrfDoStmt' et (LastStmt body _ _) = AST.UExprStmt <$> et body
   36 gTrfDoStmt' et (RecStmt { recS_stmts = stmts }) = AST.URecStmt <$> trfAnnList "," (gTrfDoStmt' et) stmts
   37 gTrfDoStmt' _ stmt = unhandledElement "simple statement" stmt

   39 trfListCompStmts :: TransformName n r => [Located (Stmt n (LHsExpr n))] -> Trf (AnnListG AST.UListCompBody (Dom r) RangeStage)
   40 trfListCompStmts [unLoc -> ParStmt blocks _ _ _, unLoc -> (LastStmt {})]
   41   = nonemptyAnnList
   42       <$> trfScopedSequence (\(ParStmtBlock stmts _ _) ->
   43                                 let ann = collectLocs $ getNormalStmts stmts
   44                                  in annLocNoSema (pure ann) (AST.UListCompBody <$> makeList "," (pure $ srcSpanStart ann) (concat <$> trfScopedSequence trfListCompStmt stmts))
   45                             ) blocks
   46 trfListCompStmts others
   47   = let ann = (collectLocs $ getNormalStmts others)
   48      in makeList "|" (pure $ srcSpanStart ann)
   49           ((:[]) <$> annLocNoSema (pure ann)
   50                                   (AST.UListCompBody <$> makeList "," (pure $ srcSpanStart ann) (concat <$> trfScopedSequence trfListCompStmt others)))
   51 
   52 trfListCompStmt :: TransformName n r => Located (Stmt n (LHsExpr n)) -> Trf [Ann AST.UCompStmt (Dom r) RangeStage]
   53 trfListCompStmt (L _ trst@(TransStmt { trS_stmts = stmts }))
   54   = (++) <$> (concat <$> local (\s -> s { contRange = mkSrcSpan (srcSpanStart (contRange s)) (srcSpanEnd (getLoc (last stmts))) }) (trfScopedSequence trfListCompStmt stmts))
   55          <*> ((:[]) <$> extractActualStmt trst)
   56 -- last statement is extracted
   57 trfListCompStmt (unLoc -> LastStmt _ _ _) = pure []
   58 trfListCompStmt other = (:[]) <$> copyAnnot AST.UCompStmt (trfDoStmt other)
   59 
   60 extractActualStmt :: TransformName n r => Stmt n (LHsExpr n) -> Trf (Ann AST.UCompStmt (Dom r) RangeStage)
   61 extractActualStmt = \case
   62   TransStmt { trS_form = ThenForm, trS_using = using, trS_by = by }
   63     -> addAnnotation by using (AST.UThenStmt <$> trfExpr using <*> trfMaybe "," "" trfExpr by)
   64   TransStmt { trS_form = GroupForm, trS_using = using, trS_by = by }
   65     -> addAnnotation by using (AST.UGroupStmt <$> trfMaybe "," "" trfExpr by <*> (makeJust <$> trfExpr using))
   66   _ -> convertionProblem "extractActualStmt: called on a statement that is not then or group"
   67   where addAnnotation by using
   68           = annLocNoSema (combineSrcSpans (getLoc using) . combineSrcSpans (maybe noSrcSpan getLoc by)
   69                             <$> tokenLocBack AnnThen)
   70 
   71 getNormalStmts :: [Located (Stmt n (LHsExpr n))] -> [Located (Stmt n (LHsExpr n))]
   72 getNormalStmts (L _ (LastStmt _ _ _) : rest) = getNormalStmts rest
   73 getNormalStmts (stmt : rest) = stmt : getNormalStmts rest
   74 getNormalStmts [] = []
   75 
   76 getLastStmt :: [Located (Stmt n (LHsExpr n))] -> Located (HsExpr n)
   77 getLastStmt (L _ (LastStmt body _ _) : _) = body
   78 getLastStmt (_ : rest) = getLastStmt rest
   79 getLastStmt [] = convProblem "getLastStmt: empty"
