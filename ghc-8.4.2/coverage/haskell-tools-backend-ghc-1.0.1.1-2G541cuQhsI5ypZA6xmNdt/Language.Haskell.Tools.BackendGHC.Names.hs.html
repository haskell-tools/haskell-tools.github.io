   26 trfOperator :: forall n r . TransformName n r => Located (IdP n) -> Trf (Ann AST.UOperator (Dom r) RangeStage)
   27 trfOperator = trfLocNoSema (trfOperator' @n)
   28 
   29 trfOperator' :: forall n r . TransformName n r => IdP n -> Trf (AST.UOperator (Dom r) RangeStage)
   30 trfOperator' n
   31   | isSymOcc (occName @n n) = AST.UNormalOp <$> (trfQualifiedNameFocus @n) True n
   32   | otherwise = AST.UBacktickOp <$> (trfQualifiedNameFocus @n) True n
   33 
   34 trfName :: forall n r . TransformName n r => Located (IdP n) -> Trf (Ann AST.UName (Dom r) RangeStage)
   35 trfName = trfLocNoSema (trfName' @n)
   36 
   37 trfName' :: forall n r . TransformName n r => IdP n -> Trf (AST.UName (Dom r) RangeStage)
   38 trfName' n
   39   | isSymOcc (occName @n n) = (if isSpecKind then AST.UNormalName else AST.UParenName) <$> (trfQualifiedNameFocus @n) isSpecKind n
   40   | otherwise = AST.UNormalName <$> (trfQualifiedNameFocus @n) False n
   41   where -- special names that are operators, but appear in name context
   42     isSpecKind = occNameString (occName @n n) `elem` ["*", "#", "?", "??"]
   43 
   44 trfAmbiguousFieldName :: TransformName n r => Located (AmbiguousFieldOcc n) -> Trf (Ann AST.UName (Dom r) RangeStage)
   45 trfAmbiguousFieldName (L l af) = trfAmbiguousFieldName' l af
   46 
   47 trfAmbiguousFieldName' :: forall n r . TransformName n r => SrcSpan -> AmbiguousFieldOcc n -> Trf (Ann AST.UName (Dom r) RangeStage)
   48 trfAmbiguousFieldName' l (Unambiguous (L _ rdr) pr) = annLocNoSema (pure l) $ trfName' @n (unpackPostRn @n rdr pr)
   49 -- no Id transformation is done, so we can basically ignore the postTC value
   50 trfAmbiguousFieldName' _ (Ambiguous (L l rdr) _)
   51   = annLocNoSema (pure l)
   52       $ (if (isSymOcc (occName @GhcPs rdr)) then AST.UParenName else AST.UNormalName)
   53           <$> (annLoc (createAmbigousNameInfo rdr l) (pure l) $ AST.nameFromList <$> trfNameStr (isSymOcc (occName @GhcPs rdr)) (rdrNameStr rdr))
   54 
   55 trfAmbiguousOperator' :: forall n r . TransformName n r => SrcSpan -> AmbiguousFieldOcc n -> Trf (Ann AST.UOperator (Dom r) RangeStage)
   56 trfAmbiguousOperator' l (Unambiguous (L _ rdr) pr) = annLocNoSema (pure l) $ trfOperator' @n (unpackPostRn @n rdr pr)
   57 -- no Id transformation is done, so we can basically ignore the postTC value
   58 trfAmbiguousOperator' _ (Ambiguous (L l rdr) _)
   59   = annLocNoSema (pure l)
   60       $ (if (isSymOcc (occName @GhcPs rdr)) then AST.UNormalOp else AST.UBacktickOp)
   61           <$> (annLoc (createAmbigousNameInfo rdr l) (pure l) $ AST.nameFromList <$> trfOperatorStr (not $ isSymOcc (occName @GhcPs rdr)) (rdrNameStr rdr))

   76 -- | This class allows us to use the same transformation code for multiple variants of the GHC AST.
   77 -- GHC UName annotated with 'name' can be transformed to our representation with semantic annotations of 'res'.
   78 class (TransformableName name, HsHasName (IdP name), FromGHCName (IdP res), Eq (IdP name) {-, TransformableName res, HsHasName res -}, GHCName res, NameOrRdrName (IdP name) ~ (IdP name))
   79         => TransformName name res where
   80   -- | Demote a given name
   81   transformName :: IdP name -> IdP res

   83 instance TransformName GhcPs GhcPs where transformName = id

   88 trfNameText :: String -> Trf (Ann AST.UName (Dom r) RangeStage)
   89 trfNameText str
   90   = annContNoSema (AST.UNormalName <$> annLoc (createImplicitNameInfo str) (asks contRange) (AST.nameFromList <$> trfNameStr (isOperatorStr str) str))
   91 
   92 trfImplicitName :: HsIPName -> Trf (Ann AST.UName (Dom r) RangeStage)
   93 trfImplicitName (HsIPName fs)
   94   = let nstr = unpackFS fs
   95      in do rng <- asks contRange
   96            let rng' = mkSrcSpan (updateCol (+1) (srcSpanStart rng)) (srcSpanEnd rng)
   97            annContNoSema (AST.UImplicitName <$> annLoc (createImplicitNameInfo nstr) (pure rng')
   98                                                   (AST.nameFromList <$> trfNameStr (isOperatorStr nstr) nstr))

  100 isOperatorStr :: String -> Bool
  101 isOperatorStr = any (not . isAlphaNum)

  103 trfQualifiedName :: forall n r . TransformName n r => Bool -> Located (IdP n) -> Trf (Ann AST.UQualifiedName (Dom r) RangeStage)
  104 trfQualifiedName isOperator (L l n) = focusOn l $ (trfQualifiedNameFocus @n) isOperator n
  105 
  106 trfQualifiedNameFocus :: forall n r . TransformName n r => Bool -> IdP n -> Trf (Ann AST.UQualifiedName (Dom r) RangeStage)
  107 trfQualifiedNameFocus isOperator n
  108   = do rng <- asks contRange
  109        let rng' = if isOperator == isSymOcc (occName @n n) then rng
  110                     else mkSrcSpan (updateCol (+1) (srcSpanStart rng)) (updateCol (subtract 1) (srcSpanEnd rng))
  111        annLoc (createNameInfo (transformName @n @r n)) (pure rng') (trfQualifiedName' @n n)
  112 
  113 trfQualifiedName' :: forall n r . TransformName n r => IdP n -> Trf (AST.UQualifiedName (Dom r) RangeStage)
  114 trfQualifiedName' n = AST.nameFromList <$> ((if isSymOcc (occName @n n) then trfOperatorStr else trfNameStr) False =<< (correctNameString @n) n)
  115 
  116 trfOperatorStr :: Bool -> String -> Trf (AnnListG AST.UNamePart (Dom r) RangeStage)
  117 trfOperatorStr isInParen str = do rng <- correctSpan <$> asks contRange
  118                                   makeList "." (pure $ srcSpanStart rng)
  119                                                (pure [Ann (noSemaInfo $ AST.NodeSpan rng) (AST.UNamePart str)])
  120   where correctSpan sp = if isInParen then mkSrcSpan (updateCol (+1) (srcSpanStart sp))
  121                                                      (updateCol (subtract 1) (srcSpanEnd sp))
  122                                       else sp
  123 
  124 -- | Creates a qualified name from a name string
  125 trfNameStr :: Bool -> String -> Trf (AnnListG AST.UNamePart (Dom r) RangeStage)
  126 trfNameStr isInBackticks str = makeList "." atTheStart (trfNameStr' str . correct <$> atTheStart)
  127   where correct = if isInBackticks then updateCol (+1) else id
  128 
  129 trfNameStr' :: String -> SrcLoc -> [Ann AST.UNamePart (Dom r) RangeStage]
  130 trfNameStr' str startLoc = fst $
  131   foldl (\(r,loc) np -> let nextLoc = advanceAllSrcLoc loc np
  132                          in ( r ++ [Ann (noSemaInfo $ AST.NodeSpan (mkSrcSpan loc nextLoc)) (AST.UNamePart np)], advanceAllSrcLoc nextLoc "." ) )
  133   ([], startLoc) (splitOn "." str)
  134   where -- | Move the source location according to a string
  135         advanceAllSrcLoc :: SrcLoc -> String -> SrcLoc
  136         advanceAllSrcLoc (RealSrcLoc rl) str = RealSrcLoc $ foldl advanceSrcLoc rl str
  137         advanceAllSrcLoc oth _ = oth
