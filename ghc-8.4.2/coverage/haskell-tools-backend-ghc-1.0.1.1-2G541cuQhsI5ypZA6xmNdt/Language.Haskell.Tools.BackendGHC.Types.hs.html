   30 trfType :: forall n r . TransformName n r => Located (HsType n) -> Trf (Ann AST.UType (Dom r) RangeStage)
   31 trfType typ | RealSrcSpan loce <- getLoc typ
   32   = do othSplices <- asks typeSplices
   33        let contSplice = filter (\sp -> case getLoc sp of (RealSrcSpan spLoc) -> spLoc `containsSpan` loce; _ -> False) othSplices
   34        case contSplice of [] -> trfLocNoSema trfType' typ
   35                           _ -> let lsp@(L l sp) = minimumBy (compareSpans `on` getLoc) contSplice
   36                                 in typeSpliceInserted lsp (annLocNoSema (pure l) (AST.UTySplice <$> (trfSplice =<< rdrSplice sp)))
   37   | otherwise = trfLocNoSema trfType' typ
   38 
   39 trfType' :: forall n r . TransformName n r => HsType n -> Trf (AST.UType (Dom r) RangeStage)
   40 trfType' = trfType'' . cleanHsType where
   41   trfType'' :: HsType n -> Trf (AST.UType (Dom r) RangeStage)
   42   trfType'' (HsForAllTy [] typ) = trfType' (unLoc typ)
   43   trfType'' (HsForAllTy bndrs typ) = AST.UTyForall <$> defineTypeVars (trfBindings bndrs)
   44                                                    <*> addToScope bndrs (trfType typ)
   45   trfType'' (HsQualTy (L _ []) typ) = trfType' (unLoc typ)
   46   trfType'' (HsQualTy ctx typ) = AST.UTyCtx <$> (fromJust . (^. annMaybe) <$> trfCtx atTheStart ctx)
   47                                             <*> trfType typ
   48   trfType'' (HsTyVar _ name) = AST.UTyVar <$> transformingPossibleVar name (trfName @n name)
   49   trfType'' (HsAppsTy apps) | Just (head, args, _) <- getAppsTyHead_maybe apps
   50     = foldl (\core t -> AST.UTyApp <$> annLocNoSema (pure $ getLoc head `combineSrcSpans` getLoc t) core <*> trfType t) (trfType' (unLoc head)) args
   51   trfType'' (HsAppTy t1 t2) = AST.UTyApp <$> trfType t1 <*> trfType t2
   52   trfType'' (HsFunTy t1 t2) = AST.UTyFun <$> trfType t1 <*> trfType t2
   53   trfType'' (HsListTy typ) = AST.UTyList <$> trfType typ
   54   trfType'' (HsPArrTy typ) = AST.UTyParArray <$> trfType typ
   55   trfType'' (HsTupleTy HsBoxedOrConstraintTuple typs) = AST.UTyTuple <$> trfAnnList ", " trfType' typs
   56   trfType'' (HsTupleTy HsBoxedTuple typs) = AST.UTyTuple <$> trfAnnList ", " trfType' typs
   57   trfType'' (HsTupleTy HsUnboxedTuple typs) = AST.UTyUnbTuple <$> trfAnnList ", " trfType' typs
   58   trfType'' (HsOpTy t1 op t2) = AST.UTyInfix <$> trfType t1 <*> trfOperator @n op <*> trfType t2
   59   trfType'' (HsParTy typ) = AST.UTyParen <$> trfType typ
   60   trfType'' (HsKindSig typ kind) = AST.UTyKinded <$> trfType typ <*> trfKind kind
   61   trfType'' (HsSpliceTy qq@(HsQuasiQuote {}) _) = AST.UTyQuasiQuote <$> annContNoSema (trfQuasiQuotation' qq)
   62   trfType'' (HsSpliceTy splice _) = AST.UTySplice <$> trfSplice splice
   63   trfType'' (HsBangTy (HsSrcBang _ SrcUnpack _) typ) = AST.UTyUnpack <$> trfType typ
   64   trfType'' (HsBangTy (HsSrcBang _ SrcNoUnpack _) typ) = AST.UTyNoUnpack <$> trfType typ
   65   trfType'' (HsBangTy (HsSrcBang _ _ SrcStrict) typ) = AST.UTyBang <$> trfType typ
   66   trfType'' (HsBangTy (HsSrcBang _ _ SrcLazy) typ) = AST.UTyLazy <$> trfType typ
   67   trfType'' pt@(HsExplicitListTy {}) = AST.UTyPromoted <$> annContNoSema (trfPromoted' trfType' pt)
   68   trfType'' pt@(HsExplicitTupleTy {}) = AST.UTyPromoted <$> annContNoSema (trfPromoted' trfType' pt)
   69   trfType'' pt@(HsTyLit {}) = AST.UTyPromoted <$> annContNoSema (trfPromoted' trfType' pt)
   70   trfType'' (HsWildCardTy _) = pure AST.UTyWildcard -- TODO: named wildcards
   71   trfType'' (HsSumTy types) = AST.UUnbSumType <$> trfAnnList " | " trfType' types
   72   trfType'' t = unhandledElement "type" t

   74 trfBindings :: TransformName n r => [Located (HsTyVarBndr n)] -> Trf (AnnListG AST.UTyVar (Dom r) RangeStage)
   75 trfBindings vars = trfAnnList " " trfTyVar' vars

   77 trfTyVar :: TransformName n r => Located (HsTyVarBndr n) -> Trf (Ann AST.UTyVar (Dom r) RangeStage)
   78 trfTyVar = trfLocNoSema trfTyVar'
   79 
   80 trfTyVar' :: forall n r . TransformName n r => HsTyVarBndr n -> Trf (AST.UTyVar (Dom r) RangeStage)
   81 trfTyVar' (UserTyVar name) = AST.UTyVarDecl <$> typeVarTransform (trfName @n name)
   82                                            <*> (nothing " " "" atTheEnd)
   83 trfTyVar' (KindedTyVar name kind) = AST.UTyVarDecl <$> typeVarTransform (trfName @n name)
   84                                                   <*> trfKindSig (Just kind)

   86 trfCtx :: TransformName n r => Trf SrcLoc -> Located (HsContext n) -> Trf (AnnMaybeG AST.UContext (Dom r) RangeStage)
   87 trfCtx sp (L _ []) = nothing " " "" sp
   88 trfCtx _ (L l [L _ (HsParTy t)])
   89   = makeJust <$> annLocNoSema (combineSrcSpans l <$> tokenLoc AnnDarrow)
   90                               (AST.UContext <$> annLocNoSema (pure l) (AST.UTupleAssert <$> (trfAnnList ", " trfAssertion' [t])))
   91 trfCtx _ (L l [t])
   92   = makeJust <$> annLocNoSema (combineSrcSpans l <$> tokenLoc AnnDarrow)
   93                               (AST.UContext <$> trfAssertion t)
   94 trfCtx _ (L l ctx) = makeJust <$> annLocNoSema (combineSrcSpans l <$> tokenLoc AnnDarrow)
   95                                                (AST.UContext <$> annLocNoSema (pure l) (AST.UTupleAssert <$> (trfAnnList ", " trfAssertion' ctx)))
   96 
   97 trfAssertion :: TransformName n r => LHsType n -> Trf (Ann AST.UAssertion (Dom r) RangeStage)
   98 trfAssertion = trfLocNoSema trfAssertion'
   99 
  100 trfAssertion' :: forall n r . TransformName n r => HsType n -> Trf (AST.UAssertion (Dom r) RangeStage)
  101 trfAssertion' (cleanHsType -> HsParTy t)
  102   = trfAssertion' (unLoc t)
  103 trfAssertion' (cleanHsType -> HsOpTy left op right)
  104   = AST.UInfixAssert <$> trfType left <*> trfOperator @n op <*> trfType right
  105 trfAssertion' (cleanHsType -> HsTupleTy _ tys)
  106   = AST.UTupleAssert <$> makeList ", " (after AnnOpenP) (mapM trfAssertion tys)
  107 trfAssertion' (cleanHsType -> HsWildCardTy _)
  108   = pure AST.UWildcardAssert
  109 trfAssertion' (cleanHsType -> t) = case cleanHsType base of
  110    HsTyVar _ name -> AST.UClassAssert <$> trfName @n name <*> trfAnnList " " trfType' args
  111    HsEqTy t1 t2 -> AST.UInfixAssert <$> trfType t1 <*> annLocNoSema (tokenLoc AnnTilde) (trfOperator' @n typeEq) <*> trfType t2
  112    HsIParamTy name t -> AST.UImplicitAssert <$> define (focusOn (getLoc name) (trfImplicitName (unLoc name))) <*> trfType t
  113    t -> unhandledElement "assertion" t
  114   where (args, _, base) = getArgs t
  115 
  116         getArgs :: HsType n -> ([LHsType n], Maybe SrcSpan, HsType n)
  117         getArgs (HsAppTy (L l ft) at) = case getArgs ft of (args, sp, base) -> (args++[at], sp <|> Just l, base)
  118         getArgs t = ([], Nothing, t)
  119 
  120         typeEq :: IdP n
  121         typeEq = nameFromId @n (mkVanillaGlobal (tyConName heqTyCon) (tyConKind heqTyCon))
