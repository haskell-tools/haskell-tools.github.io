   41 trfDecls :: TransformName n r => [LHsDecl n] -> Trf (AnnListG AST.UDecl (Dom r) RangeStage)

   44 trfDeclsGroup :: forall n r . TransformName n r => HsGroup n -> Trf (AnnListG AST.UDecl (Dom r) RangeStage)
   45 trfDeclsGroup g@(HsGroup vals splices tycls derivs fixities defaults foreigns warns anns rules vects _)
   46   = do rdrSpls <- asks declSplices -- now we don't want to rename the splices, just interested in their locations to
   47                                    -- filter out the declarations that are generated from them
   48        let (sigs, bagToList -> binds) = getBindsAndSigs vals
   49            -- collect the declarations from the group
   50            alldecls :: [Located (HsDecl n)]
   51            alldecls = (map (fmap SpliceD) splices)
   52                         ++ (map (fmap ValD) binds)
   53                         ++ (map (fmap SigD) sigs)
   54                         ++ (map (fmap TyClD) (concat $ map group_tyclds tycls))
   55                         ++ (map (fmap DerivD) derivs)
   56                         ++ (map (fmap (SigD . FixSig)) (mergeFixityDefs fixities))
   57                         ++ (map (fmap DefD) defaults)
   58                         ++ (map (fmap ForD) foreigns)
   59                         ++ (map (fmap WarningD) warns)
   60                         ++ (map (fmap AnnD) anns)
   61                         ++ (map (fmap RuleD) rules)
   62                         ++ (map (fmap VectD) vects)
   63                         ++ (map (fmap InstD) (hsGroupInstDecls g))
   64        -- Declarations generated from TH should only be in scope after the splice.
   65        let (genNames, sourceNames) = partition (\d -> any (\spl -> getLoc spl `containsRealSpan` getLoc d) rdrSpls) alldecls
   66        addToCurrentScope sourceNames $ do
   67          -- use the definitions generated by previous splices when renaming one
   68          spls <- asks declSplices >>= mapM (\(L l e) -> let namesGeneratedBefore = filter ((srcSpanStart l >) . srcSpanEnd . getLoc) genNames
   69                                                          in addToCurrentScope namesGeneratedBefore ((L l) <$> transformSplice e))
   70          let actualDefinitions = removeContained $ orderElems $ replaceSpliceDecls spls alldecls
   71            in makeIndentedListNewlineBefore atTheEnd
   72                 (orderDefs <$> ((++) <$> getDeclsToInsert <*> (mapM (trfDeclsWithScope genNames) actualDefinitions)))
   73   where
   74     -- use the definitions generated by previous splices when transforming a definition
   75     trfDeclsWithScope genNames d = local (\s -> s {declSplices = []})
   76                                      $ addToCurrentScope namesGeneratedBefore (trfDecl d)
   77       where namesGeneratedBefore = filter ((srcSpanStart (getLoc d) >) . srcSpanEnd . getLoc) genNames
   78 
   79     replaceSpliceDecls :: [Located (HsSplice n)] -> [Located (HsDecl n)] -> [Located (HsDecl n)]
   80     replaceSpliceDecls splices decls = foldl mergeSplice decls splices
   81 
   82     orderElems :: [Located a] -> [Located a]
   83     orderElems = sortOn (srcSpanStart . getLoc)
   84 
   85     removeContained :: [Located (HsDecl n)] -> [Located (HsDecl n)]
   86     removeContained (fst:snd:rest) | getLoc fst `containsRealSpan` getLoc snd
   87       = removeContained (fst:rest)
   88     removeContained (fst:rest) = fst : removeContained rest
   89     removeContained [] = []
   90 
   91     (RealSrcSpan sp1) `containsRealSpan` (RealSrcSpan sp2) = sp1 `containsSpan` sp2
   92     _ `containsRealSpan` _ = False
   93 
   94     mergeSplice :: [Located (HsDecl n)] -> Located (HsSplice n) -> [Located (HsDecl n)]
   95     mergeSplice decls spl@(L spLoc@(RealSrcSpan rss) _)
   96       = L spLoc (SpliceD (SpliceDecl spl ExplicitSplice)) : filter (\(L (RealSrcSpan rdsp) _) -> not (rss `containsSpan` rdsp)) decls
   97     mergeSplice _ (L (UnhelpfulSpan {}) _) = convProblem "mergeSplice: no real span"
   98 
   99     getDeclsToInsert :: Trf [Ann AST.UDecl (Dom r) RangeStage]
  100     getDeclsToInsert = do decls <- asks declsToInsert
  101                           allLocals <- asks localsInScope
  102                           case allLocals of locals:_ -> liftGhc $ mapM (replaceNamesInDecls (map (^. _1) locals)) decls
  103                                             [] -> convertionProblem "getDeclsToInsert: empty scope"
  104        where replaceNamesInDecls :: [GHC.Name] -> Ann AST.UDecl (Dom GhcPs) RangeStage -> GHC.Ghc (Ann AST.UDecl (Dom r) RangeStage)
  105              replaceNamesInDecls locals = AST.semaTraverse $
  106                 AST.SemaTrf (pure . (AST.nameInfo .- findName)) pure pure
  107                             (pure . trfImportInfo findName) pure pure pure
  108                where findName :: RdrName -> IdP r
  109                      findName rdr = fromGHCName $ fromMaybe (convProblem $ "Data definition name not found: " ++ showSDocUnsafe (ppr rdr)
  110                                                                              ++ ", locals: " ++ (concat $ intersperse ", " $ map (showSDocUnsafe . ppr) locals))
  111                                                 $ find ((occNameString (rdrNameOcc rdr) ==) . occNameString . nameOccName) locals

  113 trfDecl :: forall n r . TransformName n r => Located (HsDecl n) -> Trf (Ann AST.UDecl (Dom r) RangeStage)
  114 trfDecl = trfLocNoSema $ \case
  115   TyClD (FamDecl (FamilyDecl (ClosedTypeFamily typeEqs) name tyVars _ kindSig inj))
  116     -> AST.UClosedTypeFamilyDecl <$> focusAfter AnnType (createDeclHead name tyVars)
  117                                 <*> trfFamilyResultSig kindSig inj
  118                                 <*> trfTypeEqs typeEqs
  119   TyClD (FamDecl fd) -> AST.UTypeFamilyDecl <$> annContNoSema (trfTypeFam' fd)
  120   TyClD (SynDecl name vars _ rhs _)
  121     -> AST.UTypeDecl <$> between AnnType AnnEqual (createDeclHead name vars) <*> trfType rhs
  122   TyClD (DataDecl name vars _ (HsDataDefn nd ctx _ kind cons derivs) _ _)
  123     -> do let ctxTok = case nd of DataType -> AnnData
  124                                   NewType -> AnnNewtype
  125               consLoc = focusBeforeIfPresent AnnDeriving atTheEnd
  126           whereLoc <- tokenLoc AnnWhere
  127           if isGoodSrcSpan whereLoc then trfGADT nd name vars ctx kind cons derivs ctxTok consLoc
  128                                     else trfDataDef nd name vars ctx cons derivs ctxTok consLoc
  129   TyClD (ClassDecl ctx name vars _ funDeps sigs defs typeFuns typeFunDefs _ _)
  130     -> AST.UClassDecl <$> trfCtx (after AnnClass) ctx
  131                      <*> betweenIfPresent AnnClass AnnWhere (createDeclHead name vars)
  132                      <*> trfFunDeps @n funDeps
  133                      <*> createClassBody sigs defs typeFuns typeFunDefs
  134   InstD (ClsInstD (ClsInstDecl typ binds sigs typefam datafam overlap))
  135     -> AST.UInstDecl <$> trfMaybeDefault " " "" trfOverlap (after AnnInstance) overlap
  136                     <*> trfInstanceRule (hsib_body typ)
  137                     <*> trfInstBody binds sigs typefam datafam
  138   InstD (DataFamInstD (DataFamInstDecl (hsib_body -> FamEqn con pats _ (HsDataDefn nd _ _ _ cons derivs))))
  139     | all ((\case ConDeclH98{} -> True; _ -> False) . unLoc) cons
  140     -> AST.UDataInstDecl <$> trfDataKeyword nd
  141                         <*> (focusAfter AnnInstance . focusBeforeIfPresent AnnEqual . focusBeforeIfPresent AnnDeriving)
  142                               (makeInstanceRuleTyVars con pats)
  143                                                        -- the location is needed when there is no = sign
  144                         <*> makeListBefore " = " " | " (pure $ srcSpanStart $ foldLocs $ getLoc con : map getLoc pats) (mapM trfConDecl cons)
  145                         <*> makeIndentedList atTheEnd (mapM trfDerivings (unLoc derivs))
  146   InstD (DataFamInstD (DataFamInstDecl (hsib_body -> FamEqn con pats _ (HsDataDefn nd _ _ kind cons _))))
  147     -> AST.UGDataInstDecl <$> trfDataKeyword nd
  148                         <*> (focusAfter AnnInstance . focusBeforeIfPresent AnnWhere)
  149                               (makeInstanceRuleTyVars con pats)
  150                         <*> focusBefore AnnWhere (trfKindSig kind)
  151                         <*> makeIndentedListBefore " where " atTheEnd (mapM trfGADTConDecl cons)
  152   InstD (TyFamInstD (TyFamInstDecl (hsib_body -> FamEqn con pats _ rhs)))
  153     -> AST.UTypeInstDecl <$> between AnnInstance AnnEqual (makeInstanceRuleTyVars con pats) <*> trfType rhs
  154   ValD bind -> trfVal bind
  155   SigD sig -> trfSig sig
  156   DerivD (DerivDecl t strat overlap) -> AST.UDerivDecl <$> trfDerivingStrategy strat <*> trfMaybeDefault " " "" trfOverlap (after AnnInstance) overlap <*> trfInstanceRule (hsib_body t)
  157   RuleD (HsRules _ rules) -> AST.UPragmaDecl <$> annContNoSema (AST.URulePragma <$> makeIndentedList (before AnnClose) (mapM trfRewriteRule rules))
  158   RoleAnnotD (RoleAnnotDecl name roles) -> AST.URoleDecl <$> trfQualifiedName @n False name <*> makeList " " atTheEnd (mapM trfRole roles)
  159   DefD (DefaultDecl types) -> AST.UDefaultDecl <$> makeList "," (after AnnOpenP) (mapM trfType types)
  160   ForD (ForeignImport name (hsib_body -> typ) _ (CImport ccall safe _ _ _))
  161     -> AST.UForeignImport <$> trfCallConv ccall <*> trfSafety (getLoc ccall) safe <*> define (trfName @n name) <*> trfType typ
  162   ForD (ForeignExport name (hsib_body -> typ) _ (CExport (L l (CExportStatic _ _ ccall)) _))
  163     -> AST.UForeignExport <$> annLocNoSema (pure l) (trfCallConv' ccall) <*> trfName @n name <*> trfType typ
  164   SpliceD (SpliceDecl (unLoc -> spl) _) -> AST.USpliceDecl <$> trfSplice spl
  165   WarningD (Warnings _ [])
  166     -> AST.UPragmaDecl <$> annContNoSema (AST.UDeprPragma <$> (makeList " " (after AnnOpen) (pure []))
  167                                                           <*> makeList ", " (before AnnClose) (pure []))
  168   WarningD (Warnings _ [L _ (Warning names (DeprecatedTxt _ stringLits))])
  169     -> AST.UPragmaDecl <$> annContNoSema (AST.UDeprPragma <$> (makeList " " (after AnnOpen) $ mapM (trfName @n) names)
  170                                                           <*> makeList ", " (before AnnClose) (mapM (\(L l (StringLiteral _ fs)) -> trfFastString (L l fs)) stringLits))
  171   WarningD (Warnings _ [L _ (Warning names (WarningTxt _ stringLits))])
  172     -> AST.UPragmaDecl <$> annContNoSema (AST.UWarningPragma <$> (makeNonemptyList " " $ mapM (trfName @n) names)
  173                                                              <*> makeList ", " (before AnnClose) (mapM (\(L l (StringLiteral _ fs)) -> trfFastString (L l fs)) stringLits))
  174   AnnD (HsAnnotation stxt subject expr)
  175     -> AST.UPragmaDecl <$> annContNoSema (AST.UAnnPragma <$> trfAnnotationSubject @n stxt subject (srcSpanStart $ getLoc expr) <*> trfExpr expr)
  176   d -> unhandledElement "declaration" d

  178 trfGADT :: TransformName n r => NewOrData -> Located (IdP n) -> LHsQTyVars n -> Located (HsContext n)
  179                                  -> Maybe (Located (HsKind n)) -> [Located (ConDecl n)]
  180                                  -> Located [LHsDerivingClause n] -> AnnKeywordId -> Trf SrcLoc -> Trf (AST.UDecl (Dom r) RangeStage)
  181 trfGADT nd name vars ctx kind cons derivs ctxTok consLoc
  182   = AST.UGDataDecl <$> trfDataKeyword nd
  183                    <*> trfCtx (after ctxTok) ctx
  184                    <*> betweenIfPresent ctxTok AnnEqual (createDeclHead name vars)
  185                    <*> focusBefore AnnWhere (trfKindSig kind)
  186                    <*> makeIndentedListBefore " where " consLoc (mapM trfGADTConDecl cons)
  187                    <*> makeIndentedList atTheEnd (mapM trfDerivings (unLoc derivs))
  188 
  189 trfDataDef :: TransformName n r => NewOrData -> Located (IdP n) -> LHsQTyVars n -> Located (HsContext n)
  190                                      -> [Located (ConDecl n)] -> Located [LHsDerivingClause n]
  191                                      -> AnnKeywordId -> Trf SrcLoc -> Trf (AST.UDecl (Dom r) RangeStage)
  192 trfDataDef nd name vars ctx cons derivs ctxTok consLoc
  193   = AST.UDataDecl <$> trfDataKeyword nd
  194                   <*> trfCtx (after ctxTok) ctx
  195                   <*> betweenIfPresent ctxTok AnnEqual (createDeclHead name vars)
  196                   <*> makeListBefore "=" " | " consLoc (mapM trfConDecl cons)
  197                   <*> makeIndentedList atTheEnd (mapM trfDerivings (unLoc derivs))

  199 trfVal :: TransformName n r => HsBindLR n n -> Trf (AST.UDecl (Dom r) RangeStage)
  200 trfVal (PatSynBind psb) = AST.UPatternSynonymDecl <$> annContNoSema (trfPatternSynonym psb)
  201 trfVal bind = AST.UValueBinding <$> (annContNoSema $ trfBind' bind)

  203 trfSig :: forall n r . TransformName n r => Sig n -> Trf (AST.UDecl (Dom r) RangeStage)
  204 trfSig (ts@TypeSig {}) = AST.UTypeSigDecl <$> defineTypeVars (annContNoSema $ trfTypeSig' ts)
  205 trfSig (FixSig fs) = AST.UFixityDecl <$> (annContNoSema $ trfFixitySig fs)
  206 trfSig (PatSynSig ids typ)
  207   = AST.UPatTypeSigDecl <$> annContNoSema (AST.UPatternTypeSignature <$> trfAnnList ", " (trfName' @n) ids <*> trfType (hsib_body typ))
  208 trfSig (InlineSig name prag)
  209   = AST.UPragmaDecl <$> annContNoSema (AST.UInlinePragmaDecl <$> trfInlinePragma @n name prag)
  210 trfSig (SpecSig name (map hsib_body -> types) (inl_act -> phase))
  211   = AST.UPragmaDecl <$> annContNoSema (AST.USpecializeDecl <$> trfSpecializePragma name types phase)
  212 trfSig (CompleteMatchSig _ names typeConstraint)
  213   = AST.UPragmaDecl <$> annContNoSema (AST.UCompletePragma <$> trfAnnList ", " (trfName' @n) (unLoc names)
  214                                                            <*> trfMaybe " :: " "" (trfName @n) typeConstraint)
  215 trfSig s = unhandledElement "signature" s

  217 trfSpecializePragma :: forall n r . TransformName n r
  218                     => Located (IdP n) -> [Located (HsType n)] -> Activation -> Trf (Ann AST.USpecializePragma (Dom r) RangeStage)
  219 trfSpecializePragma name types phase
  220   = annContNoSema $ AST.USpecializePragma <$> trfPhase (pure $ srcSpanStart (getLoc name)) phase
  221                                           <*> trfName @n name
  222                                           <*> (orderAnnList <$> trfAnnList ", " trfType' types)

  227 trfConDecl' :: forall n r . TransformName n r => ConDecl n -> Trf (AST.UConDecl (Dom r) RangeStage)
  228 trfConDecl' (ConDeclH98 { con_name = name, con_qvars = tyVars, con_cxt = ctx, con_details = PrefixCon args })
  229   = AST.UConDecl <$> trfConTyVars tyVars <*> trfConCtx ctx <*> define (trfName @n name) <*> makeList " " atTheEnd (mapM trfType args)
  230 trfConDecl' (ConDeclH98 { con_name = name, con_qvars = tyVars, con_cxt = ctx, con_details = RecCon (unLoc -> flds) })
  231   = AST.URecordDecl <$> trfConTyVars tyVars <*> trfConCtx ctx <*> define (trfName @n name) <*> (between AnnOpenC AnnCloseC $ trfAnnList ", " trfFieldDecl' flds)
  232 trfConDecl' (ConDeclH98 { con_name = name, con_qvars = tyVars, con_cxt = ctx, con_details = InfixCon t1 t2 })
  233   = AST.UInfixConDecl <$> trfConTyVars tyVars <*> trfConCtx ctx <*> trfType t1 <*> define (trfOperator @n name) <*> trfType t2
  234 trfConDecl' gadt@(ConDeclGADT {}) = unhandledElement "normal constructor declaration" gadt

  244 trfGADTConDecl :: TransformName n r => Located (ConDecl n) -> Trf (Ann AST.UGadtConDecl (Dom r) RangeStage)

  247 trfGADTConDecl' :: forall n r . TransformName n r => ConDecl n -> Trf (AST.UGadtConDecl (Dom r) RangeStage)
  248 trfGADTConDecl' (ConDeclGADT { con_names = names, con_type = hsib_body -> typ })
  249   = let nameLoc = collectLocs names
  250         typLoc = getLoc typ
  251         (vars, ctx, t) = getTypeVarsAndCtx typ
  252      in AST.UGadtConDecl <$> define (trfAnnList ", " (trfName' @n) names)
  253                          <*> focusOn (mkSrcSpan (srcSpanEnd nameLoc) (srcSpanStart typLoc)) (trfBindings vars)
  254                          <*> updateFocus (return . updateEnd (\_ -> srcSpanStart typLoc)) (focusAfterIfPresent AnnDot (trfCtx atTheStart ctx))
  255                          <*> trfGadtConType t
  256   where getTypeVarsAndCtx :: LHsType n -> ([LHsTyVarBndr n], LHsContext n, LHsType n)
  257         getTypeVarsAndCtx (L _ (HsForAllTy [] typ)) = getTypeVarsAndCtx typ
  258         getTypeVarsAndCtx (L _ (HsForAllTy bndrs typ)) = let (_,ctx,t) = getTypeVarsAndCtx typ in (bndrs, ctx, t)
  259         getTypeVarsAndCtx (L _ (HsQualTy ctx typ)) = let (vars,_,t) = getTypeVarsAndCtx typ in (vars, ctx, t)
  260         getTypeVarsAndCtx t = ([], L noSrcSpan [], t)

  275 trfDerivings :: TransformName n r => Located (HsDerivingClause n) -> Trf (Ann AST.UDeriving (Dom r) RangeStage)

  286                                               NewtypeStrategy -> return AST.UNewtypeStrategy)

  289 trfInstanceRule :: TransformName n r => Located (HsType n) -> Trf (Ann AST.UInstanceRule (Dom r) RangeStage)

  292 trfInstanceRule' :: forall n r . TransformName n r => HsType n -> Trf (AST.UInstanceRule (Dom r) RangeStage)
  293 trfInstanceRule' (HsForAllTy bndrs (unLoc -> HsQualTy ctx typ))
  294   = AST.UInstanceRule <$> (makeJust <$> annLocNoSema (pure $ collectLocs bndrs) (trfBindings bndrs))
  295                       <*> trfCtx (after AnnDot) ctx
  296                       <*> trfInstanceHead typ
  297 trfInstanceRule' (HsQualTy ctx typ) = AST.UInstanceRule <$> nothing "" " . " atTheStart
  298                                                         <*> trfCtx atTheStart ctx
  299                                                         <*> trfInstanceHead typ
  300 trfInstanceRule' (HsParTy typ) = instanceHead $ annContNoSema (AST.UInstanceHeadParen <$> trfInstanceHead typ)
  301 trfInstanceRule' (HsTyVar _ tv) = instanceHead $ annContNoSema (AST.UInstanceHeadCon <$> trfName @n tv)
  302 trfInstanceRule' (HsAppTy t1 t2) = instanceHead $ annContNoSema (AST.UInstanceHeadApp <$> trfInstanceHead t1 <*> trfType t2)
  303 trfInstanceRule' (HsOpTy t1 op t2) = instanceHead $ annContNoSema (AST.UInstanceHeadApp <$> annLocNoSema (pure $ getLoc t1 `combineSrcSpans` getLoc op) (AST.UInstanceHeadInfix <$> trfType t1 <*> trfOperator @n op) <*> trfType t2)
  304 trfInstanceRule' t = unhandledElement "instance rule" t

  306 instanceHead :: Trf (Ann AST.UInstanceHead (Dom r) RangeStage) -> Trf (AST.UInstanceRule (Dom r) RangeStage)
  307 instanceHead hd = AST.UInstanceRule <$> (nothing "" " . " atTheStart) <*> (nothing " " "" atTheStart) <*> hd

  309 makeInstanceRuleTyVars :: forall n r . TransformName n r => Located (IdP n) -> [LHsType n] -> Trf (Ann AST.UInstanceRule (Dom r) RangeStage)
  310 makeInstanceRuleTyVars n vars
  311   | isSymOcc (occName @n (unLoc n))
  312   , leftOp:rest <- vars
  313   , srcSpanStart (getLoc n) > srcSpanEnd (getLoc leftOp)
  314   = annContNoSema
  315       $ AST.UInstanceRule <$> nothing "" " . " atTheStart
  316                           <*> nothing " " "" atTheStart
  317                           <*> foldl foldTypeArgs
  318                                     (annLocNoSema (pure $ combineSrcSpans (getLoc leftOp) (getLoc n))
  319                                       (AST.UInstanceHeadInfix <$> trfType leftOp <*> trfOperator @n n)) rest
  320   | otherwise
  321   = annContNoSema
  322       $ AST.UInstanceRule <$> nothing "" " . " atTheStart
  323                           <*> nothing " " "" atTheStart

  373 createDeclHead :: forall n r . TransformName n r => Located (IdP n) -> LHsQTyVars n -> Trf (Ann AST.UDeclHead (Dom r) RangeStage)

  389 -- | Get the parentheses directly before and after (for parenthesized application)
  390 addParenLocs :: SrcSpan -> Trf SrcSpan

  392   = let possibleSpan = mkSrcSpan (updateCol (subtract 1) (srcSpanStart sp)) (updateCol (+1) (srcSpanEnd sp))
  393      in local (\s -> s { contRange = possibleSpan })
  394               (combineSrcSpans <$> (combineSrcSpans sp <$> tokenLoc AnnOpenP) <*> tokenLocBack AnnCloseP)

  397 createClassBody :: TransformName n r => [LSig n] -> LHsBinds n -> [LFamilyDecl n]
  398                                -> [LTyFamDefltEqn n] -> Trf (AnnMaybeG AST.UClassBody (Dom r) RangeStage)

  400   = do isThereWhere <- isGoodSrcSpan <$> (tokenLoc AnnWhere)

  402          then makeJust <$> annLocNoSema (combinedLoc <$> tokenLoc AnnWhere)

  407         allLocs = map getLoc sigs ++ map getLoc (bagToList binds) ++ map getLoc typeFams ++ map getLoc typeFamDefs

  409         getSigs = mapM trfClassElemSig sigs
  410         getBinds = mapM (copyAnnot AST.UClsDef . trfBind) (bagToList binds)
  411         getFams = mapM (copyAnnot AST.UClsTypeFam . trfTypeFam) typeFams
  412         getFamDefs = mapM trfTypeFamDef typeFamDefs

  472 trfInstDataFam :: forall n r . TransformName n r => Located (DataFamInstDecl n) -> Trf (Ann AST.UInstBodyDecl (Dom r) RangeStage)
  473 trfInstDataFam = trfLocNoSema $ \case
  474   (DataFamInstDecl (hsib_body -> FamEqn tc pats _ (HsDataDefn dn ctx _ ks cons derivs)))
  475     | all ((\case ConDeclH98{} -> True; _ -> False) . unLoc) cons
  476     -> AST.UInstBodyDataDecl
  477          <$> trfDataKeyword dn
  478          <*> annLocNoSema (pure $ collectLocs pats `combineSrcSpans` getLoc tc `combineSrcSpans` getLoc ctx)
  479                           (AST.UInstanceRule <$> nothing "" " . " atTheStart
  480                                              <*> trfCtx atTheStart ctx
  481                                              <*> transformNameAndPats tc pats)
  482          <*> trfAnnList "" trfConDecl' cons
  483          <*> makeIndentedList atTheEnd (mapM trfDerivings (unLoc derivs))
  484     | otherwise
  485     -> AST.UInstBodyGadtDataDecl
  486         <$> trfDataKeyword dn
  487         <*> annLocNoSema (pure $ collectLocs pats `combineSrcSpans` getLoc tc `combineSrcSpans` getLoc ctx)
  488                          (AST.UInstanceRule <$> nothing "" " . " atTheStart
  489                                             <*> trfCtx atTheStart ctx
  490                                             <*> transformNameAndPats tc pats)
  491         <*> trfKindSig ks
  492         <*> trfAnnList "" trfGADTConDecl' cons
  493         <*> makeIndentedList atTheEnd (mapM trfDerivings (unLoc derivs))
  494   where transformNameAndPats :: Located (IdP n) -> [LHsType n] -> Trf (Ann AST.UInstanceHead (Dom r) RangeStage)
  495         transformNameAndPats tc pats -- TODO: this is simpler with lexical fixity
  496           | all (\p -> srcSpanEnd (getLoc tc) < srcSpanStart (getLoc p)) pats -- prefix instance head application
  497           = foldl (\r t -> annLocNoSema (combineSrcSpans (getLoc t) . getRange <$> r)
  498                                           (AST.UInstanceHeadApp <$> r <*> (trfType t)))
  499                   (copyAnnot AST.UInstanceHeadCon (trfName @n tc)) pats
  500         transformNameAndPats tc (p:rest)
  501           | otherwise -- infix instance head application
  502           = foldl (\r t -> annLocNoSema (combineSrcSpans (getLoc t) . getRange <$> r)
  503                                           (AST.UInstanceHeadApp <$> r <*> (trfType t)))
  504                   (annLocNoSema (pure $ getLoc p `combineSrcSpans` getLoc tc)
  505                           (AST.UInstanceHeadInfix <$> trfType p <*> trfOperator @n tc)) rest

  507 trfPatternSynonym :: forall n r . TransformName n r => PatSynBind n n -> Trf (AST.UPatternSynonym (Dom r) RangeStage)
  508 trfPatternSynonym (PSB id _ lhs def dir)
  509   = let sep = case dir of ImplicitBidirectional -> AnnEqual
  510                           _                     -> AnnLarrow
  511         rhsLoc = combineSrcSpans (getLoc def) <$> tokenLoc sep
  512         -- we use the selector name instead of the pattern variable name
  513         rewrites = case lhs of RecCon flds -> map (\r -> (unLoc (recordPatSynPatVar r), unLoc (recordPatSynSelectorId r))) flds
  514                                _           -> []
  515         changedRhs = biplateRef .- (\n -> case lookup n rewrites of Just x -> x; Nothing -> n) $ def
  516      in AST.UPatternSynonym <$> trfPatSynLhs id lhs sep
  517                             <*> annLocNoSema rhsLoc (trfPatSynRhs dir changedRhs)
  518 
  519   where trfPatSynLhs :: Located (IdP n) -> HsPatSynDetails (Located (IdP n)) -> AnnKeywordId -> Trf (Ann AST.UPatSynLhs (Dom r) RangeStage)
  520         trfPatSynLhs id (PrefixCon args) _
  521           = annLocNoSema (pure $ foldLocs (getLoc id : map getLoc args)) $ AST.UNormalPatSyn <$> define (trfName @n id) <*> trfAnnList " " (trfName' @n) args
  522         trfPatSynLhs op (InfixCon lhs rhs) _
  523           = annLocNoSema (pure $ getLoc lhs `combineSrcSpans` getLoc rhs) $ AST.UInfixPatSyn <$> define (trfName @n lhs) <*> trfOperator @n op <*> trfName @n rhs
  524         trfPatSynLhs id (RecCon flds) kw
  525           = annLocNoSema (mkSrcSpan (srcSpanStart (getLoc id)) <$> before kw)
  526               $ AST.URecordPatSyn <$> define (trfName @n id) <*> trfAnnList ", " (trfName' @n) (map recordPatSynSelectorId flds)
  527 
  528         trfPatSynRhs :: HsPatSynDir n -> Located (Pat n) -> Trf (AST.UPatSynRhs (Dom r) RangeStage)
  529         trfPatSynRhs ImplicitBidirectional pat = AST.UBidirectionalPatSyn <$> trfPattern pat <*> nothing " where " "" atTheEnd

  531         trfPatSynRhs Unidirectional pat = AST.UOneDirectionalPatSyn <$> trfPattern pat

  533         trfPatSynWhere :: MatchGroup n (LHsExpr n) -> Trf (Ann AST.UPatSynWhere (Dom r) RangeStage)
  534         trfPatSynWhere (MG { mg_alts = alts }) = annLocNoSema (pure $ getLoc alts) (AST.UPatSynWhere <$> makeIndentedList (after AnnWhere) (mapM (trfMatch (unLoc id)) (unLoc alts)))
