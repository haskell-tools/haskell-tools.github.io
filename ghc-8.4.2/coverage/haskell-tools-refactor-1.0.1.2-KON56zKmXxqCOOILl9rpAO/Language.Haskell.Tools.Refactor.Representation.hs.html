   20 -- | A type for the input and result of refactoring a module
   21 type UnnamedModule = Ann AST.UModule IdDom SrcTemplateStage
   22 
   23 -- | The name of the module and the AST
   24 type ModuleDom = (SourceFileKey, UnnamedModule)
   25 
   26 -- | Module name and marker to separate .hs-boot module definitions. Specifies a source file in a working directory.
   27 data SourceFileKey = SourceFileKey { _sfkFileName :: FilePath
   28                                    , _sfkModuleName :: String
   29                                    }
   30   deriving (Eq, Ord, Show)
   31 
   32 -- | Change in the project, modification or removal of a module.
   33 data RefactorChange = ContentChanged { fromContentChanged :: ModuleDom }
   34                     | ModuleRemoved { removedModuleName :: String }
   35                     | ModuleCreated { createdModuleName :: String
   36                                     , createdModuleContent :: UnnamedModule
   37                                     , sameLocation :: SourceFileKey
   38                                     }
   39 instance Show RefactorChange where
   40   show (ContentChanged (n, _)) = "ContentChanged (" ++ show n  ++ ")"
   41   show (ModuleRemoved n) = "ModuleRemoved " ++ n
   42   show (ModuleCreated n _ other) = "ModuleCreated " ++ n ++ " (" ++ show other ++ ")"
   43 
   44 -- | Exceptions that can occur while loading modules or during internal operations (not during performing the refactor).
   45 data RefactorException = IllegalExtensions [String]
   46                        | SourceCodeProblem ErrorMessages
   47                        | UnknownException String
   48   deriving (Show, Typeable)
   49 
   50 instance Show ErrorMessages where
   51   show = show . bagToList
   52 
   53 instance Exception RefactorException where
   54   displayException (SourceCodeProblem prob)
   55     = "Source code problem: " ++ showSDocUnsafe (vcat (pprErrMsgBagWithLoc prob))
   56   displayException (IllegalExtensions exts)
   57     = "The following extensions are not allowed: " ++ (concat $ intersperse ", " exts) ++ "."
   58   displayException (UnknownException ex) = "An unexpected problem appeared: " ++ ex ++ "."
   59 
   60 -- | Transforms module name to a .hs file name relative to the source root directory.
   61 moduleSourceFile :: String -> FilePath
   62 moduleSourceFile m = (foldl1 (</>) (splitOn "." m)) <.> "hs"
   63 
   64 -- | Transforms a source root relative file name into module name.
   65 sourceFileModule :: FilePath -> String
   66 sourceFileModule fp = intercalate "." $ splitDirectories $ dropExtension fp
