   30 -- | Performs the given refactoring, transforming it into a Ghc action
   31 runRefactor :: ModuleDom -> [ModuleDom] -> Refactoring -> Ghc (Either String [RefactorChange])
   32 runRefactor mod mods trf = runExceptT $ trf mod mods

   34 -- | Wraps a refactoring that only affects one module. Performs the per-module finishing touches.
   35 localRefactoring :: LocalRefactoring -> Refactoring
   36 localRefactoring ref (name, mod) _
   37   = (\m -> [ContentChanged (name, m)]) <$> localRefactoringRes id mod (ref mod)
   38 
   39 -- | Transform the result of the local refactoring
   40 localRefactoringRes :: ((UnnamedModule -> UnnamedModule) -> a -> a)
   41                           -> UnnamedModule -> LocalRefactor a -> Refactor a
   42 localRefactoringRes access mod trf
   43   = let init = RefactorCtx (semanticsModule $ mod ^. semantics) mod (mod ^? modImports&annList)
   44      in flip runReaderT init $ do (mod, recorded) <- runWriterT (fromRefactorT trf)
   45                                   return $ access (insertText (rights recorded) . addGeneratedImports (lefts recorded)) mod

   47 -- | Re-inserts the elements removed from the AST that should be kept (for example preprocessor directives)
   48 insertText :: SourceInfoTraversal p => [(SrcSpan,String,String)] -> p dom SrcTemplateStage -> p dom SrcTemplateStage
   49 insertText [] p = p
   50 insertText inserted p
   51   -- this traverses the AST and finds the positions where the removed elements can be added
   52   = evalState (sourceInfoTraverseUp (SourceInfoTrf
   53                 (\stn -> sourceTemplateNodeElems !~ takeWhatPrecedesElem (stn ^. sourceTemplateNodeRange) $ stn)
   54                 (srcTmpSeparators !~ takeWhatPrecedesSep)
   55                 pure) (return ()) (return ()) p) (map Right $ sortOn (^. _1) inserted)
   56   where
   57    -- insert fragments into list separators
   58    takeWhatPrecedesSep :: [([SourceTemplateTextElem], SrcSpan)] -> State [Either SrcSpan (SrcSpan,String,String)] [([SourceTemplateTextElem], SrcSpan)]
   59    takeWhatPrecedesSep seps = takeWhatPrecedes Nothing (Just . (^. _2))
   60                                                (\str -> _1 .- (++ [StayingText str ""]))
   61                                                (\str -> _1 .- ([StayingText str ""] ++))
   62                                                seps
   63 
   64    -- insert fragments into AST elements
   65    takeWhatPrecedesElem :: SrcSpan -> [SourceTemplateElem] -> State [Either SrcSpan (SrcSpan,String,String)] [SourceTemplateElem]
   66    takeWhatPrecedesElem rng elems = takeWhatPrecedes (Just rng) (^? sourceTemplateTextRange)
   67                                                                 (\s -> sourceTemplateTextElem .- (++ [StayingText s ""]))
   68                                                                 (\s -> sourceTemplateTextElem .- ([StayingText s ""] ++))
   69                                                                 elems
   70 
   71    -- finds the position of the fragment where there are elements in the template both before and after the fragment
   72    -- puts holes into the list of inserted fragments where child elements are located
   73    -- uses these holes to determine where should the fragment be added
   74    takeWhatPrecedes :: Maybe SrcSpan -> (a -> Maybe SrcSpan) -> (String -> a -> a) -> (String -> a -> a) -> [a] -> State [Either SrcSpan (SrcSpan,String,String)] [a]
   75    takeWhatPrecedes rng access append prepend elems
   76      | ranges <- mapMaybe access elems
   77      , not (null ranges)
   78      = do let start = srcSpanStart $ fromMaybe (head ranges) rng
   79               end = srcSpanEnd $ fromMaybe (last ranges) rng
   80           toInsert <- get
   81           let (prefix,rest) = break ((>= start) . srcSpanStart . either id (\(sp,_,_) -> sp)) toInsert
   82               (middle,suffix) = break ((> end) . srcSpanEnd . either id (\(sp,_,_) -> sp)) rest
   83           put $ prefix ++ Left (mkSrcSpan start end) : suffix
   84           return $ mergeInserted access append prepend False middle elems
   85      where mergeInserted :: (a -> Maybe SrcSpan) -> (String -> a -> a) -> (String -> a -> a) -> Bool -> [Either SrcSpan (SrcSpan,String,String)] -> [a] -> [a]
   86            -- no fragments left
   87            mergeInserted _ _ _ _ [] elems = elems
   88            mergeInserted access append prepend prep insert@(Right (insertSpan,insertStr,ln):toInsert) (fstElem:elems)
   89               -- insert a fragment to the end of the current element if the next elment is after the fragment
   90               | Just _ <- access fstElem -- TODO: is this needed?
   91               , not prep && case mapMaybe access elems of sp:_ -> srcSpanStart sp >= srcSpanEnd insertSpan
   92                                                           _ -> True
   93               = mergeInserted access append prepend prep toInsert (append (ln ++ insertStr ++ ln) fstElem : elems)
   94               -- insert the fragment before the current elem if we need an element before (we skipped a child)
   95               -- and the current element is after the inserted fragment
   96               | Just fstElemSpace <- access fstElem
   97               , prep && srcSpanStart fstElemSpace >= srcSpanEnd insertSpan
   98               = mergeInserted access append prepend prep toInsert (prepend (ln ++ insertStr ++ ln) fstElem : elems)
   99               | isJust (access fstElem) && prep
  100               = mergeInserted access append prepend False insert (fstElem : elems) -- switch back to append mode
  101               | otherwise
  102               = fstElem : mergeInserted access append prepend (if isJust (access fstElem) then False else prep) insert elems -- move on and switch back to append mode
  103            -- when found a hole
  104            mergeInserted access append prepend prep insert@(Left sp : toInsert) (fstElem:elems)
  105               | Just fstElemSpace <- access fstElem
  106               = if srcSpanStart fstElemSpace > srcSpanEnd sp
  107                   -- switch to prepend mode
  108                   then mergeInserted access append prepend True toInsert (fstElem:elems)
  109                   -- skip elements that are not after the fragment
  110                   else fstElem : mergeInserted access append prepend prep insert elems
  111               | otherwise
  112               = fstElem : mergeInserted access append prepend True toInsert elems -- switch to prepend mode and move on
  113            mergeInserted _ _ _ _ _ [] = [] -- maybe error
  114    takeWhatPrecedes _ _ _ _ elems = return elems

  116 -- | Adds the imports that bring names into scope that are needed by the refactoring
  117 addGeneratedImports :: [GHC.Name] -> HT.Module -> HT.Module
  118 addGeneratedImports names m = modImports&annListElems .- (++ addImports names) $ m
  119   where addImports :: [GHC.Name] -> [HT.ImportDecl]
  120         addImports names = map createImport $ groupBy ((==) `on` GHC.nameModule) $ filter (isJust . GHC.nameModule_maybe) $ nub $ sort names
  121 
  122         -- TODO: group names like constructors into correct IESpecs
  123         createImport :: [GHC.Name] -> HT.ImportDecl
  124         -- works on groupby result, so list is nonempty
  125         createImport names = mkImportDecl False True False Nothing (mkModuleName $ GHC.moduleNameString $ GHC.moduleName $ GHC.nameModule $ head names)
  126                                           Nothing (Just $ mkImportSpecList (map (\n -> mkIESpec (mkUnqualName' n) Nothing) names))

  138 qualifiedName :: GHC.Name -> String
  139 qualifiedName name = case GHC.nameModule_maybe name of
  140   Just mod -> GHC.moduleNameString (GHC.moduleName mod) ++ "." ++ GHC.occNameString (GHC.nameOccName name)
  141   Nothing -> GHC.occNameString (GHC.nameOccName name)

  149 -- | Create a name that references the definition. Generates an import if the definition is not yet imported.
  150 referenceName' :: ([String] -> GHC.Name -> Ann nt IdDom SrcTemplateStage) -> GHC.Name
  151                     -> LocalRefactor (Ann nt IdDom SrcTemplateStage)
  152 referenceName' makeName name
  153   | name `elem` registeredNamesFromPrelude || qualifiedName name `elem` otherNamesFromPrelude
  154   = return $ makeName [] name -- imported from prelude
  155   | otherwise
  156   = do RefactorCtx {refCtxRoot = mod, refCtxImports = imports, refModuleName = thisModule} <- ask
  157        if maybe True (thisModule ==) (GHC.nameModule_maybe name)
  158          then return $ makeName [] name -- in the same module, use simple name
  159          else let possibleImports = filter ((name `elem`) . (\imp -> semanticsImported $ imp ^. semantics)) imports
  160                   fromPrelude = name `elem` semanticsImplicitImports (mod ^. semantics)
  161                in if | fromPrelude -> return $ makeName [] name
  162                      | null possibleImports -> do tell [Left name]
  163                                                   return $ makeName (moduleParts name) name
  164                      | otherwise -> return $ referenceBy makeName name possibleImports
  165                                      -- use it according to the best available import
  166   where moduleParts = maybe [] (splitOn "." . GHC.moduleNameString . GHC.moduleName) . GHC.nameModule_maybe
  167 
  168 -- | Reference the name by the shortest suitable import
  169 referenceBy :: ([String] -> GHC.Name -> Ann nt IdDom SrcTemplateStage) -> GHC.Name
  170                  -> [Ann UImportDecl IdDom SrcTemplateStage] -> Ann nt IdDom SrcTemplateStage
  171 referenceBy makeName name imps =
  172   let prefixes = map importQualifier imps
  173    in makeName (minimumBy (compare `on` (length . concat)) prefixes) name
  174   where importQualifier :: Ann UImportDecl IdDom SrcTemplateStage -> [String]
  175         importQualifier imp
  176           = if isJust (imp ^? importQualified&annJust)
  177               then case imp ^? importAs&annJust&importRename of
  178                       Nothing -> splitOn "." (imp ^. importModule&moduleNameString) -- fully qualified import
  179                       Just asName -> splitOn "." (asName ^. moduleNameString) -- the name given by as clause
  180               else [] -- unqualified import
