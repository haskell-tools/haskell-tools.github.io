    4 -- | Types and instances for monadic refactorings. The refactoring monad provides automatic

   25 -- | A monad that can be used to refactor
   26 class Monad m => RefactorMonad m where
   27   refactError :: String -> m a
   28   liftGhc :: Ghc a -> m a
   29 
   30 -- | A refactoring that only affects one module
   31 type LocalRefactoring = UnnamedModule -> LocalRefactor UnnamedModule
   32 
   33 -- | The type of a refactoring
   34 type Refactoring = ModuleDom -> [ModuleDom] -> Refactor [RefactorChange]
   35 
   36 -- | The type of a refactoring that affects the whole project.
   37 type ProjectRefactoring = [ModuleDom] -> Refactor [RefactorChange]
   38 
   39 -- | The refactoring monad for a given module
   40 type LocalRefactor = LocalRefactorT Refactor
   41 
   42 -- | The refactoring monad for the whole project
   43 type Refactor = ExceptT String Ghc
   44 
   45 -- | Input and output information for the refactoring
   46 -- TODO: use multiple states instead of Either
   47 newtype LocalRefactorT m a
   48   = LocalRefactorT { fromRefactorT :: WriterT [Either GHC.Name (SrcSpan, String, String)]
   49                                               (ReaderT RefactorCtx m) a
   50                    }
   51   deriving ( Functor, Applicative, Monad, MonadReader RefactorCtx
   52            , MonadWriter [Either GHC.Name (SrcSpan, String, String)]
   53            , MonadIO, HasDynFlags, ExceptionMonad, GhcMonad )
   54 
   55 -- | The information a refactoring can use
   56 data RefactorCtx
   57   = RefactorCtx { refModuleName :: GHC.Module -- ^ The name of the module being refactored. Used for accessing implicit imports.
   58                 , refCtxRoot :: Ann UModule IdDom SrcTemplateStage
   59                 , refCtxImports :: [Ann UImportDecl IdDom SrcTemplateStage]
   60                 }
   61 
   62 instance MonadTrans LocalRefactorT where
   63   lift = LocalRefactorT . lift . lift
   64 
   65 instance RefactorMonad Refactor where
   66   refactError = throwE
   67   liftGhc = lift
   68 
   69 instance RefactorMonad LocalRefactor where
   70   refactError = lift . refactError
   71   liftGhc = lift . liftGhc
   72

   77 instance RefactorMonad m => RefactorMonad (LazySt.StateT s m) where
