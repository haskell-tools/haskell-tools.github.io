    4 -- | This module declares the messages that can be sent from the client to the
    5 -- daemon engine and from the engine to the client.
    6 module Language.Haskell.Tools.Daemon.Protocol

    8   , Marker(..), Severity(..)

   20 -- | The messages expected from the client.
   21 data ClientMessage
   22   = KeepAlive -- ^ A simple ping message to check that the server is running.
   23   | Reset -- ^ A message that instructs the server to reset its internal state and re-load loaded packages.
   24   | Handshake { clientVersion :: [Int] }
   25     -- ^ Tells the client version and asks the servers version.
   26   | SetPackageDB { pkgDB :: PackageDB }
   27     -- ^ Sets the package database for the engine to use.
   28   | AddPackages { addedPathes :: [FilePath] }
   29     -- ^ Registers packages to the engine. They will be subject to subsequent
   30     -- refactorings. Will cause the packages to be loaded, resulting in
   31     -- LoadingModules, LoadedModule or CompilationProblem responses.
   32   | RemovePackages { removedPathes :: [FilePath] }
   33     -- ^ Deregisters the given packages from the engine. They will not be
   34     -- subjects of further refactorings.
   35   | SetWorkingDir { newWorkingDir :: FilePath }
   36     -- ^ Sets the working directory for the compilation. Important when
   37     -- compiling code that loads resources based on relative pathes.
   38   | SetGHCFlags { ghcFlags :: [String] }
   39     -- ^ Sets the compilation flags. The unused flags are returned via the
   40     -- UnusedFlags response.
   41   | PerformRefactoring { refactoring :: String
   42                        , modulePath :: FilePath
   43                        , editorSelection :: String
   44                        , details :: [String] -- ^ Additional details for the refactoring like the
   45                                              -- names of generated definitions.
   46                        , shutdownAfter :: Bool -- ^ Stop the daemon after performing the refactoring.
   47                        , diffMode :: Bool -- ^ Don't change the files, send back the result as
   48                                           -- a unified diff.
   49                        }
   50     -- ^ Orders the engine to perform the refactoring on the module given
   51     -- with the selection and details. Successful refactorings will cause re-loading of modules.
   52     -- If 'shutdownAfter' or 'diffMode' is not set, after the refactoring,
   53     -- modules are re-loaded, LoadingModules, LoadedModule responses are sent.
   54   | PerformQuery { query :: String
   55                  , modulePath :: FilePath
   56                  , editorSelection :: String
   57                  , details :: [String] -- ^ Additional details for the refactoring like the
   58                                        -- names of generated definitions.
   59                  , shutdownAfter :: Bool -- ^ Stop the daemon after performing the query.
   60                  }
   61     -- ^ Orders the engine to perform a query on the module given with the selection and details.
   62   | UndoLast
   63     -- ^ Asks the daemon to undo the last refactoring.
   64   | Disconnect
   65     -- ^ Stops the engine. It replies with Disconnected.
   66   | ReLoad { addedModules :: [FilePath]
   67            , changedModules :: [FilePath]
   68            , removedModules :: [FilePath]
   69            }
   70     -- ^ Instructs the engine to re-load a changed module.
   71     -- LoadingModules, LoadedModule responses may be sent.
   72   | Stop -- TODO: remove
   73     -- ^ Stops the server. OBSOLATE
   74   deriving (Show, Generic)
   75 
   76 instance FromJSON ClientMessage
   77 
   78 -- | The possible responses that the server can give.
   79 data ResponseMsg
   80   = KeepAliveResponse -- ^ A response to KeepAlive
   81   | HandshakeResponse { serverVersion :: [Int] }
   82     -- ^ Tells the version of the server.
   83   | ErrorMessage { errorMsg :: String }
   84     -- ^ An error message marking internal problems or user mistakes.
   85     -- TODO: separate internal problems and user mistakes.
   86   | CompilationProblem { markers :: [Marker]
   87                        , errorHints :: [String]
   88                        }
   89     -- ^ A response that tells there are errors in the source code given.
   90   | DiffInfo { diffInfo :: String }
   91     -- ^ Information about changes that would be caused by the refactoring.
   92   | LoadingModules { modulesToLoad :: [FilePath] }
   93     -- ^ The traversal of the project is done, now the engine is loading the
   94     -- given modules.
   95   | LoadedModule { loadedModulePath :: FilePath
   96                  , loadedModuleName :: String
   97                  }
   98     -- ^ The engine has loaded the given module.
   99   | QueryResult { queryName :: String
  100                 , queryType :: String
  101                 , queryResult :: Value
  102                 }
  103     -- ^ The result of querying the program representation.
  104   | UnusedFlags { unusedFlags :: [String] }
  105     -- ^ Returns the flags that are not used by the engine.
  106   | Disconnected
  107     -- ^ The engine has closed the connection.
  108   deriving (Show, Generic)
  109 
  110 instance ToJSON ResponseMsg

  112 data UndoRefactor = RemoveAdded { undoRemovePath :: FilePath }
  113                   | RestoreRemoved { undoRestorePath :: FilePath
  114                                    , undoRestoreContents :: String

  116                   | UndoChanges { undoChangedPath :: FilePath
  117                                 , undoDiff :: FileDiff
  118                                 }
  119   deriving (Show, Generic, NFData)
