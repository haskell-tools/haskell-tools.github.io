    1 -- | Generation of pattern-level AST fragments for refactorings.
    2 -- The bindings defined here create a the annotated version of the AST constructor with the same name.
    3 -- For example, @mkVarPat@ creates the annotated version of the @VarPat@ AST constructor.
    4 {-# LANGUAGE MonoLocalBinds #-}
    5 {-# LANGUAGE OverloadedStrings #-}
    6 
    7 module Language.Haskell.Tools.Rewrite.Create.Patterns where
    8 
    9 import Language.Haskell.Tools.AST (UFieldWildcard(..), UPatternField(..), UPattern(..))
   10 import Language.Haskell.Tools.PrettyPrint.Prepare
   11 import Language.Haskell.Tools.Rewrite.Create.Utils (mkAnn, mkAnnList)
   12 import Language.Haskell.Tools.Rewrite.ElementTypes
   13 
   14 -- | Pattern name binding
   15 mkVarPat :: Name -> Pattern
   16 mkVarPat = mkAnn child . UVarPat
   17 
   18 -- | Literal pattern
   19 mkLitPat :: Literal -> Pattern
   20 mkLitPat = mkAnn child . ULitPat
   21 
   22 -- | Infix constructor application pattern (@ a :+: b @)
   23 mkInfixAppPat :: Pattern -> Operator -> Pattern -> Pattern
   24 mkInfixAppPat lhs op rhs = mkAnn (child <> " " <> child <> " " <> child) $ UInfixAppPat lhs op rhs
   25 
   26 -- | Constructor application pattern (@ Point x y @)
   27 mkAppPat :: Name -> [Pattern] -> Pattern
   28 mkAppPat n pat = mkAnn (child <> child) $ UAppPat n (mkAnnList (after " " $ separatedBy " " list) pat)
   29 
   30 -- | Tuple pattern (@ (x,y) @)
   31 mkTuplePat :: [Pattern] -> Pattern
   32 mkTuplePat pats = mkAnn ("(" <> child <> ")") $ UTuplePat (mkAnnList (separatedBy ", " list) pats)
   33 
   34 -- | Unboxed tuple pattern (@ (\# x, y \#) @)
   35 mkUnboxTuplePat :: [Pattern] -> Pattern
   36 mkUnboxTuplePat pats = mkAnn ("(# " <> child <> " #)") $ UUnboxTuplePat (mkAnnList (separatedBy ", " list) pats)
   37 
   38 -- | List pattern (@ [1,2,a,x] @)
   39 mkListPat :: [Pattern] -> Pattern
   40 mkListPat pats = mkAnn ("[" <> child <> "]") $ UListPat (mkAnnList (separatedBy ", " list) pats)
   41 
   42 -- | Parallel array pattern (@ [:1,2,a,x:] @)
   43 mkParArrayPat :: [Pattern] -> Pattern
   44 mkParArrayPat pats = mkAnn ("[:" <> child <> ":]") $ UParArrPat (mkAnnList (separatedBy ", " list) pats)
   45 
   46 -- | Parenthesised patterns
   47 mkParenPat :: Pattern -> Pattern
   48 mkParenPat = mkAnn ("(" <> child <> ")") . UParenPat
   49 
   50 -- | Record pattern (@ Point { x = 3, y } @)
   51 mkRecPat :: Name -> [PatternField] -> Pattern
   52 mkRecPat name flds = mkAnn (child <> "{ " <> child <> " }") $ URecPat name (mkAnnList (separatedBy ", " list) flds)
   53 
   54 -- | As-pattern (explicit name binding) (@ ls\@(hd:_) @)
   55 mkAsPat :: Name -> Pattern -> Pattern
   56 mkAsPat name pat = mkAnn (child <> "@" <> child) $ UAsPat name pat
   57 
   58 -- | Wildcard pattern: (@ _ @)
   59 mkWildPat :: Pattern
   60 mkWildPat = mkAnn "_" UWildPat
   61 
   62 -- | Irrefutable pattern (@ ~(x:_) @)
   63 mkIrrefutablePat :: Pattern -> Pattern
   64 mkIrrefutablePat = mkAnn ("~" <> child) . UIrrefutablePat
   65 
   66 -- | Bang pattern (@ !x @)
   67 mkBangPat :: Pattern -> Pattern
   68 mkBangPat = mkAnn ("!" <> child) . UBangPat
   69 
   70 -- | Pattern with explicit type signature (@ x :: Int @)
   71 mkTypeSigPat :: Pattern -> Type -> Pattern
   72 mkTypeSigPat pat typ = mkAnn (child <> " :: " <> child) $ UTypeSigPat pat typ
   73 
   74 -- | View pattern (@ f -> Just 1 @)
   75 mkViewPat :: Expr -> Pattern -> Pattern
   76 mkViewPat name pat = mkAnn (child <> " -> " <> child) $ UViewPat name pat
   77 
   78 -- | Splice patterns: @$(generateX inp)@
   79 mkSplicePat :: Splice -> Pattern
   80 mkSplicePat = mkAnn child . USplicePat
   81 
   82 -- | Quasi-quoted patterns: @[| 1 + 2 |]@
   83 mkQuasiQuotePat :: QuasiQuote -> Pattern
   84 mkQuasiQuotePat = mkAnn child . UQuasiQuotePat
   85 
   86 -- | Named field pattern (@ p = Point 3 2 @)
   87 mkPatternField :: Name -> Pattern -> PatternField
   88 mkPatternField name pat = mkAnn (child <> " = " <> child) $ UNormalFieldPattern name pat
   89 
   90 -- | Named field pun (@ p @)
   91 mkFieldPunPattern :: Name -> PatternField
   92 mkFieldPunPattern name = mkAnn child $ UFieldPunPattern name
   93 
   94 -- | Wildcard field pattern (@ .. @)
   95 mkFieldWildcardPattern :: PatternField
   96 mkFieldWildcardPattern = mkAnn child $ UFieldWildcardPattern $ mkAnn ".." FldWildcard
