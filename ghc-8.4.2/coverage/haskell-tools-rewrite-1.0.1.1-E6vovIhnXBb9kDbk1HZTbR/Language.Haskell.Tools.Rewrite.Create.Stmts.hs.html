    1 -- | Generation of statement-level AST fragments for refactorings.
    2 -- The bindings defined here are the AST constructor names with an "mk" prefix.
    3 {-# LANGUAGE MonoLocalBinds #-}
    4 {-# LANGUAGE OverloadedStrings #-}
    5 
    6 module Language.Haskell.Tools.Rewrite.Create.Stmts where
    7 
    8 import Language.Haskell.Tools.AST (UCompStmt(..), UListCompBody(..), UStmt'(..))
    9 import Language.Haskell.Tools.PrettyPrint.Prepare
   10 import Language.Haskell.Tools.Rewrite.Create.Utils (mkAnn, mkAnnList, mkAnnMaybe)
   11 import Language.Haskell.Tools.Rewrite.ElementTypes
   12 
   13 -- | Creates a binding statement (@ x <- action @)
   14 mkBindStmt :: Pattern -> Expr -> Stmt
   15 mkBindStmt bound expr = mkAnn (child <> " <- " <> child) $ UBindStmt bound expr
   16 
   17 -- | Creates a non-binding statement (@ action @)
   18 mkExprStmt :: Expr -> Stmt
   19 mkExprStmt = mkAnn child . UExprStmt
   20 
   21 -- | Creates a let statement (@ let x = 3; y = 4 @)
   22 mkLetStmt :: [LocalBind] -> Stmt
   23 mkLetStmt = mkAnn ("let " <> child) . ULetStmt . mkAnnList (indented list)
   24 
   25 -- | Creates a recursive binding statement with (@ rec b <- f a c; c <- f b a @)
   26 mkRecStmt :: [Stmt] -> Stmt
   27 mkRecStmt = mkAnn ("rec " <> child) . URecStmt . mkAnnList (indented list)
   28 
   29 -- * List comprehensions
   30 
   31 -- | Body of a list comprehension: (@ | x <- [1..10] @)
   32 mkListCompBody :: [CompStmt] -> ListCompBody
   33 mkListCompBody = mkAnn child . UListCompBody . mkAnnList (separatedBy " " list)
   34 
   35 -- | Normal monadic statement of a list comprehension
   36 mkCompStmt :: Stmt -> CompStmt
   37 mkCompStmt = mkAnn child . UCompStmt
   38 
   39 -- | Then statements by @TransformListComp@ (@ then sortWith by (x + y) @)
   40 mkThenStmt :: Expr -> Maybe Expr -> CompStmt
   41 mkThenStmt th by = mkAnn ("then " <> child) 
   42                      $ UThenStmt th $ mkAnnMaybe (after " by " opt) by
   43 
   44 -- | Grouping statements by @TransformListComp@ (@ then group by (x + y) using groupWith @) 
   45 mkGroupStmt :: Maybe Expr -> Maybe Expr -> CompStmt
   46 mkGroupStmt by using = mkAnn ("then group" <> child) 
   47                          $ UGroupStmt (mkAnnMaybe (after " by " opt) by)
   48                                       (mkAnnMaybe (after " using " opt) using)
   49 
   50 -- * Commands

   52 -- | Creates a binding command (@ x <- action @)
   53 mkBindCmd :: Pattern -> Cmd -> CmdStmt
   54 mkBindCmd bound expr = mkAnn (child <> " <- " <> child) $ UBindStmt bound expr
   55 
   56 -- | Creates a non-binding command (@ action @)
   57 mkExprCmd :: Cmd -> CmdStmt
   58 mkExprCmd = mkAnn child . UExprStmt
   59 
   60 -- | Creates a let command (@ let x = 3; y = 4 @)
   61 mkLetStmtCmd :: [LocalBind] -> CmdStmt
   62 mkLetStmtCmd = mkAnn ("let " <> child) . ULetStmt . mkAnnList (indented list)
   63 
   64 -- | Creates a recursive binding command with (@ rec b <- f a c; c <- f b a @)
   65 mkRecCmd :: [CmdStmt] -> CmdStmt
   66 mkRecCmd = mkAnn ("rec " <> child) . URecStmt . mkAnnList (indented list)
