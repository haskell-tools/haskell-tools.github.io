    1 -- | Generation of UModule-level AST fragments for refactorings.
    2 -- The bindings defined here create a the annotated version of the AST constructor with the same name.
    3 -- For example, @mkModule@ creates the annotated version of the @UModule@ AST constructor.
    4 {-# LANGUAGE MonoLocalBinds, OverloadedStrings #-}
    5 
    6 module Language.Haskell.Tools.Rewrite.Create.Modules where

    8 import Data.String (IsString(..), String)

   15 -- | The representation of a haskell module, that is a separate compilation unit.
   16 -- It may or may not have a header.
   17 mkModule :: [FilePragma] -> Maybe ModuleHead -> [ImportDecl] -> [Decl] -> Module
   18 mkModule filePrags head imps decls
   19   = mkAnn (child <> child <> child <> child)
   20       $ UModule (mkAnnList (followedBy "\n" $ separatedBy "\n" list) filePrags) (mkAnnMaybe opt head)
   21                 (mkAnnList (after "\n" $ indented list) imps) (mkAnnList (after "\n" $ indented list) decls)
   22 
   23 -- | Module declaration with name and (optional) exports
   24 mkModuleHead :: ModuleName -> Maybe ModulePragma -> Maybe ExportSpecs -> ModuleHead
   25 mkModuleHead n pr es = mkAnn ("module " <> child <> child <> child <> " where")
   26                          $ UModuleHead n (mkAnnMaybe (after "\n" opt) pr) (mkAnnMaybe opt es)
   27 
   28 -- | A list of export specifications surrounded by parentheses
   29 mkExportSpecs :: [ExportSpec] -> ExportSpecs
   30 mkExportSpecs = mkAnn ("(" <> child <> ")") . UExportSpecs . mkAnnList (separatedBy ", " list)
   31 
   32 -- | Export a name and related names
   33 mkExportSpec :: IESpec -> ExportSpec
   34 mkExportSpec = mkAnn child . UDeclExport
   35 
   36 -- | The export of an imported module (@ module A @)
   37 mkModuleExport :: ModuleName -> ExportSpec
   38 mkModuleExport = mkAnn ("module " <> child) . UModuleExport
   39 
   40 -- | Marks a name to be imported or exported with related names (subspecifier)
   41 mkIESpec :: Name -> Maybe SubSpec -> IESpec
   42 mkIESpec name ss = mkAnn (child <> child <> child) (UIESpec noth name (mkAnnMaybe (after "(" $ followedBy ")" opt) ss))
   43 
   44 -- | Marks a pattern synonym to be imported or exported
   45 mkPatternIESpec :: Name -> IESpec
   46 mkPatternIESpec name = mkAnn (child <> child) (UIESpec (justVal $ mkAnn child UImportPattern) name noth)
   47 
   48 -- | @(a,b,c)@: a class exported with some of its methods, or a datatype exported with some of its constructors.
   49 mkSubList :: [Name] -> SubSpec
   50 mkSubList = mkAnn child . USubSpecList . mkAnnList (separatedBy ", " list)
   51 
   52 -- | @(..)@: a class exported with all of its methods, or a datatype exported with all of its constructors.
   53 mkSubAll :: SubSpec
   54 mkSubAll = mkAnn ".." USubSpecAll
   55 
   56 -- | An import declaration: @import Module.Name@
   57 mkImportDecl :: Bool -> Bool -> Bool -> Maybe String -> ModuleName -> Maybe ModuleName -> Maybe ImportSpec
   58                   -> ImportDecl
   59 mkImportDecl source qualified safe pkg name rename spec
   60   = mkAnn ("import " <> child <> child <> child <> child <> child <> child <> child) $
   61       UImportDecl (if source then justVal (mkAnn "{-# SOURCE #-} " UImportSource) else noth)
   62                   (if qualified then justVal (mkAnn "qualified " UImportQualified) else noth)
   63                   (if safe then justVal (mkAnn "safe " UImportSafe) else noth)
   64                   (case pkg of Just str -> justVal (mkStringNode str); _ -> noth)
   65                   name (mkAnnMaybe opt (fmap (mkAnn (" as " <> child) . UImportRenaming) rename)) (mkAnnMaybe opt spec)
   66 
   67 -- | Restrict the import definition to ONLY import the listed names
   68 mkImportSpecList :: [IESpec] -> ImportSpec
   69 mkImportSpecList = mkAnn ("(" <> child <> ")") . UImportSpecList . mkAnnList (separatedBy ", " list)
   70 
   71 -- | Restrict the import definition to DONT import the listed names
   72 mkImportHidingList :: [IESpec] -> ImportSpec
   73 mkImportHidingList = mkAnn (" hiding (" <> child <> ")") . UImportSpecHiding . mkAnnList (separatedBy ", " list)
   74 
   75 -- | The name of a module
   76 mkModuleName :: String -> ModuleName
   77 mkModuleName s = mkAnn (fromString s) (UModuleName s)
   78

   81 mkFilePragmas :: [FilePragma] -> FilePragmaList
   82 mkFilePragmas = mkAnnList (separatedBy "\n" list)
   83 
   84 -- | @LANGUAGE@ pragma, listing the enabled language extensions in that file
   85 mkLanguagePragma :: [String] -> FilePragma
   86 mkLanguagePragma extensions
   87   = mkAnn ("{-# LANGUAGE " <> child <> " #-}") $ ULanguagePragma
   88       $ mkAnnList (separatedBy ", " list) (map (\ext -> mkAnn (fromString ext) (ULanguageExtension ext)) extensions)
   89 
   90 -- | @OPTIONS@ pragma, possibly qualified with a tool, e.g. OPTIONS_GHC
   91 mkOptionsGHC :: String -> FilePragma
   92 mkOptionsGHC opts
   93   = mkAnn ("{-# OPTIONS_GHC " <> child <> " #-}") $ UOptionsPragma
   94       $ mkStringNode opts
   95 
   96 -- | A warning pragma attached to the module
   97 mkModuleWarningPragma :: [String] -> ModulePragma
   98 mkModuleWarningPragma msg
   99   = mkAnn ("{-# WARNING " <> child <> " #-}") $ UModuleWarningPragma
  100       $ mkAnnList (separatedBy " " list) $ map mkStringNode msg
  101 
  102 -- | A deprecated pragma attached to the module
  103 mkModuleDeprecatedPragma :: [String] -> ModulePragma
  104 mkModuleDeprecatedPragma msg
  105   = mkAnn ("{-# DEPRECATED " <> child <> " #-}") $ UModuleDeprecatedPragma
  106       $ mkAnnList (separatedBy " " list) $ map mkStringNode msg
