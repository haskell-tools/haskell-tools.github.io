   46 -- | Creates a data or newtype declaration.
   47 mkDataDecl :: DataOrNewtypeKeyword -> Maybe Context -> DeclHead -> [ConDecl] -> [Deriving] -> Decl
   48 mkDataDecl keyw ctx dh cons derivs
   49   = mkAnn (child <> " " <> child <> child <> child <> child)
   50       $ UDataDecl keyw (mkAnnMaybe (after " " opt) ctx) dh
   51                  (mkAnnList (after " = " $ separatedBy " | " list) cons) (mkAnnList (indented list) derivs)

   53 -- | Creates a GADT-style data or newtype declaration.
   54 mkGADTDataDecl :: DataOrNewtypeKeyword -> Maybe Context -> DeclHead -> Maybe (KindConstraint)
   55                     -> [GadtConDecl] -> [Deriving] -> Decl
   56 mkGADTDataDecl keyw ctx dh kind cons derivs
   57   = mkAnn (child <> " " <> child <> child <> child <> child <> child)
   58       $ UGDataDecl keyw (mkAnnMaybe (after " " opt) ctx) dh
   59                   (mkAnnMaybe (after " " opt) kind) (mkAnnList (after " = " $ separatedBy " | " list) cons)
   60                   (mkAnnList (indented list) derivs)

   62 -- | Creates a GADT constructor declaration (@ D1 :: Int -> T String @)
   63 mkGadtConDecl :: [Name] -> Type -> GadtConDecl
   64 mkGadtConDecl names typ
   65   = mkAnn (child <> " :: " <> child <> child <> child)
   66       $ UGadtConDecl (mkAnnList (separatedBy ", " list) names) emptyList noth (mkAnn child $ UGadtNormalType typ)
   67 
   68 -- | Creates a GADT constructor declaration with record syntax (@ D1 :: { val :: Int } -> T String @)
   69 mkGadtRecordConDecl :: [Name] -> [FieldDecl] -> Type -> GadtConDecl
   70 mkGadtRecordConDecl names flds typ
   71   = mkAnn (child <> " :: " <> child <> child <> child) $ UGadtConDecl (mkAnnList (separatedBy ", " list) names) emptyList noth
   72       $ mkAnn (child <> " -> " <> child)
   73       $ UGadtRecordType (mkAnnList (after "{ " $ separatedBy ", " $ followedBy " }" list) flds) typ

  107 -- | Creates a type class declaration (@ class X a where f = ... @)
  108 mkClassDecl :: Maybe Context -> DeclHead -> [FunDep] -> Maybe ClassBody -> Decl
  109 mkClassDecl ctx dh funDeps body
  110   = let fdeps = case funDeps of [] -> Nothing
  111                                 _ -> Just $ mkAnn child $ UFunDeps $ mkAnnList (separatedBy ", " list) funDeps
  112      in mkAnn ("class " <> child <> child <> child <> child)
  113           $ UClassDecl (mkAnnMaybe (followedBy " " opt) ctx) dh (mkAnnMaybe (after " | " opt) fdeps) (mkAnnMaybe opt body)
  114 
  115 -- | Creates the list of declarations that can appear in a typeclass
  116 mkClassBody :: [ClassElement] -> ClassBody
  117 mkClassBody = mkAnn (" where " <> child) . UClassBody . mkAnnList (indented list)

  444   = mkAnn ("{-# INLINEABLE " <> child <> child <> " #-}") $ UInlinePragmaDecl
  445       $ mkAnn child $ UInlinablePragma (mkAnnMaybe (followedBy " " opt) phase) name
  446 
  447 -- | A pragma for maintaining line numbers in generated sources (@ {-\# LINE 123 "somefile" \#-} @)
  448 mkLinePragma :: Int -> Maybe (StringNode) -> TopLevelPragma
  449 mkLinePragma line filename
  450   = mkAnn ("{-# LINE " <> child <> child <> " #-}")
  451      $ ULinePragma (mkAnn child $ LineNumber line) (mkAnnMaybe (after " " opt) filename)
  452 
  453 -- | A pragma that tells the compiler that a polymorph function should be optimized for a given type (@ {-\# SPECIALISE f :: Int -> b -> b \#-} @)
  454 mkSpecializePragma :: Maybe PhaseControl -> Name -> [Type] -> TopLevelPragma
  455 mkSpecializePragma phase def specTypes
  456   = mkAnn child (USpecializeDecl
  457                   $ mkAnn ("{-# SPECIALIZE " <> child <> child <> " " <> child <> " #-}")
  458                     $ USpecializePragma (mkAnnMaybe (after " " opt) phase) def $ mkAnnList (separatedBy ", " list) specTypes)
  459 
  460 -- | Marks that the pragma should be applied from a given compile phase (@ [2] @)
  461 mkPhaseControlFrom :: Integer -> PhaseControl
  462 mkPhaseControlFrom phaseNum
  463   = mkAnn ("[" <> child <> child <> "]") $ UPhaseControl (mkAnnMaybe opt Nothing) (mkAnnMaybe opt $ Just $ mkAnn child $ PhaseNumber phaseNum)
  464 
  465 -- | Marks that the pragma should be applied until a given compile phase (@ [~2] @)
  466 mkPhaseControlUntil :: Integer -> PhaseControl
  467 mkPhaseControlUntil phaseNum
  468   = mkAnn ("[" <> child <> child <> "]") $ UPhaseControl (mkAnnMaybe opt $ Just $ mkAnn "~" PhaseInvert)
  469                                                          (mkAnnMaybe opt $ Just $ mkAnn child $ PhaseNumber phaseNum)
  470 
  471 -- | A rewrite rule (@ "map/map" forall f g xs. map f (map g xs) = map (f.g) xs @)
  472 mkRewriteRule :: String -> Maybe PhaseControl -> [RuleVar] -> Expr -> Expr -> Rule
  473 mkRewriteRule name phase vars lhs rhs
  474   = mkAnn (child <> " " <> child <> child <> child <> " = " <> child)
  475       $ URule (mkAnn ("\"" <> child <> "\"") $ UStringNode name) (mkAnnMaybe (followedBy " " opt) phase)
  476               (mkAnnList (after "forall " $ separatedBy " " $ followedBy ". " list) (vars)) lhs rhs
  477 
  478 mkRuleVar :: Name -> RuleVar
  479 mkRuleVar name = mkAnn child (URuleVar name)
  480 
  481 -- | The definition with the given name is annotated
  482 mkNameAnnotation :: Name -> AnnotationSubject
  483 mkNameAnnotation name = mkAnn child $ UNameAnnotation name
  484 
  485 -- | A type with the given name is annotated
  486 mkTypeAnnotation :: Name -> AnnotationSubject
  487 mkTypeAnnotation name = mkAnn ("type " <> child) $ UTypeAnnotation name
  488 
  489 -- | The whole module is annotated
  490 mkModuleAnnotation :: AnnotationSubject
  491 mkModuleAnnotation = mkAnn "module" UModuleAnnotation
  492 
  493 -- | A @CONLIKE@ modifier for an @INLINE@ pragma.
  494 mkConlikeAnnotation :: ConlikeAnnot
  495 mkConlikeAnnotation = mkAnn "CONLIKE" UConlikeAnnot
