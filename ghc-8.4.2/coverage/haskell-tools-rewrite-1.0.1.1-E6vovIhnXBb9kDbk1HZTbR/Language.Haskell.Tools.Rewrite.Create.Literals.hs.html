   12 -- | Character literal: @'c'@
   13 mkCharLit :: Char -> Literal
   14 mkCharLit c = mkAnn (fromString $ show c) $ UCharLit c
   15 
   16 -- | String literal: @"abc"@
   17 mkStringLit :: String -> Literal
   18 mkStringLit s = mkAnn (fromString $ show s) $ UStringLit s
   19 
   20 -- | Integer literal: @12@
   21 mkIntLit :: Integer -> Literal
   22 mkIntLit i = mkAnn (fromString $ show i) $ UIntLit i
   23 
   24 -- | Fractional literal: @3.14@
   25 mkFracLit :: Rational -> Literal
   26 mkFracLit f = mkAnn (fromString $ show f) $ UFracLit f
   27 
   28 -- | Primitive integer literal (of type @Int#@): @32#@
