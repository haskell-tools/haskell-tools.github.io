    9 -- * Do-notation
   10 
   11 -- | Binding statement (@ x <- action @)
   12 pattern BindStmt :: Pattern -> Expr -> Stmt
   13 pattern BindStmt bound expr <- Ann _ (UBindStmt bound expr)
   14 
   15 -- | Non-binding statement (@ action @)
   16 pattern ExprStmt :: Expr -> Stmt
   17 pattern ExprStmt expr <- Ann _ (UExprStmt expr)
   18 
   19 -- | Let statement (@ let x = 3; y = 4 @)
   20 pattern LetStmt :: LocalBindList -> Stmt
   21 pattern LetStmt binds <- Ann _ (ULetStmt binds)
   22 
   23 -- | A recursive binding statement with (@ rec b <- f a c; c <- f b a @)
   24 pattern RecStmt :: StmtList -> Stmt
   25 pattern RecStmt stmts <- Ann _ (URecStmt stmts)
   26 
   27 pattern DoKeyword :: DoKind
   28 pattern DoKeyword <- Ann _ UDoKeyword
   29 
   30 pattern MDoKeyword :: DoKind
   31 pattern MDoKeyword <- Ann _ UMDoKeyword
   32 
   33 -- * List comprehensions
   34 
   35 -- | Body of a list comprehension: (@ | x <- [1..10] @)
   36 pattern ListCompBody :: CompStmtList -> ListCompBody
   37 pattern ListCompBody stmts <- Ann _ (UListCompBody stmts)
   38 
   39 -- | Normal monadic statement of a list comprehension
   40 pattern CompStmt :: Stmt -> CompStmt
   41 pattern CompStmt stmt <- Ann _ (UCompStmt stmt)
   42 
   43 -- | Then statements by @TransformListComp@ (@ then sortWith by (x + y) @)
   44 pattern ThenStmt :: Expr -> MaybeExpr -> CompStmt
   45 pattern ThenStmt then_ by <- Ann _ (UThenStmt then_ by)
   46 
   47 -- | Grouping statements by @TransformListComp@ (@ then group by (x + y) using groupWith @) 
   48 pattern GroupStmt :: MaybeExpr -> MaybeExpr -> CompStmt
   49 pattern GroupStmt by using <- Ann _ (UGroupStmt by using)

   57 -- | Non-binding statement command (@ action @)

   62 pattern LetStmtCmd :: LocalBindList -> CmdStmt
