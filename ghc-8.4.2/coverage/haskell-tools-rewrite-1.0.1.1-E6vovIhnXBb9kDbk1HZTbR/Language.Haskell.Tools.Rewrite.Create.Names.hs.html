    1 -- | Generation of names for refactorings
    2 {-# LANGUAGE MonoLocalBinds, OverloadedStrings #-}
    3 
    4 module Language.Haskell.Tools.Rewrite.Create.Names where
    5 
    6 import Data.String (IsString(..), String)
    7 import Language.Haskell.Tools.AST
    8 import Language.Haskell.Tools.PrettyPrint.Prepare
    9 import Language.Haskell.Tools.Rewrite.Create.Utils (emptyList, mkAnn, mkAnnList)
   10 import Language.Haskell.Tools.Rewrite.ElementTypes
   11 import qualified Name as GHC
   12 
   13 -- | Creates a simple, unqualified name
   14 mkName :: String -> Name
   15 mkName = mkNormalName . mkSimpleName
   16 
   17 mkQualOp :: [String] -> String -> Operator
   18 mkQualOp quals = mkAnn child . UNormalOp . mkQualifiedName quals
   19 
   20 mkBacktickOp :: [String] -> String -> Operator
   21 mkBacktickOp quals = mkAnn ("`" <> child <> "`") . UBacktickOp . mkQualifiedName quals
   22 
   23 -- | Creates an annotated qualified operator: @A.B.+@ or @\`A.B.mod\`@.
   24 mkQualOp' :: [String] -> GHC.Name -> Operator
   25 mkQualOp' quals n | GHC.isSymOcc (GHC.getOccName n) = mkAnn child $ UNormalOp $ mkQualifiedName' quals n
   26                   | otherwise                       = mkAnn ("`" <> child <> "`") $ UBacktickOp $ mkQualifiedName' quals n
   27 
   28 -- | Creates an annotated unqualified operator: @+@ or @\`mod\`@.
   29 mkUnqualOp' :: GHC.Name -> Operator
   30 mkUnqualOp' n | GHC.isSymOcc (GHC.getOccName n) = mkAnn child $ UNormalOp $ mkSimpleName' n
   31               | otherwise                       = mkAnn ("`" <> child <> "`") $ UBacktickOp $ mkSimpleName' n
   32   
   33 mkUnqualOp :: String -> Operator
   34 mkUnqualOp = mkAnn child . UNormalOp . mkSimpleName
   35 
   36 -- | Creates an annotated qualified (non-operator) binding name: @A.B.f@ or @(A.B.+)@
   37 mkQualName' :: [String] -> GHC.Name -> Name
   38 mkQualName' quals n | GHC.isSymOcc (GHC.getOccName n) = mkAnn ("(" <> child <> ")") $ UParenName $ mkQualifiedName' quals n
   39                     | otherwise                       = mkAnn child $ UNormalName $ mkQualifiedName' quals n
   40 
   41 -- | Creates an annotated unqualified (non-operator) binding name: @f@ or @(+)@
   42 mkUnqualName' :: GHC.Name -> Name
   43 mkUnqualName' n | GHC.isSymOcc (GHC.getOccName n) = mkAnn ("(" <> child <> ")") $ UParenName $ mkSimpleName' n
   44                 | otherwise                       = mkAnn child $ UNormalName $ mkSimpleName' n
   45 
   46 mkNormalName :: QualifiedName -> Name
   47 mkNormalName = mkAnn child . UNormalName
   48 
   49 -- | Creates a parenthesized name: @ foldl (+) 0 @
   50 mkParenName :: QualifiedName -> Name
   51 mkParenName = mkAnn ("(" <> child <> ")") . UParenName
   52 
   53 -- | Creates an implicit name: @ ?var @
   54 mkImplicitName :: QualifiedName -> Name
   55 mkImplicitName = mkAnn ("?" <> child) . UImplicitName
   56 
   57 -- | Creates an annotated qualified simple name
   58 mkQualifiedName' :: [String] -> GHC.Name -> QualifiedName
   59 mkQualifiedName' quals n = mkQualifiedName quals (GHC.occNameString $ GHC.getOccName n)
   60 
   61 mkQualifiedName :: [String] -> String -> QualifiedName
   62 mkQualifiedName [] n = mkSimpleName n
   63 mkQualifiedName quals name
   64   = mkAnn (child <> "." <> child)
   65           (UQualifiedName (mkAnnList (separatedBy "." list) $ map mkNamePart quals) (mkNamePart name))
   66 
   67 -- | Creates a part of a qualified name.         
   68 mkNamePart :: String -> NamePart
   69 mkNamePart s = mkAnn (fromString s) (UNamePart s)
   70 
   71 -- | Creates a simple (unqualified) name
   72 mkSimpleName' :: GHC.Name -> QualifiedName
   73 mkSimpleName' = mkSimpleName . GHC.occNameString . GHC.getOccName
   74 
   75 -- | Creates a simple (unqualified) name
   76 mkSimpleName :: String -> QualifiedName
   77 mkSimpleName n = mkAnn (child <> child) 
   78                        (UQualifiedName emptyList (mkNamePart n))
   79 
   80 -- | Creates a quoted text
   81 mkStringNode :: String -> StringNode
   82 mkStringNode s = mkAnn (fromString s) (UStringNode s)
