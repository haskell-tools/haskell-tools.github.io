   26 -- | Does the import declaration import only the explicitly listed elements?
   27 importIsExact :: Ann UImportDecl dom stage -> Bool
   28 importIsExact = isJust . (^? importSpec&annJust&importSpecList)

   30 -- | Does the import declaration import all elements that are not excluded explicitly?
   31 importIsHiding :: Ann UImportDecl dom stage -> Bool
   32 importIsHiding = isJust . (^? importSpec&annJust&importSpecHiding)

   34 -- | Accesses the name of a function or value binding
   35 bindingName :: Simple Traversal (Ann UValueBind dom stage) (Ann UQualifiedName dom stage)
   36 bindingName = (valBindPat&patternName&simpleName
   37                         &+& funBindMatches&annList&matchLhs
   38                               &(matchLhsName&simpleName &+& matchLhsOperator&operatorName))
   39 
   40 -- | Accesses that name of a declaration through the declaration head.
   41 declHeadNames :: Simple Traversal (Ann UDeclHead dom stage) (Ann UQualifiedName dom stage)
   42 declHeadNames = (dhName&simpleName &+& dhBody&declHeadNames &+& dhAppFun&declHeadNames &+& dhOperator&operatorName)

   44 -- | A reference to access type arguments to a type constructor call that may be universally qualified
   45 -- or parenthesized.
   46 typeParams :: Simple Traversal (Ann UType dom stage) (Ann UType dom stage)
   47 typeParams = fromTraversal typeParamsTrav
   48   where typeParamsTrav f (Ann a (UTyFun p r)) = Ann a <$> (UTyFun <$> f p <*> typeParamsTrav f r)
   49         typeParamsTrav f (Ann a (UTyForall vs t)) = Ann a <$> (UTyForall vs <$> typeParamsTrav f t)
   50         typeParamsTrav f (Ann a (UTyCtx ctx t)) = Ann a <$> (UTyCtx ctx <$> typeParamsTrav f t)
   51         typeParamsTrav f (Ann a (UTyParen t)) = Ann a <$> (UTyParen <$> typeParamsTrav f t)
   52         typeParamsTrav f t = f t
   53 
   54 valBindPats :: Simple Traversal (Ann UValueBind dom stage) (Ann UPattern dom stage)
   55 valBindPats = valBindPat &+& funBindMatches & annList & matchLhs & (matchLhsArgs & annList &+& matchLhsLhs &+& matchLhsRhs)

   61 -- | Get all nodes that contain a given source range
   62 nodesContaining :: (HasRange (inner dom stage), Biplate (node dom stage) (inner dom stage))
   63                 => RealSrcSpan -> Simple Traversal (node dom stage) (inner dom stage)
   64 nodesContaining rng = biplateRef & filtered (isInside rng)
   65 
   66 -- | Return true if the node contains a given range
   67 isInside :: HasRange (inner dom stage) => RealSrcSpan -> inner dom stage -> Bool
   68 isInside rng nd = case getRange nd of RealSrcSpan sp -> sp `containsSpan` rng
   69                                       _              -> False
   70 
   71 -- | Get all nodes that are contained in a given source range
   72 nodesContained :: (HasRange (inner dom stage), Biplate (node dom stage) (inner dom stage))
   73                     => RealSrcSpan -> Simple Traversal (node dom stage) (inner dom stage)
   74 nodesContained rng = biplateRef & filtered (isContained rng)
   75 
   76 -- | Return true if the node contains a given range
   77 isContained :: HasRange (inner dom stage) => RealSrcSpan -> inner dom stage -> Bool
   78 isContained rng nd = case getRange nd of RealSrcSpan sp -> rng `containsSpan` sp
   79                                          _              -> False

   81 -- | Get the nodes that have exactly the given range
   82 nodesWithRange :: (Biplate (Ann node dom stage) (Ann inner dom stage), SourceInfo stage)
   83                => RealSrcSpan -> Simple Traversal (Ann node dom stage) (Ann inner dom stage)
   84 nodesWithRange rng = biplateRef & filtered (hasRange rng)
   85   where -- True, if the node has the given range
   86         hasRange :: SourceInfo stage => RealSrcSpan -> Ann inner dom stage -> Bool
   87         hasRange rng node = case getRange node of RealSrcSpan sp -> sp == rng
   88                                                   _              -> False
   89 
   90 -- | Get the shortest source range that contains the given
   91 getNodeContaining :: (Biplate (Ann node dom stage) (Ann inner dom stage), SourceInfo stage, HasRange (Ann inner dom stage))
   92                   => RealSrcSpan -> Ann node dom stage -> Maybe (Ann inner dom stage)
   93 getNodeContaining sp node = case node ^? nodesContaining sp of
   94   [] -> Nothing
   95   results -> Just $ minimumBy (compareRangeLength `on` getRange) results
   96 
   97 -- | Compares two source spans based on their lengths. Can only used for NESTED spans.
   98 compareRangeLength :: SrcSpan -> SrcSpan -> Ordering
   99 compareRangeLength (RealSrcSpan sp1) (RealSrcSpan sp2)
  100   = (lineDiff sp1 `compare` lineDiff sp2) `mappend` (colDiff sp1 `compare` colDiff sp2)
  101   where lineDiff sp = srcLocLine (realSrcSpanStart sp) - srcLocLine (realSrcSpanEnd sp)
  102         colDiff sp = srcLocCol (realSrcSpanStart sp) - srcLocCol (realSrcSpanEnd sp)

  104 -- | A class to access the names of named elements. Have to locate where does the AST element store its name.
  105 -- The returned name will be the one that was marked isDefining.
  106 class NamedElement elem where
  107   elementName :: Simple Traversal (Ann elem dom st) (Ann UQualifiedName dom st)
  108 
  109 instance NamedElement UDecl where
  110   elementName = (declHead & declHeadNames)
  111                   &+& (declTypeFamily & tfHead & declHeadNames)
  112                   &+& (declValBind & bindingName)
  113                   &+& (declName & simpleName)
  114                   &+& (declTypeSig & tsName & annList & simpleName)
  115                   &+& (declPatSyn & patLhs & (patName & simpleName &+& patSynOp & operatorName))
  116 
  117 instance NamedElement ULocalBind where
  118   elementName = localVal&bindingName
  119                   &+& localSig&tsName&annList&simpleName
  120                   &+& localFixity&fixityOperators&annList&operatorName

  122 inScope :: GHC.Name -> Scope -> Bool
  123 inScope n sc = any ((n `elem`) . map (^. _1)) sc
