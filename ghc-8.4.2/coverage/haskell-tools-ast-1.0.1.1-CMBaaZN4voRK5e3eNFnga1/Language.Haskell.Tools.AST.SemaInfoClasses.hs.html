   23 -- | Domains that have semantic information for names
   24 type HasNameInfo dom = (Domain dom, HasNameInfo' (SemanticInfo dom UQualifiedName))
   25 
   26 -- | Infos that may have a name that can be extracted
   27 class HasNameInfo' si where
   28   semanticsName :: si -> Maybe GHC.Name
   29 
   30 instance HasNameInfo' (NameInfo GhcRn) where
   31   semanticsName = (^? nameInfo)
   32 
   33 instance HasNameInfo' CNameInfo where
   34   semanticsName = fmap idName . (^? cnameInfo)
   35 
   36 instance HasNameInfo dom => HasNameInfo' (Ann UQualifiedName dom st) where
   37   semanticsName = semanticsName . (^. annotation&semanticInfo)
   38 
   39 instance HasNameInfo dom => HasNameInfo' (Ann UName dom st) where
   40   semanticsName = semanticsName . _simpleName . _element
