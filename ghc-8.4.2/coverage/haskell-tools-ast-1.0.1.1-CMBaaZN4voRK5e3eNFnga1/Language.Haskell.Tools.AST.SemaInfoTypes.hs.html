  134 trfPNames :: (IdP n -> IdP n') -> PName n -> PName n'
  135 trfPNames f (PName name parent) = PName (f name) (fmap f parent)
  136 
  137 trfPNamesM :: Monad m => (IdP n -> m (IdP n')) -> PName n -> m (PName n')
  138 trfPNamesM f (PName name (Just parent)) = PName <$> f name <*> (Just <$> f parent)
  139 trfPNamesM f (PName name Nothing) = PName <$> f name <*> return Nothing

  141 -- | Info for the module element
  142 data ModuleInfo n = ModuleInfo { _defModuleName :: GHC.Module
  143                                , _defDynFlags :: DynFlags -- ^ The compilation flags that are set up when the module was compiled
  144                                , _defIsBootModule :: Bool -- ^ True if this module is created from a hs-boot file
  145                                , _implicitNames :: [PName n] -- ^ implicitly imported names
  146                                , _prelTransMods :: [GHC.Module] -- ^ Modules imported transitively.
  147                                }
  148 
  149 trfModuleInfoM :: Monad m => (IdP n -> m (IdP n')) -> ModuleInfo n -> m (ModuleInfo n')
  150 trfModuleInfoM f (ModuleInfo mn df bm impl tm)
  151   = ModuleInfo mn df bm <$> mapM (trfPNamesM f) impl <*> return tm

  155 instance Data DynFlags where

  170 -- | Info corresponding to an import declaration
  171 data ImportInfo n = ImportInfo { _importedModule :: GHC.Module -- ^ The name and package of the imported module
  172                                , _availableNames :: [IdP n] -- ^ Names available from the imported module
  173                                , _importedNames :: [PName n] -- ^ Names actually imported from the module.
  174                                , _importTransMods :: [GHC.Module] -- ^ Modules imported transitively.
  175                                }
  176 
  177 trfImportInfo :: (IdP n -> IdP n') -> ImportInfo n -> ImportInfo n'
  178 trfImportInfo f (ImportInfo mod avail imp trm)
  179   = ImportInfo mod (map f avail) (map (trfPNames f) imp) trm
  180 
  181 trfImportInfoM :: Monad m => (IdP n -> m (IdP n')) -> ImportInfo n -> m (ImportInfo n')
  182 trfImportInfoM f (ImportInfo mod avail imp trm)
  183   = ImportInfo mod <$> (mapM f avail) <*> (mapM (trfPNamesM f) imp) <*> return trm
  184

  190 -- | Creates semantic information for an import declaration
  191 -- Strict in the list of the used and imported declarations, orphan and family instances.
  192 mkImportInfo :: GHC.Module -> [IdP n] -> [PName n] -> [GHC.Module] -> ImportInfo n

  197 -- | Gets the class and family instances from a module.
  198 getInstances :: GHC.GhcMonad m => [GHC.Module] -> m ([ClsInst], [FamInst])
  199 getInstances mods = do
  200   env <- GHC.getSession
  201   eps <- liftIO $ hscEPS env
  202   let (hptInsts, hptFamInsts) = hptInstances env (`elem` map GHC.moduleName mods)
  203       isFromMods inst = maybe False (`elem` mods) $ nameModule_maybe $ Var.varName $ is_dfun inst
  204       famIsFromMods inst = maybe False (`elem` mods) $ nameModule_maybe $ co_ax_name $ fi_axiom inst
  205       epsInsts = filter isFromMods $ instEnvElts $ eps_inst_env eps
  206       epsFamInsts = filter famIsFromMods $ famInstEnvElts $ eps_fam_inst_env eps
  207   return (hptInsts ++ epsInsts, hptFamInsts ++ epsFamInsts)
