   13 -- | Character literal: @'c'@
   14 mkCharLit :: Char -> Literal dom
   15 mkCharLit c = mkAnn (fromString $ show c) $ UCharLit c
   16 
   17 -- | String literal: @"abc"@
   18 mkStringLit :: String -> Literal dom
   19 mkStringLit s = mkAnn (fromString $ show s) $ UStringLit s
   20 
   21 -- | Integer literal: @12@
