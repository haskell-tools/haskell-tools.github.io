   47                            }

   50 instance Outputable UsageSpec where
   51   ppr (UsageSpec q useQ asQ)
   52     = GHC.text $ (if q then "qualified " else "") ++ "as "
   53         ++ (if useQ == asQ || q then asQ else asQ ++ " or " ++ useQ)
   54   pprPrec _ (UsageSpec q useQ asQ)
   55     = GHC.text $ (if q then "qualified " else "") ++ "as "
   56         ++ (if useQ == asQ || q then asQ else asQ ++ " or " ++ useQ)

  123 -- | Info for the module element
  124 data ModuleInfo n = ModuleInfo { _defModuleName :: GHC.Module
  125                                , _defDynFlags :: DynFlags -- ^ The compilation flags that are set up when the module was compiled
  126                                , _defIsBootModule :: Bool -- ^ True if this module is created from a hs-boot file
  127                                , _implicitNames :: [PName n] -- ^ implicitly imported names
  128                                , _prelOrphanInsts :: [ClsInst] -- ^ Class instances implicitly passed from Prelude.
  129                                , _prelFamInsts :: [FamInst] -- ^ Family instances implicitly passed from Prelude.
  130                                }
  131   deriving Data
  132 
  133 instance Data DynFlags where
  134   gunfold _ _ _ = error "Cannot construct dyn flags"
  135   toConstr _ = dynFlagsCon
  136   dataTypeOf _ = dynFlagsType
  137 
  138 dynFlagsType = mkDataType "DynFlags.DynFlags" [dynFlagsCon]
  139 dynFlagsCon = mkConstr dynFlagsType "DynFlags" [] Data.Prefix
  140 
  141 -- | Creates semantic information for the module element
  142 mkModuleInfo :: GHC.Module -> DynFlags -> Bool -> [PName n] -> [ClsInst] -> [FamInst] -> ModuleInfo n
  143 mkModuleInfo mod dfs boot !imported !orphan !family = ModuleInfo mod dfs boot imported orphan family
  144 
  145 -- | Info corresponding to an import declaration
  146 data ImportInfo n = ImportInfo { _importedModule :: GHC.Module -- ^ The name and package of the imported module
  147                                , _availableNames :: [n] -- ^ Names available from the imported module
  148                                , _importedNames :: [PName n] -- ^ Names actually imported from the module.
  149                                , _importedOrphanInsts :: [ClsInst] -- ^ Class instances implicitly passed.
  150                                , _importedFamInsts :: [FamInst] -- ^ Family instances implicitly passed.
  151                                }
  152   deriving Data

  157 -- | Creates semantic information for an import declaration
  158 mkImportInfo :: GHC.Module -> [n] -> [PName n] -> [ClsInst] -> [FamInst] -> ImportInfo n

  172 instance Show ScopeInfo where
  173   show (ScopeInfo locals) = "(ScopeInfo " ++ showSDocUnsafe (ppr locals) ++ ")"
  174 
  175 instance Outputable n => Show (NameInfo n) where
  176   show (NameInfo locals defined nameInfo)
  177     = "(NameInfo " ++ showSDocUnsafe (ppr locals) ++ " " ++ show defined ++ " "
  178         ++ showSDocUnsafe (ppr nameInfo) ++ ")"
  179   show (AmbiguousNameInfo locals defined nameInfo span)
  180     = "(AmbiguousNameInfo " ++ showSDocUnsafe (ppr locals) ++ " " ++ show defined ++ " "
  181         ++ showSDocUnsafe (ppr nameInfo) ++ " " ++ show span ++ ")"
  182   show (ImplicitNameInfo locals defined nameInfo span)
  183     = "(ImplicitNameInfo " ++ showSDocUnsafe (ppr locals) ++ " " ++ show defined ++ " "
  184         ++ showSDocUnsafe (ppr nameInfo) ++ " " ++ show span ++ ")"
  185 
  186 instance Show CNameInfo where
  187   show (CNameInfo locals defined nameInfo fixity)
  188     = "(CNameInfo " ++ showSDocUnsafe (ppr locals) ++ " " ++ show defined ++ " "
  189         ++ showSDocUnsafe (ppr nameInfo) ++ showSDocUnsafe (ppr fixity) ++ ")"
  190 
  191 instance Outputable n => Show (PName n) where
  192   show (PName n (Just parent))
  193     = showSDocUnsafe (ppr n) ++ "[in " ++ showSDocUnsafe (ppr parent) ++ "]"
  194   show (PName n Nothing) = showSDocUnsafe (ppr n)
  195 
  196 instance Outputable n => Show (ModuleInfo n) where
  197   show (ModuleInfo mod _ isboot imp clsInsts famInsts)
  198     = "(ModuleInfo " ++ showSDocUnsafe (ppr mod) ++ " " ++ show isboot ++ " " ++ show imp ++ " "
  199           ++ showSDocUnsafe (ppr clsInsts) ++ " " ++ showSDocUnsafe (ppr famInsts) ++ ")"
  200 
  201 instance Outputable n => Show (ImportInfo n) where
  202   show (ImportInfo mod avail imported clsInsts famInsts)
  203     = "(ImportInfo " ++ showSDocUnsafe (ppr mod) ++ " " ++ showSDocUnsafe (ppr avail) ++ " "
  204         ++ show imported ++ " " ++ showSDocUnsafe (ppr clsInsts) ++ " "
  205         ++ showSDocUnsafe (ppr famInsts) ++ ")"
  206 
  207 instance Show ImplicitFieldInfo where
  208   show (ImplicitFieldInfo bnds)
  209     = "(ImplicitFieldInfo [" ++ concat (intersperse "," (map showImplicitFld bnds)) ++ "])"
  210     where showImplicitFld (from, to) = showSDocUnsafe (ppr from) ++ "->" ++ showSDocUnsafe (ppr to)
  211 
  212 instance Show NoSemanticInfo where
  213   show NoSemanticInfo = "NoSemanticInfo"

  224 instance Functor NameInfo where
  225   fmap f = nameInfo .- f
  226 
  227 instance Functor PName where
  228   fmap f (PName n p) = PName (f n) (fmap f p)
  229 
  230 instance Functor ModuleInfo where
  231   fmap f = implicitNames .- fmap (fmap f)
  232 
  233 instance Functor ImportInfo where
  234   fmap f (ImportInfo mod avail imps clsInsts famInsts)
  235     = ImportInfo mod (fmap f avail) (fmap (fmap f) imps) clsInsts famInsts
  236 
  237 instance Foldable NameInfo where
  238   foldMap f si = maybe mempty f (si ^? nameInfo)
  239 
  240 instance Foldable ModuleInfo where
  241   foldMap f si = foldMap (foldMap f) (si ^. implicitNames)
  242 
  243 instance Foldable ImportInfo where
  244   foldMap f si = foldMap f (((si ^. availableNames)
  245                    ++ (si ^? importedNames & traversal & (pName &+& pNameParent & just) )))
  246 
  247 instance Foldable PName where
  248   foldMap f (PName n p) = f n `mappend` foldMap f p
  249 
  250 instance Traversable PName where
  251   traverse f (PName n p) = PName <$> f n <*> traverse f p
  252 
  253 instance Traversable NameInfo where
  254   traverse f (NameInfo locals defined nameInfo) = NameInfo locals defined <$> f nameInfo
  255   traverse _ (AmbiguousNameInfo locals defined nameInfo span)
  256     = pure $ AmbiguousNameInfo locals defined nameInfo span
  257   traverse _ (ImplicitNameInfo locals defined nameInfo span)
  258     = pure $ ImplicitNameInfo locals defined nameInfo span
  259 
  260 instance Traversable ModuleInfo where
  261   traverse f (ModuleInfo mod dfs isboot imp clsInsts famInsts)
  262     = ModuleInfo mod dfs isboot <$> traverse (traverse f) imp <*> pure clsInsts <*> pure famInsts
  263 
  264 instance Traversable ImportInfo where
  265   traverse f (ImportInfo mod avail imps clsInsts famInsts)
  266     = ImportInfo mod <$> traverse f avail <*> traverse (traverse f) imps <*> pure clsInsts
  267                      <*> pure famInsts
