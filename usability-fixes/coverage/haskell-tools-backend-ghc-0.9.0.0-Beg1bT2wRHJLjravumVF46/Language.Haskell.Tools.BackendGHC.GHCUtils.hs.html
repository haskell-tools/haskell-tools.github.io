   29 class OutputableBndrId name => GHCName name where
   30   rdrName :: name -> RdrName
   31   getFromNameUsing :: Applicative f => (Name -> Ghc (f Id)) -> Name -> Ghc (f name)
   32   getBindsAndSigs :: HsValBinds name -> ([LSig name], LHsBinds name)
   33   nameFromId :: Id -> name
   34   unpackPostRn :: RdrName -> PostRn name name -> name
   35 
   36   gunpackPostRn :: a -> (name -> a) -> PostRn name name -> a
   37 
   38 instance GHCName RdrName where
   39   rdrName = id
   40   getFromNameUsing _ n = return $ pure (nameRdrName n)
   41   getBindsAndSigs (ValBindsIn binds sigs) = (sigs, binds)
   42   getBindsAndSigs _ = error "ValBindsOut: ValBindsOut in parsed source"
   43   nameFromId = nameRdrName . getName
   44   unpackPostRn rdr _ = rdr
   45 
   46   gunpackPostRn a _ _ = a

   48 occName :: GHCName n => n -> OccName
   49 occName = rdrNameOcc . rdrName

   51 instance GHCName GHC.Name where
   52   rdrName = nameRdrName
   53   getFromNameUsing f n = fmap nameFromId <$> f n
   54   getBindsAndSigs (ValBindsOut bindGroups sigs) = (sigs, unionManyBags (map snd bindGroups))
   55   getBindsAndSigs _ = error "getBindsAndSigs: ValBindsIn in renamed source"
   56   nameFromId = getName
   57   unpackPostRn _ a = a

   59   gunpackPostRn _ f pr = f pr

   69 -- | Loading ids for top-level ghc names
   70 getTopLevelId :: GHC.Name -> Ghc (Maybe GHC.Id)
   71 getTopLevelId name =
   72     lookupName name >>= \case
   73       Just (AnId id) -> return (Just id)
   74       Just (AConLike (RealDataCon dc)) -> return $ Just $ mkVanillaGlobal name (dataConUserType dc)
   75       Just (AConLike (PatSynCon ps)) -> return $ Just $ mkVanillaGlobal name (createPatSynType ps)
   76       Just (ATyCon tc) -> return $ Just $ mkVanillaGlobal name (tyConKind tc)
   77       _ -> return Nothing
   78   where createPatSynType patSyn = case patSynSig patSyn of (_, _, _, _, args, res) -> mkFunTys args res

  233 -- | Tries to simplify the type that has HsAppsTy before renaming. Does not always provide the correct form.
  234 -- Treats each operator as if they are of equivalent precedence and always left-associative.
  235 cleanHsType :: forall n . (OutputableBndrId n) => HsType n -> HsType n
  236 -- for some reason * is considered infix
  237 cleanHsType (HsAppsTy apps) = unLoc $ guessType apps
  238   where guessType :: OutputableBndrId n => [LHsAppType n] -> LHsType n
  239         guessType (L l (HsAppInfix n) : rest) -- must be a prefix actually
  240           = guessType' (L l (HsTyVar NotPromoted n)) rest
  241         guessType (L _ (HsAppPrefix t) : rest) = guessType' t rest
  242         guessType [] = error $ "guessType: empty: " ++ showSDocUnsafe (ppr apps)
  243 
  244         guessType' :: LHsType n -> [LHsAppType n] -> LHsType n
  245         guessType' fun (L _ (HsAppPrefix t) : rest) = guessType' (hsAppTy fun t) rest
  246         guessType' fun (L l (HsAppInfix n) : rest)
  247           -- TODO: find a better check
  248           | showSDocUnsafe (ppr n) == "*" = guessType' (hsAppTy fun (L l (HsTyVar NotPromoted n))) rest
  249         guessType' left (L _ (HsAppInfix n) : right) = hsOpTy left n (guessType right)
  250         guessType' t [] = t
  251 
  252         hsAppTy :: LHsType n -> LHsType n -> LHsType n
  253         hsAppTy t1 t2 = L (getLoc t1 `combineSrcSpans` getLoc t2) $ HsAppTy t1 t2
  254 
  255         hsOpTy :: LHsType n -> Located n -> LHsType n -> LHsType n
  256         hsOpTy t1 n t2 = L (getLoc t1 `combineSrcSpans` getLoc t2) $ HsOpTy t1 n t2
  257 cleanHsType t = t

  259 mergeFixityDefs :: [Located (FixitySig n)] -> [Located (FixitySig n)]
  260 mergeFixityDefs (s@(L l _) : rest)
  261   = let (same, different) = partition ((== l) . getLoc) rest
  262      in foldl mergeWith s (map unLoc same) : mergeFixityDefs different
  263   where mergeWith (L l (FixitySig names fixity)) (FixitySig otherNames _) = L l (FixitySig (names ++ otherNames) fixity)
  264 mergeFixityDefs [] = []

  266 getGroupRange :: HsGroup n -> SrcSpan
  267 getGroupRange (HsGroup {..})
  268   = foldr combineSrcSpans noSrcSpan locs
  269   where locs = [getHsValRange hs_valds] ++ map getLoc hs_splcds ++ map getLoc (concatMap group_tyclds hs_tyclds) ++ map getLoc (concatMap group_roles hs_tyclds)
  270                  ++ map getLoc hs_derivds ++ map getLoc hs_fixds ++ map getLoc hs_defds
  271                  ++ map getLoc hs_fords ++ map getLoc hs_warnds ++ map getLoc hs_annds ++ map getLoc hs_ruleds ++ map getLoc hs_vects
  272                  ++ map getLoc hs_docs

  274 getHsValRange :: HsValBinds n -> SrcSpan
  275 getHsValRange (ValBindsIn vals sig) = foldr combineSrcSpans noSrcSpan $ map getLoc (bagToList vals) ++ map getLoc sig
  276 getHsValRange (ValBindsOut vals sig) = foldr combineSrcSpans noSrcSpan $ concatMap (map getLoc . bagToList . snd) vals ++ map getLoc sig
