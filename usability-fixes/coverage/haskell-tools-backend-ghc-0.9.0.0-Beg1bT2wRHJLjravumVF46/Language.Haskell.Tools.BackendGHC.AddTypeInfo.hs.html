   40 addTypeInfos :: LHsBinds Id -> Ann AST.UModule (Dom GHC.Name) RangeStage -> Ghc (Ann AST.UModule IdDom RangeStage)
   41 addTypeInfos bnds mod = do
   42   ut <- liftIO mkUnknownType
   43   let getType = getType' ut
   44   fixities <- getFixities
   45   -- fixities to normal form before going on
   46   liftIO $ evaluate fixities
   47   let createCName sc def id = mkCNameInfo sc def id fixity
   48         where fixity = if any (any ((getOccName id ==) . getOccName . (^. _1))) (drop 1 sc)
   49                           then Nothing
   50                           else fmap (snd . snd) $ List.find (\(mod,(occ,_)) -> Just mod == (nameModule_maybe $ varName id) && occ == getOccName id) fixities
   51   evalStateT (semaTraverse
   52     (AST.SemaTrf
   53       (\ni -> case (AST.semanticsSourceInfo ni, AST.semanticsName ni) of
   54                 (_, Just name) -> lift $ createCName (AST.semanticsScope ni) (AST.semanticsDefining ni) <$> getType name
   55                 (Just l@(RealSrcSpan loc), _)
   56                   -> case Map.lookup l locMapping of
   57                             Just id -> return $ createCName (AST.semanticsScope ni) (AST.semanticsDefining ni) id
   58                             _ -> do (none,rest) <- gets (break ((\case (RealSrcSpan sp) -> sp `containsSpan` loc) . fst))
   59                                     case rest of [] -> throw $ ConvertionProblem (RealSrcSpan loc) "Ambiguous or implicit name missing"
   60                                                  ((_,id):more) -> do put (none ++ more)
   61                                                                      return $ createCName (AST.semanticsScope ni) (AST.semanticsDefining ni) id
   62                 _ -> convProblem "addTypeInfos: Cannot access a the semantics of a name.")
   63       pure (traverse (lift . getType)) (traverse (lift . getType)) pure
   64         pure) mod) (extractSigIds bnds ++ extractSigBindIds bnds)
   65   where locMapping = Map.fromList $ map (\(L l id) -> (l, id)) $ extractExprIds bnds
   66         getType' ut name = fromMaybe (mkVanillaGlobal name ut) <$> ((<|> Map.lookup name ids) <$> getTopLevelId name)
   67         ids = Map.fromList $ map (\id -> (getName id, id)) $ extractTypes bnds
   68         extractTypes :: LHsBinds Id -> [Id]
   69         extractTypes = concatMap universeBi . bagToList
   70 
   71         mkUnknownType :: IO Type
   72         mkUnknownType = do
   73           tUnique <- mkSplitUniqSupply 'x'
   74           return $ mkTyVarTy $ mkVanillaGlobal (mkSystemName (uniqFromSupply tUnique) (mkDataOcc "TypeNotFound")) (mkTyConTy starKindTyCon)
   75 
   76         getFixities :: Ghc [(Module, (OccName, GHC.Fixity))]
   77         getFixities = do env <- getSession
   78                          pit <- liftIO $ eps_PIT <$> hscEPS env
   79                          let hpt = hsc_HPT env
   80                              ifaces = moduleEnvElts pit ++ map hm_iface (eltsUDFM hpt)
   81                          return $ concatMap (\mi -> map (mi_module mi, ) $ mi_fixities mi) ifaces
   82 
   83 extractExprIds :: LHsBinds Id -> [Located Id]
   84         -- expressions like HsRecFld are removed from the typechecked representation, they are replaced by HsVar
   85 extractExprIds = catMaybes . map (\case L l (HsVar (L _ n)) -> Just (L l n)
   86                                         L l (HsWrap _ (HsVar (L _ n))) -> Just (L l n)
   87                                         _ -> Nothing
   88                                  ) . concatMap universeBi . bagToList
   89 
   90 extractSigIds :: LHsBinds Id -> [(SrcSpan,Id)]
   91 extractSigIds = filter (isGoodSrcSpan . fst)
   92                   . concat
   93                   . map (\case L l bs@(AbsBindsSig {} :: HsBind Id) -> map (l,) $ getImplVars (abs_sig_ev_bind bs)
   94                                _                                    -> [] )
   95                   . concatMap universeBi . bagToList
   96   where getImplVars (EvBinds evbnds) = catMaybes $ map getEvVar $ bagToList evbnds
   97         getImplVars _                = []
   98         getEvVar (EvBind lhs _ False) = Just lhs
   99         getEvVar _                    = Nothing
  100 
  101 extractSigBindIds :: LHsBinds Id -> [(SrcSpan,Id)]
  102 extractSigBindIds = filter (isGoodSrcSpan . fst)
  103                       . catMaybes
  104                       . map (\case L l (IPBind (Right id) _) -> Just (l,id)
  105                                    _                         -> Nothing )
  106                       . concatMap universeBi . bagToList
