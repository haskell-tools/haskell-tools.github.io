   55 createModuleInfo :: ModSummary -> SrcSpan -> [LImportDecl n] -> Trf (Sema.ModuleInfo GHC.Name)
   56 createModuleInfo mod nameLoc (filter (not . ideclImplicit . unLoc) -> imports) = do
   57   let prelude = (xopt ImplicitPrelude $ ms_hspp_opts mod)
   58                   && all (\idecl -> ("Prelude" /= (GHC.moduleNameString $ unLoc $ ideclName $ unLoc idecl))
   59                                       || nameLoc == getLoc idecl) imports
   60   (_, preludeImports) <- if prelude then getImportedNames "Prelude" Nothing else return (ms_mod mod, [])
   61   (insts, famInsts)
   62     <- if prelude then lift $ getOrphanAndFamInstances (Module baseUnitId (GHC.mkModuleName "Prelude"))
   63                   else return ([], [])
   64   return $ mkModuleInfo (ms_mod mod) (ms_hspp_opts mod) (case ms_hsc_src mod of HsSrcFile -> False; _ -> True)
   65                         (forceElements preludeImports) (forceElements insts) (forceElements famInsts)

   93 -- | Adds semantic information to an impord declaration. See ImportInfo.
   94 createImportData :: (GHCName r, HsHasName n) => GHC.ImportDecl n -> Trf (ImportInfo r)
   95 createImportData (GHC.ImportDecl _ name pkg _ _ _ _ _ declHiding) =
   96   do (mod,importedNames) <- getImportedNames (GHC.moduleNameString $ unLoc name) (fmap (unpackFS . sl_fs) pkg)
   97      names <- liftGhc $ filterM (checkImportVisible declHiding . (^. pName)) importedNames
   98      -- TODO: only use getFromNameUsing once
   99      -- elements are forced to prevent unevaluated references remaining and causing a space leak
  100      lookedUpNames <- liftGhc $ mapM translatePName $ names
  101      lookedUpImported <- liftGhc $ mapM (getFromNameUsing getTopLevelId . (^. pName)) $ importedNames
  102      (insts, famInsts) <- lift $ getOrphanAndFamInstances mod
  103      return $ mkImportInfo mod (forceElements $ catMaybes lookedUpImported)
  104                                (forceElements $ catMaybes lookedUpNames)
  105                                (forceElements insts) (forceElements famInsts)
  106   where translatePName (PName n p) = do n' <- getFromNameUsing getTopLevelId n
  107                                         p' <- maybe (return Nothing) (getFromNameUsing getTopLevelId) p
  108                                         return (PName <$> n' <*> Just p')

  110 getOrphanAndFamInstances :: Module -> Ghc ([ClsInst], [FamInst])
  111 getOrphanAndFamInstances mod = do
  112   env <- getSession
  113   eps <- liftIO (readIORef (hsc_EPS env))
  114   let ifc = lookupIfaceByModule (hsc_dflags env) (hsc_HPT env) (eps_PIT eps) mod
  115       hp = lookupHptByModule (hsc_HPT env) mod
  116       uses = catMaybes $ map getModule $ maybe [] (\ifc -> dep_orphs (mi_deps ifc) `union` dep_finsts (mi_deps ifc)) ifc
  117       getModule mod = if moduleUnitId mod == mainUnitId
  118                         then fmap Right $ lookupHptByModule (hsc_HPT env) mod
  119                         else fmap Left $ lookupIfaceByModule (hsc_dflags env) (hsc_HPT env) (eps_PIT eps) mod
  120       usedMods = lefts uses
  121       usedDetails = map hm_details (maybeToList hp ++ rights uses)
  122       hptInstances = filter (isOrphan . is_orphan) $ concatMap md_insts usedDetails
  123       hptFamilyInstances = concatMap md_fam_insts usedDetails
  124       allInstances = instEnvElts (eps_inst_env eps)
  125       relevantInstances = hptInstances ++ filter ((\n -> nameModule n `elem` (mod : map mi_module usedMods)) . idName . instanceDFunId) (filter (isOrphan . is_orphan) allInstances)
  126       allFamilyInstances = famInstEnvElts (eps_fam_inst_env eps)
  127       relevantFamilyInstances = hptFamilyInstances ++ filter ((\n -> nameModule n `elem` (mod : map mi_module usedMods)) . co_ax_name . fi_axiom) allFamilyInstances
  128   return (relevantInstances, relevantFamilyInstances)

  131 -- | Get names that are imported from a given import
  132 getImportedNames :: String -> Maybe String -> Trf (GHC.Module, [PName GHC.Name])
  133 getImportedNames name pkg = liftGhc $ do
  134   hpt <- hsc_HPT <$> getSession
  135   eps <- getSession >>= liftIO . readIORef . hsc_EPS
  136   mod <- findModule (mkModuleName name) (fmap mkFastString pkg)
  137   -- load exported names from interface file
  138   let ifaceNames = maybe [] mi_exports $ flip lookupModuleEnv mod
  139                                        $ eps_PIT eps
  140   let homeExports = maybe [] (md_exports . hm_details) (lookupHptByModule hpt mod)
  141   -- TODO: Why selectors are added in one case and not added in the other?
  142   return (mod, concatMap (availToPName availNames) ifaceNames ++ concatMap (availToPName availNamesWithSelectors) homeExports)
  143     where availToPName f a = map (\n -> if n == availName a then PName n Nothing else PName n (Just (availName a))) (f a)

  145 -- | Check is a given name is imported from an import with given import specification.
  146 checkImportVisible :: (HsHasName name, GhcMonad m) => Maybe (Bool, Located [LIE name]) -> GHC.Name -> m Bool
  147 checkImportVisible (Just (isHiding, specs)) name
  148   | isHiding  = not . or @[] <$> mapM (`ieSpecMatches` name) (map unLoc (unLoc specs))
  149   | otherwise = or @[] <$> mapM (`ieSpecMatches` name) (map unLoc (unLoc specs))
  150 checkImportVisible _ _ = return True

  152 forceElements :: [a] -> [a]
  153 forceElements [] = []
  154 forceElements (a : ls) = let res = forceElements ls
  155                           in a `seq` res `seq` (a : ls)

  158 ieSpecMatches :: (HsHasName name, GhcMonad m) => IE name -> GHC.Name -> m Bool
  159 ieSpecMatches (concatMap hsGetNames' . HsSyn.ieNames -> ls) name
  160   | name `elem` ls = return True
  161 -- ieNames does not consider field names
  162 ieSpecMatches (IEThingWith thing _ with flds) name
  163   | name `elem` concatMap hsGetNames' (map (ieWrappedName . unLoc) (thing : with) ++ map (flSelector . unLoc) flds)
  164   = return True
  165 ieSpecMatches ie@(IEThingAll _) name | [n] <- hsGetNames' (HsSyn.ieName ie), isTyConName n
  166   = do entity <- lookupName n
  167        return $ case entity of Just (ATyCon tc)
  168                                  | Just cls <- tyConClass_maybe tc
  169                                      -> name `elem` map getName (classMethods cls)
  170                                  | otherwise -> name `elem` concatMap (\dc -> getName dc : map flSelector (dataConFieldLabels dc))
  171                                                                       (tyConDataCons tc)
  172                                _             -> False
  173 ieSpecMatches _ _ = return False

  222 -- | Get the elements where there is no ; before
  223 elementsWithoutSemi :: [Ann e (Dom n) RangeStage] -> Trf [Bool]
  224 elementsWithoutSemi [] = return []
  225 elementsWithoutSemi (fst:rest) = indentedElements' (srcSpanEnd $ getRange fst) rest
  226   where indentedElements' lastEnd (elem:rest)
  227           = let sepRange = mkSrcSpan lastEnd (srcSpanStart $ getRange elem)
  228              in (:) <$> (not . (\l -> isGoodSrcSpan l && srcSpanStart l < srcSpanEnd l) <$> focusOn sepRange (tokenLoc AnnSemi))
  229                     <*> indentedElements' (srcSpanEnd $ getRange elem) rest
  230         indentedElements' _ [] = return []

  280 -- | Annotates a node with the given location and focuses on the given source span.
  281 annLoc :: Trf (SemanticInfo (Dom n) b) -> Trf SrcSpan -> Trf (b (Dom n) RangeStage) -> Trf (Ann b (Dom n) RangeStage)
  282 annLoc semam locm nodem = do loc <- locm

  285                              return (Ann (NodeInfo sema (NodeSpan loc)) node)

  305 -- | Focuses the transformation to go between tokens. The tokens must be found inside the current range.
  306 between :: AnnKeywordId -> AnnKeywordId -> Trf a -> Trf a
  307 between firstTok lastTok = focusAfter firstTok . focusBefore lastTok
  308 
  309 -- | Focuses the transformation to go between tokens if they are present
  310 betweenIfPresent :: AnnKeywordId -> AnnKeywordId -> Trf a -> Trf a
  311 betweenIfPresent firstTok lastTok = focusAfterIfPresent firstTok . focusBeforeIfPresent lastTok
  312 
  313 -- | Focuses the transformation to be performed after the given token. The token must be found inside the current range.
  314 focusAfter :: AnnKeywordId -> Trf a -> Trf a
  315 focusAfter firstTok trf
  316   = do firstToken <- tokenLoc firstTok
  317        if (isGoodSrcSpan firstToken)
  318           then local (\s -> s { contRange = mkSrcSpan (srcSpanEnd firstToken) (srcSpanEnd (contRange s))}) trf
  319           else do rng <- asks contRange
  320                   convertionProblem $ "focusAfter: token not found in " ++ show rng ++ ": " ++ show firstTok
  321 
  322 focusAfterIfPresent :: AnnKeywordId -> Trf a -> Trf a
  323 focusAfterIfPresent firstTok trf
  324   = do firstToken <- tokenLoc firstTok
  325        if (isGoodSrcSpan firstToken)
  326           then local (\s -> s { contRange = mkSrcSpan (srcSpanEnd firstToken) (srcSpanEnd (contRange s))}) trf
  327           else trf
  328 
  329 -- | Focuses the transformation to be performed before the given token. The token must be found inside the current range.
  330 focusBefore :: AnnKeywordId -> Trf a -> Trf a
  331 focusBefore lastTok trf
  332   = do lastToken <- tokenLocBack lastTok
  333        if (isGoodSrcSpan lastToken)
  334           then local (\s -> s { contRange = mkSrcSpan (srcSpanStart (contRange s)) (srcSpanStart lastToken)}) trf
  335           else do rng <- asks contRange
  336                   convertionProblem $ "focusBefore: token not found in " ++ show rng ++ ": " ++ show lastTok
  337 
  338 focusBeforeIfPresent :: AnnKeywordId -> Trf a -> Trf a
  339 focusBeforeIfPresent lastTok trf
  340   = do lastToken <- tokenLocBack lastTok
  341        if (isGoodSrcSpan lastToken)
  342           then local (\s -> s { contRange = mkSrcSpan (srcSpanStart (contRange s)) (srcSpanStart lastToken)}) trf
  343           else trf

  391 -- | Searches for tokens in the given order inside the parent element and returns their combined location
  392 tokensLoc :: [AnnKeywordId] -> Trf SrcSpan
  393 tokensLoc keys = foldLocs <$> eachTokenLoc keys
  394 
  395 -- | Searches for tokens in the given order inside the parent element and returns their location
  396 eachTokenLoc :: [AnnKeywordId] -> Trf [SrcSpan]
  397 eachTokenLoc (keyw:rest)
  398   = do spanFirst <- tokenLoc keyw
  399        spanRest <- focusAfterLoc (srcSpanEnd spanFirst) (eachTokenLoc rest)
  400        return (spanFirst : spanRest)
  401 eachTokenLoc [] = pure []

  414 -- | Annotates the element with the same annotation that is on the other element
  415 copyAnnot :: SemanticInfo (Dom n) a ~ SemanticInfo (Dom n) b
  416                => (Ann a (Dom n) RangeStage -> b (Dom n) RangeStage) -> Trf (Ann a (Dom n) RangeStage) -> Trf (Ann b (Dom n) RangeStage)

  431 updateCol f (RealSrcLoc loc) = mkSrcLoc (srcLocFile loc) (srcLocLine loc) (f $ srcLocCol loc)

  445 -- | Rearrange definitions to appear in the order they are defined in the source file.
  446 orderDefs :: [Ann e (Dom n) RangeStage] -> [Ann e (Dom n) RangeStage]
  447 orderDefs = sortBy (compare `on` srcSpanStart . (^. AST.annotation & AST.sourceInfo & AST.nodeSpan))

  453 -- | Only keeps one of the elements that are on the same source location
  454 removeDuplicates :: [Located e] -> [Located e]
  455 removeDuplicates (fst:rest) = fst : removeDuplicates (filter ((/= getLoc fst) . getLoc) rest)

  462 -- | Transform a list of definitions where the defined names are in scope for subsequent definitions
  463 trfScopedSequence :: HsHasName d => (d -> Trf e) -> [d] -> Trf [e]
  464 trfScopedSequence f (def:rest) = (:) <$> f def <*> addToScope def (trfScopedSequence f rest)

  467 -- | Splits a given string at whitespaces while calculating the source location of the fragments
  468 splitLocated :: Located String -> [Located String]
  469 splitLocated (L (RealSrcSpan l) str) = splitLocated' str (realSrcSpanStart l) Nothing
  470   where splitLocated' :: String -> RealSrcLoc -> Maybe (RealSrcLoc, String) -> [Located String]
  471         splitLocated' (c:rest) currLoc (Just (startLoc, str)) | isSpace c
  472           = L (RealSrcSpan $ mkRealSrcSpan startLoc currLoc) (reverse str) : splitLocated' rest (advanceSrcLoc currLoc c) Nothing
  473         splitLocated' (c:rest) currLoc Nothing | isSpace c = splitLocated' rest (advanceSrcLoc currLoc c) Nothing
  474         splitLocated' (c:rest) currLoc (Just (startLoc, str)) = splitLocated' rest (advanceSrcLoc currLoc c) (Just (startLoc, c:str))
  475         splitLocated' (c:rest) currLoc Nothing = splitLocated' rest (advanceSrcLoc currLoc c) (Just (currLoc, [c]))
  476         splitLocated' [] currLoc (Just (startLoc, str)) = [L (RealSrcSpan $ mkRealSrcSpan startLoc currLoc) (reverse str)]
  477         splitLocated' [] _ Nothing = []
  478 splitLocated _ = convProblem "splitLocated: unhelpful span given"

  480 compareSpans :: SrcSpan -> SrcSpan -> Ordering
  481 compareSpans (RealSrcSpan a) (RealSrcSpan b)
  482   | a `containsSpan` b = GT
  483   | b `containsSpan` a = LT
  484 compareSpans _ _ = EQ
