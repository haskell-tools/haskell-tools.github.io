   62 -- | This function does the real job of acting upon client messages in a stateful environment of a
   63 -- client.
   64 updateClient :: DaemonOptions -> [RefactoringChoice IdDom] -> (ResponseMsg -> IO ()) -> ClientMessage
   65                   -> DaemonSession Bool
   66 updateClient options refactors resp = updateClient' (UpdateCtx options refactors resp)
   67 
   68 updateClient' :: UpdateCtx -> ClientMessage -> DaemonSession Bool
   69 -- resets the internal state of Haskell-tools (but keeps options)
   70 updateClient' UpdateCtx{..} Reset
   71   = do roots <- gets (^? refSessMCs & traversal & mcRoot)
   72        ghcflags <- gets (^. ghcFlagsSet)
   73        modify' $ resetSession
   74        Session sess <- liftIO $ initGhcSession (generateCode (sharedOptions options))
   75        env <- liftIO $ readIORef sess
   76        liftIO $ unload env [] -- clear (unload everything) the global state of the linker
   77        lift $ setSession env
   78        addPackages response roots
   79        return True
   80 
   81 updateClient' UpdateCtx{..} (Handshake _)
   82   = do liftIO (response $ HandshakeResponse $ versionBranch version)
   83        return True
   84 
   85 updateClient' UpdateCtx{..} KeepAlive
   86   = do liftIO (response KeepAliveResponse)
   87        return True

   89 updateClient' UpdateCtx{..} Disconnect
   90   = do liftIO (response Disconnected)
   91        return False

  111 updateClient' UpdateCtx{..} (RemovePackages packagePathes) = do
  112     mcs <- gets (^. refSessMCs)
  113     let existingFiles = concatMap @[] (map (^. sfkFileName) . Map.keys) (mcs ^? traversal & filtered isRemoved & mcModules)
  114     lift $ forM_ existingFiles (\fs -> removeTarget (TargetFile fs Nothing))
  115     lift $ deregisterDirs (mcs ^? traversal & filtered isRemoved & mcSourceDirs & traversal)
  116     modify' $ refSessMCs .- filter (not . isRemoved)
  117     modifySession (\s -> s { hsc_mod_graph = filter ((`notElem` existingFiles) . getModSumOrig) (hsc_mod_graph s) })
  118     mcs <- gets (^. refSessMCs)
  119     when (null mcs) $ modify' (packageDBSet .= False)
  120     return True
  121   where isRemoved mc = (mc ^. mcRoot) `elem` packagePathes

  123 updateClient' UpdateCtx{..} UndoLast | disableHistory $ sharedOptions options
  124   = do liftIO $ response $ ErrorMessage "Recording history has been disabled from command line."
  125        return True
  126 updateClient' UpdateCtx{..} UndoLast =
  127   do undos <- gets (^. undoStack)
  128      case undos of
  129        [] -> do liftIO $ response $ ErrorMessage "There is nothing to undo."
  130                 return True
  131        lastUndo:_ -> do
  132          modify (undoStack .- tail)
  133          liftIO $ mapM_ performUndo lastUndo
  134          reloadModules response (getUndoAdded lastUndo)
  135                                 (getUndoChanged lastUndo)
  136                                 (getUndoRemoved lastUndo) -- reload the reverted files
  137 
  138 updateClient' UpdateCtx{..} (ReLoad added changed removed) =
  139   -- TODO: check for changed cabal files and reload their packages
  140   do modify (undoStack .= []) -- clear the undo stack
  141      reloadModules response added changed removed

  147 -- TODO: perform refactorings without selected modules
  148 updateClient' UpdateCtx{..} (PerformRefactoring refact modPath selection args shutdown diffMode)
  149   = do (selectedMod, otherMods) <- getFileMods modPath
  150        performRefactoring (refact:selection:args)
  151                           (maybe (Left modPath) Right selectedMod) otherMods
  152        when shutdown $ liftIO $ response Disconnected
  153        return (not shutdown)
  154   where performRefactoring cmd actualMod otherMods = do
  155           res <- lift $ performCommand refactorings cmd actualMod otherMods
  156           case res of
  157             Left err -> liftIO $ response $ ErrorMessage err
  158             Right diff -> do changedMods <- applyChanges diff
  159                              if not diffMode
  160                                then when (not (disableHistory $ sharedOptions options)) $ updateHistory changedMods
  161                                else liftIO $ response $ DiffInfo (concatMap (either snd (^. _4)) changedMods)
  162                              isWatching <- gets (isJust . (^. watchProc))
  163                              when (not isWatching && not shutdown && not diffMode)
  164                               -- if watch is on, then it will automatically
  165                               -- reload changed files, otherwise we do it manually
  166                               $ void $ reloadChanges (map ((^. sfkModuleName) . (^. _1)) (rights changedMods))
  167 
  168         applyChanges changes = do
  169           forM changes $ \case
  170             ModuleCreated n m otherM -> do
  171               mcs <- gets (^. refSessMCs)
  172               Just (_, otherMR) <- gets (lookupModInSCs otherM . (^. refSessMCs))
  173               let Just otherMS = otherMR ^? modRecMS
  174                   Just mc = lookupModuleColl (otherM ^. sfkModuleName) mcs
  175               otherSrcDir <- liftIO $ getSourceDir otherMS
  176               let loc = toFileName otherSrcDir n
  177               let newCont = prettyPrint m
  178               when (not diffMode) $ do
  179                 modify' $ refSessMCs & traversal & filtered (\mc' -> (mc' ^. mcId) == (mc ^. mcId)) & mcModules
  180                             .- Map.insert (SourceFileKey loc n) (ModuleNotLoaded False False)
  181                 liftIO $ withBinaryFile loc WriteMode $ \handle -> do
  182                   hSetEncoding handle utf8
  183                   hPutStr handle newCont
  184                   hFlush handle
  185                 lift $ addTarget (Target (TargetFile loc Nothing) True Nothing)
  186               return $ Right (SourceFileKey loc n, loc, RemoveAdded loc, createUnifiedDiff loc "" newCont)
  187             ContentChanged (n,m) -> do
  188               let newCont = prettyPrint m
  189                   file = n ^. sfkFileName
  190               origCont <- liftIO $ withBinaryFile file ReadMode $ \handle -> do
  191                 hSetEncoding handle utf8
  192                 StrictIO.hGetContents handle
  193               let undo = createUndo 0 $ getGroupedDiff origCont newCont
  194               let unifiedDiff = createUnifiedDiff file origCont newCont
  195               when (not diffMode) $ do
  196                liftIO $ withBinaryFile file WriteMode $ \handle -> do
  197                  hSetEncoding handle utf8
  198                  hPutStr handle newCont
  199                  hFlush handle
  200               return $ Right (n, file, UndoChanges file undo, unifiedDiff)
  201             ModuleRemoved mod -> do
  202               Just (sfk,_) <- gets (lookupModuleInSCs mod . (^. refSessMCs))
  203               let file = sfk ^. sfkFileName
  204               origCont <- liftIO (StrictBS.unpack <$> StrictBS.readFile file)
  205               when (not diffMode) $ do
  206                 lift $ removeTarget (TargetFile file Nothing)
  207                 modify' $ (refSessMCs .- removeModule mod)
  208                 liftIO $ removeFile file
  209               return $ Left (RestoreRemoved file origCont, createUnifiedDiff file origCont "")
  210 
  211         reloadChanges changedMods
  212           = reloadChangedModules (\ms -> response (LoadedModules [(getModSumOrig ms, getModSumName ms)]))
  213                                  (\mss -> response (LoadingModules (map getModSumOrig mss)))
  214                                  (\ms -> getModSumName ms `elem` changedMods)
  215 
  216         updateHistory :: [Either (UndoRefactor, b) (SourceFileKey, FilePath, UndoRefactor, String)] -> DaemonSession ()
  217         updateHistory changedMods
  218           = do modify (undoStack .- (map (either fst (^. _3)) changedMods :))
  219                -- force the evaluation of the undo stack to prevent older versions of
  220                -- modules seeming to be used when they could be garbage collected
  221                us <- gets (^. undoStack)
  222                liftIO $ evaluate $ force us
  223                return ()

  225 addPackages :: (ResponseMsg -> IO ()) -> [FilePath] -> DaemonSession ()
  226 addPackages _ [] = return ()
  227 addPackages resp packagePathes = do
  228   nonExisting <- filterM ((return . not) <=< liftIO . doesDirectoryExist) packagePathes
  229   forM_ packagePathes watchNew -- put a file system watch on each package
  230   DaemonSessionState {..} <- get
  231   if (not (null nonExisting))
  232     then liftIO $ resp $ ErrorMessage $ "The following packages are not found: " ++ concat (intersperse ", " nonExisting)
  233     else do
  234       -- clear existing removed packages
  235       existingMCs <- gets (^. refSessMCs)
  236       let existing = (existingMCs ^? traversal & filtered isTheAdded & mcModules & traversal & modRecMS)
  237           existingModNames = map ms_mod existing
  238       needToReload <- (filter (\ms -> not $ ms_mod ms `elem` existingModNames))
  239                         <$> getReachableModules (\_ -> return ()) (\ms -> ms_mod ms `elem` existingModNames)
  240       modify' $ refSessMCs .- filter (not . isTheAdded) -- remove the added package from the database
  241       forM_ existing $ \ms -> removeTarget (TargetFile (getModSumOrig ms) Nothing)
  242       modifySession (\s -> s { hsc_mod_graph = filter (not . (`elem` existingModNames) . ms_mod) (hsc_mod_graph s) })
  243       -- load new modules
  244       pkgDBok <- initializePackageDBIfNeeded
  245       if pkgDBok then do
  246         loadPackagesFrom
  247           (\ms -> resp (LoadedModules [(getModSumOrig ms, getModSumName ms)])
  248                     >> return (getModSumOrig ms))
  249           (resp . LoadingModules . map getModSumOrig)
  250           (\st fp -> maybeToList <$> detectAutogen fp (st ^. packageDB)) packagePathes
  251         mapM_ (reloadModule (\_ -> return ())) needToReload -- don't report consequent reloads (not expected)
  252       else liftIO $ resp $ ErrorMessage $ "Attempted to load two packages with different package DB. "
  253                                             ++ "Stack, cabal-sandbox and normal packages cannot be combined"
  254   where isTheAdded mc = (mc ^. mcRoot) `elem` packagePathes
  255         initializePackageDBIfNeeded = do
  256           pkgDBAlreadySet <- gets (^. packageDBSet)
  257           pkgDB <- gets (^. packageDB)
  258           locs <- liftIO $ mapM (packageDBLoc pkgDB) packagePathes
  259           case locs of
  260             firstLoc:rest ->
  261               if | not (all (== firstLoc) rest)
  262                      -> return False
  263                  | pkgDBAlreadySet -> do
  264                      pkgDBLocs <- gets (^. packageDBLocs)
  265                      return (pkgDBLocs == firstLoc)
  266                  | otherwise -> do
  267                      usePackageDB firstLoc
  268                      modify' ((packageDBSet .= True) . (packageDBLocs .= firstLoc))
  269                      return True
  270             [] -> return True

  272 -- | Reloads changed modules to have an up-to-date version loaded
  273 reloadModules :: (ResponseMsg -> IO ()) -> [FilePath] -> [FilePath] -> [FilePath] -> DaemonSession Bool
  274 reloadModules resp added changed removed = do
  275   lift $ forM_ removed (\src -> removeTarget (TargetFile src Nothing))
  276   -- remove targets deleted
  277   modify' $ refSessMCs & traversal & mcModules
  278               .- Map.filter (\m -> maybe True ((`notElem` removed) . getModSumOrig) (m ^? modRecMS))
  279   modifySession (\s -> s { hsc_mod_graph = filter (\mod -> getModSumOrig mod `notElem` removed) (hsc_mod_graph s) })
  280   -- reload changed modules
  281   -- TODO: filter those that are in reloaded packages
  282   reloadChangedModules (\ms -> resp (LoadedModules [(getModSumOrig ms, getModSumName ms)]))
  283                        (\mss -> resp (LoadingModules (map getModSumOrig mss)))
  284                        (\ms -> getModSumOrig ms `elem` changed)
  285   mcs <- gets (^. refSessMCs)
  286   let mcsToReload = filter (\mc -> any ((mc ^. mcRoot) `isPrefixOf`) added && isNothing (moduleCollectionPkgId (mc ^. mcId))) mcs
  287   addPackages resp (map (^. mcRoot) mcsToReload) -- reload packages containing added modules
  288   return True

  290 -- | Creates a compressed set of changes in one file
  291 createUndo :: Eq a => Int -> [Diff [a]] -> [(Int, Int, [a])]
  292 createUndo i (Both str _ : rest) = createUndo (i + length str) rest
  293 createUndo i (First rem : Second add : rest)
  294   = (i, i + length add, rem) : createUndo (i + length add) rest
  295 createUndo i (First rem : rest) = (i, i, rem) : createUndo i rest
  296 createUndo i (Second add : rest)
  297   = (i, i + length add, []) : createUndo (i + length add) rest
  298 createUndo _ [] = []

  300 -- | Creates a unified-style diff of two texts. Only used when the user wants to know what would change.
  301 createUnifiedDiff :: FilePath -> String -> String -> String
  302 createUnifiedDiff name left right
  303   = render $ prettyContextDiff (PP.text name) (PP.text name) PP.text $ getContextDiff 3 (lines left) (lines right)

  305 -- | Undo a refactoring change using the information that was saved earlier.
  306 performUndo :: UndoRefactor -> IO ()
  307 performUndo (RemoveAdded fp) = removeFile fp
  308 performUndo (RestoreRemoved fp cont)
  309   = liftIO $ withBinaryFile fp WriteMode $ \handle -> do
  310       hSetEncoding handle utf8
  311       hPutStr handle cont
  312 performUndo (UndoChanges fp changes) = do
  313   cont <- liftIO $ withBinaryFile fp ReadMode $ \handle -> do
  314     hSetEncoding handle utf8
  315     StrictIO.hGetContents handle
  316   liftIO $ withBinaryFile fp WriteMode $ \handle -> do
  317       hSetEncoding handle utf8
  318       hPutStr handle (performUndoChanges 0 changes cont)
  319 
  320 -- | Undo the changes in one file using the information that was saved earlier. See 'createUndo'.
  321 performUndoChanges :: Int -> FileDiff -> String -> String
  322 performUndoChanges i ((start,end,replace):rest) str | i == start
  323   = replace ++ performUndoChanges end rest (drop (end-start) str)
  324 performUndoChanges i diffs (c:str) = c : performUndoChanges (i+1) diffs str
  325 performUndoChanges _ _ [] = []

  327 -- | Get the files added by a refactoring.
  328 getUndoAdded :: [UndoRefactor] -> [FilePath]
  329 getUndoAdded = catMaybes . map (\case RestoreRemoved fp _ -> Just fp
  330                                       _                   -> Nothing)
  331 
  332 -- | Get the files changed by a refactoring.
  333 getUndoChanged :: [UndoRefactor] -> [FilePath]
  334 getUndoChanged = catMaybes . map (\case UndoChanges fp _ -> Just fp
  335                                         _                -> Nothing)
  336 
  337 -- | Get the files removed by a refactoring.
  338 getUndoRemoved :: [UndoRefactor] -> [FilePath]
  339 getUndoRemoved = catMaybes . map (\case RemoveAdded fp -> Just fp
  340                                         _              -> Nothing)

  345 usePackageDB :: GhcMonad m => [FilePath] -> m ()
  346 usePackageDB [] = return ()
  347 usePackageDB pkgDbLocs
  348   = do dfs <- getSessionDynFlags
  349        dfs' <- liftIO $ fmap fst $ initPackages
  350                  $ dfs { packageDBFlags = map (PackageDB . PkgConfFile) pkgDbLocs ++ packageDBFlags dfs
  351                        , pkgDatabase = Nothing
  352                        }
  353        void $ setSessionDynFlags dfs'
