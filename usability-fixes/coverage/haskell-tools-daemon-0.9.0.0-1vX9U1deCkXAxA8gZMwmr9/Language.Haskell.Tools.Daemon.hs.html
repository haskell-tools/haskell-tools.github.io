   47 -- | Starts the daemon process. This will not return until the daemon stops. You can use this entry
   48 -- point when the other endpoint of the client connection is not needed, for example, when you use
   49 -- socket connection to connect to the daemon process.
   50 runDaemon' :: [RefactoringChoice IdDom] -> DaemonOptions -> IO ()
   51 runDaemon' refactorings args = do store <- newEmptyMVar
   52                                   runDaemon refactorings socketMode store args
   53 
   54 -- | Starts the daemon process. This will not return until the daemon stops.
   55 -- The daemon process is parameterized by the refactorings you can use in it. This entry point gives
   56 -- back the other endpoint of the connection so it can be used to run the daemon in the same process.
   57 runDaemon :: [RefactoringChoice IdDom] -> WorkingMode a -> MVar a -> DaemonOptions -> IO ()
   58 runDaemon _ _ _ DaemonOptions{..} | daemonVersion
   59   = putStrLn $ showVersion version
   60 runDaemon refactorings mode connStore config@DaemonOptions{..} = withSocketsDo $
   61     do when (not silentMode) $ putStrLn $ "Starting Haskell Tools daemon"
   62        hSetBuffering stdout LineBuffering
   63        hSetBuffering stderr LineBuffering
   64        conn <- daemonConnect mode portNumber
   65        putMVar connStore conn
   66        when (not silentMode) $ putStrLn $ "Connection established"
   67        ghcSess <- initGhcSession (generateCode sharedOptions)
   68        state <- newMVar initSession
   69        (wp,th) <- if noWatch sharedOptions
   70                     then return (Nothing, [])
   71                     else createWatchProcess'
   72                            (watchExe sharedOptions) ghcSess state (daemonSend mode conn)
   73        modifyMVarMasked_ state ( \s -> return s { _watchProc = wp, _watchThreads = th })
   74        serverLoop refactorings mode conn config ghcSess state
   75        case wp of Just watchProcess -> stopWatch watchProcess th
   76                   Nothing -> return ()
   77        daemonDisconnect mode conn
   78 
   79 -- | Starts the server loop, receiving requests from the client and updated the server state
   80 -- according to these.
   81 serverLoop :: [RefactoringChoice IdDom] -> WorkingMode a -> a -> DaemonOptions -> Session
   82                 -> MVar DaemonSessionState -> IO ()
   83 serverLoop refactorings mode conn options ghcSess state =
   84   do msgs <- daemonReceive mode conn
   85      continue <- mapM respondToMsg msgs
   86      sessionData <- readMVar state
   87      when (not (sessionData ^. exiting) && all (== True) continue)
   88        $ serverLoop refactorings mode conn options ghcSess state
   89    `catches` exceptionHandlers (serverLoop refactorings mode conn options ghcSess state)
   90                                (daemonSend mode conn . ErrorMessage)
   91   where respondToMsg (Right req)
   92           = do when (not (silentMode options)) $ putStrLn $ "Message received: " ++ show req
   93                respondTo options refactorings ghcSess state (daemonSend mode conn) req
   94            `catches` userExceptionHandlers
   95                         (\s -> daemonSend mode conn (ErrorMessage s) >> return True)
   96                         (\err hint -> daemonSend mode conn (CompilationProblem err hint) >> return True)
   97         respondToMsg (Left msg) = do daemonSend mode conn $ ErrorMessage $ "MALFORMED MESSAGE: " ++ msg
   98                                      return True

  100 -- | Responds to a client request by modifying the daemon and GHC state accordingly.
  101 respondTo :: DaemonOptions -> [RefactoringChoice IdDom] -> Session -> MVar DaemonSessionState
  102                -> (ResponseMsg -> IO ()) -> ClientMessage -> IO Bool
  103 respondTo options refactorings ghcSess state next req
  104   = modifyMVar state (\st -> swap <$> reflectGhc (runStateT (updateClient options refactorings next req) st) ghcSess)
