   17 -- | Get the name of a module collection from its Id.
   18 moduleCollectionPkgId :: ModuleCollectionId -> Maybe String
   19 moduleCollectionPkgId (DirectoryMC _) = Nothing
   20 moduleCollectionPkgId (LibraryMC id) = Just id
   21 moduleCollectionPkgId (ExecutableMC id _) = Just id
   22 moduleCollectionPkgId (TestSuiteMC id _) = Just id
   23 moduleCollectionPkgId (BenchmarkMC id _) = Just id

   27 -- | Find the module collection where the given module is contained. Based on module name.
   28 lookupModuleColl :: String -> [ModuleCollection SourceFileKey] -> Maybe (ModuleCollection SourceFileKey)
   29 lookupModuleColl moduleName = find (any ((moduleName ==) . (^. sfkModuleName)) . Map.keys . (^. mcModules))
   30 
   31 -- | Find the module collection where the given module is contained. Based on source file name.
   32 lookupSourceFileColl :: FilePath -> [ModuleCollection SourceFileKey] -> Maybe (ModuleCollection SourceFileKey)
   33 lookupSourceFileColl fp = find (any ((fp ==) . (^. sfkFileName)) . Map.keys . (^. mcModules))
   34 
   35 -- | Find the module with the given name. Based on module name.
   36 lookupModuleInSCs :: String -> [ModuleCollection SourceFileKey] -> Maybe (SourceFileKey, ModuleRecord)
   37 lookupModuleInSCs moduleName = find ((moduleName ==) . (^. sfkModuleName) . fst) . concatMap (Map.assocs . (^. mcModules))
   38 
   39 -- | Find the module with the given name. Based on source file name.
   40 lookupModInSCs :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Maybe (SourceFileKey, ModuleRecord)
   41 lookupModInSCs key = find (((key ^. sfkFileName) ==) . (^. sfkFileName) . fst) . concatMap (Map.assocs . (^. mcModules))
   42 
   43 -- | Get the module with the given name and source file key.
   44 lookupSFKInSCs :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Maybe ModuleRecord
   45 lookupSFKInSCs key = listToMaybe . catMaybes . map (Map.lookup key . (^. mcModules))
   46 
   47 -- | Remove a module with the given name. Based on module name.
   48 removeModule :: String -> [ModuleCollection SourceFileKey] -> [ModuleCollection SourceFileKey]
   49 removeModule moduleName = map (mcModules .- Map.filterWithKey (\k _ -> moduleName /= (k ^. sfkModuleName)))
   50 
   51 -- | Check if we already generated code for the given module. Based on both module name and source
   52 -- file name.
   53 hasGeneratedCode :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Bool
   54 hasGeneratedCode key = maybe False (\case ModuleCodeGenerated {} -> True; _ -> False)
   55                          . lookupSFKInSCs key
   56 
   57 -- | Check if the given module needs code generation. Finds the module if no source file name is
   58 -- present and module names check or if both module names and source file names check.
   59 needsGeneratedCode :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Bool
   60 needsGeneratedCode key mcs = maybe False (\case ModuleCodeGenerated {} -> True; ModuleNotLoaded True _ -> True; _ -> False)
   61                               $ lookupSFKInSCs key mcs <|> lookupSFKInSCs (sfkFileName .= "" $ key) mcs
   62 
   63 -- | Marks the given module for code generation. Finds the module if no source file name is
   64 -- present and module names check or if both module names and source file names check.
   65 codeGeneratedFor :: SourceFileKey -> [ModuleCollection SourceFileKey] -> [ModuleCollection SourceFileKey]
   66 codeGeneratedFor key = map (mcModules .- Map.adjust setCodeGen (sfkFileName .= "" $ key) . Map.adjust setCodeGen key)
   67   where setCodeGen (ModuleTypeChecked mod ms) = ModuleCodeGenerated mod ms
   68         setCodeGen (ModuleNotLoaded _ exp) = ModuleNotLoaded True exp
   69         setCodeGen m = m
   70 
   71 -- | Check if the given module has been already loaded. Based on both module name and source
   72 -- file name.
   73 isAlreadyLoaded :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Bool
   74 isAlreadyLoaded key = maybe False (\case (_, ModuleNotLoaded {}) -> False; _ -> True)
   75                          . find ((key ==) . fst) . concatMap (Map.assocs . (^. mcModules))
   76 
   77 -- | Insert a module with a source file key to our database if it wasn't there already
   78 insertIfMissing :: SourceFileKey -> [ModuleCollection SourceFileKey] -> [ModuleCollection SourceFileKey]
   79 insertIfMissing sfk mods
   80   = case lookupSFKInSCs sfk mods of
   81       Just x -> mods
   82       Nothing -> element 0 & mcModules .- Map.insert sfk (ModuleNotLoaded False False) $ mods
