    1 module Language.Haskell.Tools.Daemon.Options
    2          (DaemonOptions(..), parseDaemonCLI, SharedDaemonOptions(..), sharedOptionsParser) where

    4 import Data.Semigroup ((<>))
    5 import Options.Applicative

    7 -- | Command line options for the daemon process.
    8 data DaemonOptions = DaemonOptions { daemonVersion :: Bool
    9                                    , portNumber :: Int
   10                                    , silentMode :: Bool
   11                                    , sharedOptions :: SharedDaemonOptions
   12                                    }
   13 
   14 -- | Command line options shared by CLI and daemon.
   15 data SharedDaemonOptions = SharedDaemonOptions { noWatch :: Bool
   16                                                , watchExe :: Maybe FilePath
   17                                                , generateCode :: Bool
   18                                                , disableHistory :: Bool
   19                                                }
   20 
   21 parseDaemonCLI = execParser daemonCLI
   22 
   23 daemonCLI = info (daemonOptionsParser <**> helper)
   24                  (fullDesc
   25                    <> progDesc "Start the Haskell-tools daemon process"
   26                    <> header "ht-daemon: a background process for Haskell development tools.")
   27 
   28 daemonOptionsParser :: Parser DaemonOptions
   29 daemonOptionsParser = DaemonOptions <$> version <*> port <*> silent <*> sharedOptionsParser
   30   where version = switch (long "version"
   31                             <> short 'v'
   32                             <> help "Show the version of this software")
   33         port = option auto
   34                  (long "port"
   35                    <> short 'p'
   36                    <> value 4123
   37                    <> showDefault
   38                    <> help "Set the number of port where the daemon will wait for clients."
   39                    <> metavar "PORT_NUMBER")
   40         silent = switch (long "silent"
   41                           <> short 's'
   42                           <> help "Set to disable messages from daemon.")
   43 
   44 sharedOptionsParser :: Parser SharedDaemonOptions
   45 sharedOptionsParser = SharedDaemonOptions <$> noWatch <*> watch <*> generateCode <*> noHistory
   46   where noWatch = switch (long "no-watch"
   47                             <> help "Disables file system watching.")
   48         watch = optional . strOption
   49                   $ long "watch-exe"
   50                       <> short 'w'
   51                       <> help "The file path of the watch executable that is used to monitor file system changes."
   52                       <> metavar "WATH_PATH"
   53         generateCode = switch (long "generate-code"
   54                                  <> help "Always generate code for the modules of the loaded project.")
   55         noHistory = switch (long "no-history"
   56                               <> help "Disables saving the performed refactorings.")
