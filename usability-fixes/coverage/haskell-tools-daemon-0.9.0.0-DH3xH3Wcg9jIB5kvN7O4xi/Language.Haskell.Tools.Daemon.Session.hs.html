   43 -- | Load packages from the given directories. Loads modules, performs the given callback action, warns for duplicate modules.
   44 loadPackagesFrom :: (ModSummary -> IO a)
   45                       -> ([ModSummary] -> IO ())
   46                       -> (DaemonSessionState -> FilePath -> IO [FilePath])
   47                       -> [FilePath]
   48                       -> DaemonSession [a]
   49 loadPackagesFrom report loadCallback additionalSrcDirs packages =
   50   do modColls <- liftIO $ getAllModules packages
   51      st <- get
   52      moreSrcDirs <- liftIO $ mapM (additionalSrcDirs st) packages
   53      lift $ useDirs ((modColls ^? traversal & mcSourceDirs & traversal) ++ concat moreSrcDirs)
   54      mcs' <- liftIO (traversal !~ locateModules $ modColls)
   55      modify' (refSessMCs .- (++ mcs'))
   56      mcs <- gets (^. refSessMCs)
   57      let alreadyLoadedFilesInOtherPackages
   58            = concatMap (map (^. sfkFileName) . Map.keys . Map.filter (isJust . (^? typedRecModule)) . (^. mcModules))
   59                        (filter (\mc -> (mc ^. mcRoot) `notElem` packages) mcs)
   60      currentTargets <- map targetId <$> (lift getTargets)
   61      lift $ mapM_ (\t -> when (targetId t `notElem` currentTargets) (addTarget t))
   62                   (map makeTarget $ List.nubBy ((==) `on` (^. sfkFileName))
   63                                   $ List.sort $ concatMap getExposedModules mcs')
   64      withAlteredDynFlags (liftIO . fmap (st ^. ghcFlagsSet) . setupLoadFlags (mcs ^? traversal & mcId) (mcs ^? traversal & mcRoot)
   65                                                                              (mcs ^? traversal & mcDependencies & traversal)
   66                                                                              (foldl @[] (>=>) return (mcs ^? traversal & mcLoadFlagSetup))) $ do
   67        -- need to update package state when setting the list of visible packages
   68        dfs <- getSessionDynFlags
   69        (dfs', _) <- liftIO $ initPackages dfs
   70        setSessionDynFlags dfs'
   71        modsForColls <- lift $ depanal [] True
   72        let modsToParse = flattenSCCs $ topSortModuleGraph False modsForColls Nothing
   73            actuallyCompiled = filter (\ms -> getModSumOrig ms `notElem` alreadyLoadedFilesInOtherPackages) modsToParse
   74        liftIO $ loadCallback actuallyCompiled
   75        modify (refSessMCs .- foldl (.) id (map (insertIfMissing . keyFromMS) actuallyCompiled))
   76        void $ checkEvaluatedMods (\_ -> return ()) actuallyCompiled
   77        mods <- mapM (loadModule report) actuallyCompiled
   78        return mods
   79 
   80   where getExposedModules :: ModuleCollection k -> [k]
   81         getExposedModules
   82           = Map.keys . Map.filter (\v -> fromMaybe True (v ^? recModuleExposed)) . (^. mcModules)
   83 
   84         locateModules :: ModuleCollection ModuleNameStr -> IO (ModuleCollection SourceFileKey)
   85         locateModules mc
   86           = mcModules !~ ((Map.fromList <$>)
   87                             . mapM (locateModule (mc ^. mcSourceDirs) (mc ^. mcModuleFiles))
   88                             . Map.assocs) $ mc
   89 
   90         locateModule :: [FilePath] -> [(ModuleNameStr, FilePath)]
   91                           -> (ModuleNameStr, ModuleRecord) -> IO (SourceFileKey,ModuleRecord)
   92         locateModule srcDirs modMaps (modName, record)
   93           = do candidate <- createTargetCandidate srcDirs modMaps modName
   94                return (SourceFileKey (either (const "") id candidate) modName, record)
   95 
   96 
   97         loadModule :: (ModSummary -> IO a) -> ModSummary -> DaemonSession a
   98         loadModule report ms
   99           = do needsCodeGen <- gets (needsGeneratedCode (keyFromMS ms) . (^. refSessMCs))
  100                reloadModule report (if needsCodeGen then forceCodeGen ms else ms)
  101 
  102         -- | Creates a possible target from a module name. If possible, finds the
  103         -- corresponding source file to distinguish between modules of the same name.
  104         createTargetCandidate :: [FilePath] -> [(ModuleNameStr, FilePath)] -> ModuleNameStr
  105                                     -> IO (Either ModuleName FilePath)
  106         createTargetCandidate srcFolders mapping modName
  107           = wrapEither <$> filterM doesFileExist
  108                              (map (</> toFileName modName) srcFolders)
  109           where toFileName modName
  110                   = case lookup modName mapping of
  111                       Just fileName -> fileName
  112                       Nothing -> List.intercalate [pathSeparator] (splitOn "." modName) <.> "hs"
  113                 wrapEither [] = Left (GHC.mkModuleName modName)
  114                 wrapEither (fn:_) = Right fn
  115 
  116         makeTarget (SourceFileKey "" modName) = Target (TargetModule (GHC.mkModuleName modName)) True Nothing
  117         makeTarget (SourceFileKey filePath _) = Target (TargetFile filePath Nothing) True Nothing

  121 -- | Get the module that is selected for refactoring and all the other modules.
  122 getMods :: Maybe SourceFileKey -> DaemonSession ( Maybe (SourceFileKey, UnnamedModule IdDom)
  123                                                 , [(SourceFileKey, UnnamedModule IdDom)] )
  124 getMods actMod
  125   = do mcs <- gets (^. refSessMCs)
  126        return $ ( (_2 !~ (^? typedRecModule)) =<< flip lookupModInSCs mcs =<< actMod
  127                 , filter ((actMod /=) . Just . fst) $ concatMap (catMaybes . map (_2 !~ (^? typedRecModule)) . Map.assocs . (^. mcModules)) mcs )
  128 
  129 -- | Get the module that is selected for refactoring and all the other modules.
  130 getFileMods :: String -> DaemonSession ( Maybe (SourceFileKey, UnnamedModule IdDom)
  131                                        , [(SourceFileKey, UnnamedModule IdDom)] )
  132 getFileMods fname
  133   = do mcs <- gets (^. refSessMCs)
  134        let mods = mapMaybe (\(k,m) -> fmap (,k) (m ^? modRecMS))
  135                            (concatMap @[] Map.assocs $ (mcs ^? traversal & mcModules))
  136        let sfs = catMaybes $ map (\(ms,k) -> if | Just fname == fmap normalise (ml_hs_file (ms_location ms)) -> Just (False, k)
  137                                                 | fname == getModSumName ms -> Just (True, k)
  138                                                 | otherwise -> Nothing) mods
  139        case List.sort sfs of (_,sf):_ -> getMods (Just sf)
  140                              []       -> getMods Nothing

  142 -- | Reload the modules that have been changed (given by predicate). Pefrom the callback.
  143 reloadChangedModules :: (ModSummary -> IO a) -> ([ModSummary] -> IO ()) -> (ModSummary -> Bool)
  144                            -> DaemonSession [a]
  145 reloadChangedModules report loadCallback isChanged = do
  146   reachable <- getReachableModules loadCallback isChanged
  147   void $ checkEvaluatedMods report reachable
  148   -- remove module from session before reloading it, resolves space leak
  149   let reachableMods = map ms_mod_name reachable
  150       notReloaded = (`notElem` reachableMods) . GHC.moduleName . mi_module . hm_iface
  151   lift $ modifySession (\s -> s { hsc_HPT = filterHpt notReloaded (hsc_HPT s)
  152                                 , hsc_mod_graph = filter ((`notElem` reachableMods) . ms_mod_name) (hsc_mod_graph s)
  153                                 })
  154   mapM (reloadModule report) reachable

  156 -- | Get all modules that can be accessed from a given set of modules. Can be used to select which
  157 -- modules need to be reloaded after a change.
  158 getReachableModules :: ([ModSummary] -> IO ()) -> (ModSummary -> Bool) -> DaemonSession [ModSummary]
  159 getReachableModules loadCallback selected = do
  160   mcs <- gets (^. refSessMCs)
  161   -- IMPORTANT: make sure that the module collection is not passed into the flags, they
  162   -- might not be evaluated and then the reference could prevent garbage collection
  163   -- of entire ASTs
  164   withAlteredDynFlags (liftIO . setupLoadFlags (mcs ^? traversal & mcId) (mcs ^? traversal & mcRoot)
  165                                                (mcs ^? traversal & mcDependencies & traversal)
  166                                                (foldl @[] (>=>) return (mcs ^? traversal & mcLoadFlagSetup)) ) $ do
  167     allMods <- lift $ depanal [] True
  168     let (allModsGraph, lookup) = moduleGraphNodes False allMods
  169         changedMods = catMaybes $ map (\ms -> lookup (ms_hsc_src ms) (moduleName $ ms_mod ms))
  170                         $ filter selected allMods
  171         recompMods = map (ms_mod . getModFromNode) $ reachablesG (transposeG allModsGraph) changedMods
  172         sortedMods = reverse $ topologicalSortG allModsGraph
  173         sortedRecompMods = filter ((`elem` recompMods) . ms_mod) $ map getModFromNode sortedMods
  174     liftIO $ loadCallback sortedRecompMods
  175     return sortedRecompMods

  177 -- | Reload a given module. Perform a callback.
  178 reloadModule :: (ModSummary -> IO a) -> ModSummary -> DaemonSession a
  179 reloadModule report ms = do
  180   ghcfl <- gets (^. ghcFlagsSet)
  181   mcs <- gets (^. refSessMCs)
  182   let fp = getModSumOrig ms
  183       modName = getModSumName ms
  184       codeGen = needsGeneratedCode (keyFromMS ms) mcs
  185   case lookupSourceFileColl fp mcs <|> lookupModuleColl modName mcs of
  186     Just mc -> reloadModuleIn codeGen modName ghcfl mc
  187     Nothing -> case mcs of mc:_ -> reloadModuleIn codeGen modName ghcfl mc
  188                            []   -> error "reloadModule: module collections empty"
  189   where
  190     reloadModuleIn codeGen modName ghcfl mc = do
  191       let dfs = ms_hspp_opts ms
  192       -- IMPORTANT: make sure that the module collection is not passed into the flags, they
  193       -- might not be evaluated and then the reference could prevent garbage collection
  194       -- of entire ASTs
  195       dfs' <- liftIO $ fmap ghcfl $ ((mc ^. mcFlagSetup) <=< (mc ^. mcLoadFlagSetup)) dfs
  196       let ms' = ms { ms_hspp_opts = dfs' }
  197       newm <- lift $ withAlteredDynFlags (\_ -> return dfs') $
  198         parseTyped (if codeGen then forceCodeGen ms' else ms')
  199       -- replace the module in the program database
  200       modify' $ refSessMCs & traversal & filtered (\c -> (c ^. mcId) == (mc ^. mcId)) & mcModules
  201                   .- Map.insert (keyFromMS ms') ((if codeGen then ModuleCodeGenerated else ModuleTypeChecked) newm ms')
  202                        . Map.delete (SourceFileKey "" modName)
  203       liftIO $ report ms'

  205 -- | Finds out if a newly added module forces us to generate code for another one.
  206 -- If the other is already loaded it will be reloaded.
  207 checkEvaluatedMods :: (ModSummary -> IO a) -> [ModSummary] -> DaemonSession [a]
  208 checkEvaluatedMods report mods = do
  209     mcs <- gets (^. refSessMCs)
  210     -- IMPORTANT: make sure that the module collection is not passed into the flags, they
  211     -- might not be evaluated and then the reference could prevent garbage collection
  212     -- of entire ASTs
  213     let lookupFlags ms = maybe return (^. mcFlagSetup) mc
  214           where modName = getModSumName ms
  215                 mc = lookupModuleColl modName mcs
  216     modsNeedCode <- lift (getEvaluatedMods mods lookupFlags)
  217     catMaybes <$> forM modsNeedCode (reloadIfNeeded mcs)
  218   where reloadIfNeeded mcs ms
  219           = let key = keyFromMS ms
  220               in if not (hasGeneratedCode key mcs)
  221                    then do -- mark the module for code generation
  222                            modify $ refSessMCs .- codeGeneratedFor key
  223                            if (isAlreadyLoaded key mcs) then
  224                                -- The module is already loaded but code is not generated. Need to reload.
  225                                Just <$> lift (codeGenForModule report (codeGeneratedFor key mcs) ms)
  226                              else return Nothing
  227                    else return Nothing

  229 -- | Re-load the module with code generation enabled. Must be used when the module had already been loaded,
  230 -- but code generation were not enabled by then.
  231 codeGenForModule :: (ModSummary -> IO a) -> [ModuleCollection SourceFileKey] -> ModSummary -> Ghc a
  232 codeGenForModule report mcs ms
  233   = let modName = getModSumName ms
  234         mc = fromMaybe (error $ "codeGenForModule: The following module is not found: " ++ modName) $ lookupModuleColl modName mcs
  235      in -- TODO: don't recompile, only load?
  236         do withAlteredDynFlags (liftIO . (mc ^. mcFlagSetup))
  237              $ void $ parseTyped (forceCodeGen ms)
  238            liftIO $ report ms

  240 -- | Check which modules can be reached from the module, if it uses template haskell.
  241 -- A definition that needs code generation can be inside a module that does not uses the
  242 -- TemplateHaskell extension.
  243 getEvaluatedMods :: [ModSummary] -> (ModSummary -> DynFlags -> IO DynFlags) -> Ghc [GHC.ModSummary]
  244 -- We cannot really get the modules that need to be linked, because we cannot rename splice content if the
  245 -- module is not type checked and that is impossible if the splice cannot be evaluated.
  246 getEvaluatedMods mods additionalFlags
  247   = do allMods <- getModuleGraph
  248        flags <- getSessionDynFlags
  249        let (allModsGraph, lookup) = moduleGraphNodes False allMods
  250        -- some flags are stored only in the module collection and are not recorded in the summary
  251        thmods <- liftIO $ filterM (\ms -> ((|| isTH ms) . xopt TemplateHaskell) <$> additionalFlags ms flags) mods
  252        let modsWithTH = catMaybes $ map (\ms -> lookup (ms_hsc_src ms) (moduleName $ ms_mod ms)) thmods
  253            recompMods = map (moduleName . ms_mod . getModFromNode) $ reachablesG allModsGraph modsWithTH
  254            sortedMods = map getModFromNode $ reverse $ topologicalSortG allModsGraph
  255            sortedTHMods = filter ((`elem` recompMods) . moduleName . ms_mod) sortedMods
  256        return sortedTHMods
  257   where isTH mod = fromEnum TemplateHaskell `member` extensionFlags (ms_hspp_opts mod)
