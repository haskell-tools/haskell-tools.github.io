    5 -- | Setting the package database to use when compiling modules. The daemon must have one single
    6 -- package database that cannot be changed after a package is loaded using that package database.
    7 -- Available package databases are the cabal global, the cabal sandbox, the stack or one that had
    8 -- been explicitely set by a file path.
    9 module Language.Haskell.Tools.Daemon.PackageDB (PackageDB(..), packageDBLoc, detectAutogen) where
   10

   24 -- | Possible package database configurations.
   25 data PackageDB = AutoDB -- ^ Decide the package database automatically.
   26                | DefaultDB -- ^ Use the global cabal package database (like when using ghc).
   27                | CabalSandboxDB -- ^ Use the sandboxed cabal package database.
   28                | StackDB -- ^ Use the stack package databases (local and snapshot).
   29                | ExplicitDB { packageDBPath :: FilePath } -- ^ Set the package database explicitely.
   30   deriving (Show, Generic)
   31 
   32 instance FromJSON PackageDB
   33 
   34 -- | Finds the location of the package database based on the configuration.
   35 packageDBLoc :: PackageDB -> FilePath -> IO [FilePath]
   36 packageDBLoc AutoDB path = concat <$> sequence [ packageDBLoc StackDB path
   37                                                , packageDBLoc CabalSandboxDB path
   38                                                , packageDBLoc DefaultDB path
   39                                                ]
   40 packageDBLoc DefaultDB _ = do
   41   dbs <- runCommandExpectOK "ghc-pkg list base"
   42   return $ maybe [] (filter (\l -> not (null l) && not (" " `isPrefixOf` l)) . lines) dbs
   43 packageDBLoc CabalSandboxDB path = do
   44   hasConfigFile <- doesFileExist (path </> "cabal.config")
   45   hasSandboxFile <- doesFileExist (path </> "cabal.sandbox.config")
   46   config <- if hasConfigFile then readFile (path </> "cabal.config")
   47               else if hasSandboxFile then readFile (path </> "cabal.sandbox.config")
   48                                      else return ""
   49   return $ map (drop (length "package-db: ")) $ filter ("package-db: " `isPrefixOf`) $ lines config
   50 packageDBLoc StackDB path = withCurrentDirectory path $ (fmap $ either (\(_ :: SomeException) -> []) id) $ try $ do
   51      projRoot <- runCommandExpectOK "stack path --allow-different-user --project-root"
   52      -- we only accept stack projects where the packages are (direct or indirect) subdirectories of the project root
   53      if maybe False (`isPrefixOf` path) projRoot then do
   54        globalDB <- runCommandExpectOK "stack path --allow-different-user --global-pkg-db"
   55        snapshotDB <- runCommandExpectOK "stack path --allow-different-user --snapshot-pkg-db"
   56        localDB <- runCommandExpectOK "stack path --allow-different-user --local-pkg-db"
   57        return $ maybeToList localDB ++ maybeToList snapshotDB ++ maybeToList globalDB
   58      else return []
   59 packageDBLoc (ExplicitDB dir) path = do
   60   hasDir <- doesDirectoryExist (path </> dir)
   61   if hasDir then return [path </> dir]
   62             else return []

   64 -- | Gets the (probable) location of autogen folder depending on which type of
   65 -- build we are using.
   66 detectAutogen :: FilePath -> PackageDB -> IO (Maybe FilePath)
   67 detectAutogen root AutoDB = do
   68   defDB <- detectAutogen root DefaultDB
   69   sandboxDB <- detectAutogen root CabalSandboxDB
   70   stackDB <- detectAutogen root StackDB
   71   return $ choose [ defDB, sandboxDB, stackDB ]
   72 detectAutogen root DefaultDB = ifExists (root </> "dist" </> "build" </> "autogen")
   73 detectAutogen root (ExplicitDB _) = ifExists (root </> "dist" </> "build" </> "autogen")
   74 detectAutogen root CabalSandboxDB = ifExists (root </> "dist" </> "build" </> "autogen")
   75 detectAutogen root StackDB = (fmap $ either (\(_ :: SomeException) -> Nothing) id) $ try $ do
   76   dir <- withCurrentDirectory root $ do
   77     distDir <- runCommandExpectOK "stack path --allow-different-user --dist-dir"
   78     return $ trim (fromMaybe "" distDir)
   79   genExists <- doesDirectoryExist (root </> dir </> "build" </> "autogen")
   80   buildExists <- doesDirectoryExist (root </> dir </> "build")
   81   if | genExists -> return $ Just (root </> dir </> "build" </> "autogen")
   82      | buildExists -> do -- for some packages, the autogen folder is inside a folder named after the package
   83                          cont <- filterM doesDirectoryExist . map ((root </> dir </> "build") </>)
   84                                    =<< listDirectory (root </> dir </> "build")
   85                          existing <- mapM ifExists (map (</> "autogen") cont)
   86                          return $ choose existing
   87      | otherwise -> return Nothing
   88 
   89 -- | Run a command and return its result if successful display an error message otherwise.
   90 runCommandExpectOK :: String -> IO (Maybe String)
   91 runCommandExpectOK cmd = do
   92   (exitCode, res, errs) <- readCreateProcessWithExitCode (shell cmd) ""
   93   case exitCode of ExitSuccess -> return (Just $ trim res)
   94                    ExitFailure code -> do putStrLn ("The command '" ++ cmd ++ "' exited with "
   95                                                       ++ show code ++ ":\n" ++ errs)
   96                                           return Nothing
   97 
   98 trim :: String -> String
   99 trim = f . f
  100    where f = reverse . dropWhile isSpace
  101 
  102 choose :: Alternative f => [f a] -> f a
  103 choose = foldl (<|>) empty
  104 
  105 ifExists :: FilePath -> IO (Maybe FilePath)
  106 ifExists fp = do exists <- doesDirectoryExist fp
  107                  if exists then return (Just fp)
  108                            else return Nothing
