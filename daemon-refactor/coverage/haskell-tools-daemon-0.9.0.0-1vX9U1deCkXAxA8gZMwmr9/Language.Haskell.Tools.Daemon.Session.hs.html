   52 -- | Load packages from the given directories. Loads modules, performs the given callback action, warns for duplicate modules.
   53 loadPackagesFrom :: (ModSummary -> IO a)
   54                       -> ([ModSummary] -> IO ())
   55                       -> (DaemonSessionState -> FilePath -> IO [FilePath])
   56                       -> [FilePath]
   57                       -> DaemonSession [a]
   58 loadPackagesFrom report loadCallback additionalSrcDirs packages =
   59   do modColls <- liftIO $ getAllModules packages
   60      st <- get
   61      moreSrcDirs <- liftIO $ mapM (additionalSrcDirs st) packages
   62      lift $ useDirs ((modColls ^? traversal & mcSourceDirs & traversal) ++ concat moreSrcDirs)
   63      mcs' <- liftIO (traversal !~ locateModules $ modColls)
   64      modify' (refSessMCs .- (++ mcs'))
   65      mcs <- gets (^. refSessMCs)
   66      let alreadyLoadedFilesInOtherPackages
   67            = concatMap (map (^. sfkFileName) . Map.keys . Map.filter (isJust . (^? typedRecModule)) . (^. mcModules))
   68                        (filter (\mc -> (mc ^. mcRoot) `notElem` packages) mcs)
   69      currentTargets <- map targetId <$> (lift getTargets)
   70      lift $ mapM_ (\t -> when (targetId t `notElem` currentTargets) (addTarget t))
   71                   (map makeTarget $ List.nubBy ((==) `on` (^. sfkFileName))
   72                                   $ List.sort $ concatMap getExposedModules mcs')
   73      let flagsForLoad = liftIO . fmap ((st ^. pkgDbFlags) . (st ^. ghcFlagsSet))
   74                                . setupLoadFlags (mcs ^? traversal & mcId) (mcs ^? traversal & mcRoot)
   75                                                 (mcs ^? traversal & mcDependencies & traversal)
   76                                                 (foldl @[] (>=>) return (mcs ^? traversal & mcLoadFlagSetup))
   77      mods <- withAlteredDynFlags flagsForLoad $ do
   78        loadVisiblePackages -- need to update package state when setting the list of visible packages
   79        modsForColls <- lift $ depanal [] True
   80        let modsToParse = flattenSCCs $ topSortModuleGraph False modsForColls Nothing
   81            actuallyCompiled = filter (\ms -> getModSumOrig ms `notElem` alreadyLoadedFilesInOtherPackages) modsToParse
   82        modify' (refSessMCs .- foldl (.) id (map (insertIfMissing . keyFromMS) actuallyCompiled))
   83        dfs <- getSessionDynFlags
   84        return actuallyCompiled
   85 
   86      liftIO $ loadCallback mods
   87      checkEvaluatedMods mods
   88      mapM (reloadModule report) mods
   89 
   90   where getExposedModules :: ModuleCollection k -> [k]
   91         getExposedModules
   92           = Map.keys . Map.filter (\v -> fromMaybe True (v ^? recModuleExposed)) . (^. mcModules)
   93 
   94         locateModules :: ModuleCollection ModuleNameStr -> IO (ModuleCollection SourceFileKey)
   95         locateModules mc
   96           = mcModules !~ ((Map.fromList <$>)
   97                             . mapM (locateModule (mc ^. mcSourceDirs) (mc ^. mcModuleFiles))
   98                             . Map.assocs) $ mc
   99 
  100         locateModule :: [FilePath] -> [(ModuleNameStr, FilePath)]
  101                           -> (ModuleNameStr, ModuleRecord) -> IO (SourceFileKey,ModuleRecord)
  102         locateModule srcDirs modMaps (modName, record)
  103           = do candidate <- createTargetCandidate srcDirs modMaps modName
  104                return (SourceFileKey (either (const "") id candidate) modName, record)
  105 
  106         -- | Creates a possible target from a module name. If possible, finds the
  107         -- corresponding source file to distinguish between modules of the same name.
  108         createTargetCandidate :: [FilePath] -> [(ModuleNameStr, FilePath)] -> ModuleNameStr
  109                                     -> IO (Either ModuleName FilePath)
  110         createTargetCandidate srcFolders mapping modName
  111           = wrapEither <$> filterM doesFileExist
  112                              (map (</> toFileName modName) srcFolders)
  113           where toFileName modName
  114                   = case lookup modName mapping of
  115                       Just fileName -> fileName
  116                       Nothing -> List.intercalate [pathSeparator] (splitOn "." modName) <.> "hs"
  117                 wrapEither [] = Left (GHC.mkModuleName modName)
  118                 wrapEither (fn:_) = Right fn
  119 
  120         makeTarget (SourceFileKey "" modName) = Target (TargetModule (GHC.mkModuleName modName)) True Nothing
  121         makeTarget (SourceFileKey filePath _) = Target (TargetFile filePath Nothing) True Nothing

  123 loadVisiblePackages :: DaemonSession ()
  124 loadVisiblePackages = do
  125   dfs <- getSessionDynFlags
  126   (dfs', _) <- liftIO $ initPackages dfs
  127   setSessionDynFlags dfs' -- set the package flags (only for this load session)
  128   modify' (pkgDbFlags .= \dfs -> dfs { pkgDatabase = pkgDatabase dfs'
  129                                      , pkgState = pkgState dfs'
  130                                      }) -- save the package database

  134 -- | Get the module that is selected for refactoring and all the other modules.
  135 getMods :: Maybe SourceFileKey -> DaemonSession ( Maybe (SourceFileKey, UnnamedModule IdDom)
  136                                                 , [(SourceFileKey, UnnamedModule IdDom)] )
  137 getMods actMod
  138   = do mcs <- gets (^. refSessMCs)
  139        return $ ( (_2 !~ (^? typedRecModule)) =<< flip lookupModInSCs mcs =<< actMod
  140                 , filter ((actMod /=) . Just . fst) $ concatMap (catMaybes . map (_2 !~ (^? typedRecModule)) . Map.assocs . (^. mcModules)) mcs )
  141 
  142 -- | Get the module that is selected for refactoring and all the other modules.
  143 getFileMods :: String -> DaemonSession ( Maybe (SourceFileKey, UnnamedModule IdDom)
  144                                        , [(SourceFileKey, UnnamedModule IdDom)] )
  145 getFileMods fname
  146   = do mcs <- gets (^. refSessMCs)
  147        let mods = mapMaybe (\(k,m) -> fmap (,k) (m ^? modRecMS))
  148                            (concatMap @[] Map.assocs $ (mcs ^? traversal & mcModules))
  149        let sfs = catMaybes $ map (\(ms,k) -> if | Just fname == fmap normalise (ml_hs_file (ms_location ms)) -> Just (False, k)
  150                                                 | fname == getModSumName ms -> Just (True, k)
  151                                                 | otherwise -> Nothing) mods
  152        case List.sort sfs of (_,sf):_ -> getMods (Just sf)
  153                              []       -> getMods Nothing

  155 -- | Reload the modules that have been changed (given by predicate). Pefrom the callback.
  156 reloadChangedModules :: (ModSummary -> IO a) -> ([ModSummary] -> IO ()) -> (ModSummary -> Bool)
  157                            -> DaemonSession [a]
  158 reloadChangedModules report loadCallback isChanged = do
  159   reachable <- getReachableModules loadCallback isChanged
  160   checkEvaluatedMods reachable
  161   -- remove module from session before reloading it, resolves space leak
  162   clearModules reachable
  163   mapM (reloadModule report) reachable
  164 
  165 -- | Clears the given modules from the GHC state to enable re-loading them
  166 -- From the Haskell-tools state we only clear them individually, when their module collection is determined.
  167 clearModules :: [ModSummary] -> DaemonSession ()
  168 clearModules [] = return ()
  169 clearModules mods = do
  170   let reachableMods = map ms_mod_name mods
  171       notReloaded = (`notElem` reachableMods) . GHC.moduleName . mi_module . hm_iface
  172   env <- getSession
  173   let hptStay = filterHpt notReloaded (hsc_HPT env)
  174   -- clear the symbol cache for iserv
  175   liftIO $ purgeLookupSymbolCache env
  176   -- clear the global linker state
  177   liftIO $ unload env (mapMaybe hm_linkable (eltsHpt hptStay))
  178   -- clear name cache
  179   nameCache <- liftIO $ readIORef $ hsc_NC env
  180   let nameCache' = nameCache { nsNames = delModuleEnvList (nsNames nameCache) (map ms_mod mods) }
  181   liftIO $ writeIORef (hsc_NC env) nameCache'
  182   -- clear home package table and module graph
  183   lift $ modifySession (\s -> s { hsc_HPT = hptStay
  184                                 , hsc_mod_graph = filter ((`notElem` reachableMods) . ms_mod_name) (hsc_mod_graph s)
  185                                 })

  187 -- | Get all modules that can be accessed from a given set of modules. Can be used to select which
  188 -- modules need to be reloaded after a change.
  189 getReachableModules :: ([ModSummary] -> IO ()) -> (ModSummary -> Bool) -> DaemonSession [ModSummary]
  190 getReachableModules loadCallback selected = do
  191   ghcflags <- gets (^. ghcFlagsSet)
  192   dbFlags <- gets (^. pkgDbFlags)
  193   mcs <- gets (^. refSessMCs)
  194   -- IMPORTANT: make sure that the module collection is not passed into the flags, they
  195   -- might not be evaluated and then the reference could prevent garbage collection
  196   -- of entire ASTs
  197   withAlteredDynFlags (liftIO . fmap (dbFlags . ghcflags)
  198                               . setupLoadFlags (mcs ^? traversal & mcId) (mcs ^? traversal & mcRoot)
  199                                                (mcs ^? traversal & mcDependencies & traversal)
  200                                                (foldl @[] (>=>) return (mcs ^? traversal & mcLoadFlagSetup))) $ do
  201     allMods <- lift $ depanal [] True
  202     sortedRecompMods <- lift $ dependentModules (return . selected)
  203     liftIO $ loadCallback sortedRecompMods
  204     return sortedRecompMods

  206 -- | Reload a given module. Perform a callback.
  207 reloadModule :: (ModSummary -> IO a) -> ModSummary -> DaemonSession a
  208 reloadModule report ms = do
  209   ghcfl <- gets (^. ghcFlagsSet)
  210   mcs <- gets (^. refSessMCs)
  211   let modName = getModSumName ms
  212       codeGen = needsGeneratedCode (keyFromMS ms) mcs
  213   case lookupModuleCollection ms mcs of
  214     Just mc -> reloadModuleIn codeGen modName ghcfl mc
  215     Nothing -> case mcs of mc:_ -> reloadModuleIn codeGen modName ghcfl mc
  216                            []   -> error "reloadModule: module collections empty"
  217   where
  218     reloadModuleIn codeGen modName ghcfl mc = do
  219       dbFlags <- gets (^. pkgDbFlags)
  220       let dfs = ms_hspp_opts ms
  221       -- IMPORTANT: make sure that the module collection is not passed into the flags, they
  222       -- might not be evaluated and then the reference could prevent garbage collection
  223       -- of entire ASTs
  224       newm <- lift $ withAlteredDynFlags (liftIO . fmap (dbFlags . ghcfl) . ((mc ^. mcFlagSetup) <=< (mc ^. mcLoadFlagSetup))) $ do
  225         dfs <- liftIO $ mc ^. mcFlagSetup $ ms_hspp_opts ms
  226         let ms' = ms { ms_hspp_opts = dfs }
  227         -- some flags are cached in mod summary, so we need to override
  228         parseTyped (if codeGen then forceCodeGen ms' else ms')
  229       -- replace the module in the program database
  230       modify' $ refSessMCs & traversal & filtered (\c -> (c ^. mcId) == (mc ^. mcId)) & mcModules
  231                   .- Map.insert (keyFromMS ms) ((if codeGen then ModuleCodeGenerated else ModuleTypeChecked) newm ms)
  232                        . removeModuleMS ms
  233       liftIO $ report ms

  235 -- | Finds out if a newly added module forces us to generate code for another one.
  236 -- If the other is already loaded it will be reloaded.
  237 checkEvaluatedMods :: [ModSummary] -> DaemonSession ()
  238 checkEvaluatedMods changed = do
  239     mcs <- gets (^. refSessMCs)
  240     -- IMPORTANT: make sure that the module collection is not passed into the flags, they
  241     -- might not be evaluated and then the reference could prevent garbage collection
  242     -- of entire ASTs
  243     let lookupFlags ms = maybe return (^. mcFlagSetup) mc $ ms_hspp_opts ms
  244           where mc = lookupModuleCollection ms mcs
  245     modsNeedCode <- lift (getEvaluatedMods changed lookupFlags)
  246     -- specify the need of code generation for later loading
  247     forM_ modsNeedCode (\ms -> modify $ refSessMCs .- codeGeneratedFor (keyFromMS ms))
  248     let reloaded = filter (\ms -> isAlreadyLoaded (keyFromMS ms) mcs) modsNeedCode
  249     clearModules reloaded
  250     -- reload modules that have already been loaded
  251     lift $ forM_ reloaded (\ms -> codeGenForModule mcs ms)
  252 
  253 -- | Re-load the module with code generation enabled. Must be used when the module had already been loaded,
  254 -- but code generation were not enabled by then.
  255 codeGenForModule :: [ModuleCollection SourceFileKey] -> ModSummary -> Ghc ()
  256 codeGenForModule mcs ms
  257 -- we don't need to update anything, just re-compile (we don't store the typed AST) and generate the code
  258   = do withAlteredDynFlags (liftIO . (mc ^. mcFlagSetup)) $ void $ parseTyped (forceCodeGen ms)
  259   where mc = fromMaybe (error $ "codeGenForModule: The following module is not found: " ++ getModSumName ms)
  260                $ lookupModuleCollection ms mcs

  262 -- | Check which modules can be reached from the module, if it uses template haskell.
  263 -- A definition that needs code generation can be inside a module that does not uses the
  264 -- TemplateHaskell extension.
  265 getEvaluatedMods :: [ModSummary] -> (ModSummary -> IO DynFlags) -> Ghc [ModSummary]
  266 -- We cannot really get the modules that need to be linked, because we cannot rename splice content if the
  267 -- module is not type checked and that is impossible if the splice cannot be evaluated.
  268 getEvaluatedMods changed additionalFlags
  269   = do let changedModulePathes = map getModSumOrig changed
  270        -- some flags are stored only in the module collection and are not recorded in the summary
  271        supportingModules (\ms -> (\flags -> getModSumOrig ms `elem` changedModulePathes && TemplateHaskell `xopt` flags)
  272                                     <$> liftIO (additionalFlags ms))
