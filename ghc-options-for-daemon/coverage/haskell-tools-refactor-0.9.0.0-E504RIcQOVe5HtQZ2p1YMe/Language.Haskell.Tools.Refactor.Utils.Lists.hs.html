   17 -- | Filters the elements of the list. By default it removes the separator before the element.
   18 -- Of course, if the first element is removed, the following separator is removed as well.
   19 filterList :: SourceInfoTraversal e => (Ann e dom SrcTemplateStage -> Bool) -> AnnList e dom -> AnnList e dom
   20 filterList pred = filterListIndexed (const pred)
   21 
   22 filterListIndexed :: SourceInfoTraversal e => (Int -> Ann e dom SrcTemplateStage -> Bool) -> AnnList e dom -> AnnList e dom
   23 filterListIndexed pred (AnnListG (NodeInfo sema src) elems)
   24   =  AnnListG (NodeInfo sema (srcTmpIndented .- fmap filterIndents $ srcTmpSeparators .- filterSeparators $ src)) filteredElems
   25   where elementsKept = findIndices (uncurry pred) (zip [0..] elems)
   26         filteredElems = sublist elementsKept elems
   27         filterIndents = sublist elementsKept
   28         filterSeparators = take (length elementsKept - 1) . sublist elementsKept
   29 
   30 -- | A version of filterList that cares about keeping non-removable code elements (like preprocessor pragmas)
   31 filterListSt :: SourceInfoTraversal e => (Ann e dom SrcTemplateStage -> Bool) -> AnnList e dom -> LocalRefactor dom (AnnList e dom)
   32 filterListSt pred = filterListIndexedSt (const pred)
   33 
   34 -- | A version of filterListIndexed that cares about keeping non-removable code elements (like preprocessor pragmas)
   35 filterListIndexedSt :: SourceInfoTraversal e => (Int -> Ann e dom SrcTemplateStage -> Bool) -> AnnList e dom -> LocalRefactor dom (AnnList e dom)
   36 filterListIndexedSt pred (AnnListG (NodeInfo sema src) elems)
   37   = do mapM_ removeChild removedElems
   38        mapM_ removeSeparator removedSeparators
   39        return $ AnnListG (NodeInfo sema (srcTmpIndented .- fmap filterIndents $ srcTmpSeparators .- filterSeparators $ src)) filteredElems
   40   where elementsKept = findIndices (uncurry pred) (zip [0..] elems)
   41         filteredElems = sublist elementsKept elems
   42         removedSeparators :: [([SourceTemplateTextElem], SrcSpan)]
   43         removedSeparators = notSublist elementsKept (src ^. srcTmpSeparators) ++ lastSepRemoved
   44         lastSepRemoved = if (length elems - 1) `notElem` elementsKept
   45                            then take 1 (reverse (sublist elementsKept $ src ^. srcTmpSeparators)) else []
   46         removedElems = notSublist elementsKept elems
   47         filterIndents = sublist elementsKept
   48         filterSeparators = take (length elementsKept - 1) . sublist elementsKept

   51 -- | Selects the given indices from a list
   52 sublist :: [Int] -> [a] -> [a]
   53 sublist indices = map snd . filter ((`elem` indices) . fst) . zip [0..]
   54 
   55 -- | Selects all but the given indices from a list
   56 notSublist :: [Int] -> [a] -> [a]
   57 notSublist indices = map snd . filter ((`notElem` indices) . fst) . zip [0..]

   59 -- | Inserts the element in the places where the two positioning functions (one checks the element before, one the element after)
   60 -- allows the placement.
   61 insertWhere :: Bool -> Ann e dom SrcTemplateStage -> (Maybe (Ann e dom SrcTemplateStage) -> Bool)
   62                  -> (Maybe (Ann e dom SrcTemplateStage) -> Bool) -> AnnList e dom
   63                  -> AnnList e dom
   64 insertWhere indented e before after al
   65   = let index = insertIndex before after (al ^? annList)
   66      in case index of
   67           Nothing -> al
   68           Just ind -> annListElems .- insertAt ind e
   69                         $ (if isEmptyAnnList then id else annListAnnot&sourceInfo .- setIndented ind . addDefaultSeparator ind)
   70                         $ al
   71   where setIndented i = srcTmpIndented .- fmap (insertAt i indented)
   72         addDefaultSeparator i al = srcTmpSeparators .- insertAt i ([NormalText $ al ^. srcTmpDefaultSeparator], noSrcSpan) $ al
   73         insertAt n e ls = let (bef,aft) = splitAt n ls in bef ++ [e] ++ aft
   74         isEmptyAnnList = (null :: [x] -> Bool) $ (al ^? annList)
   75 
   76 -- | Checks where the element will be inserted given the two positioning functions.
   77 insertIndex :: (Maybe (Ann e dom SrcTemplateStage) -> Bool) -> (Maybe (Ann e dom SrcTemplateStage) -> Bool) -> [Ann e dom SrcTemplateStage] -> Maybe Int
   78 insertIndex before after []
   79   | before Nothing && after Nothing = Just 0
   80   | otherwise = Nothing
   81 insertIndex before after list@(first:_)
   82   | before Nothing && after (Just first) = Just 0
   83   | otherwise = (+1) <$> insertIndex' before after list
   84   where insertIndex' before after (curr:rest@(next:_))
   85           | before (Just curr) && after (Just next) = Just 0
   86           | otherwise = (+1) <$> insertIndex' before after rest
   87         insertIndex' before after (curr:[])
   88           | before (Just curr) && after Nothing = Just 0
   89           | otherwise = Nothing
   90         insertIndex' before after []
   91           | before Nothing && after Nothing = Just 0
   92           | otherwise = Nothing

   94 -- | Gets the elements and separators from a list. The first separator is zipped to the second element.
   95 -- To the first element, the "" string is zipped.
   96 zipWithSeparators :: AnnList e dom -> [(([SourceTemplateTextElem], SrcSpan), Ann e dom SrcTemplateStage)]
   97 zipWithSeparators (AnnListG (NodeInfo _ src) elems)
   98   | [] <- src ^. srcTmpSeparators
   99   = zip (([], noSrcSpan) : repeat ([NormalText $ src ^. srcTmpDefaultSeparator], noSrcSpan)) elems
  100   | otherwise
  101   = zip (([], noSrcSpan) : seps ++ repeat (_2 .= noSrcSpan $ last seps)) elems
  102   where seps = src ^. srcTmpSeparators
