    1 -- | UPattern matching on UModule-level AST fragments for refactorings.
    2 {-# LANGUAGE PatternSynonyms #-}
    3 module Language.Haskell.Tools.Rewrite.Match.Modules where
    4 
    5 import Language.Haskell.Tools.AST

    8 -- | The representation of a haskell module, that is a separate compilation unit.
    9 -- It may or may not have a header.
   10 pattern Module :: FilePragmaList dom -> MaybeModuleHead dom
   11               -> ImportDeclList dom -> DeclList dom -> Module dom
   12 pattern Module filePrags head imps decls  <- Ann _ (UModule filePrags head imps decls )
   13 
   14 -- | Module declaration with name and (optional) exports
   15 pattern ModuleHead :: ModuleName dom -> MaybeModulePragma dom -> MaybeExportSpecs dom -> ModuleHead dom
   16 pattern ModuleHead n pr es <- Ann _ (UModuleHead n pr es)
   17 
   18 -- | A list of export specifications surrounded by parentheses
   19 pattern ExportSpecs :: ExportSpecList dom -> ExportSpecs dom
   20 pattern ExportSpecs specs <- Ann _ (UExportSpecs specs)
   21 
   22 -- | Export a name and related names
   23 pattern ExportSpec :: IESpec dom -> ExportSpec dom
   24 pattern ExportSpec ieSpec <- Ann _ (UDeclExport ieSpec)
   25 
   26 -- | The export of an imported module (@ module A @)
   27 pattern ModuleExport :: ModuleName dom -> ExportSpec dom
   28 pattern ModuleExport name <- Ann _ (UModuleExport name)
   29 
   30 -- | Marks a name to be imported or exported with related names (subspecifier)
   31 pattern IESpec :: Name dom -> MaybeSubSpec dom -> IESpec dom
   32 pattern IESpec name ss <- Ann _ (UIESpec _ name ss)
   33 
   34 -- | @(..)@: a class exported with all of its methods, or a datatype exported with all of its constructors.
   35 pattern SubAll :: SubSpec dom
   36 pattern SubAll <- Ann _ USubSpecAll
   37 
   38 -- | @(a,b,c)@: a class exported with some of its methods, or a datatype exported with some of its constructors.
   39 pattern SubList :: NameList dom -> SubSpec dom
   40 pattern SubList names <- Ann _ (USubSpecList names)
   41 
   42 
   43 pattern ImportDecl :: MaybeImportSource dom -> MaybeImportQualified dom 
   44                         -> MaybeImportSafe dom -> MaybeStringNode dom
   45                         -> ModuleName dom -> MaybeImportRenaming dom
   46                         -> MaybeImportSpec dom -> ImportDecl dom       
   47 pattern ImportDecl source qualified safe pkg name rename spec <- Ann _ (UImportDecl source qualified safe pkg name rename spec)
   48 
   49 pattern ImportRenaming :: ModuleName dom -> ImportRenaming dom
   50 pattern ImportRenaming name <- Ann _ (UImportRenaming name)
   51 
   52 pattern ImportSpecList :: IESpecList dom -> ImportSpec dom
   53 pattern ImportSpecList ieSpecs <- Ann _ (UImportSpecList ieSpecs)
   54 
   55 pattern ImportHidingList :: IESpecList dom -> ImportSpec dom
   56 pattern ImportHidingList hidings <- Ann _ (UImportSpecHiding hidings)
   57 
   58 pattern ModuleName :: String -> ModuleName dom
   59 pattern ModuleName s <- Ann _ (UModuleName s)
   60 
   61 -- * Pragmas
   62 
   63 -- | @LANGUAGE@ pragma, listing the enabled language extensions in that file
   64 pattern LanguagePragma :: LanguageExtensionList dom -> FilePragma dom
   65 pattern LanguagePragma exts <- Ann _ (ULanguagePragma exts)
   66 
   67 -- | @OPTIONS@ pragma, possibly qualified with a tool, e.g. OPTIONS_GHC
   68 pattern OptionsPragma :: String -> FilePragma dom
   69 pattern OptionsPragma opt <- Ann _ (UOptionsPragma (Ann _ (UStringNode opt)))
   70 
   71 -- | The name of the enabled language extension, for example (@ LambdaCase @)
   72 pattern LanguageExtension :: String -> LanguageExtension dom
   73 pattern LanguageExtension ext <- Ann _ (ULanguageExtension ext)
   74 
   75 -- | A warning pragma attached to the module
   76 pattern ModuleWarningPragma :: StringNodeList dom -> ModulePragma dom
   77 pattern ModuleWarningPragma msgs <- Ann _ (UModuleWarningPragma msgs)
   78 
   79 -- | A deprecated pragma attached to the module
   80 pattern ModuleDeprecatedPragma :: StringNodeList dom -> ModulePragma dom
   81 pattern ModuleDeprecatedPragma msgs <- Ann _ (UModuleDeprecatedPragma msgs)
