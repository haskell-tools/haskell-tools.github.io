   24 autoCorrectRefactoring :: RefactoringChoice
   25 autoCorrectRefactoring = SelectionRefactoring "AutoCorrect" (localRefactoring . autoCorrect)

   27 tryItOut :: String -> String -> IO ()
   28 tryItOut mod sp = tryRefactor (localRefactoring . autoCorrect) mod sp

   30 autoCorrect :: RealSrcSpan -> LocalRefactoring
   31 autoCorrect sp mod
   32   = do res <- mapM (\f -> f sp mod) [reParen, reOrder]
   33        case catMaybes res of mod':_ -> return mod'
   34                              []     -> refactError "Cannot auto-correct the selection."
   35

   38 reOrder :: RealSrcSpan -> HT.Module -> LocalRefactor (Maybe HT.Module)
   39 reOrder sp mod = do let insts = semanticsPrelInsts mod ++ concatMap semanticsInsts (mod ^? modImports & annList :: [HT.ImportDecl])
   40                         rng:_ = map getRange (mod ^? nodesContained sp :: [Expr])
   41                     (res,done) <- liftGhc $ flip runStateT False ((nodesContained sp & filtered ((==rng) . getRange) !~ reOrderExpr insts) mod)
   42                     return (if done then Just res else Nothing)
   43 
   44 reOrderExpr :: [ClsInst] -> Expr -> StateT Bool Ghc Expr
   45 reOrderExpr insts e@(App (App f a1) a2)
   46   = do funTy <- lift $ typeExpr f
   47        arg1Ty <- lift $ typeExpr a1
   48        arg2Ty <- lift $ typeExpr a2
   49        -- liftIO $ putStrLn $ (show $ isJust $ appTypeMatches insts funTy [arg1Ty, arg2Ty]) ++ " " ++ (show $ isJust $ appTypeMatches insts funTy [arg2Ty, arg1Ty]) ++ " " ++ (showSDocUnsafe $ ppr $ funTy) ++ " " ++ (showSDocUnsafe $ ppr $ arg1Ty) ++ " " ++ (showSDocUnsafe $ ppr $ arg2Ty)
   50        if not (isJust (appTypeMatches insts funTy [arg1Ty, arg2Ty])) && isJust (appTypeMatches insts funTy [arg2Ty, arg1Ty])
   51           then put True >> return (exprArg .= a1 $ exprFun&exprArg .= a2 $ e)
   52           else return e
   53 reOrderExpr insts e@(InfixApp lhs op rhs) 
   54   = do let funTy = idType $ semanticsId (op ^. operatorName)
   55        lhsTy <- lift $ typeExpr lhs
   56        rhsTy <- lift $ typeExpr rhs
   57        -- liftIO $ putStrLn $ (show $ isJust $ appTypeMatches insts funTy [lhsTy, rhsTy]) ++ " " ++ (show $ isJust $ appTypeMatches insts funTy [lhsTy, rhsTy]) ++ " " ++ (showSDocUnsafe $ ppr $ funTy) ++ " " ++ (showSDocUnsafe $ ppr $ lhsTy) ++ " " ++ (showSDocUnsafe $ ppr $ rhsTy)
   58        if not (isJust (appTypeMatches insts funTy [lhsTy, rhsTy])) && isJust (appTypeMatches insts funTy [rhsTy, lhsTy])
   59           then put True >> return (exprLhs .= rhs $ exprRhs .= lhs $ e)
   60           else return e
   61 reOrderExpr _ e = return e

   65 reParen :: RealSrcSpan -> HT.Module -> LocalRefactor (Maybe HT.Module)
   66 reParen sp mod = do let insts = semanticsPrelInsts mod ++ concatMap semanticsInsts (mod ^? modImports & annList :: [HT.ImportDecl])
   67                         rng:_ = map getRange (mod ^? nodesContained sp :: [Expr])
   68                     (res,done) <- liftGhc $ flip runStateT False ((nodesContained sp & filtered ((==rng) . getRange) !~ reParenExpr insts) mod)
   69                     return (if done then Just res else Nothing)
   70 
   71 reParenExpr :: [ClsInst] -> Expr -> StateT Bool Ghc Expr
   72 reParenExpr insts e = do atoms <- lift $ extractAtoms e
   73                          case correctParening insts $ map (_2 .- Left) atoms of
   74                            [e'] -> put True >> return (wrapAtom e')
   75                            [] -> return e
   76                            ls -> -- TODO: choose the best one
   77                                  error $ "multiple correct parentheses were found: " ++ intercalate ", " (map (either prettyPrintAtom prettyPrint) ls)
   78 
   79 data Atom = NameA { aName :: HT.Name }
   80           | OperatorA { aOperator :: Operator }
   81           | LiteralA { aLiteral :: Literal }
   82 
   83 prettyPrintAtom :: Atom -> String
   84 prettyPrintAtom (NameA n) = prettyPrint n
   85 prettyPrintAtom (OperatorA o) = prettyPrint o
   86 prettyPrintAtom (LiteralA l) = prettyPrint l
   87 
   88 type Build = Either Atom Expr
   89 
   90 extractAtoms :: Expr -> Ghc [(GHC.Type, Atom)]
   91 extractAtoms e = do lits <- mapM (\l -> (, LiteralA l) <$> literalType l) (e ^? biplateRef)
   92                     return $ sortOn (srcSpanStart . atomRange . snd) 
   93                            $ map (\n -> (idType $ semanticsId (n ^. simpleName), NameA n)) (e ^? biplateRef) 
   94                                ++ map (\o -> (idType $ semanticsId (o ^. operatorName), OperatorA o)) (e ^? biplateRef) 
   95                                ++ lits
   96 
   97 atomRange :: Atom -> SrcSpan
   98 atomRange (NameA n) = getRange n
   99 atomRange (OperatorA n) = getRange n
  100 atomRange (LiteralA n) = getRange n
  101 
  102 wrapAtom :: Build -> Expr
  103 wrapAtom (Right e) = e
  104 wrapAtom (Left (NameA n)) = mkVar n
  105 wrapAtom (Left (OperatorA (NormalOp o))) = mkVar (mkParenName o)
  106 wrapAtom (Left (OperatorA (BacktickOp n))) = mkVar (mkNormalName n)
  107 wrapAtom (Left (LiteralA l)) = mkLit l
  108 
  109 correctParening :: [ClsInst] -> [(GHC.Type, Build)] -> [Build]
  110 correctParening _ [(_,e)] = [e]
  111 correctParening insts ls = concatMap (correctParening insts) (reduceAtoms insts ls)
  112 
  113 reduceAtoms :: [ClsInst] -> [(GHC.Type, Build)] -> [[(GHC.Type, Build)]]
  114 reduceAtoms _ [(t,e)] = [[(t,e)]]
  115 reduceAtoms insts ls = concatMap (reduceBy insts ls) [0 .. length ls - 2]
  116 
  117 reduceBy :: [ClsInst] -> [(GHC.Type, Build)] -> Int -> [[(GHC.Type, Build)]]
  118 reduceBy insts (zip [0..] -> ls) i = maybeToList (reduceFunctionApp ls i) ++ maybeToList (reduceOperatorApp ls i)
  119   where reduceFunctionApp ls i | Just (funT, fun) <- lookup i ls
  120                                , Just (argT, arg) <- lookup (i+1) ls
  121                                , Just (subst, resTyp) <- appTypeMatches insts funT [argT]
  122           = Just $ map ((_1 .- substTy subst) . snd) (take i ls) 
  123                      ++ [(resTyp, mkParen' (mkApp' fun arg))] 
  124                      ++ map ((_1 .- substTy subst) . snd) (drop (i + 2) ls)
  125         reduceFunctionApp _ _ = Nothing
  126         
  127         reduceOperatorApp ls i | Just (opT, Left (OperatorA op)) <- lookup i ls
  128                                , Just (lArgT, lArg) <- lookup (i-1) ls
  129                                , Just (rArgT, rArg) <- lookup (i+1) ls
  130                                , Just (subst, resTyp) <- appTypeMatches insts opT [lArgT, rArgT]
  131           = Just $ map ((_1 .- substTy subst) . snd) (take (i - 1) ls) 
  132                      ++ [(resTyp, mkParen' (mkInfixApp' lArg op rArg))] 
  133                      ++ map ((_1 .- substTy subst) . snd) (drop (i + 2) ls)
  134         reduceOperatorApp _ _ = Nothing
  135 
  136 mkApp' :: Build -> Build -> Build
  137 mkApp' (wrapAtom -> f) (wrapAtom -> a) = Right $ mkApp f a
  138 
  139 mkInfixApp' :: Build -> Operator -> Build -> Build
  140 mkInfixApp' (wrapAtom -> lhs) op (wrapAtom -> rhs) = Right $ mkInfixApp lhs op rhs
  141 
  142 mkParen' :: Build -> Build
  143 mkParen' (wrapAtom -> e) = Right $ mkParen e
