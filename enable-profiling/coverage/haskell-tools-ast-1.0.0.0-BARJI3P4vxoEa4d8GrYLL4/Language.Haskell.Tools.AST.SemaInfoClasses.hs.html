   22 -- | Domains that have semantic information for names
   23 type HasNameInfo dom = (Domain dom, HasNameInfo' (SemanticInfo dom UQualifiedName))
   24 
   25 -- | Infos that may have a name that can be extracted
   26 class HasNameInfo' si where
   27   semanticsName :: si -> Maybe GHC.Name
   28 
   29 instance HasNameInfo' (NameInfo GHC.Name) where
   30   semanticsName = (^? nameInfo)
   31 
   32 instance HasNameInfo' CNameInfo where
   33   semanticsName = fmap idName . (^? cnameInfo)
   34 
   35 instance HasNameInfo dom => HasNameInfo' (Ann UQualifiedName dom st) where
   36   semanticsName = semanticsName . (^. annotation&semanticInfo)
