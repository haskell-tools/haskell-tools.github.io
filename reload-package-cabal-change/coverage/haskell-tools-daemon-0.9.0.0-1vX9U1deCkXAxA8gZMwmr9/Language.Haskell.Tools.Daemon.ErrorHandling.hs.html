   25 -- Handlers for exceptions specific to our application.
   26 userExceptionHandlers :: (String -> IO a) -> ([(SrcSpan, String)] -> [String] -> IO a) -> [Handler a]
   27 userExceptionHandlers sendError sendCompProblems =
   28   [ Handler (\(UnsupportedPackage e) -> sendError ("There are unsupported elements in your package: " ++ e ++ " please correct them before loading them into Haskell-tools."))
   29   , Handler (\(UnsupportedExtension e) -> sendError ("The extension you use is not supported: " ++ e ++ ". Please check your source and cabal files for the use of that language extension."))
   30   , Handler (\(SpliceInsertionProblem rng _) -> sendError ("A problem occurred while type-checking the Template Haskell splice at: " ++ shortShowSpanWithFile rng ++ ". Some complex splices cannot be type checked for reasons currently unknown. Please simplify the splice. We are working on this problem."))
   31   , Handler (\case (BreakUpProblem outer rng _) -> sendError ("The program element at " ++ (if isGoodSrcSpan rng then shortShowSpanWithFile rng else shortShowSpanWithFile (RealSrcSpan outer)) ++ " could not be prepared for refactoring. The most likely reason is preprocessor usage. Only conditional compilation is supported, includes and preprocessor macros are not. If there is no preprocessor usage at the given location, there might be a weirdly placed comment causing a problem."))
   32   , Handler (\(TransformationProblem msg) -> sendError ("A problem occurred while preparing the program for refactoring: " ++ msg))
   33   , Handler (\(PrettyPrintProblem msg) -> sendError ("A problem occurred while pretty printing the result of the refactoring: " ++ msg))
   34   , Handler (\case (ConvertionProblem rng msg) -> sendError ("An unexpected problem occurred while converting the representation of the program element at " ++ shortShowSpanWithFile rng ++ ": " ++ msg)
   35                    (UnrootedConvertionProblem msg) -> sendError ("An unexpected problem occurred while converting between different program representations: " ++ msg))
   36   , Handler (\errs -> let msgs = map (\err -> (errMsgSpan err, show err))
   37                                    $ bagToList $ srcErrorMessages errs
   38                           hints = nub $ sort $ catMaybes $ map (handleSourceProblem . snd) msgs
   39                        in sendCompProblems msgs hints)
   40   ]
   41 
   42 -- | Handlers for generic exceptions: 'IOException', 'AsyncException', 'SomeException'.
   43 exceptionHandlers :: IO () -> (String -> IO ()) -> [Handler ()]
   44 exceptionHandlers cont sendError =
   45   [ Handler (\(err :: IOException) -> hPutStrLn stderr $ "IO Exception caught: " ++ show err)
   46   , Handler (\(e :: AsyncException) -> hPutStrLn stderr $ "Asynch exception caught: " ++ show e)
   47   , Handler (\(ex :: SomeException) -> handleException ex cont)
   48   ]
   49   where handleException ex cont
   50           = case handleGHCException (show ex) of
   51               Nothing -> do hPutStrLn stderr $ "Unexpected error: " ++ show (ex :: SomeException)
   52                             sendError $ "Internal error: " ++ show ex
   53               Just (msg, doContinue) -> sendError msg >> when doContinue cont
   54 
   55 -- | Hint text and continuation suggestion for different kinds of errors based on pattern matching on error text.
   56 handleGHCException :: String -> Maybe (String, Bool)
   57 handleGHCException msg | "failed" `isInfixOf` msg && "C pre-processor" `isInfixOf` msg
   58   = Just ("Failed to load the package. The cause of that is a failure of the pre-processor. "
   59              ++ " Only conditional compilation is supported, includes and preprocessor macros are not: " ++ msg, False)
   60 handleGHCException msg | "failed" `isInfixOf` msg && "Literate pre-processor" `isInfixOf` msg
   61   = Just ("Haskell-tools does not handle Literate Haskell yet."
   62              ++ " If you get this error after refactoring, you should undo the refactoring. The error message: " ++ msg, True)
   63 handleGHCException msg | "cannot satisfy" `isInfixOf` msg
   64   = Just ("While trying to collect the modules of the project we found the a package is not in the"
   65              ++ " used package database. Check that you actually compiled this package with all of its components (tests, benchmarks). "
   66              ++ "If so, make sure that it is installed with the same tools that the project is "
   67              ++ "configured for (cabal/stack/cabal sandbox). The error message: " ++ msg, True)
   68 handleGHCException msg = Nothing
   69 
   70 -- | Hint text and continuation suggestion for different kinds of source problems based on pattern matching on error text.
   71 handleSourceProblem :: String -> Maybe String
   72 handleSourceProblem msg | "is a package module" `isInfixOf` msg
   73   = Just $ "A module is not found, check that the current working directory is the root of the module hierarchy. "
   74              ++ " Also check that none of the modules are generated by parser generators or hsc files."
   75 handleSourceProblem msg | "Failed to load interface" `isInfixOf` msg
   76   = Just $ "Some of the required (external) modules cannot be loaded, because they are not found. Make sure that the project is "
   77               ++ " already built, and that it is built using the same package database as it is used for refactoring."
   78 handleSourceProblem msg | "Ambiguous interface" `isInfixOf` msg
   79   = Just $ "Some of the required (external) modules cannot be loaded, because they are ambiguous. "
   80              ++ "Since there is no separation between packages and package components, make sure that you do not depend "
   81              ++ "on packages that contain modules with the same qualified name."
   82 handleSourceProblem _
   83   = Just $ "While loading we found a compilation error in the source code. If it compiles normally "
   84               ++ "using cabal or stack the problem might result from modules with same name, or "
   85               ++ "generated files that are not up-to-date."
