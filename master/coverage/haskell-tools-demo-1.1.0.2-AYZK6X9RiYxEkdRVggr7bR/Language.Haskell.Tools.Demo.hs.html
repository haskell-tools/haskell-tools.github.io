   63 runDemo :: [String] -> IO ()
   64 runDemo args = do

   65   wd <- case args of dir:_ -> return dir

   67   counter <- newMVar []

   71 -- | The application that is evoked for each incoming request
   72 app :: MVar [Int] -> FilePath -> Application
   73 app sessions wd = websocketsOr defaultConnectionOptions wsApp backupApp

   74   where

   75     wsApp :: ServerApp

   76     wsApp conn = do

   77         conn <- acceptRequest conn

   78         newind <- modifyMVar sessions (\sess -> let smallest = head (filter (not . (`elem` sess)) [0..])

   79                                                  in return (smallest : sess, smallest))

   80         ghcSess <- initGhcSession (userDir wd newind)

   81         state <- newMVar initSession

   82         serverLoop newind ghcSess state conn

   83 

   84     serverLoop :: Int -> Session -> MVar RefactorSessionState -> Connection -> IO ()

   85     serverLoop sessId ghcSess state conn =

   86         do Text msg _ <- receiveDataMessage conn

   87            respondTo wd sessId ghcSess state (sendTextData conn) msg

   88            currState <- readMVar state

   89            if currState ^. isDisconnecting

   90              then sendClose conn ("" :: ByteString)

   91              else serverLoop sessId ghcSess state conn

   92       `catch` \(_ :: ConnectionException) -> do

   93                  modifyMVar_ sessions (return . delete sessId)

   94                  liftIO $ removeDirectoryIfPresent (userDir wd sessId)

   95 

   96     backupApp :: Application

   97     backupApp _ respond = respond $ responseLBS status400 [] "Not a WebSocket request"

   99 respondTo :: FilePath -> Int -> Session -> MVar RefactorSessionState -> (ByteString -> IO ()) -> ByteString -> IO ()
  100 respondTo wd id ghcSess state next mess = case decode mess of

  101   Nothing -> next $ encode (ErrorMessage $ "WRONG MESSAGE FORMAT: " ++ show (BS.unpack mess))

  102   Just req -> handleErrors wd req (next . encode)

  103                 $ do resp <- modifyMVar state (\st -> swap <$> reflectGhc (runStateT (updateClient (userDir wd id) req) st) ghcSess)

  104                      case resp of Just respMsg -> next $ encode respMsg

  105                                   Nothing -> return ()

  107 -- | This function does the real job of acting upon client messages in a stateful environment of a client
  108 updateClient :: FilePath -> ClientMessage -> StateT RefactorSessionState Ghc (Maybe ResponseMsg)
  109 updateClient _ KeepAlive = return Nothing

  110 updateClient _ Disconnect = do modify $ isDisconnecting .= True

  111                                return $ Just Disconnected

  112 updateClient dir (ModuleChanged name newContent) = do

  113     liftIO $ createFileForModule dir name newContent

  114     targets <- lift getTargets

  115     when (isNothing . find ((\case (TargetModule n) -> GHC.moduleNameString n == name; _ -> False) . targetId) $ targets)

  116       $ lift $ addTarget (Target (TargetModule (GHC.mkModuleName name)) True Nothing)

  117     void $ lift $ load LoadAllTargets

  118     reloadAllMods dir

  119     return Nothing

  120 updateClient dir (ModuleDeleted name) = do

  121     lift $ removeTarget (TargetModule (GHC.mkModuleName name))

  122     modify $ refSessMods .- Map.delete (dir, name, dir </> moduleSourceFile name)

  123     return Nothing

  124 updateClient dir (InitialProject modules) = do

  125     -- clean the workspace to remove source files from earlier sessions

  126     liftIO $ removeDirectoryIfPresent dir

  127     liftIO $ createDirectoryIfMissing True dir

  128     liftIO $ forM_ modules $ \(mod, cont) -> do

  129       withBinaryFile (toFileName dir mod) WriteMode (`hPutStr` cont)

  130     lift $ setTargets (map ((\modName -> Target (TargetModule (GHC.mkModuleName modName)) True Nothing) . fst) modules)

  131     reloadAllMods dir

  132     return Nothing

  133 updateClient _ (PerformRefactoring "UpdateAST" modName _ _) = do

  134     mod <- gets (find ((modName ==) . (\(_,m,_) -> m) . fst) . Map.assocs . (^. refSessMods))

  135     case mod of Just (_,m) -> return $ Just $ ASTViewContent $ astDebug m

  136                 Nothing -> return $ Just $ ErrorMessage "The module is not found"

  137 updateClient _ (PerformRefactoring "TestErrorLogging" _ _ _) = error "This is a test"

  138 updateClient dir (PerformRefactoring refact modName selection args) = do

  139     mod <- gets (find ((modName ==) . (\(_,m,_) -> m) . fst) . Map.assocs . (^. refSessMods))

  140     otherModules <- gets (filter ((modName /=) . (^. sfkModuleName) . fst) . map moduleNameAndContent . Map.assocs . (^. refSessMods))

  141 

  142     case mod of

  143       Just m ->

  144         do res <- lift $ performCommand builtinRefactorings

  145                                         ([refact,selection] ++ args)

  146                                         (Right $ moduleNameAndContent m) otherModules

  147            case res of

  148              Right diff -> do applyChanges diff

  149                               return $ Just $ RefactorChanges (map trfDiff diff)

  150              Left err -> return $ Just $ ErrorMessage err

  151       Nothing -> return $ Just $ ErrorMessage "The module is not found"

  152   where trfDiff (ContentChanged (key,cont)) = (key ^. sfkModuleName, Just (prettyPrint cont))

  153         trfDiff (ModuleCreated name mod _) = (name, Just (prettyPrint mod))

  154         trfDiff (ModuleRemoved name) = (name, Nothing)

  155 

  156         applyChanges diff

  157           = do forM_ diff $ \case

  158                  ModuleCreated n m _ -> do

  159                    writeModule n m

  160                    lift $ addTarget (Target (TargetModule (GHC.mkModuleName n)) True Nothing)

  161                  ContentChanged (n,m) ->

  162                    writeModule (n ^. sfkModuleName) m

  163                  ModuleRemoved mod -> do

  164                    liftIO $ removeFile (toFileName dir mod)

  165                    modify $ refSessMods .- Map.delete (dir, mod, dir </> moduleSourceFile mod)

  166                    lift $ removeTarget (TargetModule (GHC.mkModuleName mod))

  167                reloadAllMods dir

  168 

  169         writeModule n m = liftIO $ withBinaryFile (toFileName dir n) WriteMode (`hPutStr` prettyPrint m)

  171 reloadAllMods :: FilePath -> StateT RefactorSessionState Ghc ()
  172 reloadAllMods dir = do

  173   void $ lift $ load LoadAllTargets

  174   targets <- lift getTargets

  175   forM_ (map ((\case (TargetModule n) -> n) . targetId) targets) $ \modName -> do

  176       mod <- lift $ getModSummary modName >>= parseTyped

  177       modify $ refSessMods .- Map.insert (dir, GHC.moduleNameString modName, dir </> moduleSourceFile (GHC.moduleNameString modName)) mod

  179 createFileForModule :: FilePath -> String -> String -> IO ()
  180 createFileForModule dir name newContent = do

  181   let fname = toFileName dir name

  182   createDirectoryIfMissing True (takeDirectory fname)

  183   withBinaryFile fname WriteMode (`hPutStr` newContent)

  201 handleErrors :: FilePath -> ClientMessage -> (ResponseMsg -> IO ()) -> IO () -> IO ()
  202 handleErrors wd req next io = io `catch` (next <=< handleException)

  203   where handleException :: SomeException -> IO ResponseMsg

  204         handleException e

  205           | Just (se :: SourceError) <- fromException e

  206           = if isReloading

  207               then do logToFile wd (show e) req

  208                       return $ ErrorMessage ("The generated code cannot be compiled. The problem had been reported. Please restart the demo or correct the results manually.")

  209               else return $ CompilationProblem (concatMap (\msg -> showMsg msg ++ "\n\n") $ bagToList $ srcErrorMessages se)

  210           | Just (ae :: AsyncException) <- fromException e = throw ae

  211           | Just (ge :: GhcException) <- fromException e = return $ ErrorMessage $ show ge

  212           | Just (re :: RefactorException) <- fromException e = return $ ErrorMessage $ displayException re

  213           | otherwise = do logToFile wd (show e) req

  214                            return $ ErrorMessage (showInternalError e)

  215 

  216         showMsg msg = showSpan (errMsgSpan msg) ++ "\n" ++ show msg

  217         showSpan (RealSrcSpan sp) = showFileName (srcLocFile (realSrcSpanStart sp)) ++ " " ++ show (srcLocLine (realSrcSpanStart sp)) ++ ":" ++ show (srcLocCol (realSrcSpanStart sp))

  218         showSpan _ = ""

  219 

  220         isReloading = case req of PerformRefactoring {} -> True; _ -> False

  221 

  222         showFileName = joinPath . drop 2 . splitPath . makeRelative wd . unpackFS

  223 

  224         showInternalError :: SomeException -> String

  225         showInternalError e = "An internal error happened. The report has been sent to the developers. " ++ displayException e

  227 logToFile :: FilePath -> String -> ClientMessage -> IO ()
  228 logToFile wd err input = do

  229   let msg = err ++ "\n with input: " ++ show input

  230   withFile logFile AppendMode $ \handle -> do

  231       size <- hFileSize handle

  232       when (size < logSizeLimit) $ hPutStrLn handle ("\n### " ++ msg)

  233     `catch` \e -> print ("The error message cannot be logged because: "

  234                              ++ show (e :: IOException) ++ "\nHere is the message:\n" ++ msg)

  235   where logFile = wd </> "error-log.txt"

  236         logSizeLimit = 100 * 1024 * 1024 -- 100 MB

  238 data ClientMessage
  239   = KeepAlive
  240   | InitialProject { initialModules :: [(String,String)] }
  241   | PerformRefactoring { refactoring :: String
  242                        , moduleName :: String
  243                        , editorSelection :: String
  244                        , details :: [String]
  245                        }
  246   | ModuleChanged { moduleName :: String
  247                   , newContent :: String
  248                   }
  249   | ModuleDeleted { moduleName :: String }
  250   | Disconnect
  251   deriving (Show, Generic)
  252 
  253 instance FromJSON ClientMessage

  255 data ResponseMsg

  257   | ASTViewContent { astContent :: String }
  258   | ErrorMessage { errorMsg :: String }
  259   | CompilationProblem { errorMsg :: String }

  261   deriving (Show, Generic)
