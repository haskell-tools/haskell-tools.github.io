    9 -- | Normal monadic statements
   10 data UStmt' expr dom stage
   11   = UBindStmt { _stmtPattern :: Ann UPattern dom stage
   12               , _stmtExpr :: Ann expr dom stage
   13               } -- ^ Binding statement (@ x <- action @)
   14   | UExprStmt { _stmtExpr :: Ann expr dom stage
   15               } -- ^ Non-binding statement (@ action @)
   16   | ULetStmt  { _stmtBinds :: AnnListG ULocalBind dom stage
   17               } -- ^ Let statement (@ let x = 3; y = 4 @)
   18   | URecStmt  { _cmdStmtBinds :: AnnListG (UStmt' expr) dom stage
   19               } -- ^ A recursive binding statement with (@ rec b <- f a c; c <- f b a @)
   20 type UStmt = UStmt' UExpr
   21 type UCmdStmt = UStmt' UCmd
   22 
   23 -- | Body of a list comprehension: (@ | x <- [1..10] @)
   24 data UListCompBody dom stage
   25   = UListCompBody { _compStmts :: AnnListG UCompStmt dom stage
   26                   } 
   27          
   28 -- | List comprehension statement
   29 data UCompStmt dom stage
   30   = UCompStmt  { _compStmt :: Ann UStmt dom stage
   31                } -- ^ Normal monadic statement of a list comprehension
   32   | UThenStmt  { _thenExpr :: Ann UExpr dom stage
   33                , _byExpr :: AnnMaybeG UExpr dom stage
   34                } -- ^ Then statements by @TransformListComp@ (@ then sortWith by (x + y) @)
   35   | UGroupStmt { _byExpr :: AnnMaybeG UExpr dom stage
   36                , _usingExpr :: AnnMaybeG UExpr dom stage
   37                } -- ^ Grouping statements by @TransformListComp@ (@ then group by (x + y) using groupWith @) 
   38                  -- Note: either byExpr or usingExpr must have a value
   39 
   40 -- | Keywords @do@ or @mdo@ to start a do-block
   41 data UDoKind dom stage
   42   = UDoKeyword
   43   | UMDoKeyword
