   12 -- | Representation of patterns for pattern bindings
   13 data UPattern dom stage
   14   = UVarPat         { _patternName :: Ann UName dom stage
   15                     } -- ^ Pattern name binding
   16   | ULitPat         { _patternLiteral :: Ann ULiteral dom stage
   17                     } -- ^ Literal pattern
   18   | UInfixAppPat    { _patternLhs :: Ann UPattern dom stage
   19                     , _patternOperator :: Ann UOperator dom stage
   20                     , _patternRhs :: Ann UPattern dom stage
   21                     } -- ^ Infix constructor application pattern (@ a :+: b @)
   22   | UAppPat         { _patternName :: Ann UName dom stage
   23                     , _patternArgs :: AnnListG UPattern dom stage
   24                     } -- ^ Constructor application pattern (@ Point x y @)
   25   | UTuplePat       { _patternElems :: AnnListG UPattern dom stage
   26                     } -- ^ Tuple pattern (@ (x,y) @)
   27   | UUnboxTuplePat  { _patternElems :: AnnListG UPattern dom stage
   28                     } -- ^ Unboxed tuple pattern (@ (# x, y #) @)
   29   | UListPat        { _patternElems :: AnnListG UPattern dom stage
   30                     } -- ^ List pattern (@ [1,2,a,x] @)
   31   | UParArrPat      { _patternElems :: AnnListG UPattern dom stage
   32                     } -- ^ Parallel array pattern (@ [:1,2,a,x:] @)
   33   | UParenPat       { _patternInner :: Ann UPattern dom stage
   34                     } -- ^ Parenthesised patterns
   35   | URecPat         { _patternName :: Ann UName dom stage
   36                     , _patternFields :: AnnListG UPatternField dom stage
   37                     } -- ^ Record pattern (@ Point { x = 3, y } @)
   38   | UAsPat          { _patternName :: Ann UName dom stage
   39                     , _patternInner :: Ann UPattern dom stage
   40                     } -- ^ As-pattern (explicit name binding) (@ ls\@(hd:_) @)
   41   | UWildPat -- ^ Wildcard pattern: (@ _ @)
   42   | UIrrefutablePat { _patternInner :: Ann UPattern dom stage
   43                     } -- ^ Irrefutable pattern (@ ~(x:_) @)
   44   | UBangPat        { _patternInner :: Ann UPattern dom stage
   45                     } -- ^ Bang pattern (@ !x @)
   46   | UTypeSigPat     { _patternInner :: Ann UPattern dom stage
   47                     , _patternType :: Ann UType dom stage
   48                     } -- ^ Pattern with explicit type signature (@ x :: Int @)
   49   | UViewPat        { _patternExpr :: Ann UExpr dom stage
   50                     , _patternInner :: Ann UPattern dom stage
   51                     } -- ^ View pattern (@ f -> Just 1 @)
   52   -- regular list pattern omitted
   53   -- xml patterns omitted
   54   | USplicePat     { _patternSplice :: Ann USplice dom stage
   55                    } -- ^ Splice patterns: @$(generateX inp)@
   56   | UQuasiQuotePat { _patQQ :: Ann UQuasiQuote dom stage
   57                    } -- ^ Quasi-quoted patterns: @[| 1 + 2 |]@
   58   | UNPlusKPat     { _patternName :: Ann UName dom stage
   59                    , _patternLit :: Ann ULiteral dom stage
   60                    }
   61                   
   62 -- Field specification of a record pattern
   63 data UPatternField dom stage
   64   = UNormalFieldPattern   { _fieldPatternName :: Ann UName dom stage
   65                           , _fieldPattern :: Ann UPattern dom stage
   66                           } -- ^ Named field pattern (@ p = Point 3 2 @)
   67   | UFieldPunPattern      { _fieldPatternName :: Ann UName dom stage
   68                           } -- ^ Named field pun (@ p @)
   69   | UFieldWildcardPattern { _fieldPatternWildcard :: Ann UFieldWildcard dom stage
   70                           } -- ^ Wildcard field pattern (@ .. @)
