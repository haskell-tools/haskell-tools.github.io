   19 tryItOut :: String -> String -> IO ()
   20 tryItOut = tryRefactor (localRefactoring . dollarApp)
   21 
   22 type DollarMonad = StateT [SrcSpan] LocalRefactor
   23 
   24 dollarApp :: RealSrcSpan -> LocalRefactoring
   25 dollarApp sp = flip evalStateT [] . ((nodesContained sp !~ (\e -> get >>= replaceExpr e))
   26                                         >=> (biplateRef !~ parenExpr))
   27 
   28 replaceExpr :: Expr -> [SrcSpan] -> DollarMonad Expr
   29 replaceExpr expr@(App _ (Paren (InfixApp _ op arg))) replacedRanges
   30   | not (getRange arg `elem` replacedRanges)
   31   , semanticsName (op ^. operatorName) /= Just dollarName
   32   , case semanticsFixity (op ^. operatorName) of Just (Fixity _ p _) | p > 0 -> False; _ -> True
   33   = return expr
   34 replaceExpr (App fun (Paren arg)) _ = do modify $ (getRange arg :)
   35                                          mkInfixApp fun <$> lift (referenceOperator dollarName) <*> pure arg
   36 replaceExpr e _ = return e
   37 
   38 parenExpr :: Expr -> DollarMonad Expr
   39 parenExpr e = (exprLhs !~ parenDollar True) =<< (exprRhs !~ parenDollar False $ e)
   40 
   41 parenDollar :: Bool -> Expr -> DollarMonad Expr
   42 parenDollar lhs expr@(InfixApp _ _ arg)
   43   = do replacedRanges <- get
   44        if getRange arg `elem` replacedRanges && (lhs || getRange expr `notElem` replacedRanges)
   45          then return $ mkParen expr
   46          else return expr
   47 parenDollar _ e = return e
   48 
   49 dollarName :: GHC.Name
   50 [dollarName] = map idName $ filter ((dollarIdKey==) . getUnique) wiredInIds
