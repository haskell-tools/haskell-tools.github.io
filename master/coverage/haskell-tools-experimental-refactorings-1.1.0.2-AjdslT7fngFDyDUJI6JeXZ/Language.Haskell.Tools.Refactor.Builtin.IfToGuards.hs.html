    6 
    7 import Control.Reference ((^.), (.-), (&))
    8 import Data.Generics.Uniplate.Data ()
    9 import Language.Haskell.Tools.Refactor
   10 import SrcLoc (RealSrcSpan)
   11 
   12 tryItOut :: String -> String -> IO ()
   13 tryItOut = tryRefactor (localRefactoring . ifToGuards)
   14 
   15 ifToGuards :: RealSrcSpan -> LocalRefactoring
   16 ifToGuards sp = return . (nodesContaining sp .- changeBindings)
   17 
   18 changeBindings :: ValueBind -> ValueBind
   19 changeBindings (SimpleBind (VarPat name) (UnguardedRhs (If pred thenE elseE)) locals)
   20   = mkFunctionBind [mkMatch (mkMatchLhs name []) (createSimpleIfRhss pred thenE elseE) (locals ^. annMaybe) ]
   21 changeBindings fbs@(FunctionBind {})
   22   = funBindMatches&annList&matchRhs .- trfRhs $ fbs
   23   where trfRhs :: Rhs -> Rhs
   24         trfRhs (UnguardedRhs (If pred thenE elseE)) = createSimpleIfRhss pred thenE elseE
   25         trfRhs e = e -- don't transform already guarded right-hand sides to avoid multiple evaluation of the same condition
   26 changeBindings b = b
   27 
   28 createSimpleIfRhss :: Expr -> Expr -> Expr -> Rhs
   29 createSimpleIfRhss pred thenE elseE = mkGuardedRhss [ mkGuardedRhs [mkGuardCheck pred] thenE
   30                                                     , mkGuardedRhs [mkGuardCheck (mkVar (mkName "otherwise"))] elseE
   31                                                     ]
