    3 -- | Setting the package database to use when compiling modules. The daemon must have one single
    4 -- package database that cannot be changed after a package is loaded using that package database.
    5 -- Available package databases are the cabal global, the cabal sandbox, the stack or one that had
    6 -- been explicitely set by a file path.
    7 module Language.Haskell.Tools.Daemon.PackageDB (PackageDB(..), decidePkgDB, packageDBLoc, detectAutogen) where
    8

   22 -- | Possible package database configurations.
   23 data PackageDB = DefaultDB -- ^ Use the global cabal package database (like when using ghc).
   24                | CabalSandboxDB -- ^ Use the sandboxed cabal package database.
   25                | StackDB -- ^ Use the stack package databases (local and snapshot).
   26                | ExplicitDB { packageDBPath :: [FilePath] } -- ^ Set the package database explicitely.
   27   deriving (Eq, Show, Generic)
   28 
   29 instance FromJSON PackageDB
   30 
   31 -- | Decide which type of project we are dealing with based on the package folders.
   32 -- Should only be invoked if the user did not select the project-type.
   33 decidePkgDB :: [FilePath] -> IO (Maybe PackageDB)
   34 decidePkgDB [] = return Nothing
   35 decidePkgDB (firstRoot:packageRoots) = do
   36   fstRes <- decidePkgDB' firstRoot
   37   res <- mapM decidePkgDB' packageRoots
   38   if any (fstRes /=) res || (fstRes == CabalSandboxDB && not (null res))
   39     then return Nothing
   40     else return (Just fstRes)
   41 
   42 decidePkgDB' :: FilePath -> IO PackageDB
   43 decidePkgDB' root = do isSandbox <- checkSandbox
   44                        if isSandbox then return CabalSandboxDB
   45                                   else do isStack <- checkStack
   46                                           if isStack then return StackDB
   47                                                        else return DefaultDB
   48   where
   49     checkStack =
   50       withCurrentDirectory root $ (fmap $ either (\(_ :: SomeException) -> False) id) $ try $ do
   51         projRoot <- runCommandExpectOK "stack path --allow-different-user --project-root"
   52         absPath <- canonicalizePath root
   53         -- we only accept stack projects where the packages are (direct or indirect) subdirectories of the project root
   54         return $ maybe False (`isPrefixOf` absPath) projRoot
   55     checkSandbox = do
   56       hasConfigFile <- doesFileExist (root </> "cabal.config")
   57       hasSandboxFile <- doesFileExist (root </> "cabal.sandbox.config")
   58       return $ hasConfigFile || hasSandboxFile
   59 
   60 -- | Finds the location of the package database based on the configuration.
   61 packageDBLoc :: PackageDB -> FilePath -> IO [FilePath]
   62 packageDBLoc DefaultDB _ = do
   63   dbs <- runCommandExpectOK "ghc-pkg list base"
   64   return $ maybe [] (filter (\l -> not (null l) && not (" " `isPrefixOf` l)) . lines) dbs
   65 packageDBLoc CabalSandboxDB path = do
   66   hasConfigFile <- doesFileExist (path </> "cabal.config")
   67   config <- if hasConfigFile then readFile (path </> "cabal.config")
   68               else readFile (path </> "cabal.sandbox.config")
   69   return $ map (drop (length "package-db: ")) $ filter ("package-db: " `isPrefixOf`) $ lines config
   70 packageDBLoc StackDB path = withCurrentDirectory path $ do
   71    -- TODO: group the 3 calls into one for speed, split the output
   72    globalDB <- runCommandExpectOK "stack path --allow-different-user --global-pkg-db"
   73    snapshotDB <- runCommandExpectOK "stack path --allow-different-user --snapshot-pkg-db"
   74    localDB <- runCommandExpectOK "stack path --allow-different-user --local-pkg-db"
   75    return $ maybeToList localDB ++ maybeToList snapshotDB ++ maybeToList globalDB
   76 packageDBLoc (ExplicitDB dirs) _ = return dirs

   78 -- | Gets the (probable) location of autogen folder depending on which type of
   79 -- build we are using.
   80 detectAutogen :: FilePath -> PackageDB -> IO (Maybe FilePath)
   81 detectAutogen root DefaultDB = ifExists (root </> "dist" </> "build" </> "autogen")
   82 detectAutogen root (ExplicitDB _) = ifExists (root </> "dist" </> "build" </> "autogen")
   83 detectAutogen root CabalSandboxDB = ifExists (root </> "dist" </> "build" </> "autogen")
   84 detectAutogen root StackDB = (fmap $ either (\(_ :: SomeException) -> Nothing) id) $ try $ do
   85   dir <- withCurrentDirectory root $ do
   86     distDir <- runCommandExpectOK "stack path --allow-different-user --dist-dir"
   87     return $ trim (fromMaybe "" distDir)
   88   genExists <- doesDirectoryExist (root </> dir </> "build" </> "autogen")
   89   buildExists <- doesDirectoryExist (root </> dir </> "build")
   90   if | genExists -> return $ Just (root </> dir </> "build" </> "autogen")
   91      | buildExists -> do -- for some packages, the autogen folder is inside a folder named after the package
   92                          cont <- filterM doesDirectoryExist . map ((root </> dir </> "build") </>)
   93                                    =<< listDirectory (root </> dir </> "build")
   94                          existing <- mapM ifExists (map (</> "autogen") cont)
   95                          return $ choose existing
   96      | otherwise -> return Nothing

   98 -- | Run a command and return its result if successful display an error message otherwise.
   99 runCommandExpectOK :: String -> IO (Maybe String)
  100 runCommandExpectOK cmd = do
  101   (exitCode, res, errs) <- readCreateProcessWithExitCode (shell cmd) ""
  102   case exitCode of ExitSuccess -> return (Just $ trim res)
  103                    ExitFailure code -> do putStrLn ("The command '" ++ cmd ++ "' exited with "
  104                                                       ++ show code ++ ":\n" ++ errs)
  105                                           return Nothing
  106 
  107 trim :: String -> String
  108 trim = f . f
  109    where f = reverse . dropWhile isSpace
  110 
  111 -- take the first nonempty
  112 choose :: (Eq (f a), Alternative f) => [f a] -> f a
  113 choose = fromMaybe empty . find (/= empty)
  114 
  115 ifExists :: FilePath -> IO (Maybe FilePath)
  116 ifExists fp = do exists <- doesDirectoryExist fp
  117                  if exists then return (Just fp)
  118                            else return Nothing
