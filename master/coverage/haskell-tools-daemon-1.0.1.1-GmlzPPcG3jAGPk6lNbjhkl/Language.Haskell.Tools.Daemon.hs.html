   29 -- | Starts the daemon process. This will not return until the daemon stops. You can use this entry
   30 -- point when the other endpoint of the client connection is not needed, for example, when you use
   31 -- socket connection to connect to the daemon process.
   32 runDaemon' :: [RefactoringChoice] -> [QueryChoice] -> DaemonOptions -> IO ()
   33 runDaemon' refactorings queries args
   34   = do store <- newEmptyMVar
   35        runDaemon refactorings queries socketMode store args

   37 -- | Starts the daemon process. This will not return until the daemon stops.
   38 -- The daemon process is parameterized by the refactorings you can use in it. This entry point gives
   39 -- back the other endpoint of the connection so it can be used to run the daemon in the same process.
   40 runDaemon :: [RefactoringChoice] -> [QueryChoice] -> WorkingMode a -> MVar a -> DaemonOptions -> IO ()
   41 runDaemon _ _ _ _ DaemonOptions{..} | daemonVersion
   42   = putStrLn $ showVersion version
   43 runDaemon refactorings queries mode connStore config@DaemonOptions{..} = withSocketsDo $
   44     do when (not silentMode) $ putStrLn $ "Starting Haskell Tools daemon"
   45        hSetBuffering stdout LineBuffering
   46        hSetBuffering stderr LineBuffering
   47        conn <- daemonConnect mode portNumber
   48        putMVar connStore conn
   49        when (not silentMode) $ putStrLn $ "Connection established"
   50        (ghcSess, warnMVar) <- initGhcSession (generateCode sharedOptions)
   51        state <- newMVar initSession
   52        -- set the ghc flags given by command line
   53        case Options.ghcFlags sharedOptions of
   54          Just flags -> void $ respondTo config refactorings queries ghcSess state (daemonSend mode conn) warnMVar (SetGHCFlags flags)
   55          Nothing -> return ()
   56        case projectType sharedOptions of
   57          Just t -> void $ respondTo config refactorings queries ghcSess state (daemonSend mode conn) warnMVar (SetPackageDB t)
   58          Nothing -> return ()
   59        -- set up the file watch
   60        (wp,th) <- if noWatch sharedOptions
   61                     then return (Nothing, [])
   62                     else createWatchProcess'
   63                            (watchExe sharedOptions) ghcSess state warnMVar (daemonSend mode conn)
   64        modifyMVarMasked_ state ( \s -> return s { _watchProc = wp, _watchThreads = th })
   65        -- start the server loop
   66        serverLoop refactorings queries mode conn config ghcSess state warnMVar
   67        -- free allocated resources
   68        case wp of Just watchProcess -> stopWatch watchProcess th
   69                   Nothing -> return ()
   70        daemonDisconnect mode conn

   72 -- | Starts the server loop, receiving requests from the client and updated the server state
   73 -- according to these.
   74 serverLoop :: [RefactoringChoice] -> [QueryChoice] -> WorkingMode a -> a -> DaemonOptions -> Session
   75                 -> MVar DaemonSessionState -> MVar [Marker] -> IO ()
   76 serverLoop refactorings queries mode conn options ghcSess state warnMVar =
   77   do msgs <- daemonReceive mode conn
   78      continue <- mapM respondToMsg msgs
   79      sessionData <- readMVar state
   80      when (not (sessionData ^. exiting) && all (== True) continue)
   81        $ serverLoop refactorings queries mode conn options ghcSess state warnMVar
   82    `catches` exceptionHandlers (serverLoop refactorings queries mode conn options ghcSess state warnMVar)
   83                                (daemonSend mode conn . ErrorMessage)
   84   where respondToMsg (Right req)
   85           = do when (not (silentMode options)) $ putStrLn $ "Message received: " ++ show req
   86                respondTo options refactorings queries ghcSess state (daemonSend mode conn) warnMVar req
   87            `catches` userExceptionHandlers
   88                         (\s -> daemonSend mode conn (ErrorMessage s) >> return True)
   89                         (\err hint -> daemonSend mode conn (CompilationProblem err hint) >> return True)
   90         respondToMsg (Left msg) = do daemonSend mode conn $ ErrorMessage $ "MALFORMED MESSAGE: " ++ msg
   91                                      return True

   93 -- | Responds to a client request by modifying the daemon and GHC state accordingly.
   94 respondTo :: DaemonOptions -> [RefactoringChoice] -> [QueryChoice] -> Session
   95                -> MVar DaemonSessionState -> (ResponseMsg -> IO ()) -> MVar [Marker]
   96                -> ClientMessage -> IO Bool
   97 respondTo options refactorings queries ghcSess state next warnMVar req
   98   = modifyMVar state (\st -> swap <$> reflectGhc (runStateT upClient st) ghcSess)
   99   where upClient = updateClient options warnMVar refactorings queries next req
