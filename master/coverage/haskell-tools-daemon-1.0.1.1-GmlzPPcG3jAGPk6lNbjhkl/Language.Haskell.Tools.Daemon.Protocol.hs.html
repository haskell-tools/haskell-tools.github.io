    4 -- daemon engine and from the engine to the client.

   18 -- | The messages expected from the client.
   19 data ClientMessage
   20   = KeepAlive -- ^ A simple ping message to check that the server is running.
   21   | Reset -- ^ A message that instructs the server to reset its internal state and re-load loaded packages.
   22   | Handshake { clientVersion :: [Int] }
   23     -- ^ Tells the client version and asks the servers version.
   24   | SetPackageDB { pkgDB :: PackageDB }
   25     -- ^ Sets the package database for the engine to use.
   26   | AddPackages { addedPathes :: [FilePath] }
   27     -- ^ Registers packages to the engine. They will be subject to subsequent
   28     -- refactorings. Will cause the packages to be loaded, resulting in
   29     -- LoadingModules, LoadedModule or CompilationProblem responses.
   30   | RemovePackages { removedPathes :: [FilePath] }
   31     -- ^ Deregisters the given packages from the engine. They will not be
   32     -- subjects of further refactorings.
   33   | SetWorkingDir { newWorkingDir :: FilePath }
   34     -- ^ Sets the working directory for the compilation. Important when
   35     -- compiling code that loads resources based on relative pathes.
   36   | SetGHCFlags { ghcFlags :: [String] }
   37     -- ^ Sets the compilation flags. The unused flags are returned via the
   38     -- UnusedFlags response.
   39   | PerformRefactoring { refactoring :: String
   40                        , modulePath :: FilePath
   41                        , editorSelection :: String
   42                        , details :: [String] -- ^ Additional details for the refactoring like the
   43                                              -- names of generated definitions.
   44                        , shutdownAfter :: Bool -- ^ Stop the daemon after performing the refactoring.
   45                        , diffMode :: Bool -- ^ Don't change the files, send back the result as
   46                                           -- a unified diff.
   47                        }
   48     -- ^ Orders the engine to perform the refactoring on the module given
   49     -- with the selection and details. Successful refactorings will cause re-loading of modules.
   50     -- If 'shutdownAfter' or 'diffMode' is not set, after the refactoring,
   51     -- modules are re-loaded, LoadingModules, LoadedModule responses are sent.
   52   | PerformQuery { query :: String
   53                  , modulePath :: FilePath
   54                  , editorSelection :: String
   55                  , details :: [String] -- ^ Additional details for the refactoring like the
   56                                        -- names of generated definitions.
   57                  , shutdownAfter :: Bool -- ^ Stop the daemon after performing the query.
   58                  }
   59     -- ^ Orders the engine to perform a query on the module given with the selection and details.
   60   | UndoLast
   61     -- ^ Asks the daemon to undo the last refactoring.
   62   | Disconnect
   63     -- ^ Stops the engine. It replies with Disconnected.
   64   | ReLoad { addedModules :: [FilePath]
   65            , changedModules :: [FilePath]
   66            , removedModules :: [FilePath]
   67            }
   68     -- ^ Instructs the engine to re-load a changed module.
   69     -- LoadingModules, LoadedModule responses may be sent.
   70   | Stop -- TODO: remove
   71     -- ^ Stops the server. OBSOLATE
   72   deriving (Show, Generic)
   73 
   74 instance FromJSON ClientMessage
   75 
   76 -- | The possible responses that the server can give.
   77 data ResponseMsg
   78   = KeepAliveResponse -- ^ A response to KeepAlive
   79   | HandshakeResponse { serverVersion :: [Int] }
   80     -- ^ Tells the version of the server.
   81   | ErrorMessage { errorMsg :: String }
   82     -- ^ An error message marking internal problems or user mistakes.
   83     -- TODO: separate internal problems and user mistakes.
   84   | CompilationProblem { markers :: [Marker]
   85                        , errorHints :: [String]
   86                        }
   87     -- ^ A response that tells there are errors in the source code given.
   88   | DiffInfo { diffInfo :: String }
   89     -- ^ Information about changes that would be caused by the refactoring.
   90   | LoadingModules { modulesToLoad :: [FilePath] }
   91     -- ^ The traversal of the project is done, now the engine is loading the
   92     -- given modules.
   93   | LoadedModule { loadedModulePath :: FilePath
   94                  , loadedModuleName :: String
   95                  }
   96     -- ^ The engine has loaded the given module.
   97   | QueryResult { queryName :: String
   98                 , queryResult :: Value
   99                 }
  100     -- ^ The result of querying the program representation.
  101   | UnusedFlags { unusedFlags :: [String] }
  102     -- ^ Returns the flags that are not used by the engine.
  103   | Disconnected
  104     -- ^ The engine has closed the connection.
  105   deriving (Show, Generic)
  106 
  107 data Marker = Marker { location :: SrcSpan
  108                      , severity :: Severity
  109                      , message :: String
  110                      } deriving (Generic, Eq)
  111 
  112 instance Show Marker where
  113   show marker = show (severity marker) ++ " at " ++ shortShowSpanWithFile (location marker) ++ ": " ++ message marker
  114 
  115 data Severity = Error | Warning | Info
  116   deriving (Show, Generic, Eq)
  117 
  118 instance ToJSON ResponseMsg
  119 instance ToJSON Marker
  120 instance ToJSON Severity
  121 
  122 instance ToJSON SrcSpan where
  123   toJSON (RealSrcSpan sp) = object [ "file" A..= unpackFS (srcSpanFile sp)
  124                                    , "startRow" A..= srcLocLine (realSrcSpanStart sp)
  125                                    , "startCol" A..= srcLocCol (realSrcSpanStart sp)
  126                                    , "endRow" A..= srcLocLine (realSrcSpanEnd sp)
  127                                    , "endCol" A..= srcLocCol (realSrcSpanEnd sp)
  128                                    ]
  129   toJSON _ = Null

  131 data UndoRefactor = RemoveAdded { undoRemovePath :: FilePath }
  132                   | RestoreRemoved { undoRestorePath :: FilePath
  133                                    , undoRestoreContents :: String
  134                                    }
  135                   | UndoChanges { undoChangedPath :: FilePath
  136                                 , undoDiff :: FileDiff
  137                                 }
  138   deriving (Show, Generic, NFData)

  140 instance ToJSON UndoRefactor
