   16 -- | The modules of a library, executable, test or benchmark. A package contains one or more module collection.
   17 data ModuleCollection k
   18   = ModuleCollection { _mcId :: ModuleCollectionId
   19                      , _mcLoadDone :: Bool
   20                      , _mcRoot :: FilePath
   21                      , _mcSourceDirs :: [FilePath]
   22                      , _mcModuleFiles :: [(ModuleNameStr, FilePath)]
   23                      , _mcModules :: (Map.Map k ModuleRecord)
   24                      , _mcFlagSetup :: (DynFlags -> IO DynFlags) -- ^ Sets up the ghc environment for compiling the modules of this collection
   25                      , _mcLoadFlagSetup :: (DynFlags -> IO DynFlags) -- ^ Sets up the ghc environment for dependency analysis
   26                      , _mcDependencies :: [ModuleCollectionId]
   27                      }
   28 
   29 modCollToSfk :: ModuleCollection ModuleNameStr -> ModuleCollection SourceFileKey
   30 modCollToSfk ModuleCollection{..} = ModuleCollection{ _mcModules = Map.mapKeys (SourceFileKey "") _mcModules, ..}
   31 
   32 -- | The state of a module.
   33 data ModuleRecord
   34       = ModuleNotLoaded { _modRecCodeGen :: CodeGenPolicy
   35                         , _recModuleExposed :: Bool
   36                         }
   37       | ModuleParsed { _parsedRecModule :: UnnamedModule
   38                      , _modRecMS :: ModSummary
   39                      }
   40       | ModuleRenamed { _renamedRecModule :: UnnamedModule
   41                       , _modRecMS :: ModSummary
   42                       }
   43       | ModuleTypeChecked { _typedRecModule :: UnnamedModule
   44                           , _modRecMS :: ModSummary
   45                           , _modRecCodeGen :: CodeGenPolicy
   46                           }
   47 
   48 isLoaded :: ModuleRecord -> Bool
   49 isLoaded ModuleTypeChecked{} = True
   50 isLoaded _ = False
   51 
   52 data CodeGenPolicy = NoCodeGen | InterpretedCode | GeneratedCode
   53   deriving (Eq, Ord, Show)
   54 
   55 -- | An alias for module names
   56 type ModuleNameStr = String
   57 
   58 -- | This data structure identifies a module collection.
   59 data ModuleCollectionId = DirectoryMC FilePath
   60                         | LibraryMC String
   61                         | ExecutableMC String String
   62                         | TestSuiteMC String String
   63                         | BenchmarkMC String String
   64  deriving (Eq, Ord, Show)
   65 
   66 instance Eq (ModuleCollection k) where
   67   (==) = (==) `on` _mcId
   68 
   69 instance Show k => Show (ModuleCollection k) where
   70   show (ModuleCollection id loaded root srcDirs mapping mods _ _ deps)
   71     = "ModuleCollection (" ++ show id ++ ") " ++ show loaded ++ " " ++ root ++ " " ++ show srcDirs ++ " " ++ show mapping
   72         ++ " (" ++ show mods ++ ") " ++ show deps
   73 
   74 makeReferences ''ModuleCollection
   75 makeReferences ''ModuleRecord
   76 
   77 instance Show ModuleRecord where
   78   show (ModuleNotLoaded code exposed) = "ModuleNotLoaded " ++ show code ++ " " ++ show exposed
   79   show mr@(ModuleParsed {}) = "ModuleParsed (" ++ (GHC.moduleNameString $ GHC.moduleName $ GHC.ms_mod $ fromJust $ mr ^? modRecMS) ++ ")"
   80   show mr@(ModuleRenamed {}) = "ModuleRenamed (" ++ (GHC.moduleNameString $ GHC.moduleName $ GHC.ms_mod $ fromJust $ mr ^? modRecMS) ++ ")"
   81   show mr@(ModuleTypeChecked {}) = "ModuleTypeChecked (" ++ (GHC.moduleNameString $ GHC.moduleName $ GHC.ms_mod $ fromJust $ mr ^? modRecMS) ++ ")"
