   54                            , warnMVar :: MVar [Marker]

   57 -- | This function does the real job of acting upon client messages in a stateful environment of a
   58 -- client.
   59 updateClient :: DaemonOptions -> MVar [Marker] -> [RefactoringChoice] -> [QueryChoice]
   60                   -> (ResponseMsg -> IO ()) -> ClientMessage
   61                   -> DaemonSession Bool
   62 updateClient options warnMVar refactors queries resp
   63   = updateClient' (UpdateCtx options refactors queries resp warnMVar)
   64 
   65 updateClient' :: UpdateCtx -> ClientMessage -> DaemonSession Bool
   66 -- resets the internal state of Haskell-tools (but keeps options)
   67 updateClient' UpdateCtx{..} Reset
   68   = do roots <- gets (^? refSessMCs & traversal & mcRoot)
   69        modify' $ resetSession
   70        Session sess <- liftIO $ reinitGhcSession warnMVar (generateCode (sharedOptions options))
   71        env <- liftIO $ readIORef sess
   72        dfs <- getSessionDynFlags
   73        when (not $ gopt Opt_ExternalInterpreter dfs) $
   74          liftIO $ unload env [] -- clear (unload everything) the global state of the linker
   75        lift $ setSession env
   76        addPackages response warnMVar roots
   77        return True
   78 
   79 updateClient' UpdateCtx{..} (Handshake _)
   80   = do liftIO (response $ HandshakeResponse $ versionBranch version)
   81        return True
   82 
   83 updateClient' UpdateCtx{..} KeepAlive
   84   = do liftIO (response KeepAliveResponse)
   85        return True

   88   = do liftIO (response Disconnected)

   92   = do mcs <- gets (^. refSessMCs)

   95          else liftIO $ response $ ErrorMessage "The package database is already in use and cannot be changed."
   96        return True

  112 updateClient' UpdateCtx{..} (RemovePackages packagePathes) = do
  113     mcs <- gets (^. refSessMCs)
  114     let existingFiles = concatMap @[] (map (^. sfkFileName) . Map.keys) (mcs ^? traversal & filtered isRemoved & mcModules)
  115     lift $ forM_ existingFiles (\fs -> removeTarget (TargetFile fs Nothing))
  116     lift $ deregisterDirs (mcs ^? traversal & filtered isRemoved & mcSourceDirs & traversal)
  117     modify' $ refSessMCs .- filter (not . isRemoved)
  118     modifySession (\s -> s { hsc_mod_graph = filter ((`notElem` existingFiles) . getModSumOrig) (hsc_mod_graph s) })
  119     mcs <- gets (^. refSessMCs)
  120     when (null mcs) $ modify' (packageDB .= Nothing)
  121     return True
  122   where isRemoved mc = (mc ^. mcRoot) `elem` packagePathes

  124 updateClient' UpdateCtx{..} UndoLast | disableHistory $ sharedOptions options
  125   = do liftIO $ response $ ErrorMessage "Recording history has been disabled from command line."
  126        return True
  127 updateClient' UpdateCtx{..} UndoLast =
  128   do undos <- gets (^. undoStack)
  129      case undos of
  130        [] -> do liftIO $ response $ ErrorMessage "There is nothing to undo. Please note that the refactoring history is cleared after manual changes in the refactored files."
  131                 return True
  132        lastUndo:_ -> do
  133          modify (undoStack .- tail)
  134          liftIO $ mapM_ performUndo lastUndo
  135          reloadModules response warnMVar (getUndoAdded lastUndo)
  136                                          (getUndoChanged lastUndo)
  137                                          (getUndoRemoved lastUndo) -- reload the reverted files
  138          return True

  141   = do updateForFileChanges response warnMVar added changed removed
  142        return True

  148 updateClient' UpdateCtx{..} (PerformQuery query modPath selection args shutdown)
  149   = do (selectedMod, otherMods) <- getFileMods modPath
  150        res <- lift $ performQuery queries (query:selection:args) (maybe (Left modPath) Right selectedMod) otherMods
  151        case res of
  152          Left err -> liftIO $ response $ ErrorMessage err
  153          Right res -> liftIO $ response $ QueryResult query res
  154        when shutdown $ liftIO $ response Disconnected
  155        return (not shutdown)
  156 
  157 updateClient' UpdateCtx{..} (PerformRefactoring refact modPath selection args shutdown diffMode)
  158   = do (selectedMod, otherMods) <- getFileMods modPath
  159        performRefactoring (refact:selection:args)
  160                           (maybe (Left modPath) Right selectedMod) otherMods
  161        when shutdown $ liftIO $ response Disconnected
  162        return (not shutdown)
  163   where performRefactoring cmd actualMod otherMods = do
  164           res <- lift $ performCommand refactorings cmd actualMod otherMods
  165           case res of
  166             Left err -> liftIO $ response $ ErrorMessage err
  167             Right diff -> do changedMods <- applyChanges diff
  168                              if not diffMode
  169                                then when (not (disableHistory $ sharedOptions options)) $ updateHistory changedMods
  170                                else liftIO $ response $ DiffInfo (concatMap (either snd (^. _4)) changedMods)
  171                              isWatching <- gets (isJust . (^. watchProc))
  172                              if not isWatching && not shutdown && not diffMode
  173                               -- if watch is on, then it will automatically
  174                               -- reload changed files, otherwise we do it manually
  175                                then do reloadChanges (map ((^. sfkFileName) . (^. _1)) (rights changedMods))
  176                                        reportWarnings response warnMVar
  177                                else modify (touchedFiles .= Set.fromList (map ((^. sfkFileName) . (^. _1)) (rights changedMods)))
  178 
  179         applyChanges changes = do
  180           forM changes $ \case
  181             ModuleCreated n m otherM -> do
  182               mcs <- gets (^. refSessMCs)
  183               Just (_, otherMR) <- gets (lookupModInSCs otherM . (^. refSessMCs))
  184               let Just otherMS = otherMR ^? modRecMS
  185                   Just mc = lookupModuleColl (otherM ^. sfkModuleName) mcs
  186               otherSrcDir <- liftIO $ getSourceDir otherMS
  187               let loc = toFileName otherSrcDir n
  188               let newCont = prettyPrint m
  189               when (not diffMode) $ do
  190                 modify' $ refSessMCs & traversal & filtered (\mc' -> (mc' ^. mcId) == (mc ^. mcId)) & mcModules
  191                             .- Map.insert (SourceFileKey loc n) (ModuleNotLoaded NoCodeGen False)
  192                 liftIO $ withBinaryFile loc WriteMode $ \handle -> do
  193                   hSetEncoding handle utf8
  194                   hPutStr handle newCont
  195                   hFlush handle
  196                 lift $ addTarget (Target (TargetFile loc Nothing) True Nothing)
  197               return $ Right (SourceFileKey loc n, loc, RemoveAdded loc, createUnifiedDiff loc "" newCont)
  198             ContentChanged (n,m) -> do
  199               let newCont = prettyPrint m
  200                   file = n ^. sfkFileName
  201               origCont <- liftIO $ withBinaryFile file ReadMode $ \handle -> do
  202                 hSetEncoding handle utf8
  203                 StrictIO.hGetContents handle
  204               let undo = createUndo 0 $ getGroupedDiff origCont newCont
  205               let unifiedDiff = createUnifiedDiff file origCont newCont
  206               when (not diffMode) $ do
  207                liftIO $ withBinaryFile file WriteMode $ \handle -> do
  208                  hSetEncoding handle utf8
  209                  hPutStr handle newCont
  210                  hFlush handle
  211               return $ Right (n, file, UndoChanges file undo, unifiedDiff)
  212             ModuleRemoved mod -> do
  213               Just (sfk,_) <- gets (lookupModuleInSCs mod . (^. refSessMCs))
  214               let file = sfk ^. sfkFileName
  215               origCont <- liftIO (StrictBS.unpack <$> StrictBS.readFile file)
  216               when (not diffMode) $ do
  217                 lift $ removeTarget (TargetFile file Nothing)
  218                 modify' $ (refSessMCs .- removeModule mod)
  219                 liftIO $ removeFile file
  220               return $ Left (RestoreRemoved file origCont, createUnifiedDiff file origCont "")
  221 
  222         reloadChanges changedMods
  223           = reloadChangedModules (\ms -> response (LoadedModule (getModSumOrig ms) (getModSumName ms)))
  224                                  (\mss -> response (LoadingModules (map getModSumOrig mss)))
  225                                  (\ms -> getModSumOrig ms `elem` changedMods)

  227         updateHistory :: [Either (UndoRefactor, b) (SourceFileKey, FilePath, UndoRefactor, String)] -> DaemonSession ()
  228         updateHistory changedMods
  229           = do modify (undoStack .- (map (either fst (^. _3)) changedMods :))
  230                -- force the evaluation of the undo stack to prevent older versions of
  231                -- modules seeming to be used when they could be garbage collected
  232                us <- gets (^. undoStack)
  233                liftIO $ evaluate $ force us
  234                return ()

  236 addPackages :: (ResponseMsg -> IO ()) -> MVar [Marker] -> [FilePath] -> DaemonSession ()
  237 addPackages _ _ [] = return ()
  238 addPackages resp warnMVar packagePathes = do
  239   roots <- liftIO $ mapM canonicalizePath packagePathes
  240   nonExisting <- filterM ((return . not) <=< liftIO . doesDirectoryExist) roots
  241   DaemonSessionState {..} <- get
  242   if (not (null nonExisting))
  243     then liftIO $ resp $ ErrorMessage $ "The following packages are not found: " ++ concat (intersperse ", " nonExisting)
  244     else do
  245       forM_ roots watchNew -- put a file system watch on each package
  246       -- clear existing removed packages
  247       existingMCs <- gets (^. refSessMCs)
  248       let existing = (existingMCs ^? traversal & filtered (isTheAdded roots) & mcModules & traversal & modRecMS)
  249           existingModNames = map ms_mod existing
  250       needToReload <- (filter (\ms -> not $ ms_mod ms `elem` existingModNames))
  251                         <$> getReachableModules (\_ -> return ()) (\ms -> ms_mod ms `elem` existingModNames)
  252       modify' $ refSessMCs .- filter (not . isTheAdded roots) -- remove the added package from the database
  253       forM_ existing $ \ms -> removeTarget (TargetFile (getModSumOrig ms) Nothing)
  254       modifySession (\s -> s { hsc_mod_graph = filter (not . (`elem` existingModNames) . ms_mod) (hsc_mod_graph s) })
  255       -- load new modules
  256       pkgDBok <- initializePackageDBIfNeeded roots
  257       if pkgDBok then do
  258         errs <- loadPackagesFrom
  259                   (\ms -> resp (LoadedModule (getModSumOrig ms) (getModSumName ms)))
  260                   (resp . LoadingModules . map getModSumOrig)
  261                   (\st fp -> maybe (return []) (fmap maybeToList . detectAutogen fp . fst) (st ^. packageDB)) roots
  262         -- handle source errors here to prevent rollback on the tool state
  263         let errors = foldl mappend ([],[]) (map getProblems errs)
  264         if null (fst errors)
  265           then do mapM_ (reloadModule (\_ -> return ())) needToReload -- don't report consequent reloads (not expected)
  266                   reportWarnings resp warnMVar
  267           else liftIO $ resp $ uncurry CompilationProblem errors
  268       else liftIO $ resp $ ErrorMessage $ "Attempted to load two packages with different package DB. "
  269                                             ++ "Stack, cabal-sandbox and normal packages cannot be combined"
  270   where isTheAdded roots mc = (mc ^. mcRoot) `elem` roots
  271         initializePackageDBIfNeeded roots = do
  272           db <- dbSetup roots
  273           case db of
  274             Nothing -> return False
  275             Just (pkgDB, _) -> do
  276               locs <- liftIO $ packageDBLoc pkgDB (head roots)
  277               usePackageDB locs
  278               modify' (packageDBLocs .= locs)
  279               return True
  280         dbSetup roots = do
  281           pkgDB <- gets (^. packageDB)
  282           case pkgDB of Nothing          -> do db <- liftIO $ decidePkgDB roots
  283                                                modify (packageDB .= fmap (, False) db)
  284                                                return (fmap (, False) db)
  285                         Just (_, True)   -> return pkgDB
  286                         Just (db, False) -> do newDB <- liftIO $ decidePkgDB roots
  287                                                if newDB == Just db then return pkgDB
  288                                                                    else return Nothing

  290 -- | Updates the state of the tool after some files have been changed (possibly by another application)
  291 updateForFileChanges :: (ResponseMsg -> IO ()) -> MVar [Marker] -> [FilePath] -> [FilePath] -> [FilePath] -> DaemonSession ()
  292 updateForFileChanges resp warnMVar added changed removed = do
  293   -- clear undo stack if the changes are from another application
  294   refactoredFiles <- gets (^. touchedFiles)
  295   let changeSet = Set.fromList (added ++ changed ++ removed)
  296   when (not $ changeSet `Set.isSubsetOf` refactoredFiles)
  297     $ modify (undoStack .= []) -- clear the undo stack, if this changeset is not a result of a refactoring
  298   -- check for module collections that failed to load
  299   mcs <- gets (^. refSessMCs)
  300   let packagesNotLoaded = filter (not . (^. mcLoadDone)) mcs
  301       tryLoadAgain = filter (\r -> List.any (r `isPrefixOf`) (added ++ changed ++ removed)) $ map (^. mcRoot) packagesNotLoaded
  302   -- check for changes in .cabal files
  303   modify (touchedFiles .- (Set.\\ changeSet))
  304   let changedPackages = map takeDirectory
  305                           $ filter (\fp -> takeExtension fp == ".cabal")
  306                           $ added ++ changed ++ removed
  307       reloadedPackages = tryLoadAgain ++ changedPackages
  308       packageNotReloaded fp = not $ any (`isPrefixOf` fp) reloadedPackages
  309   addPackages resp warnMVar reloadedPackages -- reload packages if needed
  310   -- reload the rest of the modules
  311   reloadModules resp warnMVar (filter packageNotReloaded added) (filter packageNotReloaded changed)
  312                               (filter packageNotReloaded removed)

  314 -- | Reloads changed modules to have an up-to-date version loaded
  315 reloadModules :: (ResponseMsg -> IO ()) -> MVar [Marker] -> [FilePath] -> [FilePath] -> [FilePath] -> DaemonSession ()
  316 reloadModules _ _ [] [] [] = return ()
  317 reloadModules resp warnMVar added changed removed = do
  318   lift $ forM_ removed (\src -> removeTarget (TargetFile src Nothing))
  319   -- remove targets deleted
  320   modify' $ refSessMCs & traversal & mcModules
  321               .- Map.filter (\m -> maybe True ((`notElem` removed) . getModSumOrig) (m ^? modRecMS))
  322   modifySession (\s -> s { hsc_mod_graph = filter (\mod -> getModSumOrig mod `notElem` removed) (hsc_mod_graph s) })
  323   -- reload changed modules
  324   -- TODO: filter those that are in reloaded packages
  325   reloadChangedModules (\ms -> resp (LoadedModule (getModSumOrig ms) (getModSumName ms)))
  326                        (\mss -> resp (LoadingModules (map getModSumOrig mss)))
  327                        (\ms -> getModSumOrig ms `elem` changed)
  328   reportWarnings resp warnMVar
  329   mcs <- gets (^. refSessMCs)
  330   let mcsToReload = filter (\mc -> any ((mc ^. mcRoot) `isPrefixOf`) added && isNothing (moduleCollectionPkgId (mc ^. mcId))) mcs
  331   addPackages resp warnMVar (map (^. mcRoot) mcsToReload) -- reload packages containing added modules

  333 reportWarnings :: (ResponseMsg -> IO ()) -> MVar [Marker] -> DaemonSession ()
  334 reportWarnings resp warnMVar = liftIO $ do
  335   warns <- modifyMVar warnMVar (return . ([],))
  336   when (not (null warns)) $ (resp $ CompilationProblem warns [])

  338 -- | Creates a compressed set of changes in one file
  339 createUndo :: Eq a => Int -> [Diff [a]] -> [(Int, Int, [a])]
  340 createUndo i (Both str _ : rest) = createUndo (i + length str) rest
  341 createUndo i (First rem : Second add : rest)
  342   = (i, i + length add, rem) : createUndo (i + length add) rest
  343 createUndo i (First rem : rest) = (i, i, rem) : createUndo i rest
  344 createUndo i (Second add : rest)
  345   = (i, i + length add, []) : createUndo (i + length add) rest
  346 createUndo _ [] = []

  348 -- | Creates a unified-style diff of two texts. Only used when the user wants to know what would change.
  349 createUnifiedDiff :: FilePath -> String -> String -> String
  350 createUnifiedDiff name left right
  351   = render $ prettyContextDiff (PP.text name) (PP.text name) PP.text $ getContextDiff 3 (lines left) (lines right)

  353 -- | Undo a refactoring change using the information that was saved earlier.
  354 performUndo :: UndoRefactor -> IO ()
  355 performUndo (RemoveAdded fp) = removeFile fp
  356 performUndo (RestoreRemoved fp cont)
  357   = liftIO $ withBinaryFile fp WriteMode $ \handle -> do
  358       hSetEncoding handle utf8
  359       hPutStr handle cont
  360 performUndo (UndoChanges fp changes) = do
  361   cont <- liftIO $ withBinaryFile fp ReadMode $ \handle -> do
  362     hSetEncoding handle utf8
  363     StrictIO.hGetContents handle
  364   liftIO $ withBinaryFile fp WriteMode $ \handle -> do
  365       hSetEncoding handle utf8
  366       hPutStr handle (performUndoChanges 0 changes cont)
  367 
  368 -- | Undo the changes in one file using the information that was saved earlier. See 'createUndo'.
  369 performUndoChanges :: Int -> FileDiff -> String -> String
  370 performUndoChanges i ((start,end,replace):rest) str | i == start
  371   = replace ++ performUndoChanges end rest (drop (end-start) str)
  372 performUndoChanges i diffs (c:str) = c : performUndoChanges (i+1) diffs str
  373 performUndoChanges _ _ [] = []

  376 getUndoAdded :: [UndoRefactor] -> [FilePath]
  377 getUndoAdded = catMaybes . map (\case RestoreRemoved fp _ -> Just fp

  390 initGhcSession :: Bool -> IO (Session, MVar [Marker])
  391 initGhcSession genCode = do
  392   mv <- newMVar []
  393   sess <- Session <$> (newIORef =<< runGhc (Just libdir) (initGhcFlags' genCode True >> setupLogging mv >> getSession))
  394   return (sess, mv)

  396 reinitGhcSession :: MVar [Marker] -> Bool -> IO Session

  400 setupLogging :: MVar [Marker] -> Ghc ()
  401 setupLogging mv = modifySession $ \s -> s { hsc_dflags = (hsc_dflags s) { log_action = logger } }
  402   where logger dfs reason sev sp _ msg = modifyMVar_ mv (return . (Marker sp (severity sev reason) (showSDoc dfs msg) :))
  403         severity SevError _ = Error
  404         severity SevWarning (Reason Opt_WarnDeferredTypeErrors) = Error
  405         severity SevWarning (Reason Opt_WarnDeferredOutOfScopeVariables) = Error
  406         severity SevWarning _ = HT.Warning
  407         severity _ _ = Info

  409 usePackageDB :: GhcMonad m => [FilePath] -> m ()
