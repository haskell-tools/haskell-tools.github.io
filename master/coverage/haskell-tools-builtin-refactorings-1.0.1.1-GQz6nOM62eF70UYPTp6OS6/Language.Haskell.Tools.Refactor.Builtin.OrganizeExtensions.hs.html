   44 organizeExtensions :: LocalRefactoring
   45 organizeExtensions moduleAST = do
   46   exts <- liftGhc $ reduceExtensions moduleAST
   47   let langExts = mkLanguagePragma . map show $ exts
   48       ghcOpts  = moduleAST ^? filePragmas & annList & opStr & stringNodeStr
   49       ghcOpts' = map (mkOptionsGHC . unwords . filter (isPrefixOf "-") . words) ghcOpts
   50 
   51       newPragmas = mkFilePragmas $ langExts:ghcOpts'
   52 
   53   (filePragmas != newPragmas)
   54     -- remove empty {-# LANGUAGE #-} pragmas
   55     >=> filePragmas !~ filterListSt (\case LanguagePragma (AnnList []) -> False; _ -> True)
   56     $ moduleAST
   57 
   58 -- | Reduces default extension list (keeps unsupported extensions)
   59 reduceExtensions :: UnnamedModule -> Ghc [Extension]
   60 reduceExtensions moduleAST = do
   61   let defaults = map replaceDeprecated . collectDefaultExtensions $ moduleAST
   62       expanded = expandExtensions defaults
   63       (xs, ys) = partition isSupported expanded
   64   -- we can't say anything about generated code
   65   if TemplateHaskell `notElem` expanded
   66     then do
   67       xs' <- flip execStateT SMap.empty . flip runReaderT xs . traverseModule $ moduleAST
   68       -- Merging is needed because there might be unsopported extensions
   69       -- that are implied by supported extensions (TypeFamilies -> MonoLocalBinds)
   70       return . sortBy (compare `on` show) . nub . mergeImplied $ (determineExtensions xs' ++ ys)
   71     else
   72       return . mergeImplied $ defaults
   73 
   74 -- | Collects extensions induced by the source code (with location info)
   75 collectExtensions :: UnnamedModule -> Ghc ExtMap
   76 collectExtensions = collectExtensionsWith traverseModule
   77 
   78 collectExtensionsWith :: CheckNode UnnamedModule -> UnnamedModule -> Ghc ExtMap
   79 collectExtensionsWith trvModule moduleAST = do
   80   let expanded = expandExtensions . collectDefaultExtensions $ moduleAST
   81   flip execStateT SMap.empty . flip runReaderT expanded . trvModule $ moduleAST
   82

   84 -- | Expands every extension in a list, while not producing any duplicates.
   85 expandExtensions :: [Extension] -> [Extension]
   86 expandExtensions = nub . concatMap expandExtension
   87 
   88 -- | Collects extensions enabled by default
   89 collectDefaultExtensions :: UnnamedModule -> [Extension]
   90 collectDefaultExtensions = mapMaybe toExt . getExtensions
   91   where
   92   getExtensions :: UnnamedModule -> [String]
   93   getExtensions = flip (^?) (filePragmas & annList & lpPragmas & annList & langExt)
   94 
   95 toExt :: String -> Maybe Extension
   96 toExt str = case map fst . reads . canonExt . takeWhile isAlpha $ str of
   97               e:_ -> Just e
   98               []  -> fail $ "Extension '" ++ takeWhile isAlpha str ++ "' is not known."
