   22 tryItOut :: String -> String -> IO ()
   23 tryItOut mod sp = tryRefactor (localRefactoring . autoCorrect) mod sp

   25 autoCorrect :: RealSrcSpan -> LocalRefactoring
   26 autoCorrect sp mod
   27   = do res <- mapM (\f -> f sp mod) [reParen, reOrder]
   28        case catMaybes res of mod':_ -> return mod'
   29                              []     -> refactError "Cannot auto-correct the selection."

   33 reOrder :: RealSrcSpan -> HT.Module -> LocalRefactor (Maybe HT.Module)
   34 reOrder sp mod = do let accessibleMods = semanticsModule mod : semanticsPrelTransMods mod ++ concatMap semanticsTransMods (mod ^? modImports & annList :: [HT.ImportDecl])
   35                         rng:_ = map getRange (mod ^? nodesContained sp :: [Expr])
   36                     insts <- liftGhc $ fst <$> getInstances accessibleMods
   37                     (res,done) <- liftGhc $ flip runStateT False ((nodesContained sp & filtered ((==rng) . getRange) !~ reOrderExpr insts) mod)
   38                     return (if done then Just res else Nothing)
   39 
   40 reOrderExpr :: [ClsInst] -> Expr -> StateT Bool Ghc Expr
   41 reOrderExpr insts e@(App (App f a1) a2)
   42   = do funTy <- lift $ typeExpr f
   43        arg1Ty <- lift $ typeExpr a1
   44        arg2Ty <- lift $ typeExpr a2
   45        -- liftIO $ putStrLn $ (show $ isJust $ appTypeMatches insts funTy [arg1Ty, arg2Ty]) ++ " " ++ (show $ isJust $ appTypeMatches insts funTy [arg2Ty, arg1Ty]) ++ " " ++ (showSDocUnsafe $ ppr $ funTy) ++ " " ++ (showSDocUnsafe $ ppr $ arg1Ty) ++ " " ++ (showSDocUnsafe $ ppr $ arg2Ty)
   46        if not (isJust (appTypeMatches insts funTy [arg1Ty, arg2Ty])) && isJust (appTypeMatches insts funTy [arg2Ty, arg1Ty])
   47           then put True >> return (exprArg .= a1 $ exprFun&exprArg .= a2 $ e)
   48           else return e
   49 reOrderExpr insts e@(InfixApp lhs op rhs) 
   50   = do let funTy = idType $ semanticsId (op ^. operatorName)
   51        lhsTy <- lift $ typeExpr lhs
   52        rhsTy <- lift $ typeExpr rhs
   53        -- liftIO $ putStrLn $ (show $ isJust $ appTypeMatches insts funTy [lhsTy, rhsTy]) ++ " " ++ (show $ isJust $ appTypeMatches insts funTy [lhsTy, rhsTy]) ++ " " ++ (showSDocUnsafe $ ppr $ funTy) ++ " " ++ (showSDocUnsafe $ ppr $ lhsTy) ++ " " ++ (showSDocUnsafe $ ppr $ rhsTy)
   54        if not (isJust (appTypeMatches insts funTy [lhsTy, rhsTy])) && isJust (appTypeMatches insts funTy [rhsTy, lhsTy])
   55           then put True >> return (exprLhs .= rhs $ exprRhs .= lhs $ e)
   56           else return e
   57 reOrderExpr _ e = return e

   61 reParen :: RealSrcSpan -> HT.Module -> LocalRefactor (Maybe HT.Module)
   62 reParen sp mod = do let accessibleMods = semanticsModule mod : semanticsPrelTransMods mod ++ concatMap semanticsTransMods (mod ^? modImports & annList :: [HT.ImportDecl])
   63                         rng:_ = map getRange (mod ^? nodesContained sp :: [Expr])
   64                     insts <- liftGhc $ fst <$> getInstances accessibleMods
   65                     (res,done) <- liftGhc $ flip runStateT False ((nodesContained sp & filtered ((==rng) . getRange) !~ reParenExpr insts) mod)
   66                     return (if done then Just res else Nothing)
   67 
   68 reParenExpr :: [ClsInst] -> Expr -> StateT Bool Ghc Expr
   69 reParenExpr insts e = do atoms <- lift $ extractAtoms e
   70                          case correctParening insts $ map (_2 .- Left) atoms of
   71                            [e'] -> put True >> return (wrapAtom e')
   72                            [] -> return e
   73                            ls -> -- TODO: choose the best one
   74                                  error $ "multiple correct parentheses were found: " ++ intercalate ", " (map (either prettyPrintAtom prettyPrint) ls)
   75 
   76 data Atom = NameA HT.Name
   77           | OperatorA Operator
   78           | LiteralA Literal
   79 
   80 prettyPrintAtom :: Atom -> String
   81 prettyPrintAtom (NameA n) = prettyPrint n
   82 prettyPrintAtom (OperatorA o) = prettyPrint o
   83 prettyPrintAtom (LiteralA l) = prettyPrint l
   84 
   85 type Build = Either Atom Expr
   86 
   87 extractAtoms :: Expr -> Ghc [(GHC.Type, Atom)]
   88 extractAtoms e = do lits <- mapM (\l -> (, LiteralA l) <$> literalType l) (e ^? biplateRef)
   89                     return $ sortOn (srcSpanStart . atomRange . snd) 
   90                            $ map (\n -> (idType $ semanticsId (n ^. simpleName), NameA n)) (e ^? biplateRef) 
   91                                ++ map (\o -> (idType $ semanticsId (o ^. operatorName), OperatorA o)) (e ^? biplateRef) 
   92                                ++ lits
   93 
   94 atomRange :: Atom -> SrcSpan
   95 atomRange (NameA n) = getRange n
   96 atomRange (OperatorA n) = getRange n
   97 atomRange (LiteralA n) = getRange n
   98 
   99 wrapAtom :: Build -> Expr
  100 wrapAtom (Right e) = e
  101 wrapAtom (Left (NameA n)) = mkVar n
  102 wrapAtom (Left (OperatorA (NormalOp o))) = mkVar (mkParenName o)
  103 wrapAtom (Left (OperatorA (BacktickOp n))) = mkVar (mkNormalName n)
  104 wrapAtom (Left (LiteralA l)) = mkLit l
  105 
  106 correctParening :: [ClsInst] -> [(GHC.Type, Build)] -> [Build]
  107 correctParening _ [(_,e)] = [e]
  108 correctParening insts ls = concatMap (correctParening insts) (reduceAtoms insts ls)
  109 
  110 reduceAtoms :: [ClsInst] -> [(GHC.Type, Build)] -> [[(GHC.Type, Build)]]
  111 reduceAtoms _ [(t,e)] = [[(t,e)]]
  112 reduceAtoms insts ls = concatMap (reduceBy insts ls) [0 .. length ls - 2]
  113 
  114 reduceBy :: [ClsInst] -> [(GHC.Type, Build)] -> Int -> [[(GHC.Type, Build)]]
  115 reduceBy insts (zip [0..] -> ls) i = maybeToList (reduceFunctionApp ls i) ++ maybeToList (reduceOperatorApp ls i)
  116   where reduceFunctionApp ls i | Just (funT, fun) <- lookup i ls
  117                                , Just (argT, arg) <- lookup (i+1) ls
  118                                , Just (subst, resTyp) <- appTypeMatches insts funT [argT]
  119           = Just $ map ((_1 .- substTy subst) . snd) (take i ls) 
  120                      ++ [(resTyp, mkParen' (mkApp' fun arg))] 
  121                      ++ map ((_1 .- substTy subst) . snd) (drop (i + 2) ls)
  122         reduceFunctionApp _ _ = Nothing
  123         
  124         reduceOperatorApp ls i | Just (opT, Left (OperatorA op)) <- lookup i ls
  125                                , Just (lArgT, lArg) <- lookup (i-1) ls
  126                                , Just (rArgT, rArg) <- lookup (i+1) ls
  127                                , Just (subst, resTyp) <- appTypeMatches insts opT [lArgT, rArgT]
  128           = Just $ map ((_1 .- substTy subst) . snd) (take (i - 1) ls) 
  129                      ++ [(resTyp, mkParen' (mkInfixApp' lArg op rArg))] 
  130                      ++ map ((_1 .- substTy subst) . snd) (drop (i + 2) ls)
  131         reduceOperatorApp _ _ = Nothing
  132 
  133 mkApp' :: Build -> Build -> Build
  134 mkApp' (wrapAtom -> f) (wrapAtom -> a) = Right $ mkApp f a
  135 
  136 mkInfixApp' :: Build -> Operator -> Build -> Build
  137 mkInfixApp' (wrapAtom -> lhs) op (wrapAtom -> rhs) = Right $ mkInfixApp lhs op rhs
  138 
  139 mkParen' :: Build -> Build
  140 mkParen' (wrapAtom -> e) = Right $ mkParen e
