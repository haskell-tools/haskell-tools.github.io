   66 chkOperatorForTypeEq' :: CheckNode Operator
   67 chkOperatorForTypeEq' op
   68   | Just name <- semanticsName (op ^. operatorName)
   69   , name == eqTyConName
   70   = addRelation (TypeFamilies `lOr` GADTs) op
   71   | otherwise = return op

   73 chkNameForTyEqn :: CheckNode Name
   74 chkNameForTyEqn name = do
   75   mty <- runMaybeT . lookupTypeFromId $ name
   76   case mty of
   77     Just ty -> do
   78       let ty'    = GHC.expandTypeSynonyms ty
   79           tycons = universeBi ty' :: [GHC.TyCon]
   80       if any isEqTyCon tycons then addRelation (TypeFamilies `lOr` GADTs) name
   81                               else return name
   82     Nothing -> return name
   83 
   84   --traceShow (showName name ++ " -- " ++ showOutputable ty') $
   85 
   86   where isEqTyCon tc = tc `hasKey` eqTyConKey
   87                     || tc `hasKey` heqTyConKey
   88                     || tc `hasKey` eqPrimTyConKey
   89                     || tc `hasKey` eqReprPrimTyConKey
   90                     || tc `hasKey` eqPhantPrimTyConKey

   92 globalChkNamesForTypeEq' :: CheckNode Module
   93 globalChkNamesForTypeEq' m = do
   94   let origNames   = universeBi (m ^. modDecl) :: [Name]
   95       pairedNames = catMaybes . zipWith zf (map semanticsName origNames) $ origNames
   96       uniqueNames = SMap.elems . SMap.fromList . reverse $ pairedNames
   97   mapM_ chkNameForTyEqn uniqueNames
   98   return m
   99   where zf (Just x) y = Just (x,y)
  100         zf _ _        = Nothing
