   51 addRelation_ :: (MonadState ExtMap m, HasRange node) =>

   53 addRelation_ rel element = modify $ addOccurence' rel element

   59 isTurnedOn :: Extension -> ExtMonad Bool
   60 isTurnedOn ext = do
   61   defaults <- ask
   62   return $! ext `elem` defaults
   63 
   64 conditional :: (node -> ExtMonad node) ->
   65                Extension ->
   66                node ->
   67                ExtMonad node
   68 conditional checker ext = conditionalAny checker [ext]
   69 
   70 conditionalNot :: (node -> ExtMonad node) ->
   71                   Extension ->
   72                   node ->
   73                   ExtMonad node
   74 conditionalNot checker ext node = do
   75   b <-isTurnedOn ext
   76   if b then return node else checker node
   77 
   78 conditionalAny :: (node -> ExtMonad node) ->
   79                    [Extension] ->
   80                    node ->
   81                    ExtMonad node
   82 conditionalAny checker exts node = do
   83   bs <- mapM isTurnedOn exts
   84   if or bs then checker node else return node
   85 
   86 conditionalAdd :: HasRange node => Extension -> node -> ExtMonad node
   87 conditionalAdd ext = conditional (addOccurence ext) ext
