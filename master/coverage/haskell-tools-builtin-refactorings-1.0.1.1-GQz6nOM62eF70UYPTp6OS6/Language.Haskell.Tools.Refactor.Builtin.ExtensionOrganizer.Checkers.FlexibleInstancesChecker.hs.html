   33 -- | We need to check declarations because we want to avoid checking type family instances
   34 chkFlexibleInstancesDecl :: CheckNode Decl

   43 -- | Checks every single type argument in an instance declaration
   44 -- If the class being instantiated could not have been looked up, it keeps FlexibleInstances
   45 chkInstanceHead :: CheckNode InstanceHead
   46 chkInstanceHead ih = do
   47   let types = collectTyArgs ih
   48   mCls <- runMaybeT . lookupClassFromInstance $ ih
   49   case mCls of
   50     Just cls -> mapM_ (chkTypeArg cls) types >> return ih
   51     Nothing  -> addOccurence FlexibleInstances ih

   53 -- | Checks a type argument of class whether it needs FlexibleInstances
   54 -- First checks the structure of the type argument opaquely
   55 -- Then, for type synonyms, it checks whether their GHC representation itself
   56 -- needs the extension.
   57 --
   58 -- Might find false positive occurences for phantom types:
   59 -- > type Phatom a b = [a]
   60 -- > instance C (Phatnom a a)
   61 -- > instance C (Phantom a Int)
   62 chkTypeArg :: GHC.Class -> Type -> ExtMonad Type
   63 chkTypeArg cls ty = do
   64   chkNormalTypeArg ty
   65   maybeTM (return ty) (chkSynonymTypeArg cls) (semanticsTypeSynRhs ty)
   66   where chkSynonymTypeArg :: GHC.Class -> GHC.Type -> ExtMonad Type
   67         chkSynonymTypeArg cls' ty'
   68           | tyArgNeedsFI cls' ty' = addOccurence FlexibleInstances ty
   69           | otherwise             = return ty
   70 
   71 -- | Checks a type argument of class whether it has only (distinct) type variable arguments.
   72 chkNormalTypeArg :: CheckNode Type
   73 chkNormalTypeArg vars = performCheck . refact rmTypeMisc $ vars
   74 
   75   where performCheck vars = do
   76           (isOk, (_, vs)) <- runStateT (runMaybeT (chkAll vars)) ([],[])
   77           case isOk of
   78             Just isOk ->
   79               unless (isOk && length vs == (length . nubBy ((==) `on` (semanticsName . (^. simpleName))) $ vs)) --tyvars are different
   80                 (addOccurence_ FlexibleInstances vars)
   81             Nothing   -> error "chkNormalTypeArg: Couldn't look up something"
   82           return vars
   83 
   84         chkAll x =
   85           ifM (chkTopLevel x) $
   86             chkOnlyApp x
   87 
   88         chkTopLevel x = -- NOTE: this resembles a monadic bind ... (Cont?)
   89           ifM (chkListType x) .
   90             ifM (chkTupleType x) .
   91               ifM (chkUnitTyCon x) $
   92                 return False
   93 
   94         ifM cond f = do b <- cond; if b then (return b) else f
   95 
   96         chkUnitTyCon (VarType x) = do
   97           sname <- tyVarSemNameM x
   98           -- standalone top-level type variables are not accepted
   99           -- NOTE: -XHaskell98 operator type variables??
  100           -- NOTE VarType is either TyCon or TyVar
  101           --      if it is a TyCon, it cannot be wired in (Int, Char, etc)
  102           if | GHC.isTyVarName   sname -> addTyVarM x >> return False
  103              | GHC.isWiredInName sname -> addTyConM x >> return False
  104              | GHC.isTyConName   sname -> addTyConM x >> return True
  105              | otherwise               -> return True -- NEVER
  106         chkUnitTyCon _ = return False
  107 
  108 
  109         chkSingleTyVar (VarType x) = do
  110           sname <- tyVarSemNameM x
  111           if (GHC.isTyVarName sname)
  112             then addTyVarM x >> return True
  113             else addTyConM x >> return False
  114         chkSingleTyVar _ = return False
  115 
  116 
  117         chkTupleType (TupleType args) = do
  118           let xs  = args ^. annListElems
  119           bs <- mapM chkSingleTyVar xs
  120           return $! and bs
  121         chkTupleType _ = return False
  122 
  123         chkListType (ListType v) = chkSingleTyVar v
  124         chkListType _            = return False
  125 
  126         chkOnlyApp :: (MonadState ([Name],[Name]) (m1 m2),
  127                        MonadTrans m1,
  128                        MonadState ExtMap m2) =>
  129                        Type -> MaybeT (m1 m2) Bool
  130         chkOnlyApp (TypeApp f v@(VarType _)) = do
  131           isTyVar <- chkSingleTyVar v
  132           if isTyVar
  133             then case f of
  134               (VarType c) -> addTyConM c >> return True
  135               _           -> chkOnlyApp f
  136             else return False
  137         chkOnlyApp (InfixTypeApp lhs op rhs) = do
  138           addTyConM . mkNormalName $ (op ^. operatorName)
  139           lOK <- chkSingleTyVar lhs
  140           rOK <- chkSingleTyVar rhs
  141           return $! lOK && rOK
  142         chkOnlyApp _ = return False
  143 
  144         addTyCon  n (ctors, vars) = (n:ctors, vars)
  145         addTyVar  n (ctors, vars) = (ctors, n:vars)
  146         addTyConM n               = modify $ addTyCon n
  147         addTyVarM n               = modify $ addTyVar  n
  148 
  149         tyVarSemNameM x = MaybeT . return . semanticsName $ x ^. simpleName
  150 
  151 rmTypeMisc :: Type -> Type
  152 rmTypeMisc (KindedType t _) = t
  153 rmTypeMisc (ParenType x)    = x
  154 rmTypeMisc x                = x

  156 -- Decides whether a type argument of a type class constructor need FlexibleInstances
  157 tyArgNeedsFI :: GHC.Class -> GHC.Type -> Bool
  158 tyArgNeedsFI cls arg = not . hasOnlyDistinctTyVars $ tyArg
  159   where [tyArg] = GHC.filterOutInvisibleTypes (GHC.classTyCon cls) [arg]
  160 
  161 -- | Checks whether a GHC.Type is an application, and has only (distinct) type variable arguments
  162 -- Logic from TcValidity.tcInstHeadTyAppAllTyVars
  163 hasOnlyDistinctTyVars :: GHC.Type -> Bool
  164 hasOnlyDistinctTyVars ty
  165   | Just (tc, tys) <- GHC.tcSplitTyConApp_maybe (dropCasts ty)
  166   , tys'           <- GHC.filterOutInvisibleTypes tc tys
  167   , tyVars         <- mapMaybe GHC.tcGetTyVar_maybe tys'
  168   = tyVars `equalLength` tys' && hasNoDups tyVars
  169   | otherwise = False

  172 -- | A local helper function from TcValidity
  173 dropCasts :: GHC.Type -> GHC.Type
  174 dropCasts (GHC.CastTy ty _)     = dropCasts ty
  175 dropCasts (GHC.AppTy t1 t2)     = GHC.mkAppTy (dropCasts t1) (dropCasts t2)
  176 dropCasts (GHC.FunTy t1 t2)     = GHC.mkFunTy (dropCasts t1) (dropCasts t2)
  177 dropCasts (GHC.TyConApp tc tys) = GHC.mkTyConApp tc (map dropCasts tys)
  178 dropCasts (GHC.ForAllTy b ty)   = GHC.ForAllTy (dropCastsB b) (dropCasts ty)
  179 dropCasts ty                    = ty
  180 
  181 -- | A local helper function from TcValidity
  182 dropCastsB :: GHC.TyVarBinder -> GHC.TyVarBinder
  183 dropCastsB b = b
