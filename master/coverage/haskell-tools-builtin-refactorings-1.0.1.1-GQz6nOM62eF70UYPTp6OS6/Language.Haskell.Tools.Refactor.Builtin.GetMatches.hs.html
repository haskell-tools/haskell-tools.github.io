   19 getMatches :: RealSrcSpan -> ModuleDom -> [ModuleDom] -> QueryMonad Value
   20 getMatches sp (_,mod) _
   21   = case selectedName of [n] -> do ctors <- getCtors $ idType $ semanticsId n
   22                                    return $ toJSON ctors
   23                          []  -> queryError "No name is selected."
   24                          _   -> queryError "Multiple names are selected."
   25   where
   26     selectedName :: [QualifiedName]
   27     selectedName = mod ^? nodesContaining sp
   28 
   29 getCtors :: GHC.Type -> QueryMonad [(String, [String])]
   30 -- | TODO: unpack forall, context types
   31 -- | TODO: care for infix constructors
   32 getCtors t | Just (tc, _) <- splitTyConApp_maybe t
   33   = maybe (queryError (noSuccessMsg t)) (return . map formatCtor) (tyConDataCons_maybe tc)
   34 getCtors t = queryError (noSuccessMsg t)
   35 
   36 noSuccessMsg :: GHC.Type -> String
   37 noSuccessMsg t = "Cannot find the constructors of type " ++ showSDocUnsafe (ppr t)
   38 
   39 formatCtor :: DataCon -> (String, [String])
   40 formatCtor dc = (showSDocUnsafe $ ppr $ dataConName dc, createArgNames (dataConOrigArgTys dc))

   42 -- | TODO: Check for names in scope
