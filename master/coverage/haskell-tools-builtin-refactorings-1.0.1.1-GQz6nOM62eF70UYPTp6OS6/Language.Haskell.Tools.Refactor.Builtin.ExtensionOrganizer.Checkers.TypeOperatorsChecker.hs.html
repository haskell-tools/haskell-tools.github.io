   46 chkTypeOperatorsDecl' :: CheckNode Decl
   47 chkTypeOperatorsDecl' d = do
   48   let dhs = universeBi d :: [DeclHead]
   49   anyNeedsTO <- liftM or $ mapM isOperatorM dhs
   50   if anyNeedsTO then addOccurence TypeOperators d
   51                 else return d

   53 -- OccName: [Type and class operator definitions]
   54   -- We are not just looking for a *syntactically-infix* declaration,
   55   -- but one that uses an operator OccName.
   56 isOperatorM :: DeclHead -> ExtMonad Bool
   57 isOperatorM dh = do
   58   mSemName <- runMaybeT . declHeadSemName $ dh
   59   case mSemName of
   60     Just semName
   61       | occ <- GHC.nameOccName semName
   62       , GHC.isTcOcc occ && GHC.isSymOcc occ
   63       -> return True
   64       | otherwise -> return False
   65     Nothing -> return True
