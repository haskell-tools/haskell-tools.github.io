   16 floatOutRefactoring :: RefactoringChoice
   17 floatOutRefactoring = SelectionRefactoring "FloatOut" (localRefactoring . floatOut)
   18 
   19 floatOut :: RealSrcSpan -> LocalRefactoring
   20 floatOut sp mod
   21   = do (mod', st) <- runStateT (nodesContaining sp !~ extractAndInsert sp $ mod) NotEncountered
   22        case st of NotEncountered -> refactError "No definition is selected. The selection range must be inside the definition."
   23                   Extracted bnds -> -- insert it to the global definition list
   24                                     return $ modDecl & annListElems .- (++ map toTopLevel bnds) $ removeEmpties mod'
   25                   Inserted -> -- already inserted to a local scope
   26                                return (removeEmpties mod')
   27   where toTopLevel :: LocalBind -> Decl
   28         toTopLevel (LocalValBind vb) = mkValueBinding vb
   29         toTopLevel (LocalTypeSig sg) = mkTypeSigDecl sg
   30         toTopLevel (LocalFixity fx) = mkFixityDecl fx
   31 
   32         removeEmpties = removeEmptyBnds (nodesContaining sp) (nodesContaining sp)
   33 
   34 data FloatState = NotEncountered | Extracted [LocalBind] | Inserted
   35 
   36 extractAndInsert :: RealSrcSpan -> LocalBindList -> StateT FloatState LocalRefactor LocalBindList
   37 extractAndInsert sp locs
   38   | hasSharedSig = refactError "Cannot float out a definition, since it has a signature shared with other bindings that stay in the scope."
   39   | not (null nameConflicts) = refactError $ "Cannot float out a definition, since it would cause a name conflicts in the target scope: "
   40                                                 ++ concat (intersperse ", " nameConflicts)
   41   | not (null implicitConflicts) = refactError $ "Cannot float out a definition, since it uses the implicit parameters: "
   42                                                    ++ concat (intersperse ", " implicitConflicts)
   43   | otherwise = get >>= \case NotEncountered -> put (Extracted floated) >> return filteredLocs
   44                               Extracted binds -> put Inserted >> (return $ annListElems .- (++ binds) $ locs)
   45                               Inserted -> return locs
   46   where selected = locs ^? annList & filtered (isInside sp)
   47         floated = normalizeElements $ selected ++ (locs ^? annList & filtered (nameIsSelected . map semanticsName . (^? elementName)))
   48           where nameIsSelected [n] = n `elem` concatMap (map semanticsName . (^? elementName)) selected
   49                 nameIsSelected _ = False
   50 
   51         filteredLocs = filterList (\e -> not (getRange e `elem` floatedElemRanges)) locs
   52           where floatedElemRanges = map getRange floated
   53         hasSharedSig = any (\e -> not $ null (map semanticsName (filteredLocs ^? annList & elementName) `intersect` map semanticsName (e ^? elementName))) selected
   54         conflicts = map checkConflict selected
   55 
   56         nameConflicts = concat $ map fst conflicts
   57         implicitConflicts = concat $ map snd conflicts
   58 
   59 checkConflict :: LocalBind -> ([String], [String])
   60 checkConflict bnd = (concatMap @[] getConflict bndNames, implicits)
   61   where bndNames = bnd ^? elementName
   62         getConflict bndName = filter ((== nameStr) . Just) $ map (occNameString . getOccName) outerScope
   63           where outerScope = map (^. _1) $ concat $ take 1 $ drop 2 $ semanticsScope bndName
   64                 nameStr = fmap (occNameString . getOccName) $ semanticsName bndName
   65         implicits = map (occNameString . getOccName)
   66                         (concatMap getPossibleImplicits bndNames `intersect` getQNames (bnd ^? biplateRef))
   67 
   68         getQNames :: [QualifiedName] -> [GHC.Name]
   69         getQNames = catMaybes . map semanticsName
   70 
   71         getPossibleImplicits :: QualifiedName -> [GHC.Name]
   72         getPossibleImplicits qn = concat (map (map (^. _1)) $ take 2 $ semanticsScope qn) \\ catMaybes (map semanticsName bndNames)
