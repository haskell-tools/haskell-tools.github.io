   22 trfKindSig :: TransformName n r => Maybe (LHsKind n) -> Trf (AnnMaybeG AST.UKindConstraint (Dom r) RangeStage)
   23 trfKindSig = trfMaybe "" "" trfKindSig'
   24 
   25 trfKindSig' :: TransformName n r => Located (HsKind n) -> Trf (Ann AST.UKindConstraint (Dom r) RangeStage)
   26 trfKindSig' k = annLocNoSema (combineSrcSpans (getLoc k) <$> (tokenBefore (srcSpanStart (getLoc k)) AnnDcolon))
   27                              (AST.UKindConstraint <$> trfLocNoSema trfKind' k)

   29 trfKind :: TransformName n r => Located (HsKind n) -> Trf (Ann AST.UKind (Dom r) RangeStage)
   30 trfKind = trfLocNoSema (trfKind' . cleanHsType)
   31 
   32 trfKind' ::TransformName n r => HsKind n -> Trf (AST.UKind (Dom r) RangeStage)
   33 trfKind' = trfKind'' . cleanHsType where
   34   trfKind'' (HsTyVar _ (rdrName . unLoc -> Exact n))
   35     | isWiredInName n && occNameString (nameOccName n) == "*"
   36     = pure AST.UStarKind
   37     | isWiredInName n && occNameString (nameOccName n) == "#"
   38     = pure AST.UUnboxKind
   39   trfKind'' (HsParTy kind) = AST.UParenKind <$> trfKind kind
   40   trfKind'' (HsFunTy k1 k2) = AST.UFunKind <$> trfKind k1 <*> trfKind k2
   41   trfKind'' (HsAppTy k1 k2) = AST.UAppKind <$> trfKind k1 <*> trfKind k2
   42   trfKind'' (HsOpTy k1 op k2) = AST.UInfixAppKind <$> trfKind k1 <*> trfOperator op <*> trfKind k2
   43   trfKind'' (HsTyVar _ kv) = transformingPossibleVar kv (AST.UVarKind <$> trfName kv)
   44   trfKind'' (HsListTy kind) = AST.UListKind <$> trfKind kind
   45   trfKind'' (HsTupleTy _ kinds) = AST.UTupleKind <$> makeList ", " atTheStart (mapM trfKind kinds)
   46   trfKind'' (HsAppsTy [unLoc -> HsAppPrefix t]) = trfKind' (unLoc t)
   47   trfKind'' (HsAppsTy [unLoc -> HsAppInfix n]) = AST.UVarKind <$> trfName n
   48   trfKind'' pt@(HsExplicitListTy {}) = AST.UPromotedKind <$> annContNoSema (trfPromoted' trfKind' pt)
   49   trfKind'' pt@(HsExplicitTupleTy {}) = AST.UPromotedKind <$> annContNoSema (trfPromoted' trfKind' pt)
   50   trfKind'' pt@(HsTyLit {}) = AST.UPromotedKind <$> annContNoSema (trfPromoted' trfKind' pt)
   51   trfKind'' t = AST.UTypeKind <$> annContNoSema (trfType' t)
   52 
   53 trfPromoted' :: (TransformName n r, HasNoSemanticInfo (Dom r) a)
   54                   => (HsType n -> Trf (a (Dom r) RangeStage)) -> HsType n -> Trf (AST.UPromoted a (Dom r) RangeStage)
   55 trfPromoted' _ (HsTyLit (HsNumTy _ int)) = pure $ AST.UPromotedInt int
   56 trfPromoted' _ (HsTyLit (HsStrTy _ str)) = pure $ AST.UPromotedString (unpackFS str)
   57 trfPromoted' _ (HsTyVar _ name) = AST.UPromotedCon <$> trfName name
   58 trfPromoted' f (HsExplicitListTy _ _ elems) = AST.UPromotedList <$> between AnnOpenS AnnCloseS (trfAnnList ", " f elems)
   59 trfPromoted' f (HsExplicitTupleTy _ elems) = AST.UPromotedTuple <$> between AnnOpenP AnnCloseP (trfAnnList ", " f elems)
   60 trfPromoted' _ t = unhandledElement "promoted type/kind" t
