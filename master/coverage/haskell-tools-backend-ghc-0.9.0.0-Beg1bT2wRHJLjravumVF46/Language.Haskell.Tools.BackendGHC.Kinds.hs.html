   23 trfKindSig :: TransformName n r => Maybe (LHsKind n) -> Trf (AnnMaybeG AST.UKindConstraint (Dom r) RangeStage)
   24 trfKindSig = trfMaybe "" "" trfKindSig'
   25 
   26 trfKindSig' :: TransformName n r => Located (HsKind n) -> Trf (Ann AST.UKindConstraint (Dom r) RangeStage)
   27 trfKindSig' k = annLocNoSema (combineSrcSpans (getLoc k) <$> (tokenBefore (srcSpanStart (getLoc k)) AnnDcolon))
   28                              (AST.UKindConstraint <$> trfLocNoSema trfKind' k)

   30 trfKind :: TransformName n r => Located (HsKind n) -> Trf (Ann AST.UKind (Dom r) RangeStage)
   31 trfKind = trfLocNoSema (trfKind' . cleanHsType)
   32 
   33 trfKind' ::TransformName n r => HsKind n -> Trf (AST.UKind (Dom r) RangeStage)
   34 trfKind' = trfKind'' . cleanHsType where
   35   trfKind'' (HsTyVar _ (rdrName . unLoc -> Exact n))
   36     | isWiredInName n && occNameString (nameOccName n) == "*"
   37     = pure AST.UStarKind
   38     | isWiredInName n && occNameString (nameOccName n) == "#"
   39     = pure AST.UUnboxKind
   40   trfKind'' (HsParTy kind) = AST.UParenKind <$> trfKind kind
   41   trfKind'' (HsFunTy k1 k2) = AST.UFunKind <$> trfKind k1 <*> trfKind k2
   42   trfKind'' (HsAppTy k1 k2) = AST.UAppKind <$> trfKind k1 <*> trfKind k2
   43   trfKind'' (HsOpTy k1 op k2) = AST.UInfixAppKind <$> trfKind k1 <*> trfOperator op <*> trfKind k2
   44   trfKind'' (HsTyVar _ kv) = transformingPossibleVar kv (AST.UVarKind <$> trfName kv)
   45   trfKind'' (HsListTy kind) = AST.UListKind <$> trfKind kind
   46   trfKind'' (HsTupleTy _ kinds) = AST.UTupleKind <$> makeList ", " atTheStart (mapM trfKind kinds)
   47   trfKind'' (HsAppsTy [unLoc -> HsAppPrefix t]) = trfKind' (unLoc t)
   48   trfKind'' (HsAppsTy [unLoc -> HsAppInfix n]) = AST.UVarKind <$> trfName n
   49   trfKind'' pt@(HsExplicitListTy {}) = AST.UPromotedKind <$> annContNoSema (trfPromoted' trfKind' pt)
   50   trfKind'' pt@(HsExplicitTupleTy {}) = AST.UPromotedKind <$> annContNoSema (trfPromoted' trfKind' pt)
   51   trfKind'' pt@(HsTyLit {}) = AST.UPromotedKind <$> annContNoSema (trfPromoted' trfKind' pt)
   52   trfKind'' t = AST.UTypeKind <$> annContNoSema (trfType' t)
   53 
   54 trfPromoted' :: (TransformName n r, HasNoSemanticInfo (Dom r) a)
   55                   => (HsType n -> Trf (a (Dom r) RangeStage)) -> HsType n -> Trf (AST.UPromoted a (Dom r) RangeStage)
   56 trfPromoted' _ (HsTyLit (HsNumTy _ int)) = pure $ AST.UPromotedInt int
   57 trfPromoted' _ (HsTyLit (HsStrTy _ str)) = pure $ AST.UPromotedString (unpackFS str)
   58 trfPromoted' _ (HsTyVar _ name) = AST.UPromotedCon <$> trfName name
   59 trfPromoted' f (HsExplicitListTy _ _ elems) = AST.UPromotedList <$> between AnnOpenS AnnCloseS (trfAnnList ", " f elems)
   60 trfPromoted' f (HsExplicitTupleTy _ elems) = AST.UPromotedTuple <$> between AnnOpenP AnnCloseP (trfAnnList ", " f elems)
   61 trfPromoted' _ t = unhandledElement "promoted type/kind" t
