   14 trfLiteral' :: HsLit -> Trf (AST.ULiteral (Dom r) RangeStage)
   15 trfLiteral' (HsChar _ ch) = pure $ AST.UCharLit ch
   16 trfLiteral' (HsCharPrim _ ch) = pure $ AST.UPrimCharLit ch
   17 trfLiteral' (HsString _ str) = pure $ AST.UStringLit (unpackFS str)
   18 trfLiteral' (HsStringPrim _ str) = pure $ AST.UPrimStringLit (BS.foldr (:) "" str)
   19 trfLiteral' (HsInt _ i) = pure $ AST.UIntLit i
   20 trfLiteral' (HsIntPrim _ i) = pure $ AST.UPrimIntLit i
   21 trfLiteral' (HsWordPrim _ i) = pure $ AST.UPrimWordLit i
   22 trfLiteral' (HsInt64Prim _ i) = pure $ AST.UPrimIntLit i
   23 trfLiteral' (HsWord64Prim _ i) = pure $ AST.UPrimWordLit i
   24 trfLiteral' (HsInteger _ i _) = pure $ AST.UPrimIntLit i
   25 trfLiteral' (HsRat frac _) = pure $ AST.UFracLit (fl_value frac)
   26 trfLiteral' (HsFloatPrim frac) = pure $ AST.UPrimFloatLit (fl_value frac)
   27 trfLiteral' (HsDoublePrim frac) = pure $ AST.UPrimDoubleLit (fl_value frac)
   28 
   29 trfOverloadedLit :: OverLitVal -> Trf (AST.ULiteral (Dom r) RangeStage)
   30 trfOverloadedLit (HsIntegral _ i) = pure $ AST.UIntLit i
   31 trfOverloadedLit (HsFractional frac) = pure $ AST.UFracLit (fl_value frac)
   32 trfOverloadedLit (HsIsString _ str) = pure $ AST.UStringLit (unpackFS str)
