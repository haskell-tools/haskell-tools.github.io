   17 chkSynonym :: CheckNode Type
   18 chkSynonym t = do
   19   mtycon <- runMaybeT . lookupType $ t
   20   case mtycon of
   21     Nothing    -> return t
   22     Just tycon -> chkSynonym' tycon
   23   where chkSynonym' x = case lookupSynDef x of
   24                           Nothing -> return t
   25                           Just _  -> addOccurence TypeSynonymInstances t
   26 
   27 lookupSynDefM :: Type -> MaybeT ExtMonad GHC.TyCon
   28 lookupSynDefM t = do
   29   tything <- lookupType t
   30   liftMaybe $ lookupSynDef tything
   31   where liftMaybe = MaybeT . return
   32 
   33 -- NOTE: Returns Nothing if it is not a type synonym
   34 --       (or has some weird structure I didn't think of)
   35 lookupSynDef :: GHC.TyThing -> Maybe GHC.TyCon
   36 lookupSynDef syn = do
   37   tycon <- tyconFromTyThing syn
   38   rhs   <- GHC.synTyConRhs_maybe tycon
   39   tyconFromGHCType rhs
   40 
   41 tyconFromTyThing :: GHC.TyThing -> Maybe GHC.TyCon
   42 tyconFromTyThing (GHC.ATyCon tycon) = Just tycon
   43 tyconFromTyThing _ = Nothing
   44 
   45 -- won't bother
   46 tyconFromGHCType :: GHC.Type -> Maybe GHC.TyCon
   47 tyconFromGHCType (GHC.AppTy t1 _) = tyconFromGHCType t1
   48 tyconFromGHCType (GHC.TyConApp tycon _) = Just tycon
   49 tyconFromGHCType _ = Nothing

   52 -- NOTE: Return false if the type is certainly not a newtype
   53 --       Returns true if it is a newtype or it could not have been looked up
   54 isNewtype :: Type -> ExtMonad Bool
   55 isNewtype t = do
   56   tycon <- runMaybeT . lookupType $ t
   57   return $! maybe True isNewtypeTyCon tycon

   61 lookupType :: Type -> MaybeT ExtMonad GHC.TyThing
   62 lookupType t = do
   63   name  <- liftMaybe . nameFromType $ t
   64   sname <- liftMaybe . getSemName   $ name
   65   MaybeT . GHC.lookupName $ sname
   66     where liftMaybe = MaybeT . return
   67 
   68 -- NOTE: gives just name if the type being scrutinised can be newtype
   69 --       else it gives nothing
   70 nameFromType :: Type -> Maybe Name
   71 nameFromType (TypeApp f _)    = nameFromType f
   72 nameFromType (ParenType x)    = nameFromType x
   73 nameFromType (KindedType t _) = nameFromType t
   74 nameFromType (VarType x)      = Just x
   75 nameFromType _                = Nothing
   76 
   77 isNewtypeTyCon :: GHC.TyThing -> Bool
   78 isNewtypeTyCon (GHC.ATyCon tycon) = GHC.isNewTyCon tycon
   79 isNewtypeTyCon _ = False
   80 
   81 getSemName :: Name -> Maybe GHC.Name
   82 getSemName x = semanticsName (x ^. simpleName)
