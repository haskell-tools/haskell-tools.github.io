   51 organizeImports :: LocalRefactoring
   52 organizeImports mod
   53   = do usedTyThings <- catMaybes <$> mapM lookupName usedNames
   54        let dfs = semanticsDynFlags mod
   55            noNarrowingImports
   56              = xopt TemplateHaskell dfs -- no narrowing if TH is present (we don't know what will be used)
   57                 || xopt QuasiQuotes dfs -- no narrowing if TH quotes are present (we don't know what will be used)
   58                 || (xopt FlexibleInstances dfs && noNarrowingSubspecs) -- arbitrary ctors might be needed when using imported data families
   59                 || hasCoerce -- the presence of coerce implicitely requires the constructors to be present
   60            noNarrowingSubspecs
   61              = -- both standalone deriving and FFI marshalling can cause constructors to be used in the generated code
   62                xopt GHC.StandaloneDeriving dfs || hasMarshalling
   63                    -- while pattern synonyms are not handled correctly, we disable subspec narrowing to be safe
   64                 || patternSynonymAreUsed usedTyThings
   65        if noNarrowingImports
   66          then -- we don't know what definitions the generated code will use
   67               return $ modImports .- sortImports $ mod
   68          else do (prelInstances, prelFamInsts) <- liftGhc $ getInstances preludeAccessible
   69                  modImports !~ fmap sortImports . narrowImports noNarrowingSubspecs exportedModules (addFromString dfs usedNames) exportedNames prelInstances prelFamInsts $ mod
   70   where preludeAccessible = semanticsPrelTransMods mod
   71         addFromString dfs = if xopt OverloadedStrings dfs then (GHC.fromStringName :) else id
   72         usedNames = map getName $ (catMaybes $ map semanticsName
   73                         -- obviously we don't want the names in the imports to be considered, but both from
   74                         -- the declarations (used), both from the module head (re-exported) will count as usage
   75                       (universeBi (mod ^. modHead) ++ universeBi (mod ^. modDecl) :: [QualifiedName]))
   76                         ++ concatMap (map fst . semanticsImplicitFlds) (universeBi (mod ^. modDecl) :: [FieldWildcard])
   77         -- Prelude is not actually exported, but we don't want to remove it if it is explicitly there
   78         -- otherwise, we might add new imported elements that cause conflicts.
   79         exportedModules = "Prelude" : (mod ^? modHead & annJust & mhExports & annJust
   80                                                 & espExports & annList & exportModuleName & moduleNameString)
   81 
   82         -- Exported definitions must be kept imported
   83         exports = mod ^? modHead & annJust & mhExports & annJust & espExports & annList & exportDecl
   84         exportedNames = catMaybes $ map getExported exports
   85         getExported e = fmap (,hasChild) name
   86           where name = semanticsName (e ^. ieName & simpleName)
   87                 hasChild = (case e ^? ieSubspec & annJust of Just SubAll -> True; _ -> False)
   88                              || not (null @[] (e ^? ieSubspec & annJust & essList & annList))
   89 
   90         -- Checks if the module uses foreign import/export that requires marshalling. In this case no
   91         -- subspecifiers could be narrowed because constructors might be needed.
   92         hasMarshalling = not $ null @[] (mod ^? modDecl & annList & declForeignType)
   93         hasCoerce = GHC.coerceKey `elem` map getUnique usedNames
   94 
   95         -- Can this name be part of the source code?
   96         -- isSourceName n = ':' `notElem` occNameString (occName n)
   97         patternSynonymAreUsed tts = any (\case AConLike (PatSynCon _) -> True; _ -> False) tts

   99 -- | Sorts the imports in alphabetical order
  100 sortImports :: ImportDeclList -> ImportDeclList
  101 sortImports ls = srcInfo & srcTmpSeparators .= filter (not . null . fst) (concatMap (\(sep,elems) -> sep : map fst elems) reordered)
  102                    $ annListElems .= concatMap (map snd . snd) reordered
  103                    $ ls
  104   where reordered :: [(([SourceTemplateTextElem], SrcSpan), [(([SourceTemplateTextElem], SrcSpan), ImportDecl)])]
  105         reordered = map (_2 .- sortBy (compare `on` (^. _2 & importModule & AST.moduleNameString))) parts
  106 
  107         parts = map (_2 .- reverse) $ reverse $ breakApart [] imports
  108 
  109         -- break up the list of imports to import groups
  110         breakApart :: [(([SourceTemplateTextElem], SrcSpan), [(([SourceTemplateTextElem], SrcSpan), ImportDecl)])]
  111                         -> [(([SourceTemplateTextElem], SrcSpan), ImportDecl)]
  112                         -> [(([SourceTemplateTextElem], SrcSpan), [(([SourceTemplateTextElem], SrcSpan), ImportDecl)])]
  113         breakApart res [] = res
  114         breakApart res ((sep, e) : rest) | length (filter ('\n' ==) (sep ^? _1 & traversal & sourceTemplateText & traversal)) > 1
  115                                             || "\n#" `isInfixOf` (sep ^? _1 & traversal & sourceTemplateText & traversal)
  116           = breakApart ((sep, [(([], noSrcSpan),e)]) : res) rest
  117         breakApart ((lastSep, lastRes) : res) (elem : rest)
  118           = breakApart ((lastSep, elem : lastRes) : res) rest
  119         breakApart [] ((sep, e) : rest)
  120           = breakApart [(sep, [(([], noSrcSpan),e)])] rest
  121 
  122         imports = zipWithSeparators ls

  124 -- | Modify an import to only import  names that are used.
  125 narrowImports :: Bool -> [String] -> [GHC.Name] -> [(GHC.Name, Bool)] -> [ClsInst] -> [FamInst] -> ImportDeclList -> LocalRefactor ImportDeclList
  126 narrowImports noNarrowSubspecs exportedModules usedNames exportedNames prelInsts prelFamInsts imps
  127   = do impsNeeded <- liftGhc $ neededImports exportedModules (usedNames ++ map fst exportedNames) prelInsts prelFamInsts (imps ^. annListElems)
  128        (annListElems & traversal !~ narrowImport noNarrowSubspecs exportedModules usedNames exportedNames)
  129           =<< filterListIndexedSt (\i _ -> impsNeeded !! i) imps
  130 
  131 -- | Reduces the number of definitions used from an import
  132 narrowImport :: Bool -> [String] -> [GHC.Name] -> [(GHC.Name, Bool)] -> ImportDecl -> LocalRefactor ImportDecl
  133 narrowImport noNarrowSubspecs exportedModules usedNames exportedNames imp
  134   | (imp ^. importModule & moduleNameString) `elem` exportedModules
  135       || maybe False (`elem` exportedModules) (imp ^? importAs & annJust & importRename & moduleNameString)
  136   = return imp -- dont change an import if it is exported as-is (module export)
  137   | importIsExact imp
  138   = importSpec&annJust&importSpecList !~ narrowImportSpecs noNarrowSubspecs usedNames exportedNames $ imp
  139   | importIsHiding imp
  140   = return imp -- a hiding import is not changed, because the wildcard importing of class and datatype
  141                -- members could bring into scope the exact definition that was hidden
  142   | otherwise
  143   = do namedThings <- mapM lookupName actuallyImported
  144        let -- to explicitly import pattern synonyms or type operators we need to enable an extension, and the user might not expect this
  145            hasRiskyDef = any isRiskyDef namedThings
  146            groups = groupThings noNarrowSubspecs (semanticsImported imp)
  147                       (filter ((`elem` semanticsImported imp) . fst) exportedNames) (catMaybes namedThings)
  148        return $ if not hasRiskyDef && length groups < 4
  149          then importSpec .- replaceWithJust (createImportSpec groups) $ imp
  150          else imp
  151   where actuallyImported = semanticsImported imp `intersect` usedNames
  152         isRiskyDef (Just (AConLike (PatSynCon _))) = True
  153         isRiskyDef (Just (ATyCon tc)) = isSymOcc (occName (tyConName tc))
  154         isRiskyDef _ = False

  156 -- | Group things as importable definitions. The second member of the pair will be true, when there is a sub-name
  157 -- that should be imported apart from the name of the importable definition.
  158 groupThings :: Bool -> [GHC.Name] -> [(GHC.Name, Bool)] -> [TyThing] -> [(GHC.Name, Bool)]
  159 groupThings noNarrowSubspecs importable exported
  160   = map last . groupBy ((==) `on` fst) . sort . (exported ++) . map createImportFromTyThing
  161   where createImportFromTyThing :: TyThing -> (GHC.Name, Bool)
  162         createImportFromTyThing tt | Just (td, isDataType) <- getTopDef tt
  163           = if (td `elem` importable || isDataType) then (td, True)
  164                                                     else (getName tt, False)
  165         createImportFromTyThing tt@(ATyCon {}) = (getName tt, noNarrowSubspecs)
  166         createImportFromTyThing tt = (getName tt, False)
  167 
  168 -- | Gets the importable definition for a (looked up) name. The bool flag tells if it is from
  169 -- a data type.
  170 getTopDef :: TyThing -> Maybe (GHC.Name, Bool)
  171 getTopDef (AnId id) | isRecordSelector id
  172   = case recordSelectorTyCon id of RecSelData tc -> Just (getName tc, True)
  173                                    RecSelPatSyn ps -> Just (getName ps, False)
  174 getTopDef (AnId id)
  175   | Just n <- fmap (getName . dataConTyCon) (isDataConWorkId_maybe id <|> isDataConId_maybe id)
  176   = Just (n, True)
  177 getTopDef (AnId id) = fmap ((,False) . getName) (isClassOpId_maybe id)
  178 getTopDef (AConLike (RealDataCon dc))
  179   = case tyConFamInst_maybe (dataConTyCon dc) of
  180       Just (dataFam, _) -> Just (getName dataFam, True)
  181       _                 -> Just (getName $ dataConTyCon dc, True)
  182 getTopDef (AConLike (PatSynCon _)) = error "getTopDef: should not be called with pattern synonyms"
  183 getTopDef (ATyCon _) = Nothing

  185 createImportSpec :: [(GHC.Name, Bool)] -> ImportSpec
  186 createImportSpec elems = mkImportSpecList $ map createIESpec elems
  187   where createIESpec (n, False) = mkIESpec (mkUnqualName' (GHC.getName n)) Nothing
  188         createIESpec (n, True)  = mkIESpec (mkUnqualName' (GHC.getName n)) (Just mkSubAll)

  190 -- | Check each import if it is actually needed
  191 neededImports :: [String] -> [GHC.Name] -> [ClsInst] -> [FamInst] -> [ImportDecl] -> GHC.Ghc [Bool]
  192 neededImports exportedModules usedNames prelInsts prelFamInsts imps = do
  193      impsWithInsts <- mapM (\i -> (i,) <$> getInstances (semanticsTransMods i)) imps  
  194      return $ neededImports' usedNames [] prelInsts prelFamInsts impsWithInsts
  195   where neededImports' _ _ _ _ [] = []
  196         -- keep the import if any definition is needed from it
  197         neededImports' usedNames kept keptInsts keptFamInsts ((imp, (clsInsts, famInsts)) : rest)
  198           | not (null actuallyImported)
  199                || (imp ^. importModule & moduleNameString) `elem` exportedModules
  200                || maybe False (`elem` exportedModules) (imp ^? importAs & annJust & importRename & moduleNameString)
  201             = True : neededImports' usedNames (imp : kept) (clsInsts ++ keptInsts) (famInsts ++ keptFamInsts) rest
  202           where actuallyImported = semanticsImported imp `intersect` usedNames
  203         -- check if instances are needed from the import
  204         neededImports' usedNames kept keptInsts keptFamInsts ((imp, (clsInsts, famInsts)) : rest)
  205             = needed : if needed then neededImports' usedNames (imp : kept) (clsInsts ++ keptInsts) (famInsts ++ keptFamInsts) rest
  206                                  else neededImports' usedNames kept keptInsts keptFamInsts rest
  207           where needed = any (`notElem` map is_dfun keptInsts) (map is_dfun $ filter (isOrphan . is_orphan) clsInsts)
  208                            || any (`notElem` map fi_axiom keptFamInsts) (map fi_axiom famInsts)

  210 -- | Narrows the import specification (explicitly imported elements)
  211 narrowImportSpecs :: Bool -> [GHC.Name] -> [(GHC.Name, Bool)] -> IESpecList -> LocalRefactor IESpecList
  212 narrowImportSpecs noNarrowSubspecs usedNames exportedNames
  213   = (if noNarrowSubspecs then return else annList !~ narrowImportSubspecs neededNames exportedNames)
  214        >=> filterListSt isNeededSpec
  215   where neededNames = usedNames ++ map fst exportedNames
  216 
  217         isNeededSpec :: IESpec -> Bool
  218         isNeededSpec ie =
  219           (semanticsName (ie ^. ieName&simpleName)) `elem` map Just neededNames
  220           -- if the name is not used, but some of its constructors are used, it is needed
  221             || ((ie ^? ieSubspec&annJust&essList&annList) /= [])
  222             -- its a bit hard to decide if there is an element inside (for example bundled pattern synonyms)
  223             || (case ie ^? ieSubspec&annJust of Just SubAll -> True; _ -> False)
  224 
  225 -- | Reduces the number of definitions imported from a sub-specifier.
  226 narrowImportSubspecs :: [GHC.Name] -> [(GHC.Name, Bool)] -> IESpec -> LocalRefactor IESpec
  227 narrowImportSubspecs neededNames exportedNames ss | noNarrowingForThis = return ss
  228   | otherwise
  229   = ieSubspec & annJust & essList !~ filterListSt (\n -> (semanticsName (n ^. simpleName)) `elem` map Just neededNames) $ ss
  230   where noNarrowingForThis = case semanticsName (ss ^. ieName&simpleName) of
  231                                Just name -> lookup name exportedNames == Just True
  232                                _ -> False
