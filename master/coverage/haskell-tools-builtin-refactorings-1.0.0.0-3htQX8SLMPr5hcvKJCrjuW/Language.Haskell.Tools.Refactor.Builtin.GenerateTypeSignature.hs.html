   32 -- | Perform the refactoring on either local or top-level definition
   33 generateTypeSignature :: Simple Traversal Module DeclList
   34                                 -- ^ Access for a top-level definition if it is the selected definition
   35                            -> Simple Traversal Module LocalBindList
   36                                 -- ^ Access for a definition list if it contains the selected definition
   37                            -> (forall d . (BindingElem d) => AnnList d -> Maybe ValueBind)
   38                                 -- ^ Selector for either local or top-level declaration in the definition list
   39                            -> LocalRefactoring
   40 generateTypeSignature topLevelRef localRef vbAccess mod
   41   = let typeSigs = universeBi mod
   42         bindings = universeBi mod
   43         findTypeSigFor id = find (\ts -> any (id ==) $ map semanticsId (ts ^? tsName & annList & simpleName))
   44         bindsWithSigs = catMaybes $ concatMap (\b -> map (\n -> let id = semanticsId n in fmap (id,,b) (findTypeSigFor id typeSigs)) (b ^? bindingName)) bindings
   45         scopedSigs = hasScopedTypeSignatures mod
   46      in do (mod', done) <- flip runStateT False .
   47                              (topLevelRef !~ genTypeSig scopedSigs bindsWithSigs vbAccess
   48                                 <=< localRef !~ genTypeSig scopedSigs bindsWithSigs vbAccess) $ mod
   49            if done
   50              then return mod'
   51              else refactError "No binding without type signature is found at the selection."
   52 
   53 
   54 hasScopedTypeSignatures :: Module -> Bool
   55 hasScopedTypeSignatures mod = "ScopedTypeVariables" `elem` (mod ^? filePragmas & annList & lpPragmas & annList & langExt :: [String])
   56 
   57 genTypeSig :: (BindingElem d) => Bool -> [(GHC.Var, TypeSignature, ValueBind)] -> (AnnList d -> Maybe ValueBind)
   58                                    -> AnnList d -> StateT Bool LocalRefactor (AnnList d)
   59 genTypeSig scopedSigs sigBinds vbAccess ls
   60   | Just vb <- vbAccess ls
   61   , not (typeSignatureAlreadyExist ls vb)
   62     = if isSimpleBinding vb
   63         then
   64           do let id = getBindingName vb
   65                  isTheBind (Just decl)
   66                    = isBinding decl && map semanticsId (decl ^? elementName) == map semanticsId (vb ^? bindingName)
   67                  isTheBind _ = False
   68 
   69              alreadyGenerated <- get
   70              if alreadyGenerated
   71                then return ls
   72                else do put True
   73                        -- checking for possible situations when we cannot generate signature because of
   74                        -- an implicitly passed value
   75                        let dangerousTypeVars = dangerousTVs scopedSigs sigBinds
   76                            myTvs = concatMap @[] (getExternalTVs . idType . semanticsId) (vb ^? bindingName)
   77                        if not $ null @[] $ myTvs `intersect` dangerousTypeVars
   78                          then refactError $ "Could not generate type signature: the type variable(s) "
   79                                               ++ concat (intersperse ", " $ map (showSDocUnsafe . ppr) (myTvs `intersect` dangerousTypeVars))
   80                                               ++ " cannot be captured. (Use ScopedTypeVariables and forall-ed type signatures)"
   81                          else do
   82                            typeSig <- lift $ generateTSFor (getName id) (idType id)
   83                            return $ insertWhere True (createTypeSig typeSig) (const True) isTheBind ls
   84         else refactError "Signature can only be generated for simple value bindings."
   85   | otherwise = return ls
   86   where isSimpleBinding vb = case vb of SimpleBind (AST.VarPat {}) _ _ -> True
   87                                         SimpleBind _ _ _ -> False
   88                                         _ -> True
   89         dangerousTVs scopedSigs sigBinds
   90           = let dangerousDecls = if scopedSigs then filter (\(_,ts,_) -> not $ isForalledTS ts) sigBinds else sigBinds
   91                 dangerousNames = map (\(_,_,bn) -> bn ^? (valBindPats & biplateRef &+& bindingName)) dangerousDecls
   92              in concatMap (concatMap @[] (getExternalTVs . idType . semanticsId @QualifiedName)) dangerousNames
   93 
   94 generateTSFor :: GHC.Name -> GHC.Type -> LocalRefactor TypeSignature
   95 generateTSFor n t = mkTypeSignature (mkUnqualName' n) <$> generateTypeFor (-1) (dropForAlls t)
   96 
   97 -- | Generates the source-level type for a GHC internal type
   98 generateTypeFor :: Int -> GHC.Type -> LocalRefactor AST.Type
   99 generateTypeFor prec t
  100   -- context
  101   | (break (not . isPredTy) -> (preds, other), rt) <- splitFunTys t
  102   , not (null preds)
  103   = do ctx <- case preds of [pred] -> mkContext <$> generateAssertionFor pred
  104                             _      -> mkContext <$> (mkTupleAssertion <$> mapM generateAssertionFor preds)
  105        wrapParen 0 <$> (mkCtxType ctx <$> generateTypeFor 0 (mkFunTys other rt))
  106   -- function
  107   | Just (at, rt) <- splitFunTy_maybe t
  108   = wrapParen 0 <$> (mkFunctionType <$> generateTypeFor 10 at <*> generateTypeFor 0 rt)
  109   -- type operator (we don't know the precedences, so always use parentheses)
  110   | (op, [at,rt]) <- splitAppTys t
  111   , Just tc <- tyConAppTyCon_maybe op
  112   , isSymOcc (getOccName (getName tc))
  113   = wrapParen 0 <$> (mkInfixTypeApp <$> generateTypeFor 10 at <*> referenceOperator (idName $ getTCId tc) <*> generateTypeFor 10 rt)
  114   -- tuple types
  115   | Just (tc, tas) <- splitTyConApp_maybe t
  116   , isTupleTyCon tc
  117   = mkTupleType <$> mapM (generateTypeFor (-1)) tas
  118   -- string type
  119   | Just (ls, [et]) <- splitTyConApp_maybe t
  120   , Just ch <- tyConAppTyCon_maybe et
  121   , listTyCon == ls
  122   , charTyCon == ch
  123   = return $ mkVarType (mkNormalName $ mkSimpleName "String")
  124   -- list types
  125   | Just (tc, [et]) <- splitTyConApp_maybe t
  126   , listTyCon == tc
  127   = mkListType <$> generateTypeFor (-1) et
  128   -- type application
  129   | Just (tf, ta) <- splitAppTy_maybe t
  130   = wrapParen 10 <$> (mkTypeApp <$> generateTypeFor 10 tf <*> generateTypeFor 11 ta)
  131   -- type constructor
  132   | Just tc <- tyConAppTyCon_maybe t
  133   = mkVarType <$> referenceName (idName $ getTCId tc)
  134   -- type variable
  135   | Just tv <- getTyVar_maybe t
  136   = mkVarType <$> referenceName (idName tv)
  137   -- forall type
  138   | (tvs@(_:_), t') <- splitForAllTys t
  139   = wrapParen (-1) <$> (mkForallType (map (mkTypeVar' . getName) tvs) <$> generateTypeFor 0 t')
  140   | otherwise = error ("Cannot represent type: " ++ showSDocUnsafe (ppr t))
  141   where wrapParen :: Int -> AST.Type -> AST.Type
  142         wrapParen prec' node = if prec' < prec then mkParenType node else node
  143 
  144         getTCId :: GHC.TyCon -> GHC.Id
  145         getTCId tc = GHC.mkVanillaGlobal (GHC.tyConName tc) (tyConKind tc)
  146 
  147         generateAssertionFor :: GHC.Type -> LocalRefactor Assertion
  148         generateAssertionFor t
  149           | Just (tc, types) <- splitTyConApp_maybe t
  150           = mkClassAssert <$> referenceName (idName $ getTCId tc) <*> mapM (generateTypeFor 0) types
  151           | otherwise = error "generateAssertionFor: type not supported yet."
  152 
  153 -- | Check whether the definition already has a type signature
  154 typeSignatureAlreadyExist :: (BindingElem d) => AnnList d -> ValueBind -> Bool
  155 typeSignatureAlreadyExist ls vb =
  156   getBindingName vb `elem` (map semanticsId $ concatMap (^? elementName) (filter isTypeSig $ ls ^? annList))
  157 
  158 getBindingName :: ValueBind -> GHC.Id
  159 getBindingName vb = case nub $ map semanticsId $ vb ^? bindingName of
  160   [n] -> n
  161   [] -> error "Trying to generate a signature for a binding with no name"
  162   _ -> error "Trying to generate a signature for a binding with multiple names"
  163 
  164 -- * Checking for type variable constraints
  165 
  166 getExternalTVs :: GHC.Type -> [GHC.Var]
  167 getExternalTVs t
  168   | Just tv <- getTyVar_maybe t = [tv]
  169   | Just (op, arg) <- splitAppTy_maybe t = getExternalTVs op `union` getExternalTVs arg
  170   | Just (tv, t') <- splitForAllTy_maybe t = delete tv $ getExternalTVs t'
  171   | otherwise = []

  173 isForalledTS :: TypeSignature -> Bool
  174 isForalledTS ts = not $ null @[] $ ts ^? tsType & typeBounded & annList
