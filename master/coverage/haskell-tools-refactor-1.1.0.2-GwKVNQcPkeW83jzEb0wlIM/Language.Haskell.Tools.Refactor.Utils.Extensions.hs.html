   15 -- | Expands an extension into all the extensions it implies (keeps original as well)
   16 expandExtension :: GHC.Extension -> [GHC.Extension]
   17 expandExtension ext = ext : implied
   18   where fst' = (^. _1) :: (a,b,c) -> a
   19         snd' = (^. _2) :: (a,b,c) -> b
   20         trd' = (^. _3) :: (a,b,c) -> c
   21 
   22         implied = map trd' . filter snd' . filter ((== ext) . fst') $ impliedXFlags
   23

   26 replaceDeprecated GHC.NullaryTypeClasses = GHC.MultiParamTypeClasses
   27 replaceDeprecated x = x

   73 -- | Canonicalize extensions.
   74 -- This is a helper function for parsing extensions
   75 -- This way we can say @read . canonExt@ to parse any extension string
   76 canonExt :: String -> String
   77 canonExt x = fromMaybe x (lookup x irregularExtensions)
   78 
   79 -- | Serializes the extension's GHC name into its LANGUAGE pragma name.
   80 -- Should be always used in composition with show (@seriealizeExt . show@)
   81 -- when refactoring extensions.
   82 -- This function also replaces depracted extensions with their new versions.
   83 serializeExt :: String -> String
   84 serializeExt "Cpp" = "CPP"
   85 serializeExt "Rank2Types" = "RankNTypes"
   86 serializeExt "RecordPuns" = "NamedFieldPuns"
   87 serializeExt x = x
