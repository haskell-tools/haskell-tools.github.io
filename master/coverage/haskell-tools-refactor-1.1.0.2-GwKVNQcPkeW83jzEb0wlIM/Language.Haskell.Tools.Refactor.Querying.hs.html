   20 type QueryType = String
   21 type QueryMonad = ExceptT String Ghc
   22 
   23 data QueryValue = GeneralQuery Value
   24                 | MarkerQuery  [Marker]
   25   deriving (Generic, Show, Eq)
   26 
   27 data QueryChoice
   28   = LocationQuery { queryName     :: String
   29                   , locationQuery :: RealSrcSpan -> ModuleDom -> [ModuleDom] -> QueryMonad QueryValue
   30                   }
   31   | GlobalQuery   { queryName   :: String
   32                   , globalQuery :: ModuleDom -> [ModuleDom] -> QueryMonad QueryValue
   33                   }
   34 
   35 data Marker = Marker { location :: SrcSpan
   36                      , severity :: Severity
   37                      , message :: String
   38                      } deriving (Generic, Eq)
   39 
   40 data Severity = Error | Warning | Info
   41  deriving (Show, Generic, Eq)
   42 
   43 decompQuery :: QueryValue -> (QueryType, Value)
   44 decompQuery (GeneralQuery x) = ("GeneralQuery", x)
   45 decompQuery (MarkerQuery  x) = ("MarkerQuery" , toJSON x)
   46 
   47 queryCommands :: [QueryChoice] -> [String]
   48 queryCommands = map queryName
   49 
   50 queryError :: String -> QueryMonad a
   51 queryError = throwE
   52 
   53 performQuery :: [QueryChoice] -- ^ The set of available queries
   54                   -> [String] -- ^ The query command
   55                   -> Either FilePath ModuleDom -- ^ The module in which the refactoring is performed
   56                   -> [ModuleDom] -- ^ Other modules
   57                   -> Ghc (Either String (QueryType, Value))
   58 performQuery queries (name:args) modOrPath mods =
   59   case (query, modOrPath, args) of
   60     (Just (LocationQuery _ query), Right mod, sp:_)
   61       -> runExceptT $ decompQuery <$> query (correctRefactorSpan (snd mod) $ readSrcSpan sp) mod mods
   62     (Just (LocationQuery _ _), _, _)
   63       -> return $ Left $ "The query '" ++ name ++ "' needs one argument: a source range"
   64     (Just (GlobalQuery _ query), Right mod, _)
   65       -> runExceptT $ decompQuery <$> query mod mods
   66     (Nothing, _, _)
   67       -> return $ Left $ "Unknown command: " ++ name
   68   where query = find ((== name) . queryName) queries

   75 instance ToJSON SrcSpan where
   76   toJSON (RealSrcSpan sp) = object [ "file" .= unpackFS (srcSpanFile sp)
   77                                    , "startRow" .= srcLocLine (realSrcSpanStart sp)
   78                                    , "startCol" .= srcLocCol (realSrcSpanStart sp)
   79                                    , "endRow" .= srcLocLine (realSrcSpanEnd sp)
   80                                    , "endCol" .= srcLocCol (realSrcSpanEnd sp)
   81                                    ]
   82   toJSON _ = Null
