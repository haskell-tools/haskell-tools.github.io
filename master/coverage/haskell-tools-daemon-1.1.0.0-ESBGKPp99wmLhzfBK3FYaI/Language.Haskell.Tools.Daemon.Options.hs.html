   11 -- | Command line options for the daemon process.
   12 data DaemonOptions = DaemonOptions { daemonVersion :: Bool
   13                                    , portNumber :: Int
   14                                    , silentMode :: Bool
   15                                    , sharedOptions :: SharedDaemonOptions
   16                                    } deriving Show
   17 
   18 -- | Command line options shared by CLI and daemon.
   19 data SharedDaemonOptions = SharedDaemonOptions { noWatch :: Bool
   20                                                , watchExe :: Maybe FilePath
   21                                                , generateCode :: Bool
   22                                                , disableHistory :: Bool
   23                                                , ghcFlags :: Maybe [String]
   24                                                , projectType :: Maybe PackageDB
   25                                                } deriving Show

   27 parseDaemonCLI = execParser daemonCLI
   28 
   29 daemonCLI = info (daemonOptionsParser <**> helper)
   30                  (fullDesc
   31                    <> progDesc "Start the Haskell-tools daemon process"
   32                    <> header "ht-daemon: a background process for Haskell development tools.")

   34 daemonOptionsParser :: Parser DaemonOptions
   35 daemonOptionsParser = DaemonOptions <$> version <*> port <*> silent <*> sharedOptionsParser
   36   where version = switch (long "version"
   37                             <> short 'v'
   38                             <> help "Show the version of this software")
   39         port = option auto
   40                  (long "port"
   41                    <> short 'p'
   42                    <> value 4123
   43                    <> showDefault
   44                    <> help "Set the number of port where the daemon will wait for clients."
   45                    <> metavar "PORT_NUMBER")
   46         silent = switch (long "silent"
   47                           <> short 's'
   48                           <> help "Set to disable messages from daemon.")
   49 
   50 sharedOptionsParser :: Parser SharedDaemonOptions
   51 sharedOptionsParser = SharedDaemonOptions <$> noWatch <*> watch <*> generateCode <*> noHistory <*> ghcFlags <*> pkgDBFlags
   52   where noWatch = switch (long "no-watch"
   53                             <> help "Disables file system watching.")
   54         watch = optional . strOption
   55                   $ long "watch-exe"
   56                       <> short 'w'
   57                       <> help "The file path of the watch executable that is used to monitor file system changes."
   58                       <> metavar "WATH_PATH"
   59         generateCode = switch (long "generate-code"
   60                                  <> help "Always generate code for the modules of the loaded project.")
   61         noHistory = switch (long "no-history"
   62                               <> help "Disables saving the performed refactorings.")
   63         ghcFlags
   64           = optional $ option ghcFlagsParser
   65                          (long "ghc-options"
   66                            <> short 'g'
   67                            <> metavar "GHC_OPTIONS"
   68                            <> help "Flags passed to GHC when loading the packages, separated by spaces.")
   69           where ghcFlagsParser :: ReadM [String]
   70                 ghcFlagsParser
   71                   = ReadM $ do str <- ask
   72                                let str' = case str of '=':'"':rest -> init rest
   73                                                       '=':rest     -> rest
   74                                                       '"':rest     -> init rest
   75                                                       other        -> other
   76                                return ( splitOn " " str')
   77         pkgDBFlags = optional $ option typeParser
   78                        (long "project-type"
   79                          <> metavar "PROJECT_TYPE"
   80                          <> help "Set the project type (cabal, cabal-sandbox, stack, or explicit package database pathes separated by commas).")
   81           where typeParser = do str <- readerAsk
   82                                 case str of "cabal" -> return DefaultDB
   83                                             "cabal-sandbox" ->  return CabalSandboxDB
   84                                             "stack" -> return StackDB
   85                                             pathes -> return $ ExplicitDB (splitOn "," pathes)
