   41 -- | Gets all ModuleCollections from a list of source directories. It also orders the source directories that are package roots so that
   42 -- they can be loaded in the order they are defined (no backward imports). This matters in those cases because for them there can be
   43 -- special compilation flags.
   44 getAllModules :: [FilePath] -> IO [ModuleCollection ModuleNameStr]
   45 getAllModules pathes = orderMCs . concat <$> mapM getModules (map normalise pathes)
   46 
   47 -- | Sorts model collection in an order to remove all backward references.
   48 -- Works because module collections defined by directories cannot be recursive.
   49 orderMCs :: [ModuleCollection k] -> [ModuleCollection k]
   50 orderMCs = sortBy compareMCs
   51   where compareMCs :: ModuleCollection k -> ModuleCollection k -> Ordering
   52         compareMCs mc _ | DirectoryMC _ <- (mc ^. mcId) = GT
   53         compareMCs _ mc | DirectoryMC _ <- (mc ^. mcId) = LT
   54         compareMCs mc1 mc2 | (mc2 ^. mcId) `elem` (mc1 ^. mcDependencies) = GT
   55         compareMCs mc1 mc2 | (mc1 ^. mcId) `elem` (mc2 ^. mcDependencies) = LT
   56         compareMCs _ _ = EQ

   59 -- | Get modules of the project with the indicated root directory.
   60 -- If there is a cabal file, it uses that, otherwise it just scans the directory recursively for haskell sourcefiles.
   61 -- Only returns the non-boot haskell modules, the boot modules will be found during loading.
   62 getModules :: FilePath -> IO [ModuleCollection ModuleNameStr]
   63 getModules root
   64   = do files <- listDirectory root
   65        case find (\p -> takeExtension p == ".cabal") files of
   66           Just cabalFile -> modulesFromCabalFile root cabalFile
   67           Nothing        -> do mods <- modulesFromDirectory root root
   68                                return [ModuleCollection (DirectoryMC root) False root [root] [] (modKeys mods) return return []]
   69   where modKeys mods = Map.fromList $ map (, ModuleNotLoaded NoCodeGen True) mods
   70 
   71 -- | Load the module giving a directory. All modules loaded from the folder and subfolders.
   72 modulesFromDirectory :: FilePath -> FilePath -> IO [String]
   73 -- now recognizing only .hs files
   74 modulesFromDirectory root searchRoot = concat <$> (mapM goOn =<< listDirectory searchRoot)
   75   where goOn fp = let path = searchRoot </> fp
   76                    in do isDir <- doesDirectoryExist path
   77                          if isDir
   78                            then modulesFromDirectory root path
   79                            else if takeExtension path == ".hs"
   80                                   then return [concat $ intersperse "." $ splitDirectories $ dropExtension $ makeRelative root path]
   81                                   else return []
   82 
   83 srcDirFromRoot :: FilePath -> String -> FilePath
   84 srcDirFromRoot fileName "" = fileName
   85 srcDirFromRoot fileName moduleName
   86   = srcDirFromRoot (takeDirectory fileName) (dropWhile (/= '.') $ dropWhile (== '.') moduleName)
   87 
   88 -- | Load the module using a cabal file. The modules described in the cabal file will be loaded.
   89 -- The flags and extensions set in the cabal file will be used by default.
   90 modulesFromCabalFile :: FilePath -> FilePath -> IO [ModuleCollection ModuleNameStr]
   91 -- now adding all conditional entries, regardless of flags
   92 modulesFromCabalFile root cabal = (getModules . setupFlags <$> readGenericPackageDescription silent (root </> cabal))
   93   where getModules pkg = maybe [] (maybe [] (:[]) . toModuleCollection pkg) (library pkg)
   94                            ++ catMaybes (map (toModuleCollection pkg) (executables pkg))
   95                            ++ catMaybes (map (toModuleCollection pkg) (testSuites pkg))
   96                            ++ catMaybes (map (toModuleCollection pkg) (benchmarks pkg))
   97 
   98         toModuleCollection :: ToModuleCollection tmc => PackageDescription -> tmc -> Maybe (ModuleCollection ModuleNameStr)
   99         toModuleCollection PackageDescription{ buildTypeRaw = Just Custom } _
  100           = throw $ UnsupportedPackage "'build-type: custom' setting in cabal file"
  101         toModuleCollection pkg tmc
  102           = let bi = getBuildInfo tmc
  103                 packageName = pkgName $ package pkg
  104              in if buildable bi
  105                   then Just $ ModuleCollection (mkModuleCollKey packageName tmc) False
  106                                 root
  107                                 (map (normalise . (root </>)) $ hsSourceDirs bi)
  108                                 (map (\(mn, fs) -> (moduleName mn, fs)) $ getModuleSourceFiles tmc)
  109                                 (Map.fromList $ map modRecord $ getModuleNames tmc)
  110                                 (flagsFromBuildInfo bi)
  111                                 (loadFlagsFromBuildInfo bi)
  112                                 (map (\(Dependency pkgName _) -> LibraryMC (unPackageName pkgName)) (targetBuildDepends bi))
  113                   else Nothing
  114           where modRecord mn = ( moduleName mn, ModuleNotLoaded NoCodeGen (needsToCompile tmc mn) )
  115         moduleName = concat . intersperse "." . components
  116         setupFlags = either (\deps -> error $ "Missing dependencies: " ++ show deps) fst
  117                        . finalizePD (mkFlagAssignment [])
  118                                     (ComponentRequestedSpec True True) (const True) buildPlatform
  119                                     (unknownCompilerInfo buildCompilerId NoAbiTag) []

  126 -- | Extract module-related information from different kind of package components (library,
  127 -- executable, test-suite or benchmark).
  128 class ToModuleCollection t where
  129   -- | Creates a key for registering this package component.
  130   mkModuleCollKey :: PackageName -> t -> ModuleCollectionId
  131   -- | Gets the build info field from a package component.
  132   getBuildInfo :: t -> BuildInfo
  133   -- | Get the names of all the modules used by this package component.
  134   getModuleNames :: t -> [ModuleName]
  135   -- | Gets if some of the modules are defined is source files that are not in the expected
  136   -- location or named as expected.
  137   getModuleSourceFiles :: t -> [(ModuleName, FilePath)]
  138   getModuleSourceFiles _ = []
  139   -- | Checks if a module is exposed by the package component, so it is necessary to compile.
  140   -- Some of the components may have modules that are only conditionally imported by other modules.
  141   needsToCompile :: t -> ModuleName -> Bool
  142   -- | Gets the Main module in the case of executables, benchmarks and test suites.
  143   getMain :: t -> String
  144   getMain l = getMain' (getBuildInfo l)
  145 
  146 instance ToModuleCollection Library where
  147   mkModuleCollKey pn _ = LibraryMC (unPackageName pn)
  148   getBuildInfo = libBuildInfo
  149   getModuleNames = explicitLibModules
  150   needsToCompile l m = m `elem` exposedModules l
  151 
  152 instance ToModuleCollection Executable where
  153   mkModuleCollKey pn exe = ExecutableMC (unPackageName pn) (unUnqualComponentName $ exeName exe)
  154   getBuildInfo = buildInfo
  155   getModuleNames exe = fromString (getMain exe) : exeModules exe
  156   needsToCompile exe mn = components mn == [getMain exe]
  157   getModuleSourceFiles exe = [(fromString (getMain exe), modulePath exe)]
  158 
  159 instance ToModuleCollection TestSuite where
  160   mkModuleCollKey pn test = TestSuiteMC (unPackageName pn) (unUnqualComponentName $ testName test)
  161   getBuildInfo = testBuildInfo
  162   getModuleNames exe = fromString (getMain exe) : testModules exe
  163   needsToCompile exe mn = components mn == [getMain exe]
  164   getModuleSourceFiles exe
  165     = case testInterface exe of
  166         TestSuiteExeV10 _ fp -> [(fromString (getMain exe), fp)]
  167         _ -> []
  168   getMain t = case testInterface t of
  169         TestSuiteLibV09 _ mod -> intercalate "." $ components mod
  170         _ -> getMain' (getBuildInfo t)
  171 
  172 instance ToModuleCollection Benchmark where
  173   mkModuleCollKey pn test = BenchmarkMC (unPackageName pn) (unUnqualComponentName $ benchmarkName test)
  174   getBuildInfo = benchmarkBuildInfo
  175   getModuleNames exe = fromString (getMain exe) : benchmarkModules exe
  176   needsToCompile exe mn = components mn == [getMain exe]
  177   getModuleSourceFiles exe
  178     = case benchmarkInterface exe of
  179         BenchmarkExeV10 _ fp -> [(fromString (getMain exe), fp)]
  180         _ -> []

  182 -- | A default method of getting the main module using the ghc-options field, checking for the
  183 -- option -main-is.
  184 getMain' :: BuildInfo -> String
  185 getMain' bi
  186   = case ls of _:e:_ -> intercalate "." $ filter (isUpper . head) $ groupBy ((==) `on` (== '.')) e
  187                _ -> "Main"
  188   where ls = dropWhile (/= "-main-is") (concatMap snd (options bi))

  195 -- | Modify the dynamic flags to match the dependencies requested in the .cabal file.
  196 applyDependencies :: [ModuleCollectionId] -> [ModuleCollectionId] -> DynFlags -> DynFlags
  197 applyDependencies mcs deps dfs
  198   = dfs { GHC.packageFlags = GHC.packageFlags dfs ++ (catMaybes $ map (dependencyToPkgFlag mcs) deps) }
  199 
  200 -- | Only use the dependencies that are explicitely enabled. (As cabal does opposed to as ghc does.)
  201 onlyUseEnabled :: DynFlags -> DynFlags
  202 onlyUseEnabled = GHC.setGeneralFlag' GHC.Opt_HideAllPackages
  203 
  204 -- | Transform dependencies of a module collection into the package flags of the GHC API
  205 dependencyToPkgFlag :: [ModuleCollectionId] -> ModuleCollectionId -> Maybe (GHC.PackageFlag)
  206 dependencyToPkgFlag mcs lib@(LibraryMC pkgName)
  207   = if lib `notElem` mcs
  208       then Just $ GHC.ExposePackage pkgName (GHC.PackageArg pkgName) (GHC.ModRenaming True [])
  209       else Nothing
  210 dependencyToPkgFlag _ _ = Nothing

  212 -- | Sets the configuration for loading all the modules from the whole project. Combines the
  213 -- configuration of all package fragments. This solution is not perfect (it would be better to load
  214 -- all package fragments separately), but it is how it works. See 'loadFlagsFromBuildInfo'.
  215 setupLoadFlags :: [ModuleCollectionId] -> [FilePath]
  216                     -> [ModuleCollectionId] -> (DynFlags -> IO DynFlags) -> DynFlags -> IO DynFlags
  217 -- need to be strict here, otherwise the previous modules cannot be garbage collected
  218 setupLoadFlags !ids !roots !allDeps !flags dfs = applyDependencies ids allDeps . selectEnabled <$> flags dfs
  219   where selectEnabled = if any (\((mcId,mcRoot),rest) -> isDirectoryMC mcId && isIndependentMc mcRoot rest) (breaks (zip ids roots))
  220                           then id
  221                           else onlyUseEnabled
  222           where breaks :: [a] -> [(a,[a])]
  223                 breaks [] = []
  224                 breaks (e:rest) = (e,rest) : map (\(x,ls) -> (x,e:ls)) (breaks rest)
  225         isIndependentMc root rest = not $ any (`isPrefixOf` root) (map snd rest)
  226 
  227 -- | Collects the compilation options and enabled extensions from Cabal's build info representation.
  228 -- This setup will be used when loading all packages in the project. See 'setupLoadFlags'.
  229 loadFlagsFromBuildInfo :: BuildInfo -> DynFlags -> IO DynFlags
  230 loadFlagsFromBuildInfo bi@BuildInfo{ cppOptions } df
  231   = do (df',unused,warnings) <- parseDynamicFlags df (map (L noSrcSpan) $ cppOptions)
  232        mapM_ putStrLn (map (unLoc . warnMsg) warnings ++ map (("Flag is not used: " ++) . unLoc) unused)
  233        return (setupLoadExtensions df')
  234   where setupLoadExtensions = foldl (.) id (map setExtensionFlag' $ catMaybes $ map translateExtension loadExtensions)
  235         loadExtensions = [PatternSynonyms | patternSynonymsNeeded] ++ [ExplicitNamespaces | explicitNamespacesNeeded]
  236                            ++ [PackageImports | packageImportsNeeded] ++ [CPP | cppNeeded] ++ [MagicHash | magicHashNeeded]
  237         explicitNamespacesNeeded = not $ null $ map EnableExtension [ExplicitNamespaces, TypeFamilies, TypeOperators] `intersect` usedExtensions bi
  238         patternSynonymsNeeded = EnableExtension PatternSynonyms `elem` usedExtensions bi
  239         packageImportsNeeded = EnableExtension PackageImports `elem` usedExtensions bi
  240         cppNeeded = EnableExtension CPP `elem` usedExtensions bi
  241         magicHashNeeded = EnableExtension MagicHash `elem` usedExtensions bi
  242 
  243 -- | Collects the compilation options and enabled extensions from Cabal's build info representation
  244 -- for a single module. See 'compileInContext'.
  245 flagsFromBuildInfo :: BuildInfo -> DynFlags -> IO DynFlags
  246 -- the import pathes are already set globally
  247 flagsFromBuildInfo bi@BuildInfo{ options } df
  248   = do (df',unused,warnings) <- parseDynamicFlags df (map (L noSrcSpan) $ concatMap snd options)
  249        mapM_ putStrLn (map (unLoc . warnMsg) warnings ++ map (("Flag is not used: " ++) . unLoc) unused)
  250        return $ (flip lang_set (toGhcLang =<< defaultLanguage bi))
  251          $ foldl (.) id (map (\case EnableExtension ext -> setEnabled True ext
  252                                     DisableExtension ext -> setEnabled False ext
  253                         ) (usedExtensions bi))
  254          $ foldr (.) id (map (setEnabled True) (languageDefault (defaultLanguage bi)))
  255          $ df'
  256   where toGhcLang Cabal.Haskell98 = Just GHC.Haskell98
  257         toGhcLang Cabal.Haskell2010 = Just GHC.Haskell2010
  258         toGhcLang _ = Nothing
  259 
  260         -- We don't put the default settings (ImplicitPrelude, MonomorphismRestriction) here
  261         -- because that overrides the opposite extensions (NoImplicitPrelude, NoMonomorphismRestriction)
  262         -- enabled in modules.
  263         languageDefault (Just Cabal.Haskell2010)
  264           = [ DatatypeContexts, DoAndIfThenElse, EmptyDataDecls, ForeignFunctionInterface
  265             , PatternGuards, RelaxedPolyRec, TraditionalRecordSyntax ]
  266         -- Haskell 98 is the default
  267         languageDefault _
  268           = [ DatatypeContexts, NondecreasingIndentation, NPlusKPatterns, TraditionalRecordSyntax ]
  269 
  270         setEnabled enable ext
  271           = case translateExtension ext of
  272               Just e -> (if enable then setExtensionFlag' else unSetExtensionFlag') e
  273               Nothing -> id
