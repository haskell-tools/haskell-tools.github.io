   21 -- | If UndecidableInstances is turned on,
   22 -- it checks whether any type class or type family instances needs
   23 -- UndecidableInstances in the module.
   24 gblChkUndecidableInstances :: CheckNode Module
   25 gblChkUndecidableInstances = conditional gblChkUndecidableInstances' UndecidableInstances
   26 
   27 -- | Checks whether any type class or type family instances needs
   28 -- UndecidableInstances in the module.
   29 gblChkUndecidableInstances' :: CheckNode Module
   30 gblChkUndecidableInstances' m = do
   31   (clsInsts, famInsts) <- getInstances [semanticsModule m]
   32   mapM_ chkClsInst clsInsts
   33   mapM_ chkFamInst famInsts
   34   return m

   36 -- | If the type class instance requires UndecidableInstances,
   37 -- it adds the occurence with location of the instance.
   38 chkClsInst :: ClsInst -> ExtMonad ()
   39 chkClsInst inst = when (clsInstNeedsUD inst)
   40                        (addEvidenceLoc UndecidableInstances (getSrcSpan inst))
   41 
   42 -- | Decides whether a type class instance requires UndecidableInstances.
   43 clsInstNeedsUD :: ClsInst -> Bool
   44 clsInstNeedsUD inst = checkInstTermination args theta
   45   where (_, theta, _, args) = instanceSig inst
   46 
   47 -- | If the type class instance requires UndecidableInstances,
   48 -- it adds the occurence with location of the instance.
   49 chkFamInst :: FamInst -> ExtMonad ()
   50 chkFamInst inst = when (famInstNeedsUD inst)
   51                        (addEvidenceLoc UndecidableInstances (getSrcSpan inst))
   52 
   53 -- | Decides whether a family instance requires UndecidableInstances.
   54 -- If it is a data family instance, it does not need the extension.
   55 famInstNeedsUD :: FamInst -> Bool
   56 famInstNeedsUD inst
   57   | isTyFamInst inst
   58   , lhs <- fi_tys inst
   59   , rhs <- tcTyFamInsts . fi_rhs $ inst
   60   = checkFamEq lhs rhs
   61   | otherwise = False

   63 -- | Checks a declaration whether it needs UndecidableInstances.
   64 -- (If the extension is turned on)
   65 chkUndecidableInstancesDecl :: CheckNode Decl
   66 chkUndecidableInstancesDecl = conditional chkUndecidableInstancesDecl' UndecidableInstances
   67 
   68 -- | Checks a declaration whether it needs UndecidableInstances.
   69 -- If a lookup is not successful, it keeps the extension.
   70 chkUndecidableInstancesDecl' :: CheckNode Decl
   71 chkUndecidableInstancesDecl' d = do
   72   mDecl <- runMaybeT (chkUndecidableInstancesDeclMaybe d)
   73   maybe (addEvidence UndecidableInstances d) return mDecl
   74 
   75 -- | Checks a class declaration whether it has a type function in its context,
   76 -- or a closed type family declaration needs UndecidableInstances.
   77 -- For more information on the family check, see checkFamInstRhs.
   78 -- May fail on lookup.
   79 chkUndecidableInstancesDeclMaybe :: Decl -> MaybeT ExtMonad Decl
   80 chkUndecidableInstancesDeclMaybe d@(ClassDecl mCtx _ _ _)
   81   | isJust (mCtx ^. annMaybe) = do
   82     ctx <- liftMaybe $ mCtx ^. annMaybe
   83     let assert = ctx ^. contextAssertion
   84         names  = assertionQNames assert
   85     types <- mapM lookupTypeFromId names
   86     if any hasTyFunHead types then addEvidence UndecidableInstances d
   87                               else return d
   88   | otherwise = return d
   89 chkUndecidableInstancesDeclMaybe d@(ClosedTypeFamily dh _ _) = do
   90   tyFam <- lookupClosedTyFam dh
   91   let brs    = fromBranches . coAxiomBranches $ tyFam
   92       famEqs = map ((,) <$> coAxBranchLHS <*> tcTyFamInsts . coAxBranchRHS) brs
   93   mapM_ (lift . uncurry chkFamEqM) famEqs >> return d
   94   where
   95     chkFamEqM :: [GHC.Type] -> [(GHC.TyCon, [GHC.Type])] -> ExtMonad ()
   96     chkFamEqM lhs rhs = when (checkFamEq lhs rhs)
   97                              (addEvidence_ UndecidableInstances d)
   98 chkUndecidableInstancesDeclMaybe d = return d
