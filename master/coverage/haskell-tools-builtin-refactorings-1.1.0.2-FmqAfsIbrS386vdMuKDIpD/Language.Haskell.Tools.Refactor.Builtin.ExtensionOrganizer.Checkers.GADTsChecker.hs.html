   13 -- | Checks a GADT-style constructor if GADTSyntax is turned on.
   14 -- Sometimes GADTSyntax is sufficient and GADTs is not even needed.
   15 chkGADTsGadtConDecl :: CheckNode GadtConDecl
   16 chkGADTsGadtConDecl = conditional chkGADTsGadtConDecl' GADTSyntax

   18 -- | Checks a data constructor declaration if GADTs or ExistentialQuantification is turned on.
   19 -- This function is responsible for checking ExistentialQuantification as well.
   20 -- (there is no separate checker for that extension)
   21 chkConDeclForExistentials :: CheckNode ConDecl
   22 chkConDeclForExistentials = conditionalAny chkConDeclForExistentials' [GADTs, ExistentialQuantification]

   24 -- | Checks whether a GADTs-style constructor declaration requires GADTs.
   25 -- If all data constructors are vanilla Haskell 98 data constructors
   26 -- , then only GADTSyntax is needed. If any constructor's lookup fails
   27 -- , we add MissingInformation.
   28 chkGADTsGadtConDecl' :: CheckNode GadtConDecl
   29 chkGADTsGadtConDecl' conDecl = do
   30   let conNames   = conDecl ^. (gadtConNames & annListElems)
   31   mres <- mapM (runMaybeT . isVanillaDataConNameM) conNames
   32   addEvidence_ GADTSyntax conDecl
   33   if | any isNothing mres ->
   34        addRelationMI (GADTs `lOr` ExistentialQuantification) conDecl
   35      | any (not . fromJust) mres ->
   36        addRelation (GADTs `lOr` ExistentialQuantification) conDecl
   37      | otherwise -> return conDecl
   38 
   39 -- | Extracts the name from a ConDecl, and checks whether it is a vanilla
   40 -- data constructor. Ifthe lookup fails, adds MissingInformation.
   41 chkConDeclForExistentials' :: CheckNode ConDecl
   42 chkConDeclForExistentials' conDecl =
   43   fromMaybeTM (addRelationMI (GADTs `lOr` ExistentialQuantification) conDecl) $
   44   case conDecl ^. element of
   45     UConDecl _ _ n _         -> chkName n
   46     URecordDecl _ _ n _      -> chkName n
   47     UInfixConDecl _ _ _ op _ -> chkName (op ^. operatorName)
   48   where chkName :: HasNameInfo' n => n -> MaybeT ExtMonad ConDecl
   49         chkName n = do
   50           isVanilla <- isVanillaDataConNameM n
   51           if isVanilla
   52             then return conDecl
   53             else lift . addRelation (GADTs `lOr` ExistentialQuantification) $ conDecl
