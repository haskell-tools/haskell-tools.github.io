    8 chkMultiParamTypeClassesDecl :: CheckNode Decl
    9 chkMultiParamTypeClassesDecl = conditional chkMultiParamTypeClassesDecl' MultiParamTypeClasses

   12 -- | Decides whether a class or instance declaration needs MultiParamTypeClasses
   13 -- Also handles the NullaryTypeClasses case
   14 chkMultiParamTypeClassesDecl' :: CheckNode Decl
   15 chkMultiParamTypeClassesDecl' cd@(ClassDecl _ dh _ _)
   16   | n <- length . collectTyVars $ dh
   17   , n /= 1 = addEvidence MultiParamTypeClasses dh >> return cd
   18 chkMultiParamTypeClassesDecl' i@(InstanceDecl rule _)
   19   | isMultiParamNeeded (rule ^. irHead)
   20   = addEvidence MultiParamTypeClasses rule >> return i
   21 chkMultiParamTypeClassesDecl' d = return d
   22 
   23 
   24 collectTyVars :: DeclHead -> [TyVar]
   25 collectTyVars (ParenDeclHead dh)        = collectTyVars dh
   26 collectTyVars (DeclHeadApp   dh tv)     = tv : collectTyVars dh
   27 collectTyVars (InfixDeclHead lhs _ rhs) = [lhs,rhs]
   28 collectTyVars _ = []
   29

   31 -- | Decides whether an instance declaration needs MultiParamTypeClasses
   32 -- Also handles the NullaryTypeClasses case
   33 isMultiParamNeeded :: InstanceHead -> Bool
   34 isMultiParamNeeded InstanceHead{}         = True
   35 isMultiParamNeeded InfixInstanceHead{}    = True
   36 isMultiParamNeeded (ParenInstanceHead ih) = isMultiParamNeeded ih
   37 isMultiParamNeeded (AppInstanceHead f _)  = isMultiParamNeeded' f
   38   -- one level deeper
   39   where isMultiParamNeeded' InstanceHead{}         = False
   40         isMultiParamNeeded' InfixInstanceHead{}    = True
   41         isMultiParamNeeded' (ParenInstanceHead ih) = isMultiParamNeeded' ih
   42         isMultiParamNeeded' (AppInstanceHead _ _)  = True
