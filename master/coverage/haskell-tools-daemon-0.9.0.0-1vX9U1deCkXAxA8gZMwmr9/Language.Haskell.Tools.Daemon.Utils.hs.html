   18 -- | Get the name of a module collection from its Id.
   19 moduleCollectionPkgId :: ModuleCollectionId -> Maybe String
   20 moduleCollectionPkgId (DirectoryMC _) = Nothing
   21 moduleCollectionPkgId (LibraryMC id) = Just id
   22 moduleCollectionPkgId (ExecutableMC id _) = Just id
   23 moduleCollectionPkgId (TestSuiteMC id _) = Just id
   24 moduleCollectionPkgId (BenchmarkMC id _) = Just id

   28 -- | Find the module collection where the given module is contained. Based on module name.
   29 lookupModuleColl :: String -> [ModuleCollection SourceFileKey] -> Maybe (ModuleCollection SourceFileKey)
   30 lookupModuleColl moduleName = find (any ((moduleName ==) . (^. sfkModuleName)) . Map.keys . (^. mcModules))
   31 
   32 -- | Find the module collection where the given module is contained. Based on source file name.
   33 lookupSourceFileColl :: FilePath -> [ModuleCollection SourceFileKey] -> Maybe (ModuleCollection SourceFileKey)
   34 lookupSourceFileColl fp = find (any ((fp ==) . (^. sfkFileName)) . Map.keys . (^. mcModules))
   35 
   36 -- | Find the module collection where the given module is contained. Based on source file name and module name.
   37 lookupModuleCollection :: ModSummary -> [ModuleCollection SourceFileKey] -> Maybe (ModuleCollection SourceFileKey)
   38 lookupModuleCollection ms mcs = lookupSourceFileColl (getModSumOrig ms) mcs <|> lookupModuleColl (getModSumName ms) mcs
   39 
   40 -- | Find the module with the given name. Based on module name.
   41 lookupModuleInSCs :: String -> [ModuleCollection SourceFileKey] -> Maybe (SourceFileKey, ModuleRecord)
   42 lookupModuleInSCs moduleName = find ((moduleName ==) . (^. sfkModuleName) . fst) . concatMap (Map.assocs . (^. mcModules))
   43 
   44 -- | Find the module with the given name. Based on source file name.
   45 lookupModInSCs :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Maybe (SourceFileKey, ModuleRecord)
   46 lookupModInSCs key = find (((key ^. sfkFileName) ==) . (^. sfkFileName) . fst) . concatMap (Map.assocs . (^. mcModules))
   47 
   48 -- | Get the module with the given name and source file key.
   49 lookupSFKInSCs :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Maybe ModuleRecord
   50 lookupSFKInSCs key = listToMaybe . catMaybes . map (Map.lookup key . (^. mcModules))
   51 
   52 -- | Remove a module with the given name. Based on module name.
   53 removeModule :: String -> [ModuleCollection SourceFileKey] -> [ModuleCollection SourceFileKey]
   54 removeModule moduleName = map (mcModules .- Map.filterWithKey (\k _ -> moduleName /= (k ^. sfkModuleName)))
   55 
   56 -- | Remove a module with the given name from a module collection. Based on module name and file path.
   57 removeModuleMS :: ModSummary -> Map.Map SourceFileKey ModuleRecord -> Map.Map SourceFileKey ModuleRecord
   58 removeModuleMS ms = Map.filterWithKey (\k _ -> stay k)
   59   where stay k = getModSumName ms /= (k ^. sfkModuleName) || (fn /= getModSumOrig ms && fn /= "")
   60           where fn = k ^. sfkFileName
   61 
   62 -- | Check if we already generated code for the given module. Based on both module name and source
   63 -- file name.
   64 hasGeneratedCode :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Bool
   65 hasGeneratedCode key = maybe False (\case ModuleCodeGenerated {} -> True; _ -> False)
   66                          . lookupSFKInSCs key
   67 
   68 -- | Check if the given module needs code generation. Finds the module if no source file name is
   69 -- present and module names check or if both module names and source file names check.
   70 needsGeneratedCode :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Bool
   71 needsGeneratedCode key mcs = maybe False (\case ModuleCodeGenerated {} -> True; ModuleNotLoaded True _ -> True; _ -> False)
   72                               $ lookupSFKInSCs key mcs <|> lookupSFKInSCs (sfkFileName .= "" $ key) mcs
   73 
   74 -- | Marks the given module for code generation. Finds the module if no source file name is
   75 -- present and module names check or if both module names and source file names check.
   76 -- Idempotent operation.
   77 codeGeneratedFor :: SourceFileKey -> [ModuleCollection SourceFileKey] -> [ModuleCollection SourceFileKey]
   78 codeGeneratedFor key = map (mcModules .- Map.adjust setCodeGen (sfkFileName .= "" $ key) . Map.adjust setCodeGen key)
   79   where setCodeGen (ModuleTypeChecked mod ms) = ModuleCodeGenerated mod ms
   80         setCodeGen (ModuleNotLoaded _ exp) = ModuleNotLoaded True exp
   81         setCodeGen m = m
   82 
   83 -- | Check if the given module has been already loaded. Based on both module name and source
   84 -- file name.
   85 isAlreadyLoaded :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Bool
   86 isAlreadyLoaded key = maybe False (\case (_, ModuleNotLoaded {}) -> False; _ -> True)
   87                          . find ((key ==) . fst) . concatMap (Map.assocs . (^. mcModules))
   88 
   89 -- | Insert a module with a source file key to our database if it wasn't there already
   90 insertIfMissing :: SourceFileKey -> [ModuleCollection SourceFileKey] -> [ModuleCollection SourceFileKey]
   91 insertIfMissing sfk mods
   92   = case lookupSFKInSCs sfk mods of
   93       Just _ -> mods
   94       Nothing -> element 0 & mcModules .- Map.insert sfk (ModuleNotLoaded False False) $ mods
