    5 -- | Simple AST elements of Haskell

    8 import Language.Haskell.Tools.AST.Ann (Ann, AnnListG(..))

   10 data UOperator dom stage
   11   = UBacktickOp { _operatorName :: Ann UQualifiedName dom stage
   12                 } -- ^ A normal name used as an operator with backticks: @ a `mod` b @
   13   | UNormalOp { _operatorName :: Ann UQualifiedName dom stage
   14               } -- ^ A normal operator used as an operator.
   15 
   16 data UName dom stage
   17   = UParenName { _simpleName :: Ann UQualifiedName dom stage
   18                } -- ^ Parenthesized name: @ foldl (+) 0 @
   19   | UNormalName { _simpleName :: Ann UQualifiedName dom stage
   20                 } -- ^ A normal, non-operator name.
   21   | UImplicitName { _simpleName :: Ann UQualifiedName dom stage
   22                   } -- ^ Implicit name: @ ?var @
   23 
   24 -- | Possible qualified names. Contains also implicit names.
   25 -- Linear implicit parameter: @%x@. Non-linear implicit parameter: @?x@.
   26 data UQualifiedName dom stage
   27   = UQualifiedName { _qualifiers :: AnnListG UNamePart dom stage
   28                    , _unqualifiedName :: Ann UNamePart dom stage
   29                    }
   30 
   31 nameFromList :: AnnListG UNamePart dom stage -> UQualifiedName dom stage
   32 nameFromList (AnnListG a xs) | not (null xs)
   33   = UQualifiedName (AnnListG a (init xs)) (last xs)
   34 nameFromList _ = error "nameFromList: empty list"
   35 
   36 -- | Parts of a qualified name.
   37 data UNamePart dom stage
   38   = UNamePart { _simpleNameStr :: String }
   39 
   40 -- | Program elements formatted as string literals (import packages, pragma texts)
   41 data UStringNode dom stage
   42   = UStringNode { _stringNodeStr :: String }
