   21 -- Making the annotation an opaque attribute in annotated elements.
   22 -- Needed because otherwise uniplate traverses the annotations, resulting in unnecessary
   23 -- evaluation and storage.
   24 
   25 instance (DomainWith e dom, SourceInfo stage, Typeable e, Data (e dom stage))
   26       => Data (Ann e dom stage) where
   27     gfoldl k z (Ann annot elem) = z (Ann annot) `k` elem -- ANNOT IS HIDDEN
   28     gunfold k z c = case constrIndex c of 1 -> k (k (z Ann))
   29     toConstr (Ann{}) = con_Ann
   30     dataTypeOf _ = ty_Ann
   31 
   32 con_Ann = mkConstr ty_Ann "Ann" [] Prefix
   33 ty_Ann   = mkDataType "Language.Haskell.Tools.AST.Ann.Ann" [con_Ann]
   34
