  167 -- | Gets the class and family instances from a module.
  168 getInstances :: [Module] -> GHC.Ghc ([ClsInst], [FamInst])
  169 getInstances mods = do
  170   env <- GHC.getSession
  171   eps <- liftIO $ hscEPS env
  172   let homePkgs = catMaybes $ map (lookupHpt (hsc_HPT env) . GHC.moduleName) mods
  173       (hptInsts, hptFamInsts) = hptInstances env (`elem` map GHC.moduleName mods)
  174       isFromMods inst = maybe False (`elem` mods) $ nameModule_maybe $ Var.varName $ is_dfun inst
  175       famIsFromMods inst = maybe False (`elem` mods) $ nameModule_maybe $ co_ax_name $ fi_axiom inst
  176       epsInsts = filter isFromMods $ instEnvElts $ eps_inst_env eps
  177       epsFamInsts = filter famIsFromMods $ famInstEnvElts $ eps_fam_inst_env eps
  178   return (hptInsts ++ epsInsts, hptFamInsts ++ epsFamInsts)
