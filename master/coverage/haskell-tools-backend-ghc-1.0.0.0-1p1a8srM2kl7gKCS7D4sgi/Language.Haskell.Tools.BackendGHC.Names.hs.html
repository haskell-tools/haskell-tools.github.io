   25 trfOperator :: TransformName n r => Located n -> Trf (Ann AST.UOperator (Dom r) RangeStage)
   26 trfOperator = trfLocNoSema trfOperator'
   27 
   28 trfOperator' :: TransformName n r => n -> Trf (AST.UOperator (Dom r) RangeStage)
   29 trfOperator' n
   30   | isSymOcc (occName n) = AST.UNormalOp <$> trfQualifiedNameFocus True n
   31   | otherwise = AST.UBacktickOp <$> trfQualifiedNameFocus True n
   32 
   33 trfName :: TransformName n r => Located n -> Trf (Ann AST.UName (Dom r) RangeStage)
   34 trfName = trfLocNoSema trfName'
   35 
   36 trfName' :: TransformName n r => n -> Trf (AST.UName (Dom r) RangeStage)
   37 trfName' n
   38   | isSymOcc (occName n) = (if isSpecKind then AST.UNormalName else AST.UParenName) <$> trfQualifiedNameFocus isSpecKind n
   39   | otherwise = AST.UNormalName <$> trfQualifiedNameFocus False n
   40   where -- special names that are operators, but appear in name context
   41     isSpecKind = occNameString (occName n) `elem` ["*", "#", "?", "??"]
   42 
   43 trfAmbiguousFieldName :: TransformName n r => Located (AmbiguousFieldOcc n) -> Trf (Ann AST.UName (Dom r) RangeStage)
   44 trfAmbiguousFieldName (L l af) = trfAmbiguousFieldName' l af
   45 
   46 trfAmbiguousFieldName' :: forall n r . TransformName n r => SrcSpan -> AmbiguousFieldOcc n -> Trf (Ann AST.UName (Dom r) RangeStage)
   47 trfAmbiguousFieldName' l (Unambiguous (L _ rdr) pr) = annLocNoSema (pure l) $ trfName' (unpackPostRn @n rdr pr)
   48 -- no Id transformation is done, so we can basically ignore the postTC value
   49 trfAmbiguousFieldName' _ (Ambiguous (L l rdr) _)
   50   = annLocNoSema (pure l)
   51       $ (if (isSymOcc (occName rdr)) then AST.UParenName else AST.UNormalName)
   52           <$> (annLoc (createAmbigousNameInfo rdr l) (pure l) $ AST.nameFromList <$> trfNameStr (isSymOcc (occName rdr)) (rdrNameStr rdr))
   53 
   54 trfAmbiguousOperator' :: forall n r . TransformName n r => SrcSpan -> AmbiguousFieldOcc n -> Trf (Ann AST.UOperator (Dom r) RangeStage)
   55 trfAmbiguousOperator' l (Unambiguous (L _ rdr) pr) = annLocNoSema (pure l) $ trfOperator' (unpackPostRn @n rdr pr)
   56 -- no Id transformation is done, so we can basically ignore the postTC value
   57 trfAmbiguousOperator' _ (Ambiguous (L l rdr) _)
   58   = annLocNoSema (pure l)
   59       $ (if (isSymOcc (occName rdr)) then AST.UNormalOp else AST.UBacktickOp)
   60           <$> (annLoc (createAmbigousNameInfo rdr l) (pure l) $ AST.nameFromList <$> trfOperatorStr (not $ isSymOcc (occName rdr)) (rdrNameStr rdr))

   74 -- | This class allows us to use the same transformation code for multiple variants of the GHC AST.
   75 -- GHC UName annotated with 'name' can be transformed to our representation with semantic annotations of 'res'.
   76 class (TransformableName name, HsHasName name, TransformableName res, HsHasName res, GHCName res)
   77         => TransformName name res where
   78   -- | Demote a given name
   79   transformName :: name -> res
   80 
   81 instance {-# OVERLAPPABLE #-} (n ~ r, TransformableName n, HsHasName n) => TransformName n r where
   82   transformName = id
   83 
   84 instance {-# OVERLAPS #-} (TransformableName res, GHCName res, HsHasName res) => TransformName GHC.Name res where
   85   transformName = fromGHCName

   87 trfNameText :: String -> Trf (Ann AST.UName (Dom r) RangeStage)
   88 trfNameText str
   89   = annContNoSema (AST.UNormalName <$> annLoc (createImplicitNameInfo str) (asks contRange) (AST.nameFromList <$> trfNameStr (isOperatorStr str) str))
   90 
   91 trfImplicitName :: HsIPName -> Trf (Ann AST.UName (Dom r) RangeStage)
   92 trfImplicitName (HsIPName fs)
   93   = let nstr = unpackFS fs
   94      in do rng <- asks contRange
   95            let rng' = mkSrcSpan (updateCol (+1) (srcSpanStart rng)) (srcSpanEnd rng)
   96            annContNoSema (AST.UImplicitName <$> annLoc (createImplicitNameInfo nstr) (pure rng')
   97                                                   (AST.nameFromList <$> trfNameStr (isOperatorStr nstr) nstr))
   98 
   99 isOperatorStr :: String -> Bool
  100 isOperatorStr = any (not . isAlphaNum)

  102 trfQualifiedName :: TransformName n r => Bool -> Located n -> Trf (Ann AST.UQualifiedName (Dom r) RangeStage)
  103 trfQualifiedName isOperator (L l n) = focusOn l $ trfQualifiedNameFocus isOperator n
  104 
  105 trfQualifiedNameFocus :: TransformName n r => Bool -> n -> Trf (Ann AST.UQualifiedName (Dom r) RangeStage)
  106 trfQualifiedNameFocus isOperator n
  107   = do rng <- asks contRange
  108        let rng' = if isOperator == isSymOcc (occName n) then rng
  109                     else mkSrcSpan (updateCol (+1) (srcSpanStart rng)) (updateCol (subtract 1) (srcSpanEnd rng))
  110        annLoc (createNameInfo (transformName n)) (pure rng') (trfQualifiedName' n)
  111 
  112 trfQualifiedName' :: TransformName n r => n -> Trf (AST.UQualifiedName (Dom r) RangeStage)
  113 trfQualifiedName' n = AST.nameFromList <$> ((if isSymOcc (occName n) then trfOperatorStr else trfNameStr) False =<< correctNameString n)
  114 
  115 trfOperatorStr :: Bool -> String -> Trf (AnnListG AST.UNamePart (Dom r) RangeStage)
  116 trfOperatorStr isInParen str = do rng <- correctSpan <$> asks contRange
  117                                   makeList "." (pure $ srcSpanStart rng)
  118                                                (pure [Ann (noSemaInfo $ AST.NodeSpan rng) (AST.UNamePart str)])
  119   where correctSpan sp = if isInParen then mkSrcSpan (updateCol (+1) (srcSpanStart sp))
  120                                                      (updateCol (subtract 1) (srcSpanEnd sp))
  121                                       else sp
  122 
  123 -- | Creates a qualified name from a name string
  124 trfNameStr :: Bool -> String -> Trf (AnnListG AST.UNamePart (Dom r) RangeStage)
  125 trfNameStr isInBackticks str = makeList "." atTheStart (trfNameStr' str . correct <$> atTheStart)
  126   where correct = if isInBackticks then updateCol (+1) else id
  127 
  128 trfNameStr' :: String -> SrcLoc -> [Ann AST.UNamePart (Dom r) RangeStage]
  129 trfNameStr' str startLoc = fst $
  130   foldl (\(r,loc) np -> let nextLoc = advanceAllSrcLoc loc np
  131                          in ( r ++ [Ann (noSemaInfo $ AST.NodeSpan (mkSrcSpan loc nextLoc)) (AST.UNamePart np)], advanceAllSrcLoc nextLoc "." ) )
  132   ([], startLoc) (splitOn "." str)
  133   where -- | Move the source location according to a string
  134         advanceAllSrcLoc :: SrcLoc -> String -> SrcLoc
  135         advanceAllSrcLoc (RealSrcLoc rl) str = RealSrcLoc $ foldl advanceSrcLoc rl str
  136         advanceAllSrcLoc oth _ = oth

  138 trfFastString :: Located FastString -> Trf (Ann AST.UStringNode (Dom r) RangeStage)
  139 trfFastString = trfLocNoSema $ pure . AST.UStringNode . unpackFS
