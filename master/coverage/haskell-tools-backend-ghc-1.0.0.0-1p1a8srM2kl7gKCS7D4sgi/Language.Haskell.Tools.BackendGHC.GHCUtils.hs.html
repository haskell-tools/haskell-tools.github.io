   22 class OutputableBndrId name => GHCName name where
   23   rdrName :: name -> RdrName
   24   getFromNameUsing :: Applicative f => (Name -> Ghc (f Id)) -> Name -> Ghc (f name)
   25   getBindsAndSigs :: HsValBinds name -> ([LSig name], LHsBinds name)
   26   nameFromId :: Id -> name
   27   unpackPostRn :: RdrName -> PostRn name name -> name
   28 
   29   gunpackPostRn :: a -> (name -> a) -> PostRn name name -> a

   31 instance GHCName RdrName where
   32   rdrName = id
   33   getFromNameUsing _ n = return $ pure (nameRdrName n)
   34   getBindsAndSigs (ValBindsIn binds sigs) = (sigs, binds)
   35   getBindsAndSigs _ = error "ValBindsOut: ValBindsOut in parsed source"
   36   nameFromId = nameRdrName . getName
   37   unpackPostRn rdr _ = rdr
   38 
   39   gunpackPostRn a _ _ = a

   41 occName :: GHCName n => n -> OccName
   42 occName = rdrNameOcc . rdrName

   44 instance GHCName GHC.Name where
   45   rdrName = nameRdrName
   46   getFromNameUsing f n = fmap nameFromId <$> f n
   47   getBindsAndSigs (ValBindsOut bindGroups sigs) = (sigs, unionManyBags (map snd bindGroups))
   48   getBindsAndSigs _ = error "getBindsAndSigs: ValBindsIn in renamed source"
   49   nameFromId = getName
   50   unpackPostRn _ a = a
   51 
   52   gunpackPostRn _ f pr = f pr

   62 -- | Loading ids for top-level ghc names
   63 getTopLevelId :: GHC.Name -> Ghc (Maybe GHC.Id)
   64 getTopLevelId name =
   65     lookupName name >>= \case
   66       Just (AnId id) -> return (Just id)
   67       Just (AConLike (RealDataCon dc)) -> return $ Just $ mkVanillaGlobal name (dataConUserType dc)
   68       Just (AConLike (PatSynCon ps)) -> return $ Just $ mkVanillaGlobal name (createPatSynType ps)
   69       Just (ATyCon tc) -> return $ Just $ mkVanillaGlobal name (tyConKind tc)
   70       _ -> return Nothing
   71   where createPatSynType patSyn = case patSynSig patSyn of (_, _, _, _, args, res) -> mkFunTys args res

  229 -- | Tries to simplify the type that has HsAppsTy before renaming. Does not always provide the correct form.
  230 -- Treats each operator as if they are of equivalent precedence and always left-associative.
  231 cleanHsType :: forall n . (OutputableBndrId n) => HsType n -> HsType n
  232 -- for some reason * is considered infix
  233 cleanHsType (HsAppsTy apps) = unLoc $ guessType apps
  234   where guessType :: OutputableBndrId n => [LHsAppType n] -> LHsType n
  235         guessType (L l (HsAppInfix n) : rest) -- must be a prefix actually
  236           = guessType' (L l (HsTyVar NotPromoted n)) rest
  237         guessType (L _ (HsAppPrefix t) : rest) = guessType' t rest
  238         guessType [] = error $ "guessType: empty: " ++ showSDocUnsafe (ppr apps)
  239 
  240         guessType' :: LHsType n -> [LHsAppType n] -> LHsType n
  241         guessType' fun (L _ (HsAppPrefix t) : rest) = guessType' (hsAppTy fun t) rest
  242         guessType' fun (L l (HsAppInfix n) : rest)
  243           -- TODO: find a better check
  244           | showSDocUnsafe (ppr n) == "*" = guessType' (hsAppTy fun (L l (HsTyVar NotPromoted n))) rest
  245         guessType' left (L _ (HsAppInfix n) : right) = hsOpTy left n (guessType right)
  246         guessType' t [] = t
  247 
  248         hsAppTy :: LHsType n -> LHsType n -> LHsType n
  249         hsAppTy t1 t2 = L (getLoc t1 `combineSrcSpans` getLoc t2) $ HsAppTy t1 t2
  250 
  251         hsOpTy :: LHsType n -> Located n -> LHsType n -> LHsType n
  252         hsOpTy t1 n t2 = L (getLoc t1 `combineSrcSpans` getLoc t2) $ HsOpTy t1 n t2
  253 cleanHsType t = t

  255 mergeFixityDefs :: [Located (FixitySig n)] -> [Located (FixitySig n)]
  256 mergeFixityDefs (s@(L l _) : rest)
  257   = let (same, different) = partition ((== l) . getLoc) rest
  258      in foldl mergeWith s (map unLoc same) : mergeFixityDefs different
  259   where mergeWith (L l (FixitySig names fixity)) (FixitySig otherNames _) = L l (FixitySig (names ++ otherNames) fixity)
  260 mergeFixityDefs [] = []

  262 getGroupRange :: HsGroup n -> SrcSpan
  263 getGroupRange (HsGroup {..})
  264   = foldr combineSrcSpans noSrcSpan locs
  265   where locs = [getHsValRange hs_valds] ++ map getLoc hs_splcds ++ map getLoc (concatMap group_tyclds hs_tyclds) ++ map getLoc (concatMap group_roles hs_tyclds)
  266                  ++ map getLoc hs_derivds ++ map getLoc hs_fixds ++ map getLoc hs_defds
  267                  ++ map getLoc hs_fords ++ map getLoc hs_warnds ++ map getLoc hs_annds ++ map getLoc hs_ruleds ++ map getLoc hs_vects
  268                  ++ map getLoc hs_docs

  270 getHsValRange :: HsValBinds n -> SrcSpan
  271 getHsValRange (ValBindsIn vals sig) = foldr combineSrcSpans noSrcSpan $ map getLoc (bagToList vals) ++ map getLoc sig
  272 getHsValRange (ValBindsOut vals sig) = foldr combineSrcSpans noSrcSpan $ concatMap (map getLoc . bagToList . snd) vals ++ map getLoc sig
