   42 createModuleInfo :: ModSummary -> SrcSpan -> [LImportDecl n] -> Trf (Sema.ModuleInfo GHC.Name)
   43 createModuleInfo mod nameLoc (filter (not . ideclImplicit . unLoc) -> imports) = do
   44   let prelude = (xopt ImplicitPrelude $ ms_hspp_opts mod)
   45                   && all (\idecl -> ("Prelude" /= (GHC.moduleNameString $ unLoc $ ideclName $ unLoc idecl))
   46                                       || nameLoc == getLoc idecl) imports
   47   (_, preludeImports) <- if prelude then getImportedNames "Prelude" Nothing else return (ms_mod mod, [])
   48   deps <- if prelude then lift $ getDeps (Module baseUnitId (GHC.mkModuleName "Prelude"))
   49                      else return []
   50   -- This function (via getInstances) refers the ghc environment,
   51   -- we must evaluate the result or the reference may be kept preventing garbage collection.
   52   return $ mkModuleInfo (ms_mod mod) (ms_hspp_opts mod) (case ms_hsc_src mod of HsSrcFile -> False; _ -> True)
   53                         (forceElements preludeImports) deps

   81 -- | Adds semantic information to an impord declaration. See ImportInfo.
   82 createImportData :: (GHCName r, HsHasName n) => GHC.ImportDecl n -> Trf (ImportInfo r)
   83 createImportData (GHC.ImportDecl _ name pkg _ _ _ _ _ declHiding) =
   84   do (mod,importedNames) <- getImportedNames (GHC.moduleNameString $ unLoc name) (fmap (unpackFS . sl_fs) pkg)
   85      names <- liftGhc $ filterM (checkImportVisible declHiding . (^. pName)) importedNames
   86      -- TODO: only use getFromNameUsing once
   87      lookedUpNames <- liftGhc $ mapM translatePName $ names
   88      lookedUpImported <- liftGhc $ mapM (getFromNameUsing getTopLevelId . (^. pName)) $ importedNames
   89      deps <- lift $ getDeps mod
   90      -- This function (via getInstances) refers the ghc environment,
   91      -- we must evaluate the result or the reference may be kept preventing garbage collection.
   92      return $ mkImportInfo mod (forceElements $ catMaybes lookedUpImported)
   93                                (forceElements $ catMaybes lookedUpNames)
   94                                deps
   95   where translatePName (PName n p) = do n' <- getFromNameUsing getTopLevelId n
   96                                         p' <- maybe (return Nothing) (getFromNameUsing getTopLevelId) p
   97                                         return (PName <$> n' <*> Just p')

   99 getDeps :: Module -> Ghc [Module]
  100 getDeps mod = do
  101   env <- GHC.getSession
  102   eps <- liftIO $ hscEPS env
  103   case lookupIfaceByModule (hsc_dflags env) (hsc_HPT env) (eps_PIT eps) mod of
  104     Just ifc -> (mod :) <$> mapM (liftIO . getModule env . fst) (dep_mods (mi_deps ifc))
  105     Nothing -> return [mod]
  106   where getModule env modName = do 
  107           res <- findHomeModule env modName
  108           case res of Found _ m -> return m
  109                       _ -> case lookupPluginModuleWithSuggestions (hsc_dflags env) modName Nothing of
  110                              LookupFound m _ -> return m
  111                              LookupHidden hiddenPack hiddenMod -> return (head $ map fst hiddenMod ++ map fst hiddenPack)
  112                              _ -> error $ "getDeps: module not found: " ++ GHC.moduleNameString modName

  114 -- | Get names that are imported from a given import
  115 getImportedNames :: String -> Maybe String -> Trf (GHC.Module, [PName GHC.Name])
  116 getImportedNames name pkg = liftGhc $ do
  117   hpt <- hsc_HPT <$> getSession
  118   eps <- getSession >>= liftIO . readIORef . hsc_EPS
  119   mod <- findModule (mkModuleName name) (fmap mkFastString pkg)
  120   -- load exported names from interface file
  121   let ifaceNames = maybe [] mi_exports $ flip lookupModuleEnv mod
  122                                        $ eps_PIT eps
  123   let homeExports = maybe [] (md_exports . hm_details) (lookupHptByModule hpt mod)
  124   -- TODO: Why selectors are added in one case and not added in the other?
  125   return (mod, concatMap (availToPName availNames) ifaceNames ++ concatMap (availToPName availNamesWithSelectors) homeExports)
  126     where availToPName f a = map (\n -> if n == availName a then PName n Nothing else PName n (Just (availName a))) (f a)

  128 -- | Check is a given name is imported from an import with given import specification.
  129 checkImportVisible :: (HsHasName name, GhcMonad m) => Maybe (Bool, Located [LIE name]) -> GHC.Name -> m Bool
  130 checkImportVisible (Just (isHiding, specs)) name
  131   | isHiding  = not . or @[] <$> mapM (`ieSpecMatches` name) (map unLoc (unLoc specs))
  132   | otherwise = or @[] <$> mapM (`ieSpecMatches` name) (map unLoc (unLoc specs))
  133 checkImportVisible _ _ = return True
  134 
  135 -- | Forces strict evaluation of a list. Forces elements into WHNF.
  136 forceElements :: [a] -> [a]
  137 forceElements [] = []
  138 forceElements (a : ls) = let res = forceElements ls
  139                           in a `seq` res `seq` (a : ls)
  140 
  141 ieSpecMatches :: (HsHasName name, GhcMonad m) => IE name -> GHC.Name -> m Bool
  142 ieSpecMatches (concatMap hsGetNames' . HsSyn.ieNames -> ls) name
  143   | name `elem` ls = return True
  144 -- ieNames does not consider field names
  145 ieSpecMatches (IEThingWith thing _ with flds) name
  146   | name `elem` concatMap hsGetNames' (map (ieWrappedName . unLoc) (thing : with) ++ map (flSelector . unLoc) flds)
  147   = return True
  148 ieSpecMatches ie@(IEThingAll _) name | [n] <- hsGetNames' (HsSyn.ieName ie), isTyConName n
  149   = do entity <- lookupName n
  150        return $ case entity of Just (ATyCon tc)
  151                                  | Just cls <- tyConClass_maybe tc
  152                                      -> name `elem` map getName (classMethods cls)
  153                                  | otherwise -> name `elem` concatMap (\dc -> getName dc : map flSelector (dataConFieldLabels dc))
  154                                                                       (tyConDataCons tc)
  155                                _             -> False
  156 ieSpecMatches _ _ = return False

  205 -- | Get the elements where there is no ; before
  206 elementsWithoutSemi :: [Ann e (Dom n) RangeStage] -> Trf [Bool]
  207 elementsWithoutSemi [] = return []
  208 elementsWithoutSemi (fst:rest) = indentedElements' (srcSpanEnd $ getRange fst) rest
  209   where indentedElements' lastEnd (elem:rest)
  210           = let sepRange = mkSrcSpan lastEnd (srcSpanStart $ getRange elem)
  211              in (:) <$> (not . (\l -> isGoodSrcSpan l && srcSpanStart l < srcSpanEnd l) <$> focusOn sepRange (tokenLoc AnnSemi))
  212                     <*> indentedElements' (srcSpanEnd $ getRange elem) rest
  213         indentedElements' _ [] = return []

  263 -- | Annotates a node with the given location and focuses on the given source span.

  265 annLoc semam locm nodem = do loc <- locm

  288 -- | Focuses the transformation to go between tokens. The tokens must be found inside the current range.
  289 between :: AnnKeywordId -> AnnKeywordId -> Trf a -> Trf a
  290 between firstTok lastTok = focusAfter firstTok . focusBefore lastTok

  292 -- | Focuses the transformation to go between tokens if they are present
  293 betweenIfPresent :: AnnKeywordId -> AnnKeywordId -> Trf a -> Trf a
  294 betweenIfPresent firstTok lastTok = focusAfterIfPresent firstTok . focusBeforeIfPresent lastTok

  296 -- | Focuses the transformation to be performed after the given token. The token must be found inside the current range.
  297 focusAfter :: AnnKeywordId -> Trf a -> Trf a
  298 focusAfter firstTok trf
  299   = do firstToken <- tokenLoc firstTok
  300        if (isGoodSrcSpan firstToken)
  301           then local (\s -> s { contRange = mkSrcSpan (srcSpanEnd firstToken) (srcSpanEnd (contRange s))}) trf
  302           else do rng <- asks contRange
  303                   convertionProblem $ "focusAfter: token not found in " ++ show rng ++ ": " ++ show firstTok
  304 
  305 focusAfterIfPresent :: AnnKeywordId -> Trf a -> Trf a
  306 focusAfterIfPresent firstTok trf
  307   = do firstToken <- tokenLoc firstTok
  308        if (isGoodSrcSpan firstToken)
  309           then local (\s -> s { contRange = mkSrcSpan (srcSpanEnd firstToken) (srcSpanEnd (contRange s))}) trf
  310           else trf
  311 
  312 -- | Focuses the transformation to be performed before the given token. The token must be found inside the current range.
  313 focusBefore :: AnnKeywordId -> Trf a -> Trf a
  314 focusBefore lastTok trf
  315   = do lastToken <- tokenLocBack lastTok
  316        if (isGoodSrcSpan lastToken)
  317           then local (\s -> s { contRange = mkSrcSpan (srcSpanStart (contRange s)) (srcSpanStart lastToken)}) trf
  318           else do rng <- asks contRange
  319                   convertionProblem $ "focusBefore: token not found in " ++ show rng ++ ": " ++ show lastTok
  320 
  321 focusBeforeIfPresent :: AnnKeywordId -> Trf a -> Trf a
  322 focusBeforeIfPresent lastTok trf
  323   = do lastToken <- tokenLocBack lastTok
  324        if (isGoodSrcSpan lastToken)
  325           then local (\s -> s { contRange = mkSrcSpan (srcSpanStart (contRange s)) (srcSpanStart lastToken)}) trf
  326           else trf

  374 -- | Searches for tokens in the given order inside the parent element and returns their combined location

  378 -- | Searches for tokens in the given order inside the parent element and returns their location
  379 eachTokenLoc :: [AnnKeywordId] -> Trf [SrcSpan]
  380 eachTokenLoc (keyw:rest)
  381   = do spanFirst <- tokenLoc keyw
  382        spanRest <- focusAfterLoc (srcSpanEnd spanFirst) (eachTokenLoc rest)
  383        return (spanFirst : spanRest)
  384 eachTokenLoc [] = pure []

  416 -- | Update the start of the src span
  417 updateStart :: (SrcLoc -> SrcLoc) -> SrcSpan -> SrcSpan
  418 updateStart f sp = mkSrcSpan (f (srcSpanStart sp)) (srcSpanEnd sp)

  420 -- | Update the end of the src span
  421 updateEnd :: (SrcLoc -> SrcLoc) -> SrcSpan -> SrcSpan
  422 updateEnd f sp = mkSrcSpan (srcSpanStart sp) (f (srcSpanEnd sp))

  424 -- | Combine source spans of elements into one that contains them all
  425 collectLocs :: [Located e] -> SrcSpan
  426 collectLocs = foldLocs . map getLoc

  428 -- | Rearrange definitions to appear in the order they are defined in the source file.
  429 orderDefs :: [Ann e (Dom n) RangeStage] -> [Ann e (Dom n) RangeStage]
  430 orderDefs = sortBy (compare `on` srcSpanStart . (^. AST.annotation & AST.sourceInfo & AST.nodeSpan))

  432 -- | Orders a list of elements to the order they are defined in the source file.
  433 orderAnnList :: AnnListG e (Dom n) RangeStage -> AnnListG e (Dom n) RangeStage
  434 orderAnnList (AnnListG a ls) = AnnListG a (orderDefs ls)

  436 -- | Only keeps one of the elements that are on the same source location
  437 removeDuplicates :: [Located e] -> [Located e]
  438 removeDuplicates (fst:rest) = fst : removeDuplicates (filter ((/= getLoc fst) . getLoc) rest)
  439 removeDuplicates [] = []

  441 -- | Orders a list of elements to the order they are defined in the source file.
  442 orderLocated :: [Located e] -> [Located e]
  443 orderLocated = sortBy (compare `on` getLoc)

  445 -- | Transform a list of definitions where the defined names are in scope for subsequent definitions
  446 trfScopedSequence :: HsHasName d => (d -> Trf e) -> [d] -> Trf [e]
  447 trfScopedSequence f (def:rest) = (:) <$> f def <*> addToScope def (trfScopedSequence f rest)
  448 trfScopedSequence _ [] = pure []

  450 -- | Splits a given string at whitespaces while calculating the source location of the fragments
  451 splitLocated :: Located String -> [Located String]
  452 splitLocated = splitLocatedOn isSpace
  453 
  454 -- | Splits a given string while calculating the source location of the fragments
  455 splitLocatedOn :: (Char -> Bool) -> Located String -> [Located String]
  456 splitLocatedOn pred (L (RealSrcSpan l) str) = splitLocated' str (realSrcSpanStart l) Nothing
  457   where splitLocated' :: String -> RealSrcLoc -> Maybe (RealSrcLoc, String) -> [Located String]
  458         splitLocated' (c:rest) currLoc (Just (startLoc, str)) | pred c
  459           = L (RealSrcSpan $ mkRealSrcSpan startLoc currLoc) (reverse str) : splitLocated' rest (advanceSrcLoc currLoc c) Nothing
  460         splitLocated' (c:rest) currLoc Nothing | pred c = splitLocated' rest (advanceSrcLoc currLoc c) Nothing
  461         splitLocated' (c:rest) currLoc (Just (startLoc, str)) = splitLocated' rest (advanceSrcLoc currLoc c) (Just (startLoc, c:str))
  462         splitLocated' (c:rest) currLoc Nothing = splitLocated' rest (advanceSrcLoc currLoc c) (Just (currLoc, [c]))
  463         splitLocated' [] currLoc (Just (startLoc, str)) = [L (RealSrcSpan $ mkRealSrcSpan startLoc currLoc) (reverse str)]
  464         splitLocated' [] _ Nothing = []
  465 splitLocatedOn _ _ = convProblem "splitLocated: unhelpful span given"

  467 compareSpans :: SrcSpan -> SrcSpan -> Ordering
  468 compareSpans (RealSrcSpan a) (RealSrcSpan b)
  469   | a `containsSpan` b = GT
  470   | b `containsSpan` a = LT
  471 compareSpans _ _ = EQ
