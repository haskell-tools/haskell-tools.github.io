   28 trfType :: TransformName n r => Located (HsType n) -> Trf (Ann AST.UType (Dom r) RangeStage)
   29 trfType typ | RealSrcSpan loce <- getLoc typ
   30   = do othSplices <- asks typeSplices
   31        let contSplice = filter (\sp -> case getLoc sp of (RealSrcSpan spLoc) -> spLoc `containsSpan` loce; _ -> False) othSplices
   32        case contSplice of [] -> trfLocNoSema trfType' typ
   33                           _ -> let lsp@(L l sp) = minimumBy (compareSpans `on` getLoc) contSplice
   34                                 in typeSpliceInserted lsp (annLocNoSema (pure l) (AST.UTySplice <$> (trfSplice =<< rdrSplice sp)))
   35   | otherwise = trfLocNoSema trfType' typ
   36 
   37 trfType' :: TransformName n r => HsType n -> Trf (AST.UType (Dom r) RangeStage)
   38 trfType' = trfType'' . cleanHsType where
   39   trfType'' (HsForAllTy [] typ) = trfType' (unLoc typ)
   40   trfType'' (HsForAllTy bndrs typ) = AST.UTyForall <$> defineTypeVars (trfBindings bndrs)
   41                                                    <*> addToScope bndrs (trfType typ)
   42   trfType'' (HsQualTy (L _ []) typ) = trfType' (unLoc typ)
   43   trfType'' (HsQualTy ctx typ) = AST.UTyCtx <$> (fromJust . (^. annMaybe) <$> trfCtx atTheStart ctx)
   44                                             <*> trfType typ
   45   trfType'' (HsTyVar _ name) = AST.UTyVar <$> transformingPossibleVar name (trfName name)
   46   trfType'' (HsAppsTy apps) | Just (head, args, _) <- getAppsTyHead_maybe apps
   47     = foldl (\core t -> AST.UTyApp <$> annLocNoSema (pure $ getLoc head `combineSrcSpans` getLoc t) core <*> trfType t) (trfType' (unLoc head)) args
   48   trfType'' (HsAppTy t1 t2) = AST.UTyApp <$> trfType t1 <*> trfType t2
   49   trfType'' (HsFunTy t1 t2) = AST.UTyFun <$> trfType t1 <*> trfType t2
   50   trfType'' (HsListTy typ) = AST.UTyList <$> trfType typ
   51   trfType'' (HsPArrTy typ) = AST.UTyParArray <$> trfType typ
   52   trfType'' (HsTupleTy HsBoxedOrConstraintTuple typs) = AST.UTyTuple <$> trfAnnList ", " trfType' typs
   53   trfType'' (HsTupleTy HsBoxedTuple typs) = AST.UTyTuple <$> trfAnnList ", " trfType' typs
   54   trfType'' (HsTupleTy HsUnboxedTuple typs) = AST.UTyUnbTuple <$> trfAnnList ", " trfType' typs
   55   trfType'' (HsOpTy t1 op t2) = AST.UTyInfix <$> trfType t1 <*> trfOperator op <*> trfType t2
   56   trfType'' (HsParTy typ) = AST.UTyParen <$> trfType typ
   57   trfType'' (HsKindSig typ kind) = AST.UTyKinded <$> trfType typ <*> trfKind kind
   58   trfType'' (HsSpliceTy qq@(HsQuasiQuote {}) _) = AST.UTyQuasiQuote <$> annContNoSema (trfQuasiQuotation' qq)
   59   trfType'' (HsSpliceTy splice _) = AST.UTySplice <$> trfSplice splice
   60   trfType'' (HsBangTy (HsSrcBang _ SrcUnpack _) typ) = AST.UTyUnpack <$> trfType typ
   61   trfType'' (HsBangTy (HsSrcBang _ SrcNoUnpack _) typ) = AST.UTyNoUnpack <$> trfType typ
   62   trfType'' (HsBangTy (HsSrcBang _ _ SrcStrict) typ) = AST.UTyBang <$> trfType typ
   63   trfType'' (HsBangTy (HsSrcBang _ _ SrcLazy) typ) = AST.UTyLazy <$> trfType typ
   64   trfType'' pt@(HsExplicitListTy {}) = AST.UTyPromoted <$> annContNoSema (trfPromoted' trfType' pt)
   65   trfType'' pt@(HsExplicitTupleTy {}) = AST.UTyPromoted <$> annContNoSema (trfPromoted' trfType' pt)
   66   trfType'' pt@(HsTyLit {}) = AST.UTyPromoted <$> annContNoSema (trfPromoted' trfType' pt)
   67   trfType'' (HsWildCardTy _) = pure AST.UTyWildcard -- TODO: named wildcards
   68   trfType'' (HsSumTy types) = AST.UUnbSumType <$> trfAnnList " | " trfType' types
   69   trfType'' t = unhandledElement "type" t

   71 trfBindings :: TransformName n r => [Located (HsTyVarBndr n)] -> Trf (AnnListG AST.UTyVar (Dom r) RangeStage)
   72 trfBindings vars = trfAnnList " " trfTyVar' vars

   74 trfTyVar :: TransformName n r => Located (HsTyVarBndr n) -> Trf (Ann AST.UTyVar (Dom r) RangeStage)
   75 trfTyVar = trfLocNoSema trfTyVar'

   77 trfTyVar' :: TransformName n r => HsTyVarBndr n -> Trf (AST.UTyVar (Dom r) RangeStage)
   78 trfTyVar' (UserTyVar name) = AST.UTyVarDecl <$> typeVarTransform (trfName name)
   79                                            <*> (nothing " " "" atTheEnd)
   80 trfTyVar' (KindedTyVar name kind) = AST.UTyVarDecl <$> typeVarTransform (trfName name)
   81                                                   <*> trfKindSig (Just kind)

   83 trfCtx :: TransformName n r => Trf SrcLoc -> Located (HsContext n) -> Trf (AnnMaybeG AST.UContext (Dom r) RangeStage)
   84 trfCtx sp (L _ []) = nothing " " "" sp
   85 trfCtx _ (L l [L _ (HsParTy t)])
   86   = makeJust <$> annLocNoSema (combineSrcSpans l <$> tokenLoc AnnDarrow)
   87                               (AST.UContext <$> annLocNoSema (pure l) (AST.UTupleAssert <$> (trfAnnList ", " trfAssertion' [t])))
   88 trfCtx _ (L l [t])
   89   = makeJust <$> annLocNoSema (combineSrcSpans l <$> tokenLoc AnnDarrow)
   90                               (AST.UContext <$> trfAssertion t)
   91 trfCtx _ (L l ctx) = makeJust <$> annLocNoSema (combineSrcSpans l <$> tokenLoc AnnDarrow)
   92                                                (AST.UContext <$> annLocNoSema (pure l) (AST.UTupleAssert <$> (trfAnnList ", " trfAssertion' ctx)))
   93 
   94 trfAssertion :: TransformName n r => LHsType n -> Trf (Ann AST.UAssertion (Dom r) RangeStage)
   95 trfAssertion = trfLocNoSema trfAssertion'
   96 
   97 trfAssertion' :: forall n r . TransformName n r => HsType n -> Trf (AST.UAssertion (Dom r) RangeStage)
   98 trfAssertion' (cleanHsType -> HsParTy t)
   99   = trfAssertion' (unLoc t)
  100 trfAssertion' (cleanHsType -> HsOpTy left op right)
  101   = AST.UInfixAssert <$> trfType left <*> trfOperator op <*> trfType right
  102 trfAssertion' (cleanHsType -> HsTupleTy _ tys)
  103   = AST.UTupleAssert <$> makeList ", " (after AnnOpenP) (mapM trfAssertion tys)
  104 trfAssertion' (cleanHsType -> HsWildCardTy _)
  105   = pure AST.UWildcardAssert
  106 trfAssertion' (cleanHsType -> t) = case cleanHsType base of
  107    HsTyVar _ name -> AST.UClassAssert <$> trfName name <*> trfAnnList " " trfType' args
  108    HsEqTy t1 t2 -> AST.UInfixAssert <$> trfType t1 <*> annLocNoSema (tokenLoc AnnTilde) (trfOperator' typeEq) <*> trfType t2
  109    HsIParamTy name t -> AST.UImplicitAssert <$> define (focusOn (getLoc name) (trfImplicitName (unLoc name))) <*> trfType t
  110    t -> unhandledElement "assertion" t
  111   where (args, _, base) = getArgs t
  112 
  113         getArgs :: HsType n -> ([LHsType n], Maybe SrcSpan, HsType n)
  114         getArgs (HsAppTy (L l ft) at) = case getArgs ft of (args, sp, base) -> (args++[at], sp <|> Just l, base)
  115         getArgs t = ([], Nothing, t)
  116 
  117         typeEq :: n
  118         typeEq = nameFromId (mkVanillaGlobal (tyConName heqTyCon) (tyConKind heqTyCon))
