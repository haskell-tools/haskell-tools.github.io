   31 class AfterBefore i where
   32   -- | Put the given string before the element if it is not empty
   33   after :: String -> i -> i
   34   -- | The given string should follow the element if it is not empty
   35   followedBy :: String -> i -> i
   36 
   37 instance AfterBefore (ListInfo SrcTemplateStage) where
   38   after str = srcTmpListBefore .= str
   39   followedBy str = srcTmpListAfter .= str
   40 
   41 instance AfterBefore (OptionalInfo SrcTemplateStage) where
   42   after str = srcTmpOptBefore .= str
   43   followedBy str = srcTmpOptAfter .= str

   45 class RelativeIndent i where

   49 instance RelativeIndent (SpanInfo SrcTemplateStage) where

   72 -- | The elements of the list should be separated by the given string by default (might be overridden)
   73 separatedBy :: String -> ListInfo SrcTemplateStage -> ListInfo SrcTemplateStage
   74 separatedBy sep = srcTmpDefaultSeparator .= sep
   75 
   76 -- | The elements of the list should be indented on the same column
   77 indented :: ListInfo SrcTemplateStage -> ListInfo SrcTemplateStage
   78 indented = (srcTmpIndented .= Just []) . (srcTmpDefaultSeparator .= "\n")

   80 -- | Concatenates two source templates to produce a new template with all child elements.
   81 (<>) :: SpanInfo SrcTemplateStage -> SpanInfo SrcTemplateStage -> SpanInfo SrcTemplateStage
   82 SourceTemplateNode sp1 el1 _ _ <> SourceTemplateNode sp2 el2 _ _ = SourceTemplateNode (combineSrcSpans sp1 sp2) (el1 ++ el2) 0 Nothing
