   16 instance HasNameInfo' GHC.Name where
   17   semanticsName = Just <$> id
   18 
   19 instance HasNameInfo' Operator where
   20   semanticsName = opSemName
   21 
   22 instance HasNameInfo' DeclHead where
   23   semanticsName = declHeadSemName
   24 
   25 instance HasNameInfo' InstanceHead where
   26   semanticsName = instHeadSemName
   27 
   28 
   29 opSemName :: Operator -> Maybe GHC.Name
   30 opSemName = semanticsName . (^. operatorName)
   31 
   32 declHeadQName :: DeclHead -> QualifiedName
   33 declHeadQName (NameDeclHead n)       = n ^. simpleName
   34 declHeadQName (ParenDeclHead dh)     = declHeadQName dh
   35 declHeadQName (DeclHeadApp dh _)     = declHeadQName dh
   36 declHeadQName (InfixDeclHead _ op _) = op ^. operatorName
   37 
   38 declHeadSemName :: DeclHead -> Maybe GHC.Name
   39 declHeadSemName (NameDeclHead n)       = semanticsName n
   40 declHeadSemName (ParenDeclHead dh)     = declHeadSemName dh
   41 declHeadSemName (DeclHeadApp dh _)     = declHeadSemName dh
   42 declHeadSemName (InfixDeclHead _ op _) = opSemName op
   43 
   44 instHeadSemName :: InstanceHead -> Maybe GHC.Name
   45 instHeadSemName (InstanceHead n)         = semanticsName n
   46 instHeadSemName (InfixInstanceHead _ op) = opSemName op
   47 instHeadSemName (ParenInstanceHead ih)   = instHeadSemName ih
   48 instHeadSemName (AppInstanceHead ih _)   = instHeadSemName ih
   49 
   50 -- | Collects the qualified names of the class heads in an assertion.
   51 assertionQNames :: Assertion -> [QualifiedName]
   52 assertionQNames (ClassAssert n _)    = [n ^. simpleName]
   53 assertionQNames (InfixAssert _ op _) = [op ^. operatorName]
   54 assertionQNames (ImplicitAssert n _) = [n ^. simpleName]
   55 assertionQNames (TupleAssert xs)     = concatMap assertionQNames xs
   56 assertionQNames _                    = []
   57 
   58 -- | Collects the semantic names of the class heads in an assertion.
   59 assertionSemNames :: Assertion -> [GHC.Name]
   60 assertionSemNames (ClassAssert n _)    = maybeToList . semanticsName $ n
   61 assertionSemNames (InfixAssert _ op _) = maybeToList . opSemName $ op
   62 assertionSemNames (ImplicitAssert n _) = maybeToList . semanticsName $ n
   63 assertionSemNames (TupleAssert xs)     = concatMap assertionSemNames xs
   64 assertionSemNames _ = []

   66 -- | Extracts the name of a type.
   67 -- In case of a type application, it finds the type being applied.
   68 -- It works only for unambiguous types, so it won't work for tuples.
   69 nameFromType :: Type -> Maybe Name
   70 nameFromType (TypeApp f _)    = nameFromType f
   71 nameFromType (ParenType x)    = nameFromType x
   72 nameFromType (ListType t)     = nameFromType t
   73 nameFromType (KindedType t _) = nameFromType t
   74 nameFromType (BangType t)     = nameFromType t
   75 nameFromType (LazyType t)     = nameFromType t
   76 nameFromType (UnpackType t)   = nameFromType t
   77 nameFromType (NoUnpackType t) = nameFromType t
   78 nameFromType (VarType x)      = Just x
   79 nameFromType _                = Nothing
