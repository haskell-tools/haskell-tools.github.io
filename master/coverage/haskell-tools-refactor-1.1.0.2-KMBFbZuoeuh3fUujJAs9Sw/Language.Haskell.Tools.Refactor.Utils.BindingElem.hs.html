   12 -- | A type class for handling definitions that can appear as both top-level and local definitions
   13 class NamedElement d => BindingElem d where
   14 
   15   -- | Accesses a type signature definition in a local or top-level definition
   16   sigBind :: Simple Partial (Ann d IdDom SrcTemplateStage) TypeSignature
   17 
   18   -- | Accesses a value or function definition in a local or top-level definition
   19   valBind :: Simple Partial (Ann d IdDom SrcTemplateStage) ValueBind
   20 
   21   -- | Accesses a type signature definition in a local or top-level definition
   22   fixitySig :: Simple Partial (Ann d IdDom SrcTemplateStage) FixitySignature
   23 
   24   -- | Creates a new definition from a type signature
   25   createTypeSig :: TypeSignature -> Ann d IdDom SrcTemplateStage
   26 
   27   -- | Creates a new definition from a value or function definition
   28   createBinding :: ValueBind -> Ann d IdDom SrcTemplateStage
   29 
   30   -- | Creates a new fixity signature
   31   createFixitySig :: FixitySignature -> Ann d IdDom SrcTemplateStage
   32 
   33   -- | Checks if a given definition is a type signature
   34   isTypeSig :: Ann d IdDom SrcTemplateStage -> Bool
   35 
   36   -- | Checks if a given definition is a function or value binding
   37   isBinding :: Ann d IdDom SrcTemplateStage -> Bool
   38 
   39   -- | Checks if a given definition is a fixity signature
   40   isFixitySig :: Ann d IdDom SrcTemplateStage -> Bool
   41 
   42 instance BindingElem UDecl where
   43   sigBind = declTypeSig
   44   valBind = declValBind
   45   fixitySig = declFixity
   46   createTypeSig = mkTypeSigDecl
   47   createBinding = mkValueBinding
   48   createFixitySig = mkFixityDecl
   49   isTypeSig TypeSigDecl {} = True
   50   isTypeSig _ = False
   51   isBinding ValueBinding {} = True
   52   isBinding _ = False
   53   isFixitySig FixityDecl {} = True
   54   isFixitySig _ = False
   55 
   56 instance BindingElem ULocalBind where
   57   sigBind = localSig
   58   valBind = localVal
   59   fixitySig = localFixity
   60   createTypeSig = mkLocalTypeSig
   61   createBinding = mkLocalValBind
   62   createFixitySig = mkLocalFixity
   63   isTypeSig LocalTypeSig {} = True
   64   isTypeSig _ = False
   65   isBinding LocalValBind {} = True
   66   isBinding _ = False
   67   isFixitySig LocalFixity {} = True
   68   isFixitySig _ = False
   69 
   70 getValBindInList :: BindingElem d => RealSrcSpan -> AnnList d -> Maybe ValueBind
   71 getValBindInList sp ls = case ls ^? valBindsInList & filtered (isInside sp) of
   72   [] -> Nothing
   73   [n] -> Just n
   74   _ -> error "getValBindInList: Multiple nodes"
   75 
   76 valBindsInList :: BindingElem d => Simple Traversal (AnnList d) ValueBind
   77 valBindsInList = annList & valBind
