    1 -- | Representation of Haskell AST value and function bindings (both local and top-level)

   10 -- | Value binding for top-level and local bindings
   11 data UValueBind dom stage
   12   = USimpleBind { _valBindPat :: Ann UPattern dom stage
   13                 , _valBindRhs :: Ann URhs dom stage
   14                 , _valBindLocals :: AnnMaybeG ULocalBinds dom stage
   15                 } -- ^ Non-function binding (@ v = "12" @)
   16   -- TODO: use one name for a function instead of names in each match
   17   | UFunBind    { _funBindMatches :: AnnListG UMatch dom stage
   18                 } -- ^ Function binding (@ f 0 = 1; f x = x @). All matches must have the same name.
   19 
   20 -- | Clause of function binding
   21 data UMatch dom stage
   22   = UMatch { _matchLhs :: Ann UMatchLhs dom stage
   23            , _matchRhs :: Ann URhs dom stage
   24            , _matchBinds :: AnnMaybeG ULocalBinds dom stage
   25            }
   26 
   27 -- | Something on the left side of the match
   28 data UMatchLhs dom stage
   29   = UNormalLhs { _matchLhsName :: Ann UName dom stage
   30                , _matchLhsArgs :: AnnListG UPattern dom stage
   31                } -- ^ A match lhs with the function name and parameter names (@ f a b @)
   32   | UInfixLhs { _matchLhsLhs :: Ann UPattern dom stage
   33               , _matchLhsOperator :: Ann UOperator dom stage
   34               , _matchLhsRhs :: Ann UPattern dom stage
   35               , _matchLhsArgs :: AnnListG UPattern dom stage
   36               } -- ^ An infix match lhs for an operator (@ a + b @)
   37 
   38 -- | Local bindings attached to a declaration (@ where x = 42 @)
   39 data ULocalBinds dom stage
   40   = ULocalBinds { _localBinds :: AnnListG ULocalBind dom stage
   41                 }
   42 
   43 -- | Bindings that are enabled in local blocks (where or let).
   44 data ULocalBind dom stage
   45   -- TODO: check that no other signature can be inside a local binding
   46   = ULocalValBind   { _localVal :: Ann UValueBind dom stage
   47                     } -- ^ A local binding for a value
   48   | ULocalSignature { _localSig :: Ann UTypeSignature dom stage
   49                     } -- ^ A local type signature
   50   | ULocalFixity    { _localFixity :: Ann UFixitySignature dom stage
   51                     } -- ^ A local fixity declaration
   52   | ULocalInline    { _localInline :: Ann UInlinePragma dom stage
   53                     } -- ^ A local inline pragma
   54 
   55 -- | A type signature (@ f :: Int -> Int @)
   56 data UTypeSignature dom stage
   57   = UTypeSignature { _tsName :: AnnListG UName dom stage
   58                    , _tsType :: Ann UType dom stage
   59                    }
   60

   63 -- | A fixity signature (@ infixl 5 +, - @).
   64 data UFixitySignature dom stage
   65   = UFixitySignature { _fixityAssoc :: Ann Assoc dom stage
   66                      , _fixityPrecedence :: AnnMaybeG Precedence dom stage
   67                      , _fixityOperators :: AnnListG UOperator dom stage
   68                      }
   69 
   70 -- | Associativity of an operator.
   71 data Assoc dom stage
   72   = AssocNone  -- ^ non-associative operator (declared with @infix@)
   73   | AssocLeft  -- ^ left-associative operator (declared with @infixl@)
   74   | AssocRight -- ^ right-associative operator (declared with @infixr@)
   75 
   76 -- | Numeric precedence of an operator
   77 data Precedence dom stage
   78   = Precedence { _precedenceValue :: Int }
   79 
   80 -- | Right hand side of a value binding (possible with guards): (@ = 3 @ or @ | x == 1 = 3; | otherwise = 4 @)
   81 data URhs dom stage
   82   = UUnguardedRhs { _rhsExpr :: Ann UExpr dom stage
   83                   } -- ^ An unguarded right-hand-side (@ = 3 @)
   84   | UGuardedRhss  { _rhsGuards :: AnnListG UGuardedRhs dom stage
   85                   } -- ^ An unguarded right-hand-side (@ | x == 1 = 3; | otherwise = 4 @)
   86 
   87 -- | A guarded right-hand side of a value binding (@ | x > 3 = 2 @)
   88 data UGuardedRhs dom stage
   89   = UGuardedRhs { _guardStmts :: AnnListG URhsGuard dom stage -- ^ Cannot be empty.
   90                 , _guardExpr :: Ann UExpr dom stage
   91                 }
   92 
   93 -- | Guards for value bindings and pattern matches (@ Just v <- x, v > 1 @)
   94 data URhsGuard dom stage
   95   = UGuardBind  { _guardPat :: Ann UPattern dom stage
   96                 , _guardRhs :: Ann UExpr dom stage
   97                 } -- ^ A bind statement in a pattern guard (@ Just v <- x @)
   98   | UGuardLet   { _guardBinds :: AnnListG ULocalBind dom stage
   99                 } -- ^ A let statement in a pattern guard (@ let x = 3 @)
  100   | UGuardCheck { _guardCheck :: Ann UExpr dom stage
  101                 } -- ^ An expression to check for a pattern guard
  102 
  103 -- | Pragmas that control how the definitions will be inlined
  104 data UInlinePragma dom stage
  105   = UInlinePragma     { _inlineConlike :: AnnMaybeG UConlikeAnnot dom stage
  106                       , _inlinePhase :: AnnMaybeG UPhaseControl dom stage
  107                       , _inlineDef :: Ann UName dom stage
  108                       } -- ^ A pragma that marks a function for inlining to the compiler (@ {-# INLINE thenUs #-} @)
  109   | UNoInlinePragma   { _noInlineDef :: Ann UName dom stage
  110                       } -- ^ A pragma that forbids a function from being inlined by the compiler (@ {-# NOINLINE f #-} @)
  111   | UInlinablePragma  { _inlinePhase :: AnnMaybeG UPhaseControl dom stage
  112                       , _inlinableDef :: Ann UName dom stage
  113                       } -- ^ A pragma that marks a function that it may be inlined by the compiler (@ {-# INLINABLE thenUs #-} @)
  114 
  115 -- | A @CONLIKE@ modifier for an @INLINE@ pragma.
  116 data UConlikeAnnot dom stage = UConlikeAnnot
  117 
  118 -- | Controls the activation of a rewrite rule (@ [1] @)
  119 data UPhaseControl dom stage
  120   = UPhaseControl { _phaseUntil :: AnnMaybeG PhaseInvert dom stage
  121                   , _phaseNumber :: AnnMaybeG PhaseNumber dom stage
  122                   }
  123 
  124 -- | Phase number for rewrite rules
  125 data PhaseNumber dom stage
  126   = PhaseNumber { _phaseNum :: Integer }
  127 
  128 -- | A tilde that marks the inversion of the phase number
  129 data PhaseInvert dom stage = PhaseInvert
