   20 -- Making the annotation an opaque attribute in annotated elements.
   21 -- Needed because otherwise uniplate traverses the annotations, resulting in unnecessary
   22 -- evaluation and storage.
   23 
   24 instance (DomainWith e dom, SourceInfo stage, Typeable e, Data (e dom stage))
   25       => Data (Ann e dom stage) where
   26     gfoldl k z (Ann annot elem) = z (Ann annot) `k` elem -- ANNOT IS HIDDEN
   27     gunfold k z c = case constrIndex c of 1 -> k (k (z Ann))
   28     toConstr (Ann{}) = con_Ann
   29     dataTypeOf _ = ty_Ann
   30 
   31 con_Ann = mkConstr ty_Ann "Ann" [] Prefix
   32 ty_Ann   = mkDataType "Language.Haskell.Tools.AST.Ann.Ann" [con_Ann]
   33
