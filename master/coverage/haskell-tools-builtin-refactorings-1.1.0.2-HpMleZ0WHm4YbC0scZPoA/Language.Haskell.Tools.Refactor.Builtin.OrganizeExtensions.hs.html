   53 organizeExtensions :: LocalRefactoring
   54 organizeExtensions moduleAST = do
   55   exts <- liftGhc $ reduceExtensions moduleAST
   56   let langExts = map (mkLanguagePragma . pure . serializeExt . show) exts
   57       ghcOpts  = moduleAST ^? filePragmas & annList & opStr & stringNodeStr
   58       ghcOpts' = map (mkOptionsGHC . unwords . filter (isPrefixOf "-") . words) ghcOpts
   59 
   60       offExts  = map (mkLanguagePragma . pure)
   61                . sort
   62                . map (("No" ++) . serializeExt . show)
   63                . collectTurnedOffExtensions
   64                $ moduleAST
   65 
   66       newPragmas = mkFilePragmas $ offExts ++ langExts ++ ghcOpts'
   67 
   68   (filePragmas != newPragmas)
   69     -- remove empty {-# LANGUAGE #-} pragmas
   70     >=> filePragmas !~ filterListSt (\case LanguagePragma (AnnList []) -> False; _ -> True)
   71     $ moduleAST

   73 -- | Reduces default extension list (keeps unsupported extensions)
   74 reduceExtensions :: UnnamedModule -> Ghc [Extension]
   75 reduceExtensions moduleAST = do
   76   let defaults = map replaceDeprecated . collectDefaultExtensions $ moduleAST
   77       expanded = expandExtensions defaults
   78       (xs, ys) = partition isSupported expanded
   79 
   80   xs' <- flip execStateT SMap.empty . flip runReaderT xs . traverseModule $ moduleAST
   81   let filteredExts = nub . mergeImplied $ (determineExtensions xs' ++ ys)
   82   if any (`elem` filteredExts) [Cpp, TemplateHaskell, TemplateHaskellQuotes, QuasiQuotes]
   83     -- We can't say anything about generated code
   84     then return . mergeImplied $ defaults
   85     -- Merging is needed because there might be unsopported extensions
   86     -- that are implied by supported extensions (TypeFamilies -> MonoLocalBinds)
   87     else return . sortBy (compare `on` show) $ filteredExts
   88 
   89 -- | Collect the required extensions in a module and returns a markers associated with them
   90 extensionMarkers :: UnnamedModule -> Ghc [Marker]
   91 extensionMarkers = fmap (concatMap toMarkers . SMap.toList) . collectExtensions
   92   where toMarkers (rel, occs) = map (toMarker rel) occs
   93         toMarker  rel occ     = Marker (unOcc occ) Info (showWithLevel rel occ)
   94         showWithLevel rel occ = (head . words . show $ occ) ++ ": " ++ prettyPrintFormula rel
   95 
   96 -- | Collects extensions induced by the source code (with location info)
   97 collectExtensions :: UnnamedModule -> Ghc ExtMap
   98 collectExtensions = collectExtensionsWith traverseModule
   99 
  100 
  101 -- | Collects the required extensions from a module using the given traversal method
  102 collectExtensionsWith :: CheckNode UnnamedModule -> UnnamedModule -> Ghc ExtMap
  103 collectExtensionsWith trvModule moduleAST = do
  104   let expanded = expandExtensions . collectDefaultExtensions $ moduleAST
  105   flip execStateT SMap.empty . flip runReaderT expanded . trvModule $ moduleAST
  106 
  107 
  108 -- | Expands every extension in a list, while not producing any duplicates.
  109 expandExtensions :: [Extension] -> [Extension]
  110 expandExtensions = nub . concatMap expandExtension
  111 
  112 -- | Collects extensions enabled by default
  113 collectDefaultExtensions :: UnnamedModule -> [Extension]
  114 collectDefaultExtensions = mapMaybe toExt . getExtensions
  115 
  116 -- | Collects extensions enabled by default
  117 collectTurnedOffExtensions :: UnnamedModule -> [Extension]
  118 collectTurnedOffExtensions = mapMaybe (toExt . drop 2)
  119                            . filter (isPrefixOf "No")
  120                            . getExtensions
  121 
  122 -- | Collects the string representation of the extensions in the module
  123 getExtensions :: UnnamedModule -> [String]
  124 getExtensions = flip (^?) (filePragmas & annList & lpPragmas & annList & langExt)
  125 
  126 toExt :: String -> Maybe Extension
  127 toExt str = case map fst . reads . canonExt . takeWhile isAlphaNum $ str of
  128               e:_ -> Just e
  129               []  -> fail $ "Extension '" ++ takeWhile isAlphaNum str ++ "' is not known."
