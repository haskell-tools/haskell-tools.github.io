   14 gblChkCPP :: CheckNode Module
   15 gblChkCPP = conditional gblChkCPP' Cpp
   16 
   17 -- TODO (after GHC upgrade): Rewrite this with mgLookupModule
   18 gblChkCPP' :: CheckNode Module
   19 gblChkCPP' m = do
   20   ms <- getModSummary . moduleName . semanticsModule $ m
   21   let cppSrc = preprocessedSrc ms
   22   case ml_hs_file . ms_location $ ms of
   23     Just fp -> do
   24       let lines'  = lines8 . fromString
   25           cppSrc' = lines' . rmDefaultIncludes fp $ cppSrc
   26           origSrc = lines' . prettyPrint $ m
   27       when (cppSrc' /= origSrc) (addEvidenceLoc Cpp noSrcSpan)
   28     _ -> addEvidenceLoc Cpp noSrcSpan
   29   return m
   30 
   31 -- CPP inserts a line before and after the default includes,
   32 -- that contain the filepath of the module.
   33 -- We want to remove these lines, and everything between them.
   34 rmDefaultIncludes :: FilePath -> String -> String
   35 rmDefaultIncludes fp = unlines . tail
   36                      . dropWhile (not . (fp `isSubsequenceOf`))
   37                      . tail . lines
   38 
   39 -- | Returns the preprocessed source code.
   40 -- If it is not present, it returns an empty string.
   41 preprocessedSrc :: ModSummary -> String
   42 preprocessedSrc = maybe "" strBufToStr . ms_hspp_buf
   43 
   44 strBufToStr :: StringBuffer -> String
   45 strBufToStr sb@(StringBuffer _ len _) = lexemeToString sb len
