   18 -- Free variables of a type, retaining repetitions, and expanding synonyms
   19 fvType :: Type -> [TyCoVar]
   20 fvType ty | Just exp_ty <- tcView ty = fvType exp_ty
   21 fvType (TyVarTy tv)          = [tv]
   22 fvType (TyConApp _ tys)      = fvTypes tys
   23 fvType LitTy{}               = []
   24 fvType (AppTy fun arg)       = fvType fun ++ fvType arg
   25 fvType (FunTy arg res)       = fvType arg ++ fvType res
   26 fvType (ForAllTy (TvBndr tv _) ty)
   27   = fvType (tyVarKind tv) ++
   28     filter (/= tv) (fvType ty)
   29 fvType (CastTy ty co)        = fvType ty ++ fvCo co
   30 fvType (CoercionTy co)       = fvCo co
   31 
   32 fvTypes :: [Type] -> [TyVar]
   33 fvTypes = concatMap fvType
   34 
   35 fvCo :: Coercion -> [TyCoVar]
   36 fvCo (Refl _ ty)            = fvType ty
   37 fvCo (TyConAppCo _ _ args)  = concatMap fvCo args
   38 fvCo (AppCo co arg)         = fvCo co ++ fvCo arg
   39 fvCo (ForAllCo tv h co)     = filter (/= tv) (fvCo co) ++ fvCo h
   40 fvCo (FunCo _ co1 co2)      = fvCo co1 ++ fvCo co2
   41 fvCo (CoVarCo v)            = [v]
   42 fvCo (AxiomInstCo _ _ args) = concatMap fvCo args
   43 fvCo (UnivCo p _ t1 t2)     = fvProv p ++ fvType t1 ++ fvType t2
   44 fvCo (SymCo co)             = fvCo co
   45 fvCo (TransCo co1 co2)      = fvCo co1 ++ fvCo co2
   46 fvCo (NthCo _ co)           = fvCo co
   47 fvCo (LRCo _ co)            = fvCo co
   48 fvCo (InstCo co arg)        = fvCo co ++ fvCo arg
   49 fvCo (CoherenceCo co1 co2)  = fvCo co1 ++ fvCo co2
   50 fvCo (KindCo co)            = fvCo co
   51 fvCo (SubCo co)             = fvCo co
   52 fvCo (AxiomRuleCo _ cs)     = concatMap fvCo cs
   53 
   54 fvProv :: UnivCoProvenance -> [TyCoVar]
   55 fvProv UnsafeCoerceProv    = []
   56 fvProv (PhantomProv co)    = fvCo co
   57 fvProv (ProofIrrelProv co) = fvCo co
   58 fvProv (PluginProv _)      = []

   60 sizeType :: Type -> Int
   61 -- Size of a type: the number of variables and constructors
   62 sizeType ty | Just exp_ty <- tcView ty = sizeType exp_ty
   63 sizeType TyVarTy{}         = 1
   64 sizeType (TyConApp _ tys)  = sizeTypes tys + 1
   65 sizeType LitTy{}           = 1
   66 sizeType (AppTy fun arg)   = sizeType fun + sizeType arg
   67 sizeType (FunTy arg res)   = sizeType arg + sizeType res + 1
   68 sizeType (ForAllTy _ ty)   = sizeType ty
   69 sizeType (CastTy ty _)     = sizeType ty
   70 sizeType (CoercionTy _)    = 1
   71 
   72 sizeTypes :: [Type] -> Int
   73 sizeTypes = sum . map sizeType

   76 checkInstTermination :: [TcType] -> ThetaType -> Bool
   77 checkInstTermination tys = checkPreds
   78   where
   79    headFvs  = fvTypes tys
   80    headSize = sizeTypes tys
   81 
   82    checkPreds :: [PredType] -> Bool
   83    checkPreds = any check
   84 
   85    check :: PredType -> Bool
   86    check predTy
   87      = case classifyPredType predTy of
   88          EqPred {}    -> False
   89          IrredPred {} -> check2 predTy (sizeType predTy)
   90          ClassPred cls tys
   91            | isTerminatingClass cls -> False
   92            | isCTupleClass cls -> checkPreds tys
   93            | otherwise
   94            -> check2 predTy (sizeTypes . filterOutInvisibleTypes (classTyCon cls) $ tys)
   95 
   96    check2 predTy predSize = not (null badTvs) || predSize >= headSize
   97      where badTvs = fvType predTy \\ headFvs
   98      -- Tyvars occurring more often in the context than in the head
   99 
  100 isTerminatingClass :: Class -> Bool
  101 isTerminatingClass cls = isIPClass cls
  102   || cls `hasKey` typeableClassKey
  103   || cls `hasKey` coercibleTyConKey
  104   || cls `hasKey` eqTyConKey
  105   || cls `hasKey` heqTyConKey

  108 -- | Checks whether a Type has a type function application in its head.
  109 hasTyFunHead :: Type -> Bool
  110 hasTyFunHead ty = case tcSplitTyConApp_maybe ty of
  111                     Just (tc, _) -> isTypeFamilyTyCon tc
  112                     Nothing      -> False

  114 -- | Given the lhs type arguments and rhs tycon and its arguments
  115 -- of a type family instance, it determines whether it needs UndecidableInstances.
  116 checkFamEq :: [Type] -> [(TyCon, [Type])] -> Bool
  117 checkFamEq lhsTys = any check
  118   where
  119    size = sizeTypes lhsTys
  120    fvs  = fvTypes lhsTys
  121    check (_, tys)
  122      | bad_tvs <- fvTypes tys \\ fvs
  123      = not (all isTyFamFree tys) || not (null bad_tvs) || size <= sizeTypes tys
  124 
  125 isTyFamInst :: FamInst -> Bool
  126 isTyFamInst inst
  127   | SynFamilyInst <- fi_flavor inst = True
  128   | otherwise                       = False
