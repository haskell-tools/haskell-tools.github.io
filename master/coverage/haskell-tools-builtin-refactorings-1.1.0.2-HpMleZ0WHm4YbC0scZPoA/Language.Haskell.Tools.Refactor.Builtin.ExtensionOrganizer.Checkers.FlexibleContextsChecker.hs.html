   25 chkFlexibleContexts = conditional chkFlexibleContexts' FlexibleContexts

   27 chkFlexibleContexts' :: CheckNode Context
   28 chkFlexibleContexts' ctx@(Context a) = chkAssertion a >> return ctx

   30 chkAssertion :: CheckNode Assertion
   31 chkAssertion a@(ClassAssert con annTys) = do
   32   -- If a lookup fails, add MissingInformation FlexibleContexts
   33   let errDefault = addMI FlexibleContexts a >> return False
   34   isClass <- fromMaybeTM errDefault (isClassTyConNameM con)
   35   if not isClass || all hasTyVarHead (annTys ^. annListElems)
   36     then return a
   37     else addEvidence FlexibleContexts a
   38 chkAssertion a@(InfixAssert lhs op rhs)
   39   -- If the constraint is of form a ~ b, then we dont need to check
   40   | Just name <- semanticsName op
   41   , name == eqTyConName
   42   = return a
   43   -- If the constraint has only type variable heads, it doesn't need FlexibleContexts
   44   | hasTyVarHead lhs && hasTyVarHead rhs = return a
   45   -- In any other case, we keep FlexibleContexts
   46   | otherwise = addEvidence FlexibleContexts a
   47 chkAssertion a@(TupleAssert xs) = do
   48   mapM_ chkAssertion xs
   49   return a
   50 chkAssertion a@ImplicitAssert{} = return a

   55 chkFlexibleContextsDecl' :: CheckNode Decl
   56 chkFlexibleContextsDecl' d@(TypeDecl dh rhs) = do
   57   let ty = typeOrKindFromId . declHeadQName $ dh
   58   when (hasConstraintKind ty || returnsConstraintKind ty)
   59        (chkClassesInside rhs)
   60   return d
   61 chkFlexibleContextsDecl' d = return d

   63 -- | Checks whether all type class applications in a type synonym rhs
   64 -- (that has kind Constraint) have only type variable heads.
   65 -- Returns False if a lookup inside is not succesful.
   66 -- If it isn't applied to a type that has kind Constraint, it may give
   67 -- false positive results.
   68 chkClassesInside :: Type -> ExtMonad ()
   69 chkClassesInside (TupleType annTys) =
   70   mapM_ chkClassesInside (annTys ^. annListElems)
   71 chkClassesInside t = maybe (addFC t) f (splitTypeAppMaybe t)
   72   where
   73     addFC :: Type -> ExtMonad ()
   74     addFC = addEvidence_ FlexibleContexts
   75 
   76     f :: (GHC.Name, [Type]) -> ExtMonad ()
   77     f (n,args) = do
   78        isClass <- isJustT . lookupClass $ n
   79        when (n /= eqTyConName && isClass && any (not . hasTyVarHead) args)
   80             (addFC t)
   81 
   82     -- Separates the type into its head and its arguments.
   83     -- This functions should only be applied to types that have kind Constraint.
   84     splitTypeAppMaybe :: Type -> Maybe (GHC.Name, [Type])
   85     splitTypeAppMaybe (TypeApp f arg) = do
   86       (f', args) <- splitTypeAppMaybe f
   87       return (f', args ++ [arg])
   88     splitTypeAppMaybe (InfixTypeApp lhs op rhs) = do
   89       opName <- semanticsName op
   90       return (opName, [lhs,rhs])
   91     splitTypeAppMaybe (VarType n) = do
   92       sname <- semanticsName n
   93       return (sname, [])
   94     splitTypeAppMaybe (ParenType t) = splitTypeAppMaybe t
   95     splitTypeAppMaybe (KindedType t _) = splitTypeAppMaybe t
   96     splitTypeAppMaybe _ = Nothing

   99 chkQNameForFC :: QualifiedName -> ExtMonad Bool
  100 chkQNameForFC name = do
  101   mty <- runMaybeT . lookupTypeFromId $ name
  102   case mty of
  103     Just ty -> do
  104       -- First we get all subtypes of the given type, while expanding all type synonyms,
  105       -- then we remove all implicit predicates,
  106       -- and partition the types into predicate types and theta types.
  107       let expanded = expandTypeSynonyms ty
  108           subTys = universeBi expanded :: [GHC.Type]
  109           (preds, tys) = partition isPredTy . filter (not . isIPPred) $ subTys
  110 
  111           -- We don't need type equalities, so we get rid of them and their children.
  112           -- Then we extract the constraints from the types.
  113           nomEqs = universeBi (filter isNominalEqPred preds) :: [GHC.Type]
  114           preds' = preds \\ nomEqs
  115           thetas = concatMap (snd' . tcSplitNestedSigmaTys) tys
  116 
  117           -- We split each theta type into its class tycon and its arguments.
  118           -- We also filter out tuple constraints
  119           -- (we already have the constraints in it).
  120           xs = mapMaybe getClassPredTys_maybe (thetas ++ preds')
  121           xs' = filter (not . isCTupleClass . fst) xs
  122 
  123       if all (uncurry hasValidClsArgs) xs' then return False
  124                                            else return True
  125     Nothing -> return False
  126   where
  127     hasValidClsArgs :: GHC.Class -> [GHC.Type] -> Bool
  128     hasValidClsArgs cls args = cls `hasKey` eqTyConKey
  129                             || isIPClass cls
  130                             || checkValidClsArgs False cls args
  131 
  132     snd' :: (a,b,c) -> b
  133     snd' (_,x,_) = x
  134 
  135     isNominalEqPred :: GHC.Type -> Bool
  136     isNominalEqPred ty
  137       | Just (tc,_) <- splitTyConApp_maybe ty = tc `hasKey` eqTyConKey
  138       | otherwise = False

  140 gblChkQNamesForFC' :: CheckNode Module
  141 gblChkQNamesForFC' m = do
  142   -- Names appearing on the rhs of bindings are just hints,
  143   -- they don't necessarily require FlexibleContexts.
  144   let allNames   = universeBi (m ^. modDecl) :: [QualifiedName]
  145       rhs        = universeBi (m ^. modDecl) :: [Rhs]
  146       hints      = universeBi rhs            :: [QualifiedName]
  147 
  148       -- Separating hints from evidence,
  149       -- and grouping them together based on their GHC.Name
  150       evidence  = filter (isJust . semanticsName) (allNames \\ hints)
  151       hints'    = filter (isJust . semanticsName) hints
  152       groupedEs = equivalenceGroupsBy semanticsName evidence
  153       groupedHs = equivalenceGroupsBy semanticsName hints'
  154 
  155   -- Checking the representative elements, whether they need FC,
  156   -- if they do, add occurences for every node in their group.
  157   es <- filterM (chkQNameForFC . fst) groupedEs
  158   hs <- filterM (chkQNameForFC . fst) groupedHs
  159   mapM_ (addEvidence FlexibleContexts) (concatMap snd es)
  160   mapM_ (addHint     FlexibleContexts) (concatMap snd hs)
  161 
  162   return m
