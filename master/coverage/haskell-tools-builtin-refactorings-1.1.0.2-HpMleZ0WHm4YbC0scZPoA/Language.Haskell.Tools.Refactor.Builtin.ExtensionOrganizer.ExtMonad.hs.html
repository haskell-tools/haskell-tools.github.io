  112 isTurnedOn :: Extension -> ExtMonad Bool
  113 isTurnedOn ext = do
  114   defaults <- ask
  115   return $! ext `elem` defaults
  116 
  117 isTurnedOff :: Extension -> ExtMonad Bool
  118 isTurnedOff ext = not <$> isTurnedOn ext
  119 
  120 conditional :: (node -> ExtMonad node) ->
  121                Extension ->
  122                node ->
  123                ExtMonad node
  124 conditional checker ext = conditionalAny checker [ext]
  125 
  126 conditionalNot :: (node -> ExtMonad node) ->
  127                   Extension ->
  128                   node ->
  129                   ExtMonad node
  130 conditionalNot checker ext node = do
  131   b <-isTurnedOn ext
  132   if b then return node else checker node
  133 
  134 conditionalAny :: (node -> ExtMonad node) ->
  135                    [Extension] ->
  136                    node ->
  137                    ExtMonad node
  138 conditionalAny checker exts node = do
  139   bs <- mapM isTurnedOn exts
  140   if or bs then checker node else return node
  141 
  142 conditionalAdd :: HasRange node => Extension -> node -> ExtMonad node
  143 conditionalAdd ext = conditional (addEvidence ext) ext
