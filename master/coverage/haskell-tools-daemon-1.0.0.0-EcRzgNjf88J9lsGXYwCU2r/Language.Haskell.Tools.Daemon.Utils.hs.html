   16 -- | Get the name of a module collection from its Id.
   17 moduleCollectionPkgId :: ModuleCollectionId -> Maybe String
   18 moduleCollectionPkgId (DirectoryMC _) = Nothing
   19 moduleCollectionPkgId (LibraryMC id) = Just id
   20 moduleCollectionPkgId (ExecutableMC id _) = Just id
   21 moduleCollectionPkgId (TestSuiteMC id _) = Just id
   22 moduleCollectionPkgId (BenchmarkMC id _) = Just id

   26 -- | Find the module collection where the given module is contained. Based on module name.
   27 lookupModuleColl :: String -> [ModuleCollection SourceFileKey] -> Maybe (ModuleCollection SourceFileKey)
   28 lookupModuleColl moduleName = find (any ((moduleName ==) . (^. sfkModuleName)) . Map.keys . (^. mcModules))
   29 
   30 -- | Find the module collection where the given module is contained. Based on source file name.
   31 lookupSourceFileColl :: FilePath -> [ModuleCollection SourceFileKey] -> Maybe (ModuleCollection SourceFileKey)
   32 lookupSourceFileColl fp = find (any ((fp ==) . (^. sfkFileName)) . Map.keys . (^. mcModules))
   33 
   34 -- | Find the module collection where the given module is contained. Based on source file name and module name.
   35 lookupModuleCollection :: ModSummary -> [ModuleCollection SourceFileKey] -> Maybe (ModuleCollection SourceFileKey)
   36 lookupModuleCollection ms mcs = lookupSourceFileColl (getModSumOrig ms) mcs <|> lookupModuleColl (getModSumName ms) mcs
   37 
   38 -- | Find the module with the given name. Based on module name.
   39 lookupModuleInSCs :: String -> [ModuleCollection SourceFileKey] -> Maybe (SourceFileKey, ModuleRecord)
   40 lookupModuleInSCs moduleName = find ((moduleName ==) . (^. sfkModuleName) . fst) . concatMap (Map.assocs . (^. mcModules))
   41 
   42 -- | Find the module with the given name. Based on source file name.
   43 lookupModInSCs :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Maybe (SourceFileKey, ModuleRecord)
   44 lookupModInSCs key = find (((key ^. sfkFileName) ==) . (^. sfkFileName) . fst) . concatMap (Map.assocs . (^. mcModules))
   45 
   46 -- | Get the module with the given name and source file key.
   47 lookupSFKInSCs :: SourceFileKey -> [ModuleCollection SourceFileKey] -> Maybe ModuleRecord
   48 lookupSFKInSCs key = listToMaybe . catMaybes . map (Map.lookup key . (^. mcModules))
   49 
   50 -- | Remove a module with the given name. Based on module name.
   51 removeModule :: String -> [ModuleCollection SourceFileKey] -> [ModuleCollection SourceFileKey]
   52 removeModule moduleName = map (mcModules .- Map.filterWithKey (\k _ -> moduleName /= (k ^. sfkModuleName)))
   53 
   54 -- | Remove a module with the given name from a module collection. Based on module name and file path.
   55 removeModuleMS :: ModSummary -> Map.Map SourceFileKey ModuleRecord -> Map.Map SourceFileKey ModuleRecord
   56 removeModuleMS ms = Map.filterWithKey (\k _ -> stay k)
   57   where stay k = getModSumName ms /= (k ^. sfkModuleName) || (fn /= getModSumOrig ms && fn /= "")
   58           where fn = k ^. sfkFileName
   59 
   60 -- | Check if the given module needs code generation. Finds the module if no source file name is
   61 -- present and module names check or if both module names and source file names check.
   62 needsGeneratedCode :: SourceFileKey -> [ModuleCollection SourceFileKey] -> CodeGenPolicy
   63 needsGeneratedCode key mcs = fromMaybe NoCodeGen 
   64                                $ (^? modRecCodeGen)
   65                                    =<< (lookupSFKInSCs key mcs <|> lookupSFKInSCs (sfkFileName .= "" $ key) mcs)
   66 
   67 -- | Marks the given module for code generation. Finds the module if no source file name is
   68 -- present and module names check or if both module names and source file names check.
   69 -- Idempotent operation.
   70 codeGeneratedFor :: SourceFileKey -> CodeGenPolicy -> [ModuleCollection SourceFileKey] -> [ModuleCollection SourceFileKey]
   71 codeGeneratedFor key codeGen = map (mcModules .- Map.adjust (modRecCodeGen .= codeGen) (sfkFileName .= "" $ key) 
   72                                  . Map.adjust (modRecCodeGen .= codeGen) key)
   73 
   74 -- | Check if the given module has been already loaded. Based on both module name and source
   75 -- file name.
   76 isAlreadyLoaded :: SourceFileKey -> CodeGenPolicy -> [ModuleCollection SourceFileKey] -> Bool
   77 isAlreadyLoaded key codeGen = maybe False (\(_, mc) -> isLoaded mc && (mc ^? modRecCodeGen) < Just codeGen)
   78                                 . find ((key ==) . fst) . concatMap (Map.assocs . (^. mcModules))
   79 
   80 -- | Insert a module with a source file key to our database if it wasn't there already
   81 insertIfMissing :: SourceFileKey -> [ModuleCollection SourceFileKey] -> [ModuleCollection SourceFileKey]
   82 insertIfMissing sfk mods
   83   = case lookupSFKInSCs sfk mods of
   84       Just _ -> mods
   85       Nothing -> element 0 & mcModules .- Map.insert sfk (ModuleNotLoaded NoCodeGen False) $ mods
