   40 -- | Load packages from the given directories. Loads modules, performs the given callback action, warns for duplicate modules.
   41 loadPackagesFrom :: (ModSummary -> IO ())
   42                       -> ([ModSummary] -> IO ())
   43                       -> (DaemonSessionState -> FilePath -> IO [FilePath])
   44                       -> [FilePath]
   45                       -> DaemonSession [SourceError]
   46 loadPackagesFrom report loadCallback additionalSrcDirs packages =
   47   do -- collecting modules to load
   48      modColls <- liftIO $ getAllModules packages
   49      st <- get
   50      moreSrcDirs <- liftIO $ mapM (additionalSrcDirs st) packages
   51      lift $ useDirs ((modColls ^? traversal & mcSourceDirs & traversal) ++ concat moreSrcDirs)
   52      mcs' <- liftIO (traversal !~ locateModules $ modColls)
   53      modify' (refSessMCs .- (++ mcs'))
   54      mcs <- gets (^. refSessMCs)
   55      let alreadyLoadedFiles
   56            = concatMap (map (^. sfkFileName) . Map.keys . Map.filter (isJust . (^? typedRecModule)) . (^. mcModules))
   57                        (filter (\mc -> (mc ^. mcRoot) `notElem` packages) mcs)
   58      currentTargets <- map targetId <$> (lift getTargets)
   59      lift $ mapM_ (\t -> when (targetId t `notElem` currentTargets) (addTarget t))
   60                   (map makeTarget $ List.nubBy ((==) `on` (^. sfkFileName))
   61                                   $ List.sort $ concatMap getExposedModules mcs')
   62      loadRes <- gtry (loadModules mcs alreadyLoadedFiles)
   63      case loadRes of
   64        Right mods -> do 
   65          modify (refSessMCs & traversal & filtered (\mc -> (mc ^. mcId) `elem` map (^. mcId) modColls) & mcLoadDone .= True)
   66          compileModules report mods
   67        Left err -> return [err]
   68 
   69   where getExposedModules :: ModuleCollection k -> [k]
   70         getExposedModules
   71           = Map.keys . Map.filter (\v -> fromMaybe True (v ^? recModuleExposed)) . (^. mcModules)
   72 
   73         locateModules :: ModuleCollection ModuleNameStr -> IO (ModuleCollection SourceFileKey)
   74         locateModules mc
   75           = mcModules !~ ((Map.fromList <$>)
   76                             . mapM (locateModule (mc ^. mcSourceDirs) (mc ^. mcModuleFiles))
   77                             . Map.assocs) $ mc
   78 
   79         locateModule :: [FilePath] -> [(ModuleNameStr, FilePath)]
   80                           -> (ModuleNameStr, ModuleRecord) -> IO (SourceFileKey,ModuleRecord)
   81         locateModule srcDirs modMaps (modName, record)
   82           = do candidate <- createTargetCandidate srcDirs modMaps modName
   83                return (SourceFileKey (either (const "") id candidate) modName, record)
   84 
   85         -- | Creates a possible target from a module name. If possible, finds the
   86         -- corresponding source file to distinguish between modules of the same name.
   87         createTargetCandidate :: [FilePath] -> [(ModuleNameStr, FilePath)] -> ModuleNameStr
   88                                     -> IO (Either ModuleName FilePath)
   89         createTargetCandidate srcFolders mapping modName
   90           = wrapEither <$> filterM doesFileExist
   91                              (map (</> toFileName modName) srcFolders)
   92           where toFileName modName
   93                   = case lookup modName mapping of
   94                       Just fileName -> fileName
   95                       Nothing -> List.intercalate [pathSeparator] (splitOn "." modName) <.> "hs"
   96                 wrapEither [] = Left (GHC.mkModuleName modName)
   97                 wrapEither (fn:_) = Right fn
   98 
   99         makeTarget (SourceFileKey "" modName) = Target (TargetModule (GHC.mkModuleName modName)) True Nothing
  100         makeTarget (SourceFileKey filePath _) = Target (TargetFile filePath Nothing) True Nothing
  101 
  102         loadModules mcs alreadyLoaded = do
  103           mods <- withLoadFlagsForModules mcs $ do
  104             loadVisiblePackages -- need to update package state when setting the list of visible packages
  105             modsForColls <- lift $ depanal [] True
  106             let modsToParse = flattenSCCs $ topSortModuleGraph False modsForColls Nothing
  107                 actuallyCompiled = filter (\ms -> getModSumOrig ms `notElem` alreadyLoaded) modsToParse
  108             modify' (refSessMCs .- foldl (.) id (map (insertIfMissing . keyFromMS) actuallyCompiled))
  109             return actuallyCompiled
  110           liftIO $ loadCallback mods
  111           return mods
  112 
  113         compileModules report mods = do
  114             checkEvaluatedMods mods
  115             compileWhileOk mods
  116           where compileWhileOk [] = return []
  117                 compileWhileOk (mod:mods) 
  118                   = do res <- gtry (reloadModule report mod)
  119                        case res of
  120                           Left err -> do dependents <- lift $ dependentModules (return . (ms_mod mod ==) . ms_mod)
  121                                          (err :) <$> compileWhileOk (filter ((`notElem` map ms_mod dependents) . ms_mod) mods)
  122                           Right _ -> compileWhileOk mods
  123

  125 -- | Loads the packages that are declared visible (by .cabal file).
  126 loadVisiblePackages :: DaemonSession ()
  127 loadVisiblePackages = do
  128   dfs <- getSessionDynFlags
  129   (dfs', _) <- liftIO $ initPackages dfs
  130   setSessionDynFlags dfs' -- set the package flags (only for this load session)
  131   modify' (pkgDbFlags .= \dfs -> dfs { pkgDatabase = pkgDatabase dfs'
  132                                      , pkgState = pkgState dfs'
  133                                      }) -- save the package database

  135 -- | Get the module that is selected for refactoring and all the other modules.
  136 getFileMods :: String -> DaemonSession ( Maybe (SourceFileKey, UnnamedModule)
  137                                        , [(SourceFileKey, UnnamedModule)] )
  138 getFileMods fnameOrModule = do
  139   modMaps <- gets (^? refSessMCs & traversal & mcModules)
  140   let modules = mapMaybe (\(k,m) -> (\ms tc -> (ms, (k,tc))) <$> (m ^? modRecMS) <*> (m ^? typedRecModule)) -- not type checkable modules are ignored
  141                   $ concatMap @[] Map.assocs modMaps
  142       (modSel, modOthers) = List.partition (\(ms,_) -> getModSumName ms == fnameOrModule
  143                                                          && (case ms_hsc_src ms of HsSrcFile -> True; _ -> False))
  144                                           modules
  145       maxSufLength = maximum $ map sufLength modules
  146       (fnSel, fnOthers) = if null modules || maxSufLength == 0
  147                             then ([], modules)
  148                             else List.partition ((== maxSufLength) . sufLength) modules
  149       sufLength = length . commonSuffix (splitPath fnameOrModule) . splitPath . getModSumOrig . fst
  150       commonSuffix l1 l2 = takeWhile (uncurry (==)) $ zip (reverse l1) (reverse l2)
  151       backup = case fnSel of
  152                  []      -> return (Nothing, map snd fnOthers)
  153                  [(_,m)] -> return (Just m, map snd fnOthers)
  154                  _:_     -> error "getFileMods: multiple modules selected"
  155   case modSel of
  156     []      -> backup
  157     [(_,m)] -> return (Just m, map snd modOthers)
  158     _:_     -> backup

  160 -- | Reload the modules that have been changed (given by predicate). Pefrom the callback.
  161 reloadChangedModules :: (ModSummary -> IO a) -> ([ModSummary] -> IO ()) -> (ModSummary -> Bool)
  162                            -> DaemonSession [a]
  163 reloadChangedModules report loadCallback isChanged = do
  164   reachable <- getReachableModules loadCallback isChanged
  165   checkEvaluatedMods reachable
  166   -- remove module from session before reloading it, resolves space leak
  167   clearModules reachable
  168   mapM (reloadModule report) reachable
  169 
  170 -- | Clears the given modules from the GHC state to enable re-loading them
  171 -- From the Haskell-tools state we only clear them individually, when their module collection is determined.
  172 clearModules :: [ModSummary] -> DaemonSession ()
  173 clearModules [] = return ()
  174 clearModules mods = do
  175   let reachableMods = map ms_mod_name mods
  176       notReloaded = (`notElem` reachableMods) . GHC.moduleName . mi_module . hm_iface
  177   env <- getSession
  178   let hptStay = filterHpt notReloaded (hsc_HPT env)
  179   -- clear the symbol cache for iserv
  180   liftIO $ purgeLookupSymbolCache env
  181   -- clear the global linker state
  182   dfs <- getSessionDynFlags
  183   when (not $ gopt Opt_ExternalInterpreter dfs) $
  184     liftIO $ unload env (mapMaybe hm_linkable (eltsHpt hptStay))
  185   -- clear name cache
  186   nameCache <- liftIO $ readIORef $ hsc_NC env
  187   let nameCache' = nameCache { nsNames = delModuleEnvList (nsNames nameCache) (map ms_mod mods) }
  188   liftIO $ writeIORef (hsc_NC env) nameCache'
  189   -- clear home package table and module graph
  190   lift $ modifySession (\s -> s { hsc_HPT = hptStay
  191                                 , hsc_mod_graph = filter ((`notElem` reachableMods) . ms_mod_name) (hsc_mod_graph s)
  192                                 })

  194 -- | Get all modules that can be accessed from a given set of modules. Can be used to select which
  195 -- modules need to be reloaded after a change.
  196 getReachableModules :: ([ModSummary] -> IO ()) -> (ModSummary -> Bool) -> DaemonSession [ModSummary]
  197 getReachableModules loadCallback selected = do
  198   mcs <- gets (^. refSessMCs)
  199   withLoadFlagsForModules mcs $ do
  200     lift $ depanal [] True
  201     sortedRecompMods <- lift $ dependentModules (return . selected)
  202     liftIO $ loadCallback sortedRecompMods
  203     return sortedRecompMods

  205 -- | Reload a given module. Perform a callback.
  206 reloadModule :: (ModSummary -> IO a) -> ModSummary -> DaemonSession a
  207 reloadModule report ms = do
  208   mcs <- gets (^. refSessMCs)
  209   ghcfl <- gets (^. ghcFlagsSet)
  210   let codeGen = needsGeneratedCode (keyFromMS ms) mcs
  211       mc = decideMC ms mcs
  212   newm <- withFlagsForModule mc $ lift $ do
  213     dfs <- liftIO $ fmap ghcfl $ mc ^. mcFlagSetup $ ms_hspp_opts ms
  214     let ms' = ms { ms_hspp_opts = dfs }
  215     -- some flags are cached in mod summary, so we need to override
  216     parseTyped (case codeGen of NoCodeGen -> ms'
  217                                 InterpretedCode -> forceCodeGen ms'
  218                                 GeneratedCode -> forceAsmGen ms')
  219   -- replace the module in the program database
  220   modify' $ refSessMCs & traversal & filtered (\c -> (c ^. mcId) == (mc ^. mcId)) & mcModules
  221               .- Map.insert (keyFromMS ms) (ModuleTypeChecked newm ms codeGen)
  222                    . removeModuleMS ms
  223   liftIO $ report ms

  225 -- | Select which module collection we think the module is in
  226 decideMC :: ModSummary -> [ModuleCollection SourceFileKey] -> ModuleCollection SourceFileKey
  227 decideMC ms mcs =
  228   case lookupModuleCollection ms mcs of
  229     Just mc -> mc
  230     Nothing -> case filter (\mc -> (mc ^. mcRoot) `List.isPrefixOf` fileName) mcs of
  231                  mc:_ -> mc
  232                  _ -> case mcs of mc:_ -> mc
  233                                   []   -> error "reloadModule: module collections empty"
  234   where fileName = getModSumOrig ms

  236 -- | Prepares the DynFlags for the compilation of a module
  237 withFlagsForModule :: ModuleCollection SourceFileKey -> DaemonSession a -> DaemonSession a
  238 withFlagsForModule mc action = do
  239   ghcfl <- gets (^. ghcFlagsSet)
  240   dbFlags <- gets (^. pkgDbFlags)
  241   -- IMPORTANT: make sure that the module collection is not passed into the flags, they
  242   -- might not be evaluated and then the reference could prevent garbage collection
  243   -- of entire ASTs
  244   withAlteredDynFlags (liftIO . fmap (dbFlags . ghcfl) . ((mc ^. mcFlagSetup) <=< (mc ^. mcLoadFlagSetup))) action
  245 
  246 -- | Prepares the DynFlags for travesing the module graph
  247 withLoadFlagsForModules :: [ModuleCollection SourceFileKey] -> DaemonSession a -> DaemonSession a
  248 -- IMPORTANT: make sure that a module collection is not passed into the flags, they
  249 -- might not be evaluated and then the reference could prevent garbage collection
  250 -- of entire ASTs
  251 withLoadFlagsForModules mcs action = do
  252   ghcfl <- gets (^. ghcFlagsSet)
  253   dbFlags <- gets (^. pkgDbFlags)
  254   withAlteredDynFlags (liftIO . fmap (dbFlags . ghcfl)
  255                               . setupLoadFlags (mcs ^? traversal & mcId) (mcs ^? traversal & mcRoot)
  256                                                (mcs ^? traversal & mcDependencies & traversal)
  257                                                (foldl @[] (>=>) return (mcs ^? traversal & mcLoadFlagSetup))) action

  259 -- | Finds out if a newly added module forces us to generate code for another one.
  260 -- If the other is already loaded it will be reloaded.
  261 checkEvaluatedMods :: [ModSummary] -> DaemonSession ()
  262 checkEvaluatedMods changed = do
  263     mcs <- gets (^. refSessMCs)
  264     -- IMPORTANT: make sure that the module collection is not passed into the flags, they
  265     -- might not be evaluated and then the reference could prevent garbage collection
  266     -- of entire ASTs
  267     let lookupFlags ms = maybe return (^. mcFlagSetup) mc $ ms_hspp_opts ms
  268           where mc = lookupModuleCollection ms mcs
  269     (modsNeedCode, modsNeedAsm) <- lift (getEvaluatedMods changed lookupFlags)
  270     -- specify the need of code generation for later loading
  271     forM_ modsNeedCode (\ms -> modify $ refSessMCs .- codeGeneratedFor (keyFromMS ms) InterpretedCode)
  272     forM_ modsNeedAsm (\ms -> modify $ refSessMCs .- codeGeneratedFor (keyFromMS ms) GeneratedCode)
  273     let interpreted = filter (\ms -> isAlreadyLoaded (keyFromMS ms) InterpretedCode mcs) 
  274                              modsNeedCode
  275         codeGenerated = filter (\ms -> isAlreadyLoaded (keyFromMS ms) GeneratedCode mcs) modsNeedAsm
  276     clearModules (interpreted ++ codeGenerated)
  277     -- reload modules that have already been loaded
  278     forM_ interpreted (codeGenForModule mcs InterpretedCode)
  279     forM_ codeGenerated (codeGenForModule mcs GeneratedCode)
  280 
  281 -- | Re-load the module with code generation enabled. Must be used when the module had already been loaded,
  282 -- but code generation were not enabled by then.
  283 codeGenForModule :: [ModuleCollection SourceFileKey] -> CodeGenPolicy -> ModSummary -> DaemonSession ()
  284 codeGenForModule mcs codeGen ms
  285 -- we don't need to update anything, just re-compile (we don't store the typed AST) and generate the code
  286   = withFlagsForModule mc $ lift $ void $ parseTyped (case codeGen of InterpretedCode -> forceCodeGen ms
  287                                                                       GeneratedCode -> forceAsmGen ms
  288                                                                       _ -> ms)
  289   where mc = fromMaybe (error $ "codeGenForModule: The following module is not found: " ++ getModSumName ms)
  290                $ lookupModuleCollection ms mcs

  292 -- | Check which modules can be reached from the module, if it uses template haskell.
  293 -- A definition that needs code generation can be inside a module that does not uses the
  294 -- TemplateHaskell extension.
  295 getEvaluatedMods :: [ModSummary] -> (ModSummary -> IO DynFlags) -> Ghc ([ModSummary],[ModSummary])
  296 -- We cannot really get the modules that need to be linked, because we cannot rename splice content if the
  297 -- module is not type checked and that is impossible if the splice cannot be evaluated.
  298 getEvaluatedMods changed additionalFlags
  299   = do let changedModulePathes = map getModSumOrig changed
  300        -- some flags are stored only in the module collection and are not recorded in the summary
  301        eval <- supportingModules (\ms -> (\flags -> getModSumOrig ms `elem` changedModulePathes && TemplateHaskell `xopt` flags)
  302                                            <$> liftIO (additionalFlags ms))
  303        asm <- supportingModules (\ms -> (\flags -> getModSumOrig ms `elem` changedModulePathes 
  304                                                      && (StaticPointers `xopt` flags || UnboxedTuples `xopt` flags || UnboxedSums `xopt` flags))
  305                                            <$> liftIO (additionalFlags ms))
  306        let asmOrigs = map getModSumOrig asm
  307        return (filter (\ms -> getModSumOrig ms `notElem` asmOrigs) eval, asm)
