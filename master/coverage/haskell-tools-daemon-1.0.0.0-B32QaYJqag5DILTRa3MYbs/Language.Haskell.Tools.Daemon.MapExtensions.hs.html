   12 setExtensionFlag', unSetExtensionFlag' :: GHC.Extension -> DynFlags -> DynFlags
   13 setExtensionFlag' f dflags = foldr ($) (xopt_set dflags f) deps
   14   where
   15     deps = [ if turn_on then setExtensionFlag'   d
   16                         else unSetExtensionFlag' d
   17            | (f', turn_on, d) <- impliedXFlags, f' == f ]
   18 unSetExtensionFlag' f dflags = xopt_unset dflags f
