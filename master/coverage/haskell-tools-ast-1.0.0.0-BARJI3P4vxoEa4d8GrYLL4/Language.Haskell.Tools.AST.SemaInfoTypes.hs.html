  116 -- | Info for the module element
  117 data ModuleInfo n = ModuleInfo { _defModuleName :: GHC.Module
  118                                , _defDynFlags :: DynFlags -- ^ The compilation flags that are set up when the module was compiled
  119                                , _defIsBootModule :: Bool -- ^ True if this module is created from a hs-boot file
  120                                , _implicitNames :: [PName n] -- ^ implicitly imported names
  121                                , _prelOrphanInsts :: [ClsInst] -- ^ Class instances implicitly passed from Prelude.
  122                                , _prelFamInsts :: [FamInst] -- ^ Family instances implicitly passed from Prelude.
  123                                }
  124   deriving (Data, Functor, Foldable, Traversable)

  126 instance Data DynFlags where
  127   gunfold _ _ _ = error "Cannot construct dyn flags"
  128   toConstr _ = dynFlagsCon
  129   dataTypeOf _ = dynFlagsType
  130 
  131 dynFlagsType = mkDataType "DynFlags.DynFlags" [dynFlagsCon]
  132 dynFlagsCon = mkConstr dynFlagsType "DynFlags" [] Data.Prefix

  134 -- | Creates semantic information for the module element.
  135 -- Strict in the list of implicitely imported, orphan and family instances.
  136 mkModuleInfo :: GHC.Module -> DynFlags -> Bool -> [PName n] -> [ClsInst] -> [FamInst] -> ModuleInfo n
  137 -- the calculate of these fields involves a big parts of the GHC state and it causes a space leak
  138 -- if not evaluated strictly
  139 mkModuleInfo mod dfs boot !imported !orphan !family = ModuleInfo mod dfs boot imported orphan family

  141 -- | Info corresponding to an import declaration

  143                                , _availableNames :: [n] -- ^ Names available from the imported module
  144                                , _importedNames :: [PName n] -- ^ Names actually imported from the module.
  145                                , _importedOrphanInsts :: [ClsInst] -- ^ Class instances implicitly passed.
  146                                , _importedFamInsts :: [FamInst] -- ^ Family instances implicitly passed.
  147                                }
