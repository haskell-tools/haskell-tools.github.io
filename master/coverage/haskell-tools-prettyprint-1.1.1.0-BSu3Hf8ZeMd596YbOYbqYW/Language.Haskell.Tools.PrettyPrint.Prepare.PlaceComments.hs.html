   25 getNormalComments :: Map SrcSpan [Located AnnotationComment] -> Map.Map SrcSpan [Located AnnotationComment]
   26 getNormalComments = Map.map (filter (not . isPragma . unLoc))
   27 
   28 getPragmaComments :: Map SrcSpan [Located AnnotationComment] -> Map.Map String [Located String]
   29 getPragmaComments comms = Map.fromListWith (++) $ map (\(L l (AnnBlockComment str)) -> (getPragmaCommand str, [L l str]))
   30                                                 $ filter (isPragma . unLoc) $ concatMap snd $ Map.toList comms
   31   where getPragmaCommand = takeWhile (\c -> isAlphaNum c || c == '_') . dropWhile isSpace . drop 3
   32 
   33 isPragma :: AnnotationComment -> Bool
   34 isPragma (AnnBlockComment str) = take 3 str == "{-#" && take 3 (reverse str) == "}-#"
   35 isPragma _ = False

   37 -- | Puts comments in the nodes they should be attached to. Watches for lexical tokens
   38 -- that may divide the comment and the supposed element.
   39 -- Leaves the AST in a state where parent nodes does not contain all of their children.
   40 placeComments :: RangeInfo stage => Map ApiAnnKey [SrcSpan] -> Map.Map SrcSpan [Located AnnotationComment]
   41               -> Ann UModule dom stage -> Ann UModule dom stage
   42 placeComments tokens comms mod
   43   = resizeAnnots (Set.filter (\rng -> srcSpanStart rng /= srcSpanEnd rng) $ Set.fromList $ concat (Map.elems tokens))
   44       (concatMap (map nextSrcLoc . snd) (Map.toList cleanedComments)) mod
   45   where spans = allElemSpans mod
   46         sortedElemStarts = Set.fromList $ map srcSpanStart spans
   47         sortedElemEnds = Set.fromList $ map srcSpanEnd spans
   48         nextSrcLoc comm@(L sp _)
   49           = let after = fromMaybe noSrcLoc (Set.lookupLE (srcSpanStart sp) sortedElemEnds)
   50                 before = fromMaybe noSrcLoc (Set.lookupGE (srcSpanEnd sp) sortedElemStarts)
   51              in ((after,before),comm)
   52         cleanedComments = Map.map (map cleanComment) comms
   53         cleanComment (L loc (AnnLineComment txt))
   54           | last txt `elem` "\n\r" = L (mkSrcSpan (srcSpanStart loc) (decreaseCol (srcSpanEnd loc))) (AnnLineComment (init txt))
   55         cleanComment c = c
   56         decreaseCol (RealSrcLoc l) = mkSrcLoc (srcLocFile l) (srcLocLine l) (srcLocCol l - 1)
   57         decreaseCol l = l
   58 
   59 allElemSpans :: (SourceInfoTraversal node, RangeInfo stage) => Ann node dom stage -> [SrcSpan]
   60 allElemSpans = execWriter . sourceInfoTraverse (SourceInfoTrf (\ni -> tell [ni ^. nodeSpan] >> pure ni) pure pure)
   61 
   62 resizeAnnots :: RangeInfo stage => Set SrcSpan -> [((SrcLoc, SrcLoc), Located AnnotationComment)]
   63               -> Ann UModule dom stage
   64               -> Ann UModule dom stage
   65 resizeAnnots tokens comments elem
   66   = flip evalState comments $ flip runReaderT tokens $
   67         -- if a comment that could be attached to more than one documentable element (possibly nested)
   68         -- the order of different documentable elements here decide which will be chosen
   69 
   70         modImports&annList !~ expandAnnot -- expand imports to cover their comments
   71           >=> modDecl&annList !~ expandTopLevelDecl -- expand declarations to cover their comments
   72           >=> expandAnnot -- expand the module itself to cover its comments
   73       $ elem
   74 
   75 type ExpandType elem dom stage = Ann elem dom stage -> ReaderT (Set SrcSpan) (State [((SrcLoc, SrcLoc), Located AnnotationComment)]) (Ann elem dom stage)
   76 
   77 expandTopLevelDecl :: RangeInfo stage => ExpandType UDecl dom stage
   78 expandTopLevelDecl
   79   = declBody & annJust & cbElements & annList !~ expandClsElement
   80       >=> declCons & annList !~ expandConDecl
   81       >=> declGadt & annList !~ expandGadtConDecl
   82       >=> declTypeSig !~ expandTypeSig
   83       >=> expandAnnot
   84 
   85 expandTypeSig :: RangeInfo stage => ExpandType UTypeSignature dom stage
   86 expandTypeSig
   87   = tsType & typeParams !~ expandAnnot >=> expandAnnot
   88 
   89 expandClsElement :: RangeInfo stage => ExpandType UClassElement dom stage
   90 expandClsElement
   91   = ceTypeSig !~ expandTypeSig
   92       >=> ceBind !~ expandValueBind
   93       >=> expandAnnot
   94 
   95 expandValueBind :: RangeInfo stage => ExpandType UValueBind dom stage
   96 expandValueBind
   97   = valBindLocals & annJust & localBinds & annList !~ expandLocalBind
   98       >=> funBindMatches & annList & matchBinds & annJust & localBinds & annList !~ expandLocalBind
   99       >=> expandAnnot
  100 
  101 expandLocalBind :: RangeInfo stage => ExpandType ULocalBind dom stage
  102 expandLocalBind
  103   = localVal !~ expandValueBind
  104       >=> localSig !~ expandTypeSig
  105       >=> expandAnnot
  106 
  107 expandConDecl :: RangeInfo stage => ExpandType UConDecl dom stage
  108 expandConDecl
  109   = conDeclFields & annList !~ expandAnnot >=> expandAnnot
  110 
  111 expandGadtConDecl :: RangeInfo stage => ExpandType UGadtConDecl dom stage
  112 expandGadtConDecl
  113   = gadtConType & gadtConRecordFields & annList !~ expandAnnot >=> expandAnnot
  114 
  115 -- | Expands tree elements to contain the comments that should be attached to them.
  116 expandAnnot :: forall elem dom stage . RangeInfo stage => ExpandType elem dom stage
  117 expandAnnot elem
  118   = do let Just sp = elem ^? annotation&sourceInfo&nodeSpan
  119        tokens <- ask
  120        applicable <- lift $ gets (applicableComments tokens (srcSpanStart sp) (srcSpanEnd sp))
  121 
  122        -- this check is just for performance (quick return if no modification is needed)
  123        if not (null applicable) then do
  124          -- the new span is the original plus all the covered spans
  125          let newSp@(RealSrcSpan newSpan)
  126                = foldl combineSrcSpans (fromJust $ elem ^? nodeSp) (map (getLoc . snd) applicable)
  127          -- take out all comments that are now covered
  128          lift $ modify (filter (not . (\case RealSrcSpan s -> newSpan `containsSpan` s; _ -> True) . getLoc . snd))
  129          return $ nodeSp .= newSp $ elem
  130        else return elem
  131   where nodeSp :: Simple Partial (Ann elem dom stage) SrcSpan
  132         nodeSp = annotation&sourceInfo&nodeSpan
  133 
  134 -- This classification does not prefer inline comments to previous line comments, this is implicitly done
  135 -- by the order in which the elements are traversed.
  136 applicableComments :: Set SrcSpan -> SrcLoc -> SrcLoc
  137                    -> [((SrcLoc, SrcLoc), Located AnnotationComment)]
  138                    -> [((SrcLoc, SrcLoc), Located AnnotationComment)]
  139 applicableComments tokens start end = filter applicableComment
  140   where -- A comment that starts with | binds to the next documented element
  141         applicableComment ((_, before), L sp comm)
  142           | isCommentOnNext comm = before == start && noTokenBetween (srcSpanEnd sp) start
  143         -- A comment that starts with ^ binds to the previous documented element
  144         applicableComment ((after, _), L sp comm)
  145           | isCommentOnPrev comm = after == end && noTokenBetween end (srcSpanStart sp)
  146         -- All other comment binds to the previous definition if it is on the same line
  147         applicableComment ((after, _), L sp@(RealSrcSpan loc) _)
  148           | after == end && srcLocLine (realSrcSpanStart loc) == getLineLocDefault end = True
  149                          && noTokenBetween end (srcSpanStart sp)
  150         -- or the next one if that is on the next line and the columns line up
  151         applicableComment ((_, before), L sp@(RealSrcSpan loc) _)
  152           | before == start && srcLocLine (realSrcSpanEnd loc) + 1 == getLineLocDefault start
  153                             && srcLocCol (realSrcSpanStart loc) == getLineColDefault start
  154                             && noTokenBetween (srcSpanEnd sp) start
  155           = True
  156         applicableComment _ = False
  157 
  158         getLineLocDefault (RealSrcLoc l) = srcLocLine l
  159         getLineLocDefault _              = -1
  160 
  161         getLineColDefault (RealSrcLoc l) = srcLocCol l
  162         getLineColDefault _              = -1
  163 
  164         noTokenBetween start end
  165           = case Set.lookupGE (srcLocSpan start) tokens of
  166               Just tok -> srcSpanStart tok >= end
  167               Nothing -> True

  169 -- * GHC mistakenly parses -- ^ and -- | comments as simple line comments.
  170 -- These functions check if a given comment is attached to the previous or next comment.
  171 
  172 -- | Checks if a doc comment belongs to the next definition.
  173 isCommentOnNext :: AnnotationComment -> Bool
  174 isCommentOnNext (AnnDocCommentNext _) = True
  175 isCommentOnNext (AnnLineComment s) = firstNonspaceCharIs '|' s
  176 isCommentOnNext (AnnBlockComment s) = firstNonspaceCharIs '|' s
  177 isCommentOnNext _ = False
  178 
  179 -- | Checks if a doc comment belongs to the previous definition.
  180 isCommentOnPrev :: AnnotationComment -> Bool
  181 isCommentOnPrev (AnnDocCommentPrev _) = True
  182 isCommentOnPrev (AnnLineComment s) = firstNonspaceCharIs '^' s
  183 isCommentOnPrev (AnnBlockComment s) = firstNonspaceCharIs '^' s
  184 isCommentOnPrev _ = False
  185 
  186 -- the comment string contains the -- or {- characters
  187 firstNonspaceCharIs :: Char -> String -> Bool
  188 firstNonspaceCharIs c s = Just c == listToMaybe (dropWhile isSpace (drop 2 s))
