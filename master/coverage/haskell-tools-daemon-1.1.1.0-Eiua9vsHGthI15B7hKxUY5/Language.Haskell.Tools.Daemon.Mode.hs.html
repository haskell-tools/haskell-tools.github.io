    1 -- | Defines different working modes for the daemon. It can work by using a socket connection
    2 -- or channels to communicate with the client. When the daemon is used by CLI, it uses channel if
    3 -- it communicates with an editor plugin it uses the socket connection.

   18 -- | An abstraction over the connection to the client.
   19 data WorkingMode a = WorkingMode { daemonConnect :: Int -> IO a -- TODO: could we generalize this Int parameter nicely?
   20                                  , daemonDisconnect :: a -> IO ()
   21                                  , daemonSend :: a -> ResponseMsg -> IO ()
   22                                  , daemonReceive :: a -> IO [Either String ClientMessage]
   23                                  }
   24 
   25 -- | Connect to the client running in a separate process using socket connection
   26 socketMode :: WorkingMode (Socket,Socket)
   27 socketMode = WorkingMode sockConn sockDisconnect sockSend sockReceive
   28   where
   29     sockConn portNumber = do
   30       sock <- socket AF_INET Stream 0
   31       setSocketOption sock ReuseAddr 1
   32       addr:_ <- getAddrInfo Nothing Nothing (Just $ show portNumber)
   33       bind sock (addrAddress addr)
   34       listen sock 1
   35       (conn, _) <- accept sock
   36       return (sock,conn)
   37     sockDisconnect (sock,conn) = close conn >> close sock
   38     sockSend (_,conn) = sendAll conn . (`BS.snoc` '\n') . encode
   39     sockReceive (_,conn) = do
   40       msg <- recv conn 2048
   41       if not $ BS.null msg -- null on TCP means closed connection
   42         then do -- when (not isSilent) $ putStrLn $ "message received: " ++ show (unpack msg)
   43           let msgs = BS.split '\n' msg
   44            in return $ catMaybes $ map decodeMsg msgs
   45         else return []
   46       where decodeMsg :: ByteString -> Maybe (Either String ClientMessage)
   47             decodeMsg mess
   48               | BS.null mess = Nothing
   49               | otherwise = case decode mess of
   50                 Nothing -> Just $ Left $ "MALFORMED MESSAGE: " ++ unpack mess
   51                 Just req -> Just $ Right req
   52 
   53 -- | Connect to the client running in the same process using a channel
   54 channelMode :: WorkingMode (Chan ResponseMsg, Chan ClientMessage)
   55 channelMode = WorkingMode chanConn chanDisconnect chanSend chanReceive
   56   where
   57     chanConn _ = (,) <$> newChan <*> newChan
   58     chanDisconnect _ = return ()
   59     chanSend (send,_) resp = writeChan send resp
   60     chanReceive (_,recv) = (:[]) . Right <$> readChan recv
