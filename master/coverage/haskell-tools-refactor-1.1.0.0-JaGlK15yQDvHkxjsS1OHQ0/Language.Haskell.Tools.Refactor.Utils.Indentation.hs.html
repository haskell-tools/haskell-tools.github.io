   10 -- | Set the minimal indentation recursively for a part of the AST
   11 setMinimalIndent :: SourceInfoTraversal elem
   12                  => Int -> elem dom SrcTemplateStage -> elem dom SrcTemplateStage
   13 setMinimalIndent ind = runIdentity .
   14   sourceInfoTraverse (SourceInfoTrf (Identity . (sourceTemplateMinimalIndent .= ind))
   15                                     (Identity . (srcTmpListMinimalIndent .= ind))
   16                                     (Identity . (srcTmpOptMinimalIndent .= ind)))
