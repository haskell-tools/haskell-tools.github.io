    1 -- | Generation of statement-level AST fragments for refactorings.
    2 -- The bindings defined here are the AST constructor names with an "mk" prefix.
    3 {-# LANGUAGE MonoLocalBinds, OverloadedStrings #-}
    4 
    5 module Language.Haskell.Tools.Rewrite.Create.Stmts where
    6 
    7 import Language.Haskell.Tools.AST (UCompStmt(..), UListCompBody(..), UStmt'(..))
    8 import Language.Haskell.Tools.PrettyPrint.Prepare
    9 import Language.Haskell.Tools.Rewrite.Create.Utils (mkAnn, mkAnnList, mkAnnMaybe)
   10 import Language.Haskell.Tools.Rewrite.ElementTypes
   11 
   12 -- | Creates a binding statement (@ x <- action @)
   13 mkBindStmt :: Pattern -> Expr -> Stmt
   14 mkBindStmt bound expr = mkAnn (child <> " <- " <> child) $ UBindStmt bound expr
   15 
   16 -- | Creates a non-binding statement (@ action @)
   17 mkExprStmt :: Expr -> Stmt
   18 mkExprStmt = mkAnn child . UExprStmt
   19 
   20 -- | Creates a let statement (@ let x = 3; y = 4 @)
   21 mkLetStmt :: [LocalBind] -> Stmt
   22 mkLetStmt = mkAnn ("let " <> child) . ULetStmt . mkAnnList (indented list)
   23 
   24 -- | Creates a recursive binding statement with (@ rec b <- f a c; c <- f b a @)
   25 mkRecStmt :: [Stmt] -> Stmt
   26 mkRecStmt = mkAnn ("rec " <> child) . URecStmt . mkAnnList (indented list)
   27 
   28 -- * List comprehensions
   29 
   30 -- | Body of a list comprehension: (@ | x <- [1..10] @)
   31 mkListCompBody :: [CompStmt] -> ListCompBody
   32 mkListCompBody = mkAnn child . UListCompBody . mkAnnList (separatedBy " " list)
   33 
   34 -- | Normal monadic statement of a list comprehension
   35 mkCompStmt :: Stmt -> CompStmt
   36 mkCompStmt = mkAnn child . UCompStmt
   37 
   38 -- | Then statements by @TransformListComp@ (@ then sortWith by (x + y) @)
   39 mkThenStmt :: Expr -> Maybe Expr -> CompStmt
   40 mkThenStmt th by = mkAnn ("then " <> child) 
   41                      $ UThenStmt th $ mkAnnMaybe (after " by " opt) by
   42 
   43 -- | Grouping statements by @TransformListComp@ (@ then group by (x + y) using groupWith @) 
   44 mkGroupStmt :: Maybe Expr -> Maybe Expr -> CompStmt
   45 mkGroupStmt by using = mkAnn ("then group" <> child) 
   46                          $ UGroupStmt (mkAnnMaybe (after " by " opt) by)
   47                                       (mkAnnMaybe (after " using " opt) using)
   48

   51 -- | Creates a binding command (@ x <- action @)
   52 mkBindCmd :: Pattern -> Cmd -> CmdStmt
   53 mkBindCmd bound expr = mkAnn (child <> " <- " <> child) $ UBindStmt bound expr
   54 
   55 -- | Creates a non-binding command (@ action @)
   56 mkExprCmd :: Cmd -> CmdStmt
   57 mkExprCmd = mkAnn child . UExprStmt
   58 
   59 -- | Creates a let command (@ let x = 3; y = 4 @)
   60 mkLetStmtCmd :: [LocalBind] -> CmdStmt
   61 mkLetStmtCmd = mkAnn ("let " <> child) . ULetStmt . mkAnnList (indented list)
   62 
   63 -- | Creates a recursive binding command with (@ rec b <- f a c; c <- f b a @)
   64 mkRecCmd :: [CmdStmt] -> CmdStmt
   65 mkRecCmd = mkAnn ("rec " <> child) . URecStmt . mkAnnList (indented list)
