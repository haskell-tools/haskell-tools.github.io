   30 -- | Does the import declaration import only the explicitly listed elements?
   31 importIsExact :: Ann UImportDecl dom stage -> Bool
   32 importIsExact = isJust . (^? importSpec&annJust&importSpecList)

   34 -- | Does the import declaration import all elements that are not excluded explicitly?
   35 importIsHiding :: Ann UImportDecl dom stage -> Bool
   36 importIsHiding = isJust . (^? importSpec&annJust&importSpecHiding)

   38 -- | Accesses the name of a function or value binding
   39 bindingName :: Simple Traversal (Ann UValueBind dom stage) (Ann UQualifiedName dom stage)
   40 bindingName = (valBindPat&patternName&simpleName
   41                         &+& funBindMatches&annList&matchLhs
   42                               &(matchLhsName&simpleName &+& matchLhsOperator&operatorName))
   43 
   44 -- | Accesses that name of a declaration through the declaration head.
   45 declHeadNames :: Simple Traversal (Ann UDeclHead dom stage) (Ann UQualifiedName dom stage)
   46 declHeadNames = (dhName&simpleName &+& dhBody&declHeadNames &+& dhAppFun&declHeadNames &+& dhOperator&operatorName)

   48 -- | A reference to access type arguments to a type constructor call that may be universally qualified
   49 -- or parenthesized.
   50 typeParams :: Simple Traversal (Ann UType dom stage) (Ann UType dom stage)
   51 typeParams = fromTraversal typeParamsTrav
   52   where typeParamsTrav f (Ann a (UTyFun p r)) = Ann a <$> (UTyFun <$> f p <*> typeParamsTrav f r)
   53         typeParamsTrav f (Ann a (UTyForall vs t)) = Ann a <$> (UTyForall vs <$> typeParamsTrav f t)
   54         typeParamsTrav f (Ann a (UTyCtx ctx t)) = Ann a <$> (UTyCtx ctx <$> typeParamsTrav f t)
   55         typeParamsTrav f (Ann a (UTyParen t)) = Ann a <$> (UTyParen <$> typeParamsTrav f t)
   56         typeParamsTrav f t = f t
   57 
   58 valBindPats :: Simple Traversal (Ann UValueBind dom stage) (Ann UPattern dom stage)
   59 valBindPats = valBindPat &+& funBindMatches & annList & matchLhs & (matchLhsArgs & annList &+& matchLhsLhs &+& matchLhsRhs)
   60

   65 -- | Get all nodes that contain a given source range
   66 nodesContaining :: (HasRange (inner dom stage), Biplate (node dom stage) (inner dom stage))
   67                 => RealSrcSpan -> Simple Traversal (node dom stage) (inner dom stage)
   68 nodesContaining rng = biplateRef & filtered (isInside rng)
   69 
   70 -- | Return true if the node contains a given range
   71 isInside :: HasRange (inner dom stage) => RealSrcSpan -> inner dom stage -> Bool
   72 isInside rng nd = case getRange nd of RealSrcSpan sp -> sp `containsSpan` rng
   73                                       _              -> False
   74 
   75 -- | Get all nodes that are contained in a given source range
   76 nodesContained :: (HasRange (inner dom stage), Biplate (node dom stage) (inner dom stage))
   77                     => RealSrcSpan -> Simple Traversal (node dom stage) (inner dom stage)
   78 nodesContained rng = biplateRef & filtered (isContained rng)
   79 
   80 -- | Return true if the node contains a given range
   81 isContained :: HasRange (inner dom stage) => RealSrcSpan -> inner dom stage -> Bool
   82 isContained rng nd = case getRange nd of RealSrcSpan sp -> rng `containsSpan` sp
   83                                          _              -> False
   84 
   85 -- | Get the nodes that have exactly the given range
   86 nodesWithRange :: (Biplate (Ann node dom stage) (Ann inner dom stage), SourceInfo stage)
   87                => RealSrcSpan -> Simple Traversal (Ann node dom stage) (Ann inner dom stage)
   88 nodesWithRange rng = biplateRef & filtered (hasRange rng)
   89   where -- True, if the node has the given range
   90         hasRange :: SourceInfo stage => RealSrcSpan -> Ann inner dom stage -> Bool
   91         hasRange rng node = case getRange node of RealSrcSpan sp -> sp == rng
   92                                                   _              -> False
   93 
   94 -- | Get the shortest source range that contains the given
   95 getNodeContaining :: (Biplate (Ann node dom stage) (Ann inner dom stage), SourceInfo stage, HasRange (Ann inner dom stage))
   96                   => RealSrcSpan -> Ann node dom stage -> Maybe (Ann inner dom stage)
   97 getNodeContaining sp node = case node ^? nodesContaining sp of
   98   [] -> Nothing
   99   results -> Just $ minimumBy (compareRangeLength `on` getRange) results

  101 -- | Compares two source spans based on their lengths. Can only used for NESTED spans.
  102 compareRangeLength :: SrcSpan -> SrcSpan -> Ordering
  103 compareRangeLength (RealSrcSpan sp1) (RealSrcSpan sp2)
  104   = (lineDiff sp1 `compare` lineDiff sp2) `mappend` (colDiff sp1 `compare` colDiff sp2)
  105   where lineDiff sp = srcLocLine (realSrcSpanStart sp) - srcLocLine (realSrcSpanEnd sp)
  106         colDiff sp = srcLocCol (realSrcSpanStart sp) - srcLocCol (realSrcSpanEnd sp)

  108 -- | A class to access the names of named elements. Have to locate where does the AST element store its name.
  109 -- The returned name will be the one that was marked isDefining.
  110 class NamedElement elem where
  111   elementName :: Simple Traversal (Ann elem dom st) (Ann UQualifiedName dom st)
  112 
  113 instance NamedElement UDecl where
  114   elementName = (declHead & declHeadNames)
  115                   &+& (declTypeFamily & tfHead & declHeadNames)
  116                   &+& (declValBind & bindingName)
  117                   &+& (declName & simpleName)
  118                   &+& (declTypeSig & tsName & annList & simpleName)
  119                   &+& (declPatSyn & patLhs & (patName & simpleName &+& patSynOp & operatorName))
  120 
  121 instance NamedElement ULocalBind where
  122   elementName = localVal&bindingName
  123                   &+& localSig&tsName&annList&simpleName
  124                   &+& localFixity&fixityOperators&annList&operatorName

  126 inScope :: GHC.Name -> Scope -> Bool
  127 inScope n sc = any ((n `elem`) . map (^. _1)) sc
