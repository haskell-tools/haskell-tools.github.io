    8 -- | Kind constraint (@ :: * -> * @)
    9 data UKindConstraint dom stage
   10   = UKindConstraint { _kindConstr :: Ann UKind dom stage
   11                     }
   12 
   13 -- | Haskell kinds
   14 data UKind dom stage
   15   = UStarKind  -- ^ @*@, the kind of types
   16   | UUnboxKind -- ^ @#@, the kind of unboxed types
   17   | UFunKind      { _kindLeft :: Ann UKind dom stage
   18                   , _kindRight :: Ann UKind dom stage
   19                   } -- ^ @->@, the kind of type constructor
   20   | UParenKind    { _kindParen :: Ann UKind dom stage
   21                   } -- ^ A parenthesised kind
   22   | UVarKind      { _kindVar :: Ann UName dom stage
   23                   } -- ^ Kind variable (using @PolyKinds@ extension)
   24   | UAppKind      { _kindAppFun :: Ann UKind dom stage
   25                   , _kindAppArg :: Ann UKind dom stage
   26                   } -- ^ Kind application (@ k1 k2 @)
   27   | UInfixAppKind { _kindLhs ::Ann UKind dom stage
   28                   , _kindAppOp :: Ann UOperator dom stage
   29                   , _kindRhs :: Ann UKind dom stage
   30                   } -- ^ Infix kind application (@ k1 ~> k2 @)
   31   | UListKind     { _kindElem :: Ann UKind dom stage
   32                   } -- ^ A list kind (@ [k] @)
   33   | UTupleKind    { _kindElems :: AnnListG UKind dom stage
   34                   } -- ^ A tuple kind (@ (Symbol, *) @)
   35   | UPromotedKind { _kindPromoted :: Ann (UPromoted UKind) dom stage
   36                   } -- ^ A promoted kind (@ '(k1,k2,k3) @)
   37   | UTypeKind     { _kindType :: Ann UType dom stage
   38                   } -- ^ A type on the kind level with @TypeInType@
   39 
   40 data UPromoted t dom stage
   41   = UPromotedInt    { _promotedIntValue :: Integer
   42                     } -- ^ Numeric value promoted to the kind level.
   43   | UPromotedString { _promotedStringValue :: String
   44                     } -- ^ String value promoted to the kind level.
   45   | UPromotedCon    { _promotedConName :: Ann UName dom stage
   46                     } -- ^ A data constructor value promoted to the kind level.
   47   | UPromotedList   { _promotedElements :: AnnListG t dom stage
   48                     } -- ^ A list of elements as a kind.
   49   | UPromotedTuple  { _promotedElements :: AnnListG t dom stage
   50                     } -- ^ A tuple of elements as a kind.
   51   | UPromotedUnit -- ^ Kind of the unit value @()@.
