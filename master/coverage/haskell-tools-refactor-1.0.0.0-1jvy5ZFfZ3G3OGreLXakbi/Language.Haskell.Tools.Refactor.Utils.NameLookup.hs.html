   13 opSemName :: GhcMonad m => Operator -> MaybeT m GHC.Name
   14 opSemName = liftMaybe . semanticsName . (^. operatorName)

   16 declHeadSemName :: GhcMonad m => DeclHead -> MaybeT m GHC.Name

   18 declHeadSemName (ParenDeclHead dh)     = declHeadSemName dh

   20 declHeadSemName (InfixDeclHead _ op _) = opSemName op
