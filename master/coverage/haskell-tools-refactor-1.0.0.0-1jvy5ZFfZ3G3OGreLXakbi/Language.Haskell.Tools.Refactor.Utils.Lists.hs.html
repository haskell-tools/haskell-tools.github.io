   18 -- | Filters the elements of the list. By default it removes the separator before the element.
   19 -- Of course, if the first element is removed, the following separator is removed as well.
   20 filterList :: SourceInfoTraversal e => (Ann e IdDom SrcTemplateStage -> Bool) -> AnnList e -> AnnList e
   21 filterList pred = filterListIndexed (const pred)
   22 
   23 filterListIndexed :: SourceInfoTraversal e => (Int -> Ann e IdDom SrcTemplateStage -> Bool) 
   24                                                 -> AnnList e -> AnnList e
   25 filterListIndexed pred (AnnListG (NodeInfo sema src) elems)
   26   =  AnnListG (NodeInfo sema (srcTmpIndented .- fmap filterIndents $ srcTmpSeparators .- filterSeparators $ src)) filteredElems
   27   where elementsKept = findIndices (uncurry pred) (zip [0..] elems)
   28         filteredElems = sublist elementsKept elems
   29         filterIndents = sublist elementsKept
   30         filterSeparators = take (length elementsKept - 1) . sublist elementsKept
   31 
   32 -- | A version of filterList that cares about keeping non-removable code elements (like preprocessor pragmas)
   33 filterListSt :: SourceInfoTraversal e => (Ann e IdDom SrcTemplateStage -> Bool) -> AnnList e 
   34                                            -> LocalRefactor (AnnList e)
   35 filterListSt pred = filterListIndexedSt (const pred)
   36 
   37 -- | A version of filterListIndexed that cares about keeping non-removable code elements (like preprocessor pragmas)
   38 filterListIndexedSt :: SourceInfoTraversal e => (Int -> Ann e IdDom SrcTemplateStage -> Bool) 
   39                                                   -> AnnList e -> LocalRefactor (AnnList e)
   40 filterListIndexedSt pred ls@(AnnListG _ elems)
   41   | all (uncurry pred) (zip [0..] elems)
   42   = return ls
   43 filterListIndexedSt pred (AnnListG (NodeInfo sema src) elems)
   44   = do mapM_ removeChild removedElems
   45        mapM_ removeSeparator removedSeparators
   46        return $ AnnListG (NodeInfo sema (srcTmpIndented .- fmap filterIndents
   47                                           $ srcTmpSeparators .- filterSeparators
   48                                           $ srcTmpListBefore .- (++ movedBefore)
   49                                           $ src)) filteredElems
   50   where elementsKept = findIndices (uncurry pred) (zip [0..] elems)
   51         filteredElems = sublist elementsKept elems
   52         removedSeparators :: [([SourceTemplateTextElem], SrcSpan)]
   53         removedSeparators = notSublist elementsKept (src ^. srcTmpSeparators) ++ lastSepRemoved
   54         lastSepRemoved = if (length elems - 1) `notElem` elementsKept
   55                            then take 1 (reverse (sublist elementsKept $ src ^. srcTmpSeparators)) else []
   56         removedElems = notSublist elementsKept elems
   57         filterIndents = sublist elementsKept
   58         filterSeparators = take (length elementsKept - 1) . sublist elementsKept
   59         -- if only one element remains we want to keep the whitespace before the first element
   60         movedBefore = case elementsKept of
   61                         [e] | e > 0 && any @[] isStayingText (removedSeparators ^? traversal & _1 & traversal) && maybe True (not . and) (src ^. srcTmpIndented)
   62                            -> concatMap (takeWhile isSpace . (^. sourceTemplateText)) . reverse . takeWhile (not . isStayingText) . reverse $ (src ^? srcTmpSeparators & traversal & _1) !! (e - 1)
   63                         _  -> ""

   65 -- | Selects the given indices from a list
   66 sublist :: [Int] -> [a] -> [a]
   67 sublist indices = map snd . filter ((`elem` indices) . fst) . zip [0..]
   68 
   69 -- | Selects all but the given indices from a list
   70 notSublist :: [Int] -> [a] -> [a]
   71 notSublist indices = map snd . filter ((`notElem` indices) . fst) . zip [0..]

   73 -- | Inserts the element in the places where the two positioning functions (one checks the element before, one the element after)
   74 -- allows the placement.
   75 insertWhere :: Bool -> Ann e IdDom SrcTemplateStage -> (Maybe (Ann e IdDom SrcTemplateStage) -> Bool)
   76                  -> (Maybe (Ann e IdDom SrcTemplateStage) -> Bool) -> AnnList e
   77                  -> AnnList e
   78 insertWhere indented e before after al
   79   = let index = insertIndex before after (al ^? annList)
   80      in case index of
   81           Nothing -> al
   82           Just ind -> annListElems .- insertAt ind e
   83                         $ (if isEmptyAnnList then id else annListAnnot&sourceInfo .- setIndented ind . addDefaultSeparator ind)
   84                         $ al
   85   where setIndented i = srcTmpIndented .- fmap (insertAt i indented)
   86         addDefaultSeparator i al = srcTmpSeparators .- insertAt i ([NormalText $ al ^. srcTmpDefaultSeparator], noSrcSpan) $ al
   87         insertAt n e ls = let (bef,aft) = splitAt n ls in bef ++ [e] ++ aft
   88         isEmptyAnnList = (null :: [x] -> Bool) $ (al ^? annList)
   89 
   90 -- | Checks where the element will be inserted given the two positioning functions.
   91 insertIndex :: (Maybe (Ann e IdDom SrcTemplateStage) -> Bool) -> (Maybe (Ann e IdDom SrcTemplateStage) -> Bool) 
   92                  -> [Ann e IdDom SrcTemplateStage] -> Maybe Int
   93 insertIndex before after []
   94   | before Nothing && after Nothing = Just 0
   95   | otherwise = Nothing
   96 insertIndex before after list@(first:_)
   97   | before Nothing && after (Just first) = Just 0
   98   | otherwise = (+1) <$> insertIndex' before after list
   99   where insertIndex' before after (curr:rest@(next:_))
  100           | before (Just curr) && after (Just next) = Just 0
  101           | otherwise = (+1) <$> insertIndex' before after rest
  102         insertIndex' before after (curr:[])
  103           | before (Just curr) && after Nothing = Just 0
  104           | otherwise = Nothing
  105         insertIndex' before after []
  106           | before Nothing && after Nothing = Just 0
  107           | otherwise = Nothing

  109 -- | Gets the elements and separators from a list. The first separator is zipped to the second element.
  110 -- To the first element, the "" string is zipped.
  111 zipWithSeparators :: AnnList e -> [(([SourceTemplateTextElem], SrcSpan), Ann e IdDom SrcTemplateStage)]
  112 zipWithSeparators (AnnListG (NodeInfo _ src) elems)
  113   | [] <- src ^. srcTmpSeparators
  114   = zip (([], noSrcSpan) : repeat ([NormalText $ src ^. srcTmpDefaultSeparator], noSrcSpan)) elems
  115   | otherwise
  116   = zip (([], noSrcSpan) : seps ++ repeat (_2 .= noSrcSpan $ last seps)) elems
  117   where seps = src ^. srcTmpSeparators
