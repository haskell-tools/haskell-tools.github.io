    1 -- | Defines a representation to represent refactorings that can be executed on the codebase.
    2 -- Refactorings are differentiated on their signatures (inputs needed to execute).
    3 module Language.Haskell.Tools.Refactor.Refactoring where
    4 
    5 import Control.Monad.Trans.Except (runExceptT)
    6 import Data.List ((++), map, find)
    7 
    8 import GHC (RealSrcSpan, Ghc)
    9 
   10 import Language.Haskell.Tools.AST ()
   11 import Language.Haskell.Tools.Refactor.Monad (ProjectRefactoring, Refactoring)
   12 import Language.Haskell.Tools.Refactor.Prepare (correctRefactorSpan, readSrcSpan)
   13 import Language.Haskell.Tools.Refactor.Representation (RefactorChange, ModuleDom)
   14 
   15 -- | The signature and behavior of one refactoring that can be executed.
   16 data RefactoringChoice
   17   = NamingRefactoring { refactoringName :: String
   18                       , namingRefactoring :: RealSrcSpan -> String -> Refactoring
   19                       }
   20   | SelectionRefactoring { refactoringName :: String
   21                          , selectionRefactoring :: RealSrcSpan -> Refactoring
   22                          }
   23   | ModuleRefactoring { refactoringName :: String
   24                       , moduleRefactoring :: Refactoring
   25                       }
   26   | ProjectRefactoring { refactoringName :: String
   27                        , projectRefactoring :: ProjectRefactoring
   28                        }
   29 
   30 -- | Executes a given command (choosen from the set of available refactorings) on the selected
   31 -- module and given other modules.
   32 performCommand :: [RefactoringChoice] -- ^ The set of available refactorings
   33                     -> [String] -- ^ The refactoring command
   34                     -> Either FilePath ModuleDom -- ^ The module in which the refactoring is performed
   35                     -> [ModuleDom] -- ^ Other modules
   36                     -> Ghc (Either String [RefactorChange])
   37 performCommand refactorings (name:args) mod mods =
   38     case (refactoring, mod, args) of
   39       (Just (NamingRefactoring _ trf), Right mod, (sp:newName:_))
   40         -> runExceptT $ trf (correctRefactorSpan (snd mod) $ readSrcSpan sp) newName mod mods
   41       (Just (NamingRefactoring _ _), Right _, _)
   42         -> return $ Left $ "The refactoring '" ++ name
   43                              ++ "' needs two argument: a source range and a name"
   44       (Just (SelectionRefactoring _ trf), Right mod, (sp:_))
   45         -> runExceptT $ trf (correctRefactorSpan (snd mod) $ readSrcSpan sp) mod mods
   46       (Just (SelectionRefactoring _ _), Right _, _)
   47         -> return $ Left $ "The refactoring '" ++ name ++ "' needs one argument: a source range"
   48       (Just (ModuleRefactoring _ trf), Right mod, _) -> runExceptT $ trf mod mods
   49       (Just (ProjectRefactoring _ trf), _, _) -> runExceptT $ trf mods
   50       (Just _, Left modPath, _)
   51         -> return $ Left $ "The following file is not loaded to Haskell-tools: " ++ modPath
   52                              ++ ". Please add the containing package."
   53       (Nothing, _, _) -> return $ Left $ "Unknown command: " ++ name
   54   where refactoring = find ((== name) . refactoringName) refactorings
   55 
   56 -- | Gets the name of possible refactorings.
   57 refactorCommands :: [RefactoringChoice] -> [String]
   58 refactorCommands = map refactoringName
