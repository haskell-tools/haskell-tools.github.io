    1 -- | UPattern matching on UModule-level AST fragments for refactorings.
    2 {-# LANGUAGE PatternSynonyms #-}
    3 
    4 module Language.Haskell.Tools.Rewrite.Match.Modules where
    5 
    6 import Language.Haskell.Tools.AST
    7 import Language.Haskell.Tools.Rewrite.ElementTypes

    9 -- | The representation of a haskell module, that is a separate compilation unit.
   10 -- It may or may not have a header.
   11 pattern Module :: FilePragmaList -> MaybeModuleHead
   12               -> ImportDeclList -> DeclList -> Module
   13 pattern Module filePrags head imps decls  <- Ann _ (UModule filePrags head imps decls )
   14 
   15 -- | Module declaration with name and (optional) exports
   16 pattern ModuleHead :: ModuleName -> MaybeModulePragma -> MaybeExportSpecs -> ModuleHead
   17 pattern ModuleHead n pr es <- Ann _ (UModuleHead n pr es)
   18 
   19 -- | A list of export specifications surrounded by parentheses
   20 pattern ExportSpecs :: ExportSpecList -> ExportSpecs
   21 pattern ExportSpecs specs <- Ann _ (UExportSpecs specs)
   22 
   23 -- | Export a name and related names
   24 pattern ExportSpec :: IESpec -> ExportSpec
   25 pattern ExportSpec ieSpec <- Ann _ (UDeclExport ieSpec)
   26 
   27 -- | The export of an imported module (@ module A @)
   28 pattern ModuleExport :: ModuleName -> ExportSpec
   29 pattern ModuleExport name <- Ann _ (UModuleExport name)
   30 
   31 -- | Marks a name to be imported or exported with related names (subspecifier)
   32 pattern IESpec :: MaybeImportModifier -> Name -> MaybeSubSpec -> IESpec
   33 pattern IESpec modifier name ss <- Ann _ (UIESpec modifier name ss)
   34 
   35 -- | @(..)@: a class exported with all of its methods, or a datatype exported with all of its constructors.
   36 pattern SubAll :: SubSpec
   37 pattern SubAll <- Ann _ USubSpecAll
   38 
   39 -- | @(a,b,c)@: a class exported with some of its methods, or a datatype exported with some of its constructors.
   40 pattern SubList :: NameList -> SubSpec
   41 pattern SubList names <- Ann _ (USubSpecList names)
   42 
   43 
   44 pattern ImportDecl :: MaybeImportSource -> MaybeImportQualified
   45                         -> MaybeImportSafe -> MaybeStringNode
   46                         -> ModuleName -> MaybeImportRenaming
   47                         -> MaybeImportSpec -> ImportDecl
   48 pattern ImportDecl source qualified safe pkg name rename spec <- Ann _ (UImportDecl source qualified safe pkg name rename spec)
   49 
   50 pattern ImportRenaming :: ModuleName -> ImportRenaming
   51 pattern ImportRenaming name <- Ann _ (UImportRenaming name)
   52 
   53 pattern ImportSpecList :: IESpecList -> ImportSpec
   54 pattern ImportSpecList ieSpecs <- Ann _ (UImportSpecList ieSpecs)
   55 
   56 pattern ImportHidingList :: IESpecList -> ImportSpec
   57 pattern ImportHidingList hidings <- Ann _ (UImportSpecHiding hidings)
   58 
   59 pattern ModuleName :: String -> ModuleName
   60 pattern ModuleName s <- Ann _ (UModuleName s)
   61 
   62 -- * Pragmas
   63 
   64 -- | @LANGUAGE@ pragma, listing the enabled language extensions in that file
   65 pattern LanguagePragma :: LanguageExtensionList -> FilePragma
   66 pattern LanguagePragma exts <- Ann _ (ULanguagePragma exts)
   67 
   68 -- | @OPTIONS@ pragma, possibly qualified with a tool, e.g. OPTIONS_GHC
   69 pattern OptionsPragma :: String -> FilePragma
   70 pattern OptionsPragma opt <- Ann _ (UOptionsPragma (Ann _ (UStringNode opt)))
   71 
   72 -- | The name of the enabled language extension, for example (@ LambdaCase @)
   73 pattern LanguageExtension :: String -> LanguageExtension
   74 pattern LanguageExtension ext <- Ann _ (ULanguageExtension ext)
   75 
   76 -- | A warning pragma attached to the module
   77 pattern ModuleWarningPragma :: StringNodeList -> ModulePragma
   78 pattern ModuleWarningPragma msgs <- Ann _ (UModuleWarningPragma msgs)
   79 
   80 -- | A deprecated pragma attached to the module
   81 pattern ModuleDeprecatedPragma :: StringNodeList -> ModulePragma
   82 pattern ModuleDeprecatedPragma msgs <- Ann _ (UModuleDeprecatedPragma msgs)
