   91 whichExtension :: GHC.Name -> Maybe Extension
   92 whichExtension    = flip Map.lookup nameExtensionMap
   93 
   94 nameExtensionMap = Map.fromList nameExtensions
   95   where nameExtensions = [ (dataClassName,        DeriveDataTypeable)
   96                          , (typeableClassName,    DeriveDataTypeable)
   97                          , (genClassName,         DeriveGeneric)
   98                          , (gen1ClassName,        DeriveGeneric)
   99                          , (functorClassName,     DeriveFunctor)
  100                          , (foldableClassName,    DeriveFoldable)
  101                          , (traversableClassName, DeriveTraversable)
  102                          , (liftClassName,        DeriveLift)
  103                          ]

  105 chkDerivings :: CheckNode Decl
  106 chkDerivings = conditionalAny chkDerivings'         derivingExts
  107            >=> conditional    chkStandaloneDeriving Ext.StandaloneDeriving
  108 
  109       where chkDerivings' = chkDataDecl
  110                         >=> chkGADTDataDecl
  111                         >=> chkDataInstance
  112 
  113             derivingExts = [ DeriveDataTypeable
  114                            , DeriveGeneric
  115                            , DeriveFunctor
  116                            , DeriveFoldable
  117                            , DeriveTraversable
  118                            , DeriveLift
  119                            , DeriveAnyClass
  120                            , GeneralizedNewtypeDeriving
  121                            , DerivingStrategies
  122                            ]

  125 chkDataDecl :: CheckNode Decl
  126 chkDataDecl d@(DataDecl keyw _ _ _ derivs) = do
  127   annList !~ separateByKeyword keyw $ derivs
  128   return d
  129 chkDataDecl d = return d
  130 
  131 chkGADTDataDecl :: CheckNode Decl
  132 chkGADTDataDecl d@(GADTDataDecl keyw _ _ _ _ derivs) = do
  133   addOccurence_ GADTs d
  134   annList !~ separateByKeyword keyw $ derivs
  135   return d
  136 chkGADTDataDecl d = return d
  137 
  138 chkDataInstance :: CheckNode Decl
  139 chkDataInstance d@(DataInstance keyw _ _ derivs) = do
  140   addOccurence_ TypeFamilies d
  141   annList !~ separateByKeyword keyw $ derivs
  142   return d
  143 chkDataInstance d = return d

  146 separateByKeyword :: DataOrNewtypeKeyword -> CheckNode Deriving
  147 separateByKeyword keyw derivs
  148   | isNewtypeDecl keyw = chkByStrat chkClassForNewtype derivs
  149   | otherwise          = chkByStrat chkClassForData    derivs
  150   where isNewtypeDecl keyw = case keyw ^. element of
  151                                UNewtypeKeyword -> True
  152                                _               -> False

  155 getStrategy :: Deriving -> Maybe DeriveStrategy
  156 getStrategy d = d ^. (deriveStrategy & annMaybe)
  157 
  158 addExtension :: (MonadState ExtMap m, HasRange node) =>
  159                  GHC.Name -> node -> m node
  160 addExtension sname
  161   | Just ext <- whichExtension sname = addOccurence ext
  162   | otherwise                        = return
  163 
  164 addStockExtension :: CheckNode InstanceHead
  165 addStockExtension x
  166   | Just sname <- nameFromStock x = addExtension sname x
  167   | otherwise = return x
  168 
  169 chkByStrat :: CheckNode InstanceHead -> CheckNode Deriving
  170 chkByStrat checker d
  171   | Just strat <- getStrategy d = do
  172     addOccurence DerivingStrategies d
  173     chkDerivingClause (chkStrat strat) d
  174   | otherwise =
  175     chkDerivingClause checker d
  176 
  177 chkStrat :: DeriveStrategy -> CheckNode InstanceHead
  178 chkStrat (_element -> UStockStrategy)    = addStockExtension
  179 chkStrat (_element -> UNewtypeStrategy)  = addOccurence GeneralizedNewtypeDeriving
  180 chkStrat (_element -> UAnyClassStrategy) = addOccurence DeriveAnyClass
  181 
  182 chkDerivingClause :: CheckNode InstanceHead -> CheckNode Deriving
  183 chkDerivingClause checker d@(DerivingOne   x)  = checker x               >> return d
  184 chkDerivingClause checker d@(DerivingMulti xs) = (annList !~ checker) xs >> return d
  185 
  186 -- checks whether the class is stock, and if it is, returns its name
  187 nameFromStock :: InstanceHead -> Maybe GHC.Name
  188 nameFromStock x
  189   | InstanceHead name <- skipParens x,
  190     Just sname <- getSemName name,
  191     isStockClass sname
  192     = Just sname
  193   | otherwise = Nothing
  194 
  195 chkClassForData :: CheckNode InstanceHead
  196 chkClassForData x
  197   | Just sname <- nameFromStock x = addExtension sname x
  198   | otherwise = addOccurence DeriveAnyClass x
  199 
  200 -- performs check in case no explicit strategy is given
  201 chkClassForNewtype :: CheckNode InstanceHead
  202 chkClassForNewtype x
  203   | Just sname <- nameFromStock x
  204     = if | gndNotNeeded  sname -> return x
  205          | gndNeeded     sname -> addOccurence GeneralizedNewtypeDeriving x
  206          | gndNotAllowed sname -> addExtension sname x
  207   | otherwise = do
  208       gndOn       <- isTurnedOn GeneralizedNewtypeDeriving
  209       deriveAnyOn <- isTurnedOn DeriveAnyClass
  210       if | gndOn && deriveAnyOn -> addOccurence_ DeriveAnyClass x
  211          | deriveAnyOn          -> addOccurence_ DeriveAnyClass x
  212          | gndOn                -> addOccurence_ GeneralizedNewtypeDeriving x
  213          | otherwise             -> return ()
  214       return x
  215 
  216 skipParens :: InstanceHead -> InstanceHead
  217 skipParens (ParenInstanceHead x) = skipParens x
  218 skipParens x = x
  219 
  220 chkStandaloneDeriving :: CheckNode Decl
  221 chkStandaloneDeriving d@(Refact.StandaloneDeriving strat _ (decompRule -> (cls,ty)))
  222   | Just strat' <- strat = do
  223     addOccurence_  Ext.DerivingStrategies d
  224     addOccurence_  Ext.StandaloneDeriving d
  225     chkSynonym ty
  226     chkStrat strat' cls
  227     return d
  228   | otherwise = do
  229     addOccurence_  Ext.StandaloneDeriving d
  230     itIsNewType    <- isNewtype ty
  231     itIsSynNewType <- isSynNewType ty
  232     if itIsNewType || itIsSynNewType
  233       then chkClassForNewtype cls
  234       else chkClassForData    cls
  235     return d
  236 chkStandaloneDeriving d = return d
  237 
  238 decompRule :: InstanceRule -> (InstanceHead, Type)
  239 decompRule instRule = (cls, ty)
  240   where ihead = instRule  ^. irHead
  241         cls   = getClassCon   ihead
  242         ty    = rightmostType ihead
  243 
  244 getClassCon :: InstanceHead -> InstanceHead
  245 getClassCon (AppInstanceHead f _) = getClassCon f
  246 getClassCon (ParenInstanceHead x) = getClassCon x
  247 getClassCon x = x
  248 
  249 rightmostType :: InstanceHead -> Type
  250 rightmostType ihead
  251   | AppInstanceHead _ tyvar <- skipParens ihead = tyvar

  253 {-
  254   NOTE: Returns false if the type is certainly not a type synonym.
  255         Returns true if it is a synonym for a newtype or it could not have been looked up.
  256   NOTE: It always has the following side-effects:
  257         - If the input is a type synonym, then adds TypeSynonymInstances
  258           (regardless of it being a newtype or not)
  259 
  260   This behaviour will produce false positives.
  261   This is desirable since the underlying type might be a newtype
  262   in which case GeneralizedNewtypeDeriving might be necessary.
  263 -}
  264 isSynNewType :: Type -> ExtMonad Bool
  265 isSynNewType t = do
  266   mtycon <- runMaybeT . lookupType $ t
  267   case mtycon of
  268     Nothing    -> return True
  269     Just tycon -> isSynNewType' tycon
  270   where isSynNewType' x = case lookupSynDef x of
  271                             Nothing  -> return False
  272                             Just def -> do
  273                                         addOccurence_ TypeSynonymInstances t
  274                                         return (GHC.isNewTyCon def)
