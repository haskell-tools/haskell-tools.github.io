   42 organizeExtensions :: LocalRefactoring
   43 organizeExtensions moduleAST = do
   44   exts <- liftGhc $ reduceExtensions moduleAST
   45   let isRedundant e = extName `notElem` foundExts && extName `elem` handledExts
   46         where extName = unregularExts (e ^. langExt)
   47       handledExts = map show fullyHandledExtensions
   48       foundExts = map show exts
   49 
   50   -- remove unused extensions (only those that are fully handled)
   51   filePragmas & annList & lpPragmas !~ filterListSt (not . isRedundant)
   52         -- remove empty {-# LANGUAGE #-} pragmas
   53     >=> filePragmas !~ filterListSt (\case LanguagePragma (AnnList []) -> False; _ -> True)
   54     $ moduleAST
   55 
   56 -- | Reduces default extension list (keeps unsupported extensions)
   57 reduceExtensions :: UnnamedModule -> Ghc [Extension]
   58 reduceExtensions = \moduleAST -> do
   59   let expanded = expandDefaults moduleAST
   60       (xs, ys) = partition isSupported expanded
   61   xs' <- flip execStateT SMap.empty . flip runReaderT xs . traverseModule $ moduleAST
   62   return . sortBy (compare `on` show) . mergeInduced . nub $ (calcExts xs' ++ ys)
   63 
   64   where isLVar (LVar _) = True
   65         isLVar _        = False
   66 
   67         calcExts :: ExtMap -> [Extension]
   68         calcExts logRels
   69           | ks <- SMap.keys logRels
   70           , all isLVar ks
   71           = map (\(LVar x) -> x) . SMap.keys $ logRels
   72           | otherwise     = []
   73 
   74         rmInduced :: Extension -> [Extension] -> [Extension]
   75         rmInduced e = flip (\\) induced
   76           where induced = delete e $ expandExtension e
   77 
   78         mergeInduced :: [Extension] -> [Extension]
   79         mergeInduced exts = foldl (flip rmInduced) exts exts
   80 
   81 
   82 -- | Collects extensions induced by the source code (with location info)
   83 collectExtensions :: UnnamedModule -> Ghc ExtMap
   84 collectExtensions = collectExtensionsWith traverseModule
   85 
   86 collectExtensionsWith :: CheckNode UnnamedModule -> UnnamedModule -> Ghc ExtMap
   87 collectExtensionsWith trvModule moduleAST = do
   88   let expanded = expandDefaults moduleAST
   89   flip execStateT SMap.empty . flip runReaderT expanded . trvModule $ moduleAST
   90 
   91 
   92 -- | Collects default extension list, and expands each extension
   93 expandDefaults :: UnnamedModule -> [Extension]
   94 expandDefaults = nub . concatMap expandExtension . collectDefaultExtensions
   95 
   96 -- | Collects extensions enabled by default
   97 collectDefaultExtensions :: UnnamedModule -> [Extension]
   98 collectDefaultExtensions = map toExt . getExtensions
   99   where
  100   getExtensions :: UnnamedModule -> [String]
  101   getExtensions = flip (^?) (filePragmas & annList & lpPragmas & annList & langExt)
  102 
  103 toExt :: String -> Extension
  104 toExt str = case map fst . reads . unregularExts . takeWhile isAlpha $ str of
  105               e:_ -> e
  106               []  -> error $ "Extension '" ++ takeWhile isAlpha str ++ "' is not known."
