   13 chkConstrainedClassMethodsDecl :: CheckNode Decl
   14 chkConstrainedClassMethodsDecl = conditional chkCCMDecl ConstrainedClassMethods
   15   where chkCCMDecl cd@(ClassDecl _ dh _ _) = chkCCMDeclHead dh >> return cd
   16         chkCCMDecl x = return x
   17 
   18 -- | Check a DeclHead for ConstrainedClassMethods.
   19 -- Adds the extension if it is needed or the lookup fails.
   20 chkCCMDeclHead :: CheckNode DeclHead
   21 chkCCMDeclHead dh = do
   22   mNeedsCCM <- runMaybeT . chkCCMDeclHead' $ dh
   23   case mNeedsCCM of
   24     Just False -> return dh
   25     _          -> addOccurence ConstrainedClassMethods dh
   26 
   27 
   28 -- | Helper function for chkCCMDeclHead.
   29 -- True  <=> Lookup is succesful and ConstrainedClassMethods is needed
   30 -- False <=> Lookup is succesful, but CCM is not needed, or the argument is not a class DeclHead
   31 -- fails <=> Lookup is unsuccesful (either name or type lookup)
   32 chkCCMDeclHead' :: DeclHead -> MaybeT ExtMonad Bool
   33 chkCCMDeclHead' dh = do
   34   sname   <- declHeadSemName dh
   35   tything <- MaybeT . GHC.lookupName $ sname
   36   case tything of
   37     GHC.ATyCon tc | GHC.isClassTyCon tc -> liftMaybe . fmap classNeedsCCM . GHC.tyConClass_maybe $ tc
   38     _ -> return False
   39

   41 -- | Decides whether a class really needs the ConstrainedClassMethods extension
   42 -- A class needs CCM iff at least one of its class methods
   43 -- has a constraint with a non-empty type variable set, that contains only class type variables.
   44 classNeedsCCM :: GHC.Class -> Bool
   45 classNeedsCCM cls = any methodNeedsCCM methods
   46   where
   47     methods     = GHC.classMethods cls
   48     tyvars      = GHC.classTyVars cls
   49     clsTyVarSet = GHC.mkVarSet tyvars
   50 
   51     methodNeedsCCM :: GHC.Id -> Bool
   52     methodNeedsCCM methodId = any constraintNeedsCCM constraints
   53       where
   54         (_,_,tau)         = GHC.tcSplitMethodTy . GHC.idType $ methodId
   55         (_,constraints,_) = GHC.tcSplitNestedSigmaTys tau
   56 
   57         constraintNeedsCCM :: GHC.TcPredType -> Bool
   58         constraintNeedsCCM pred = not (GHC.isEmptyVarSet predTyVars)
   59                                   && predTyVars `GHC.subVarSet` clsTyVarSet
   60           where predTyVars = GHC.tyCoVarsOfType pred
