   11 import Control.Applicative ((<|>))

   14 generateExportsRefactoring :: RefactoringChoice
   15 generateExportsRefactoring = ModuleRefactoring "GenerateExports" (localRefactoring generateExports)
   16 
   17 -- | Creates an export list that imports standalone top-level definitions with all of their contained definitions
   18 generateExports :: LocalRefactoring
   19 generateExports mod = return (modHead & annJust & mhExports & annMaybe
   20                                 .= Just (createExports (getTopLevels mod)) $ mod)
   21 
   22 -- | Get all the top-level definitions with flags that mark if they can contain other top-level definitions
   23 -- (classes and data declarations).
   24 getTopLevels :: Module -> [(GHC.Name, Bool)]
   25 getTopLevels mod = catMaybes $ map (\d -> fmap (,exportContainOthers d)
   26                                                (foldl (<|>) Nothing $ map semanticsName $ d ^? elementName))
   27                                    (filter (\case TypeSigDecl{} -> False; _ -> True)
   28                                       $ mod ^? modDecl & annList)
   29   where exportContainOthers :: Decl -> Bool
   30         exportContainOthers (DataDecl {}) = True
   31         exportContainOthers (ClassDecl {}) = True
   32         exportContainOthers _ = False
   33 
   34 -- | Create the export for a give name.
   35 createExports :: [(GHC.Name, Bool)] -> ExportSpecs
   36 createExports elems = mkExportSpecs $ map (mkExportSpec . createExport) elems
   37   where createExport (n, False) = mkIESpec (mkUnqualName' (GHC.getName n)) Nothing
   38         createExport (n, True)  = mkIESpec (mkUnqualName' (GHC.getName n)) (Just mkSubAll)
