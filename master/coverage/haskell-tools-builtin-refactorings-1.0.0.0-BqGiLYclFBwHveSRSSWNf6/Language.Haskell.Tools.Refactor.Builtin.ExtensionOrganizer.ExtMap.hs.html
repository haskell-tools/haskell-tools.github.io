   62 determineExtensions :: SMap.Map (LogicalRelation Extension) v -> [Extension]
   63 {- NOTE:
   64  We calculate all the possible extension set that satisfy the logical relation,
   65  then for each extension we remove all the extensions implied by it, (mergeImplied)
   66  finally we select the minimal extension set.
   67 
   68  An extension set is minimal if:
   69   - it contains the least amount of extensions
   70   - it contains extension with minimal complexity
   71   - it is lexicographically the first one (based on the Ord instance of Extension)
   72 -}
   73 determineExtensions x = minimal . map toExts $ solution
   74   where solution = solve_all . All . LMap.keys $ x
   75         toExts   = mergeImplied . map fst . filter snd . LMap.toList
   76         minimal  = minimumBy ((compare `on` length) <> (compare `on` (sum . map complexity)) <> (compare `on` sort))
   77 
   78 rmImplied :: Extension -> [Extension] -> [Extension]
   79 rmImplied e = flip (\\) implied
   80   where implied = delete e $ expandExtension e
   81 
   82 mergeImplied :: [Extension] -> [Extension]
   83 mergeImplied exts = foldl (flip rmImplied) exts exts
