   12 -- | Checks a GADT-style constructor if GADTSyntax is turned on.
   13 -- Sometimes GADTSyntax is sufficient and GADTs is not even needed.
   14 chkGADTsGadtConDecl :: CheckNode GadtConDecl
   15 chkGADTsGadtConDecl = conditional chkGADTsGadtConDecl' GADTSyntax

   17 -- | Checks a data constructor declaration if GADTs or ExistentialQuantification is turned on.
   18 -- This function is responsible for checking ExistentialQuantification as well.
   19 -- (there is no separate checker for that extension)
   20 chkConDeclForExistentials :: CheckNode ConDecl
   21 chkConDeclForExistentials = conditionalAny chkConDeclForExistentials' [GADTs, ExistentialQuantification]

   23 -- If all data constructors are vanilla Haskell 98 data constructors, then only GADTSyntax is needed.
   24 chkGADTsGadtConDecl' :: CheckNode GadtConDecl
   25 chkGADTsGadtConDecl' conDecl = do
   26   let conNames = conDecl ^. (gadtConNames & annListElems)
   27   mres <- mapM (runMaybeT . isVanillaDataConNameM) conNames
   28   if and . catMaybes $ mres
   29     then addOccurence GADTSyntax conDecl
   30     else do addOccurence GADTSyntax conDecl
   31             addRelation (GADTs `lOr` ExistentialQuantification) conDecl
   32 
   33 -- Extracts the name from a ConDecl, and checks whether it is a vanilla
   34 -- data constructor.
   35 chkConDeclForExistentials' :: CheckNode ConDecl
   36 chkConDeclForExistentials' conDecl = liftM (fromMaybe conDecl) . runMaybeT $
   37   case conDecl ^. element of
   38     UConDecl _ _ n _         -> chkName n
   39     URecordDecl _ _ n _      -> chkName n
   40     UInfixConDecl _ _ _ op _ -> chkName (op ^. operatorName)
   41   where chkName :: HasNameInfo' n => n -> MaybeT ExtMonad ConDecl
   42         chkName n = do
   43           isVanilla <- isVanillaDataConNameM n
   44           if isVanilla
   45             then return conDecl
   46             else lift . addRelation (GADTs `lOr` ExistentialQuantification) $ conDecl
