   14 -- | We need to check declarations because we want to avoid checking type family instances
   15 chkTypeSynonymInstancesDecl :: CheckNode Decl
   16 chkTypeSynonymInstancesDecl = conditional (chkInstancesDeclWith $ chkInstanceRuleWith chkInstanceHead) TypeSynonymInstances
   17 
   18 -- | Checks an instance rule in declaration
   19 -- We need to check declarations because we want to avoid checking type family instances
   20 chkInstancesDeclWith :: CheckNode InstanceRule -> CheckNode Decl
   21 chkInstancesDeclWith chkRule d@(StandaloneDeriving _ _ rule) = chkRule rule >> return d
   22 chkInstancesDeclWith chkRule d@(InstanceDecl rule _)         = chkRule rule >> return d
   23 chkInstancesDeclWith _ d = return d
   24 
   25 -- | Checks and instance head in an instance rule
   26 chkInstanceRuleWith :: CheckNode InstanceHead -> CheckNode InstanceRule
   27 chkInstanceRuleWith chkHead r@(InstanceRule _ _ ihead) = chkHead ihead >> return r
   28 chkInstanceRuleWith _ r = return r
   29 
   30 
   31 -- | Checks every single type argument in an instance declaration whether it is a synonym
   32 chkInstanceHead :: CheckNode InstanceHead
   33 chkInstanceHead ih = do
   34   let types = collectTyArgs ih
   35   mapM_ chkTypeArg types >> return ih
   36 
   37 -- | Checks a type argument of class whether it is a synonym
   38 chkTypeArg :: Type -> ExtMonad Type
   39 chkTypeArg ty =
   40   maybeTM (return ty) (const . addOccurence TypeSynonymInstances $ ty) (semanticsTypeSynRhs ty)
   41 
   42 -- | Collects the type arguments in an instance declaration
   43 -- Type arguments are the the types that the class is being instantiated with
   44 collectTyArgs :: InstanceHead -> [Type]
   45 collectTyArgs (InstanceHead _)        = []
   46 collectTyArgs (InfixInstanceHead t _) = [t]
   47 collectTyArgs (ParenInstanceHead ih)  = collectTyArgs ih
   48 collectTyArgs (AppInstanceHead ih t)  = t : collectTyArgs ih
