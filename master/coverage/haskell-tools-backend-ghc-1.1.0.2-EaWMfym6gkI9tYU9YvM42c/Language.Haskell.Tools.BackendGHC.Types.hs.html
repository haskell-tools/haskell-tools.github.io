   31 trfType :: forall n r p . (TransformName n r, n ~ GhcPass p, HasCallStack) => Located (HsType n) -> Trf (Ann AST.UType (Dom r) RangeStage)
   32 trfType typ | RealSrcSpan loce <- getLoc typ
   33   = do othSplices <- asks typeSplices
   34        let contSplice = filter (\sp -> case getLoc sp of (RealSrcSpan spLoc) -> spLoc `containsSpan` loce; _ -> False) othSplices
   35        case contSplice of [] -> trfLocNoSema trfType' typ
   36                           _ -> let lsp@(L l sp) = minimumBy (compareSpans `on` getLoc) contSplice
   37                                 in typeSpliceInserted lsp (annLocNoSema (pure l) (AST.UTySplice <$> (trfSplice =<< rdrSplice sp)))
   38   | otherwise = trfLocNoSema trfType' typ
   39 
   40 trfType' :: forall n r p . (TransformName n r, n ~ GhcPass p, HasCallStack) => HsType n -> Trf (AST.UType (Dom r) RangeStage)
   41 trfType' = trfType'' where
   42   trfType'' :: HsType n -> Trf (AST.UType (Dom r) RangeStage)
   43   trfType'' (HsForAllTy _ [] typ) = trfType' (unLoc typ)
   44   trfType'' (HsForAllTy _ bndrs typ) = AST.UTyForall <$> defineTypeVars (trfBindings bndrs)
   45                                                      <*> addToScope bndrs (trfType typ)
   46   trfType'' (HsQualTy _ (L _ []) typ) = trfType' (unLoc typ)
   47   trfType'' (HsQualTy _ ctx typ) = AST.UTyCtx <$> (fromJust . (^. annMaybe) <$> trfCtx atTheStart ctx)
   48                                               <*> trfType typ
   49   trfType'' (HsTyVar _ _ name) = AST.UTyVar <$> transformingPossibleVar name (trfName @n name)
   50   trfType'' (HsAppTy _ t1 t2) = AST.UTyApp <$> trfType t1 <*> trfType t2
   51   trfType'' (HsFunTy _ t1 t2) = AST.UTyFun <$> trfType t1 <*> trfType t2
   52   trfType'' (HsListTy _ typ) = AST.UTyList <$> trfType typ
   53   -- trfType'' (HsPArrTy _ typ) = AST.UTyParArray <$> trfType typ
   54   trfType'' (HsTupleTy _ HsBoxedOrConstraintTuple typs) = AST.UTyTuple <$> trfAnnList ", " trfType' typs
   55   trfType'' (HsTupleTy _ HsBoxedTuple typs) = AST.UTyTuple <$> trfAnnList ", " trfType' typs
   56   trfType'' (HsTupleTy _ HsUnboxedTuple typs) = AST.UTyUnbTuple <$> trfAnnList ", " trfType' typs
   57   trfType'' (HsOpTy _ t1 op t2) = AST.UTyInfix <$> trfType t1 <*> trfOperator @n op <*> trfType t2
   58   trfType'' (HsParTy _ typ) = AST.UTyParen <$> trfType typ
   59   trfType'' (HsKindSig _ typ kind) = AST.UTyKinded <$> trfType typ <*> trfKind kind
   60   trfType'' (HsSpliceTy _ qq@(HsQuasiQuote {})) = AST.UTyQuasiQuote <$> annContNoSema (trfQuasiQuotation' qq)
   61   trfType'' (HsSpliceTy _ splice) = AST.UTySplice <$> trfSplice splice
   62   trfType'' (HsBangTy _ (HsSrcBang _ SrcUnpack _) typ) = AST.UTyUnpack <$> trfType typ
   63   trfType'' (HsBangTy _ (HsSrcBang _ SrcNoUnpack _) typ) = AST.UTyNoUnpack <$> trfType typ
   64   trfType'' (HsBangTy _ (HsSrcBang _ _ SrcStrict) typ) = AST.UTyBang <$> trfType typ
   65   trfType'' (HsBangTy _ (HsSrcBang _ _ SrcLazy) typ) = AST.UTyLazy <$> trfType typ
   66   trfType'' pt@(HsExplicitListTy {}) = AST.UTyPromoted <$> annContNoSema (trfPromoted' trfType' pt)
   67   trfType'' pt@(HsExplicitTupleTy {}) = AST.UTyPromoted <$> annContNoSema (trfPromoted' trfType' pt)
   68   trfType'' pt@(HsTyLit {}) = AST.UTyPromoted <$> annContNoSema (trfPromoted' trfType' pt)
   69   trfType'' (HsWildCardTy _) = pure AST.UTyWildcard -- TODO: named wildcards
   70   trfType'' (HsSumTy _ types) = AST.UUnbSumType <$> trfAnnList " | " trfType' types
   71   trfType'' t = unhandledElement "type" t

   73 trfBindings :: (TransformName n r, n ~ GhcPass p, HasCallStack) => [Located (HsTyVarBndr n)] -> Trf (AnnListG AST.UTyVar (Dom r) RangeStage)
   74 trfBindings [] = makeList " " atTheStart (pure [])
   75 trfBindings vars = trfAnnList " " trfTyVar' vars
   76 
   77 trfTyVar :: (TransformName n r, n ~ GhcPass p, HasCallStack) => Located (HsTyVarBndr n) -> Trf (Ann AST.UTyVar (Dom r) RangeStage)
   78 trfTyVar = trfLocNoSema trfTyVar'
   79 
   80 trfTyVar' :: forall n r p . (TransformName n r, n ~ GhcPass p, HasCallStack) => HsTyVarBndr n -> Trf (AST.UTyVar (Dom r) RangeStage)
   81 trfTyVar' (UserTyVar _ name) = AST.UTyVarDecl <$> typeVarTransform (trfName @n name)
   82                                               <*> (nothing " " "" atTheEnd)
   83 trfTyVar' (KindedTyVar _ name kind) = AST.UTyVarDecl <$> typeVarTransform (trfName @n name)
   84                                                      <*> trfKindSig (Just kind)

   86 trfCtx :: (TransformName n r, n ~ GhcPass p, HasCallStack) => Trf SrcLoc -> Located (HsContext n) -> Trf (AnnMaybeG AST.UContext (Dom r) RangeStage)
   87 trfCtx sp (L _ []) = nothing " " "" sp
   88 trfCtx _ (L l [L _ (HsParTy _ t)])
   89   = makeJust <$> annLocNoSema (combineSrcSpans l <$> tokenLoc AnnDarrow)
   90                               (AST.UContext <$> annLocNoSema (pure l) (AST.UTupleAssert <$> (trfAnnList ", " trfAssertion' [t])))
   91 trfCtx _ (L l [t])
   92   = makeJust <$> annLocNoSema (combineSrcSpans l <$> tokenLoc AnnDarrow)
   93                               (AST.UContext <$> trfAssertion t)
   94 trfCtx _ (L l ctx) = makeJust <$> annLocNoSema (combineSrcSpans l <$> tokenLoc AnnDarrow)
   95                                                (AST.UContext <$> annLocNoSema (pure l) (AST.UTupleAssert <$> (trfAnnList ", " trfAssertion' ctx)))
   96 
   97 trfAssertion :: (TransformName n r, n ~ GhcPass p, HasCallStack) => LHsType n -> Trf (Ann AST.UAssertion (Dom r) RangeStage)
   98 trfAssertion = trfLocNoSema trfAssertion'
   99 
  100 trfAssertion' :: forall n r p . (TransformName n r, n ~ GhcPass p, HasCallStack) => HsType n -> Trf (AST.UAssertion (Dom r) RangeStage)
  101 trfAssertion' (HsParTy _ t)
  102   = trfAssertion' (unLoc t)
  103 trfAssertion' (HsOpTy _ left op right)
  104   = AST.UInfixAssert <$> trfType left <*> trfOperator @n op <*> trfType right
  105 trfAssertion' (HsTupleTy _ _ tys)
  106   = AST.UTupleAssert <$> makeList ", " (after AnnOpenP) (mapM trfAssertion tys)
  107 trfAssertion' (HsWildCardTy _)
  108   = pure AST.UWildcardAssert
  109 trfAssertion' t = case base of
  110    HsTyVar _ _ name -> AST.UClassAssert <$> trfName @n name <*> trfAnnList " " trfType' args
  111    -- HsEqTy t1 t2 -> AST.UInfixAssert <$> trfType t1 <*> annLocNoSema (tokenLoc AnnTilde) (trfOperator' @n typeEq) <*> trfType t2
  112    HsIParamTy _ name t -> AST.UImplicitAssert <$> define (focusOn (getLoc name) (trfImplicitName (unLoc name))) <*> trfType t
  113    t -> unhandledElement "assertion" t
  114   where (args, _, base) = getArgs t
  115 
  116         getArgs :: HsType n -> ([LHsType n], Maybe SrcSpan, HsType n)
  117         getArgs (HsAppTy _ (L l ft) at) = case getArgs ft of (args, sp, base) -> (args++[at], sp <|> Just l, base)
  118         getArgs t = ([], Nothing, t)
