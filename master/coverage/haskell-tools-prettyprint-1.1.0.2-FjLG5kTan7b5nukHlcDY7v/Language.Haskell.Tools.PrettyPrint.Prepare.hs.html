   31 -- | Prepares the AST for pretty printing
   32 prepareAST :: StringBuffer -> Ann UModule dom RangeStage -> Ann UModule dom SrcTemplateStage
   33 prepareAST srcBuffer = rangeToSource srcBuffer . cutUpRanges . fixRanges

   35 prepareASTCpp :: StringBuffer -> Ann UModule dom RangeStage -> Ann UModule dom SrcTemplateStage

   38 fixMainRange :: StringBuffer -> Ann UModule dom RangeStage -> Ann UModule dom RangeStage
   39 fixMainRange buffer mod = setRange (mkSrcSpan (srcSpanStart $ getRange mod) (RealSrcLoc (endPos startPos buffer))) mod
   40   where startPos = mkRealSrcLoc (mkFastString "") 1 1
   41         endPos pos buf | atEnd buf = pos
   42         endPos pos buf = let (ch,buf') = nextChar buf in endPos (advanceSrcLoc pos ch) buf'
