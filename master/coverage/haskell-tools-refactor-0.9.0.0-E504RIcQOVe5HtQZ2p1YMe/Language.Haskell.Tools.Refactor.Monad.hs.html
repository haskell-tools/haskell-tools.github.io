    6 -- | Types and instances for monadic refactorings. The refactoring monad provides automatic
    7 -- importing, keeping important source fragments (such as preprocessor pragmas), and providing

   26 -- | A monad that can be used to refactor
   27 class Monad m => RefactorMonad m where
   28   refactError :: String -> m a
   29   liftGhc :: Ghc a -> m a
   30 
   31 -- | A refactoring that only affects one module
   32 type LocalRefactoring dom = UnnamedModule dom -> LocalRefactor dom (UnnamedModule dom)
   33 
   34 -- | The type of a refactoring
   35 type Refactoring dom = ModuleDom dom -> [ModuleDom dom] -> Refactor [RefactorChange dom]
   36 
   37 -- | The type of a refactoring that affects the whole project.
   38 type ProjectRefactoring dom = [ModuleDom dom] -> Refactor [RefactorChange dom]
   39 
   40 -- | The refactoring monad for a given module
   41 type LocalRefactor dom = LocalRefactorT dom Refactor
   42 
   43 -- | The refactoring monad for the whole project
   44 type Refactor = ExceptT String Ghc
   45 
   46 -- | Input and output information for the refactoring
   47 -- TODO: use multiple states instead of Either
   48 newtype LocalRefactorT dom m a
   49   = LocalRefactorT { fromRefactorT :: WriterT [Either GHC.Name (SrcSpan, String, String)]
   50                                               (ReaderT (RefactorCtx dom) m) a
   51                    }
   52   deriving ( Functor, Applicative, Monad, MonadReader (RefactorCtx dom)
   53            , MonadWriter [Either GHC.Name (SrcSpan, String, String)]
   54            , MonadIO, HasDynFlags, ExceptionMonad, GhcMonad )
   55 
   56 -- | The information a refactoring can use
   57 data RefactorCtx dom
   58   = RefactorCtx { refModuleName :: GHC.Module -- ^ The name of the module being refactored. Used for accessing implicit imports.
   59                 , refCtxRoot :: Ann UModule dom SrcTemplateStage
   60                 , refCtxImports :: [Ann UImportDecl dom SrcTemplateStage]
   61                 }
   62 
   63 instance MonadTrans (LocalRefactorT dom) where
   64   lift = LocalRefactorT . lift . lift
   65 
   66 instance RefactorMonad Refactor where
   67   refactError = throwE
   68   liftGhc = lift
   69 
   70 instance RefactorMonad (LocalRefactor dom) where
   71   refactError = lift . refactError
   72   liftGhc = lift . liftGhc
   73
