   42 -- | Performs the given refactoring, transforming it into a Ghc action
   43 runRefactor :: ModuleDom dom -> [ModuleDom dom] -> Refactoring dom -> Ghc (Either String [RefactorChange dom])
   44 runRefactor mod mods trf = runExceptT $ trf mod mods
   45 
   46 -- | Wraps a refactoring that only affects one module. Performs the per-module finishing touches.
   47 localRefactoring :: HasModuleInfo dom => LocalRefactoring dom -> Refactoring dom
   48 localRefactoring ref (name, mod) _
   49   = (\m -> [ContentChanged (name, m)]) <$> localRefactoringRes id mod (ref mod)
   50 
   51 -- | Transform the result of the local refactoring
   52 localRefactoringRes :: HasModuleInfo dom
   53                     => ((UnnamedModule dom -> UnnamedModule dom) -> a -> a)
   54                           -> UnnamedModule dom
   55                           -> LocalRefactor dom a
   56                           -> Refactor a
   57 localRefactoringRes access mod trf
   58   = let init = RefactorCtx (semanticsModule $ mod ^. semantics) mod (mod ^? modImports&annList)
   59      in flip runReaderT init $ do (mod, recorded) <- runWriterT (fromRefactorT trf)
   60                                   return $ access (insertText (rights recorded) . addGeneratedImports (lefts recorded)) mod

   62 -- | Re-inserts the elements removed from the AST that should be kept (for example preprocessor directives)
   63 insertText :: SourceInfoTraversal p => [(SrcSpan,String,String)] -> p dom SrcTemplateStage -> p dom SrcTemplateStage
   64 insertText [] p = p
   65 insertText inserted p
   66   -- this traverses the AST and finds the positions where the removed elements can be added
   67   = evalState (sourceInfoTraverseUp (SourceInfoTrf
   68                 (\stn -> sourceTemplateNodeElems !~ takeWhatPrecedesElem (stn ^. sourceTemplateNodeRange) $ stn)
   69                 (srcTmpSeparators !~ takeWhatPrecedesSep)
   70                 pure) (return ()) (return ()) p) (map Right $ sortOn (^. _1) inserted)
   71   where
   72    -- insert fragments into list separators
   73    takeWhatPrecedesSep :: [([SourceTemplateTextElem], SrcSpan)] -> State [Either SrcSpan (SrcSpan,String,String)] [([SourceTemplateTextElem], SrcSpan)]
   74    takeWhatPrecedesSep seps = takeWhatPrecedes Nothing (Just . (^. _2))
   75                                                (\str -> _1 .- (++ [StayingText str ""]))
   76                                                (\str -> _1 .- ([StayingText str ""] ++))
   77                                                seps
   78 
   79    -- insert fragments into AST elements
   80    takeWhatPrecedesElem :: SrcSpan -> [SourceTemplateElem] -> State [Either SrcSpan (SrcSpan,String,String)] [SourceTemplateElem]
   81    takeWhatPrecedesElem rng elems = takeWhatPrecedes (Just rng) (^? sourceTemplateTextRange)
   82                                                                 (\s -> sourceTemplateTextElem .- (++ [StayingText s ""]))
   83                                                                 (\s -> sourceTemplateTextElem .- ([StayingText s ""] ++))
   84                                                                 elems
   85 
   86    -- finds the position of the fragment where there are elements in the template both before and after the fragment
   87    -- puts holes into the list of inserted fragments where child elements are located
   88    -- uses these holes to determine where should the fragment be added
   89    takeWhatPrecedes :: Maybe SrcSpan -> (a -> Maybe SrcSpan) -> (String -> a -> a) -> (String -> a -> a) -> [a] -> State [Either SrcSpan (SrcSpan,String,String)] [a]
   90    takeWhatPrecedes rng access append prepend elems
   91      | ranges <- mapMaybe access elems
   92      , not (null ranges)
   93      = do let start = srcSpanStart $ fromMaybe (head ranges) rng
   94               end = srcSpanEnd $ fromMaybe (last ranges) rng
   95           toInsert <- get
   96           let (prefix,rest) = break ((>= start) . srcSpanStart . either id (\(sp,_,_) -> sp)) toInsert
   97               (middle,suffix) = break ((> end) . srcSpanEnd . either id (\(sp,_,_) -> sp)) rest
   98           put $ prefix ++ Left (mkSrcSpan start end) : suffix
   99           return $ mergeInserted access append prepend False middle elems
  100      where mergeInserted :: (a -> Maybe SrcSpan) -> (String -> a -> a) -> (String -> a -> a) -> Bool -> [Either SrcSpan (SrcSpan,String,String)] -> [a] -> [a]
  101            -- no fragments left
  102            mergeInserted _ _ _ _ [] elems = elems
  103            mergeInserted access append prepend prep insert@(Right (insertSpan,insertStr,ln):toInsert) (fstElem:elems)
  104               -- insert a fragment to the end of the current element if the next elment is after the fragment
  105               | Just _ <- access fstElem -- TODO: is this needed?
  106               , not prep && case mapMaybe access elems of sp:_ -> srcSpanStart sp >= srcSpanEnd insertSpan
  107                                                           _ -> True
  108               = mergeInserted access append prepend prep toInsert (append (ln ++ insertStr ++ ln) fstElem : elems)
  109               -- insert the fragment before the current elem if we need an element before (we skipped a child)
  110               -- and the current element is after the inserted fragment
  111               | Just fstElemSpace <- access fstElem
  112               , prep && srcSpanStart fstElemSpace >= srcSpanEnd insertSpan
  113               = mergeInserted access append prepend prep toInsert (prepend (ln ++ insertStr ++ ln) fstElem : elems)
  114               | isJust (access fstElem) && prep
  115               = mergeInserted access append prepend False insert (fstElem : elems) -- switch back to append mode
  116               | otherwise
  117               = fstElem : mergeInserted access append prepend (if isJust (access fstElem) then False else prep) insert elems -- move on and switch back to append mode
  118            -- when found a hole
  119            mergeInserted access append prepend prep insert@(Left sp : toInsert) (fstElem:elems)
  120               | Just fstElemSpace <- access fstElem
  121               = if srcSpanStart fstElemSpace > srcSpanEnd sp
  122                   -- switch to prepend mode
  123                   then mergeInserted access append prepend True toInsert (fstElem:elems)
  124                   -- skip elements that are not after the fragment
  125                   else fstElem : mergeInserted access append prepend prep insert elems
  126               | otherwise
  127               = fstElem : mergeInserted access append prepend True toInsert elems -- switch to prepend mode and move on
  128            mergeInserted _ _ _ _ _ [] = [] -- maybe error
  129    takeWhatPrecedes _ _ _ _ elems = return elems

  131 -- | Adds the imports that bring names into scope that are needed by the refactoring
  132 addGeneratedImports :: [GHC.Name] -> Ann UModule dom SrcTemplateStage -> Ann UModule dom SrcTemplateStage
  133 addGeneratedImports names m = modImports&annListElems .- (++ addImports names) $ m
  134   where addImports :: [GHC.Name] -> [Ann UImportDecl dom SrcTemplateStage]
  135         addImports names = map createImport $ groupBy ((==) `on` GHC.nameModule) $ filter (isJust . GHC.nameModule_maybe) $ nub $ sort names
  136 
  137         -- TODO: group names like constructors into correct IESpecs
  138         createImport :: [GHC.Name] -> Ann UImportDecl dom SrcTemplateStage
  139         -- works on groupby result, so list is nonempty
  140         createImport names = mkImportDecl False True False Nothing (mkModuleName $ GHC.moduleNameString $ GHC.moduleName $ GHC.nameModule $ head names)
  141                                           Nothing (Just $ mkImportSpecList (map (\n -> mkIESpec (mkUnqualName' n) Nothing) names))

  153 qualifiedName :: GHC.Name -> String
  154 qualifiedName name = case GHC.nameModule_maybe name of
  155   Just mod -> GHC.moduleNameString (GHC.moduleName mod) ++ "." ++ GHC.occNameString (GHC.nameOccName name)
  156   Nothing -> GHC.occNameString (GHC.nameOccName name)

  158 referenceName :: (HasImportInfo dom, HasModuleInfo dom) => GHC.Name -> LocalRefactor dom (Ann UName dom SrcTemplateStage)
  159 referenceName = referenceName' mkQualName'

  162 referenceOperator = referenceName' mkQualOp'

  164 -- | Create a name that references the definition. Generates an import if the definition is not yet imported.
  165 referenceName' :: (HasImportInfo dom, HasModuleInfo dom)
  166                => ([String] -> GHC.Name -> Ann nt dom SrcTemplateStage) -> GHC.Name -> LocalRefactor dom (Ann nt dom SrcTemplateStage)
  167 referenceName' makeName name
  168   | name `elem` registeredNamesFromPrelude || qualifiedName name `elem` otherNamesFromPrelude
  169   = return $ makeName [] name -- imported from prelude
  170   | otherwise
  171   = do RefactorCtx {refCtxRoot = mod, refCtxImports = imports, refModuleName = thisModule} <- ask
  172        if maybe True (thisModule ==) (GHC.nameModule_maybe name)
  173          then return $ makeName [] name -- in the same module, use simple name
  174          else let possibleImports = filter ((name `elem`) . (\imp -> semanticsImported $ imp ^. semantics)) imports
  175                   fromPrelude = name `elem` semanticsImplicitImports (mod ^. semantics)
  176                in if | fromPrelude -> return $ makeName [] name
  177                      | null possibleImports -> do tell [Left name]
  178                                                   return $ makeName (moduleParts name) name
  179                      | otherwise -> return $ referenceBy makeName name possibleImports
  180                                      -- use it according to the best available import
  181   where moduleParts = maybe [] (splitOn "." . GHC.moduleNameString . GHC.moduleName) . GHC.nameModule_maybe
  182 
  183 -- | Reference the name by the shortest suitable import
  184 referenceBy :: ([String] -> GHC.Name -> Ann nt dom SrcTemplateStage) -> GHC.Name -> [Ann UImportDecl dom SrcTemplateStage] -> Ann nt dom SrcTemplateStage
  185 referenceBy makeName name imps =
  186   let prefixes = map importQualifier imps
  187    in makeName (minimumBy (compare `on` (length . concat)) prefixes) name
  188   where importQualifier :: Ann UImportDecl dom SrcTemplateStage -> [String]
  189         importQualifier imp
  190           = if isJust (imp ^? importQualified&annJust)
  191               then case imp ^? importAs&annJust&importRename of
  192                       Nothing -> splitOn "." (imp ^. importModule&moduleNameString) -- fully qualified import
  193                       Just asName -> splitOn "." (asName ^. moduleNameString) -- the name given by as clause
  194               else [] -- unqualified import
