   42 addTypeInfos :: LHsBinds GhcTc -> Ann AST.UModule (Dom GhcRn) RangeStage -> Ghc (Ann AST.UModule IdDom RangeStage)
   43 addTypeInfos bnds mod = do
   44   ut <- liftIO mkUnknownType
   45   let getType = getType' ut
   46   fixities <- getFixities
   47   -- evaluates fixities to normal form before going on, otherwise the reference to session would be kept
   48   liftIO $ evaluate (forceElements fixities)
   49   let createCName sc def id = mkCNameInfo sc def id fixity
   50         where fixity = if any (any ((getOccName id ==) . getOccName . (^. _1))) (drop 1 sc)
   51                           then Nothing
   52                           else fmap (snd . snd) $ List.find (\(mod,(occ,_)) -> Just mod == (nameModule_maybe $ varName id) && occ == getOccName id) fixities
   53   evalStateT (semaTraverse
   54     (AST.SemaTrf
   55       (\ni -> case (AST.semanticsSourceInfo ni, AST.semanticsName ni) of
   56                 (_, Just name) -> lift $ createCName (AST.semanticsScope ni) (AST.semanticsDefining ni) <$> getType name
   57                 (Just l@(RealSrcSpan loc), _)
   58                   -> case Map.lookup l locMapping of
   59                             Just id -> return $ createCName (AST.semanticsScope ni) (AST.semanticsDefining ni) id
   60                             _ -> do (none,rest) <- gets (break ((\case (RealSrcSpan sp) -> sp `containsSpan` loc) . fst))
   61                                     case rest of [] -> throw $ ConvertionProblem emptyCallStack (RealSrcSpan loc) "Ambiguous or implicit name missing"
   62                                                  ((_,id):more) -> do put (none ++ more)
   63                                                                      return $ createCName (AST.semanticsScope ni) (AST.semanticsDefining ni) id
   64                 _ -> convProblem "addTypeInfos: Cannot access a the semantics of a name.")
   65       pure fetchLitType (lift . trfImportInfoM getType) (lift . trfModuleInfoM getType) pure
   66         pure) mod) (extractSigIds bnds ++ extractSigBindIds bnds)
   67   where locMapping = Map.fromList $ map (\(L l id) -> (l, id)) $ extractExprIds bnds
   68         getType' :: Type -> Name -> Ghc Id
   69         getType' ut name = fromMaybe (mkVanillaGlobal name ut) <$> ((<|> Map.lookup name ids) <$> getTopLevelId name)
   70         ids = Map.fromList $ map (\id -> (getName id, id)) $ extractTypes bnds
   71 
   72         extractTypes :: LHsBinds GhcTc -> [Id]
   73         extractTypes = concatMap universeBi . bagToList
   74 
   75         fetchLitType :: Monad m => PreLiteralInfo -> m LiteralInfo
   76         fetchLitType (RealLiteralInfo t) = return $ LiteralInfo t
   77         fetchLitType (PreLiteralInfo sp) = return $ LiteralInfo $ fromMaybe (convProblem $ "cannot lookup type of literal at: " ++ shortShowSpanWithFile sp) $ lookup sp decompedOverloadedLits
   78 
   79         -- NOTE: Only overloaded literals' type information is collected here
   80         --       Might not be complete
   81         --       Collecting only HsOverLits doesn't work,
   82         --       because they contain only UnhelpfulSrcSpans
   83         decompedOverloadedLits :: [(SrcSpan, Type)]
   84         decompedOverloadedLits = catMaybes $ map decompExprLit exprLits ++ map decompPatLit patLits
   85           where exprLits :: [LHsExpr GhcTc]
   86                 exprLits = concatMap universeBi . bagToList $ bnds
   87 
   88                 decompExprLit :: LHsExpr GhcTc -> Maybe (SrcSpan, Type)
   89                 decompExprLit (L loc (HsOverLit _ lit)) = Just (loc, ol_type (ol_ext lit))
   90                 decompExprLit x = Nothing
   91 
   92                 patLits :: [LPat GhcTc]
   93                 patLits = concatMap universeBi . bagToList $ bnds
   94 
   95                 decompPatLit :: LPat GhcTc -> Maybe (SrcSpan, Type)
   96                 decompPatLit (L loc (NPat llit _ _ _)) = Just (loc, llit)
   97                 decompPatLit x = Nothing
   98 
   99 
  100         mkUnknownType :: IO Type
  101         mkUnknownType = do
  102           tUnique <- mkSplitUniqSupply 'x'
  103           return $ mkTyVarTy $ mkVanillaGlobal (mkSystemName (uniqFromSupply tUnique) (mkDataOcc "TypeNotFound")) (mkTyConTy anyTyCon)
  104 
  105         getFixities :: Ghc [(Module, (OccName, GHC.Fixity))]
  106         getFixities = do env <- getSession
  107                          pit <- liftIO $ eps_PIT <$> hscEPS env
  108                          let hpt = hsc_HPT env
  109                              ifaces = moduleEnvElts pit ++ map hm_iface (eltsUDFM hpt)
  110                          return $ concatMap (\mi -> map (mi_module mi, ) $ mi_fixities mi) ifaces
  111 
  112 extractExprIds :: LHsBinds GhcTc -> [Located Id]
  113         -- expressions like HsRecFld are removed from the typechecked representation, they are replaced by HsVar
  114 extractExprIds = catMaybes . map (\case L l (HsVar _ (L _ n) :: HsExpr GhcTc) -> Just (L l n)
  115                                         L l (HsWrap _ _ (HsVar _ (L _ n))) -> Just (L l n)
  116                                         _ -> Nothing
  117                                  ) . concatMap universeBi . bagToList
  118 
  119 extractSigIds :: LHsBinds GhcTc -> [(SrcSpan,Id)]
  120 extractSigIds = filter (isGoodSrcSpan . fst)
  121                   . concat
  122                   . map (\case L l bs@(AbsBinds { abs_sig = True } :: HsBind GhcTc)
  123                                  -> map (l,) $ concatMap getImplVars (abs_ev_binds bs)
  124                                _ -> [] )
  125                   . concatMap universeBi . bagToList
  126   where getImplVars (EvBinds evbnds) = catMaybes $ map getEvVar $ bagToList evbnds
  127         getImplVars _                = []
  128         getEvVar (EvBind lhs _ False) = Just lhs
  129         getEvVar _                    = Nothing
  130 
  131 extractSigBindIds :: LHsBinds GhcTc -> [(SrcSpan,Id)]
  132 extractSigBindIds = filter (isGoodSrcSpan . fst)
  133                       . catMaybes
  134                       . map (\case L l (IPBind _ (Right id) _ :: IPBind GhcTc) -> Just (l,id)
  135                                    _                                         -> Nothing )
  136                       . concatMap universeBi
  137                       . bagToList
