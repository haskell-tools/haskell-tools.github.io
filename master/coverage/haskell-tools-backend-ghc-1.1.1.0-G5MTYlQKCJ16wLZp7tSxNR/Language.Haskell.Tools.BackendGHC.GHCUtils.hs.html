   30 class (OutputableBndrId name) => GHCName name where
   31   rdrName :: IdP name -> RdrName
   32   getFromNameUsing :: Applicative f => (Name -> Ghc (f Id)) -> Name -> Ghc (f (IdP name))
   33   getBindsAndSigs :: HsValBinds name -> ([LSig name], LHsBinds name)
   34   nameFromId :: Id -> IdP name
   35   fieldOccToId :: RdrName -> XCFieldOcc name -> IdP name
   36   nameIfThereIs :: IdP name -> Maybe Name

   38 instance GHCName GhcPs where
   39   rdrName = id
   40   getFromNameUsing _ n = return $ pure (nameRdrName n)
   41   getBindsAndSigs (ValBinds _ binds sigs) = (sigs, binds)
   42   getBindsAndSigs _ = error "ValBindsOut: ValBindsOut in parsed source"
   43   nameFromId = nameRdrName . getName
   44   fieldOccToId rdr _ = rdr
   45   nameIfThereIs _ = Nothing

   47 occName :: forall n . GHCName n => IdP n -> OccName
   48 occName = rdrNameOcc . rdrName @n

   50 instance GHCName GhcRn where
   51   rdrName = nameRdrName
   52   getFromNameUsing f n = fmap (nameFromId @GhcRn) <$> f n
   53   getBindsAndSigs (XValBindsLR (NValBinds bindGroups sigs)) = (sigs, unionManyBags (map snd bindGroups))
   54   getBindsAndSigs _ = error "getBindsAndSigs: ValBindsIn in renamed source"
   55   nameFromId = getName
   56   fieldOccToId _ name = name
   57   nameIfThereIs name = Just name

   65 -- | Loading ids for top-level ghc names
   66 getTopLevelId :: GHC.Name -> Ghc (Maybe GHC.Id)
   67 getTopLevelId name =
   68     lookupName name >>= \case
   69       Just (AnId id) -> return (Just id)
   70       Just (AConLike (RealDataCon dc)) -> return $ Just $ mkVanillaGlobal name (dataConUserType dc)
   71       Just (AConLike (PatSynCon ps)) -> return $ Just $ mkVanillaGlobal name (createPatSynType ps)
   72       Just (ATyCon tc) -> return $ Just $ mkVanillaGlobal name (tyConKind tc)
   73       _ -> return Nothing
   74   where createPatSynType patSyn = case patSynSig patSyn of (_, _, _, _, args, res) -> mkFunTys args res

  241 {-
  242 -- | Tries to simplify the type that has HsAppsTy before renaming. Does not always provide the correct form.
  243 -- Treats each operator as if they are of equivalent precedence and always left-associative.
  244 cleanHsType :: forall n . (OutputableBndrId n {-, SourceTextX n -}) => HsType n -> HsType n
  245 -- for some reason * is considered infix
  246 cleanHsType (HsAppsTy apps) = unLoc $ guessType apps
  247   where guessType :: OutputableBndrId n => [LHsAppType n] -> LHsType n
  248         guessType (L l (HsAppInfix n) : rest) -- must be a prefix actually
  249           = guessType' (L l (HsTyVar NotPromoted n)) rest
  250         guessType (L _ (HsAppPrefix t) : rest) = guessType' t rest
  251         guessType [] = error $ "guessType: empty" ++ showSDocUnsafe (ppr apps)
  252 
  253         guessType' :: LHsType n -> [LHsAppType n] -> LHsType n
  254         guessType' fun (L _ (HsAppPrefix t) : rest) = guessType' (hsAppTy fun t) rest
  255         guessType' fun (L l (HsAppInfix n) : rest)
  256           -- TODO: find a better check
  257           | showSDocUnsafe (ppr n) == "*" = guessType' (hsAppTy fun (L l (HsTyVar NotPromoted n))) rest
  258         guessType' left (L _ (HsAppInfix n) : right) = hsOpTy left n (guessType right)
  259         guessType' t [] = t
  260 
  261         hsAppTy :: LHsType n -> LHsType n -> LHsType n
  262         hsAppTy t1 t2 = L (getLoc t1 `combineSrcSpans` getLoc t2) $ HsAppTy t1 t2
  263 
  264         hsOpTy :: LHsType n -> Located (IdP n) -> LHsType n -> LHsType n
  265         hsOpTy t1 n t2 = L (getLoc t1 `combineSrcSpans` getLoc t2) $ HsOpTy t1 n t2
  266 cleanHsType t = t
  267 -}

  269 mergeFixityDefs :: [Located (FixitySig n)] -> [Located (FixitySig n)]
  270 mergeFixityDefs (s@(L l _) : rest)
  271   = let (same, different) = partition ((== l) . getLoc) rest
  272      in foldl mergeWith s (map unLoc same) : mergeFixityDefs different
  273   where mergeWith (L l (FixitySig x names fixity)) (FixitySig _ otherNames _) = L l (FixitySig x (names ++ otherNames) fixity)
  274 mergeFixityDefs [] = []

  276 getGroupRange :: HsGroup (GhcPass n) -> SrcSpan
  277 getGroupRange (HsGroup {..})
  278   = foldr combineSrcSpans noSrcSpan locs
  279   where locs = [getHsValRange hs_valds] ++ map getLoc hs_splcds ++ map getLoc (concatMap group_tyclds hs_tyclds)
  280                  ++ map getLoc (concatMap group_roles hs_tyclds)
  281                  ++ map getLoc hs_derivds ++ map getLoc hs_fixds ++ map getLoc hs_defds
  282                  ++ map getLoc hs_fords ++ map getLoc hs_warnds ++ map getLoc hs_annds ++ map getLoc hs_ruleds
  283                  ++ map getLoc hs_docs

  285 getHsValRange :: HsValBinds (GhcPass n) -> SrcSpan
  286 getHsValRange (ValBinds _ vals sig) = foldr combineSrcSpans noSrcSpan $ map getLoc (bagToList vals) ++ map getLoc sig
  287 getHsValRange ((XValBindsLR (NValBinds vals sig))) = foldr combineSrcSpans noSrcSpan $ concatMap (map getLoc . bagToList . snd) vals ++ map getLoc sig
