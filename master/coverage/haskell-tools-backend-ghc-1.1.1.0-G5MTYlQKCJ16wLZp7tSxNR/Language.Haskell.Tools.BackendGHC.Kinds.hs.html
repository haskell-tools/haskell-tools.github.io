   32 trfKindSig :: (TransformName n r, Outputable (HsType n), Data (HsType n), n ~ GhcPass p) 
   33            => Maybe (LHsKind n) -> Trf (AnnMaybeG AST.UKindConstraint (Dom r) RangeStage)
   34 trfKindSig = trfMaybe "" "" trfKindSig'
   35 
   36 trfKindSig' :: (TransformName n r, Outputable (HsType n), Data (HsType n), n ~ GhcPass p) 
   37             => Located (HsKind n) -> Trf (Ann AST.UKindConstraint (Dom r) RangeStage)
   38 trfKindSig' k = annLocNoSema (combineSrcSpans (getLoc k) <$> (tokenBefore (srcSpanStart (getLoc k)) AnnDcolon))
   39                              (AST.UKindConstraint <$> trfLocNoSema trfKind' k)

   41 trfKind :: (TransformName n r, Outputable (HsType n), Data (HsType n), n ~ GhcPass p) => Located (HsKind n) -> Trf (Ann AST.UKind (Dom r) RangeStage)
   42 trfKind = trfLocNoSema trfKind'
   43 
   44 trfKind' :: forall n r p . (TransformName n r, Outputable (HsType n), Data (HsType n), n ~ GhcPass p) => HsKind n -> Trf (AST.UKind (Dom r) RangeStage)
   45 trfKind' = trfKind'' where
   46   trfKind'' (HsTyVar _ _ (rdrName @n . unLoc -> Exact n))
   47     | isWiredInName n && occNameString (nameOccName n) == "*"
   48     = pure AST.UStarKind
   49     | isWiredInName n && occNameString (nameOccName n) == "#"
   50     = pure AST.UUnboxKind
   51   trfKind'' (HsStarTy _ _) = pure AST.UStarKind
   52   trfKind'' (HsParTy _ kind) = AST.UParenKind <$> trfKind kind
   53   trfKind'' (HsParTy _ kind) = AST.UParenKind <$> trfKind kind
   54   trfKind'' (HsFunTy _ k1 k2) = AST.UFunKind <$> trfKind k1 <*> trfKind k2
   55   trfKind'' (HsAppTy _ k1 k2) = AST.UAppKind <$> trfKind k1 <*> trfKind k2
   56   trfKind'' (HsOpTy _ k1 op k2) = AST.UInfixAppKind <$> trfKind k1 <*> trfOperator @n op <*> trfKind k2
   57   trfKind'' (HsTyVar _ _ kv) = transformingPossibleVar kv (AST.UVarKind <$> trfName @n kv)
   58   trfKind'' (HsListTy _ kind) = AST.UListKind <$> trfKind kind
   59   trfKind'' (HsTupleTy _ _ kinds) = AST.UTupleKind <$> makeList ", " atTheStart (mapM trfKind kinds)
   60   trfKind'' pt@(HsExplicitListTy {}) = AST.UPromotedKind <$> annContNoSema (trfPromoted' trfKind' pt)
   61   trfKind'' pt@(HsExplicitTupleTy {}) = AST.UPromotedKind <$> annContNoSema (trfPromoted' trfKind' pt)
   62   trfKind'' pt@(HsTyLit {}) = AST.UPromotedKind <$> annContNoSema (trfPromoted' trfKind' pt)
   63   trfKind'' t = AST.UTypeKind <$> annContNoSema (trfType' t)
   64 
   65 trfPromoted' :: forall n r a . (TransformName n r, HasNoSemanticInfo (Dom r) a, Outputable (HsType n), Data (HsType n))
   66                   => (HsType n -> Trf (a (Dom r) RangeStage)) -> HsType n -> Trf (AST.UPromoted a (Dom r) RangeStage)
   67 trfPromoted' _ (HsTyLit _ (HsNumTy _ int)) = pure $ AST.UPromotedInt int
   68 trfPromoted' _ (HsTyLit _ (HsStrTy _ str)) = pure $ AST.UPromotedString (unpackFS str)
   69 trfPromoted' _ (HsTyVar _ _ name) = AST.UPromotedCon <$> trfName @n name
   70 trfPromoted' f (HsExplicitListTy _ _ elems) = AST.UPromotedList <$> between AnnOpenS AnnCloseS (trfAnnList ", " f elems)
   71 trfPromoted' f (HsExplicitTupleTy _ elems) = AST.UPromotedTuple <$> between AnnOpenP AnnCloseP (trfAnnList ", " f elems)
   72 trfPromoted' _ t = unhandledElement "promoted type/kind" t
