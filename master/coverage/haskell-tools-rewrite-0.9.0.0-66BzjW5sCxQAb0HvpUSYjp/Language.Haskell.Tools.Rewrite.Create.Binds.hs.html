    1 -- | Generation of binding-level AST fragments for refactorings.
    2 -- The bindings defined here create a the annotated version of the AST constructor with the same name.
    3 -- For example, @mkMatch@ creates the annotated version of the @UMatch@ constructor.
    4 {-# LANGUAGE OverloadedStrings
    5            , TypeFamilies
    6            #-}
    7 module Language.Haskell.Tools.Rewrite.Create.Binds where
    8 
    9 import Data.String (IsString(..))
   10 import Language.Haskell.Tools.AST
   11 import Language.Haskell.Tools.PrettyPrint.Prepare
   12 import Language.Haskell.Tools.Rewrite.Create.Patterns (mkVarPat)
   13 import Language.Haskell.Tools.Rewrite.Create.Utils (mkAnn, mkAnnList, mkAnnMaybe)
   14 import Language.Haskell.Tools.Rewrite.ElementTypes

   16 -- | A simplified function to generate simple value bindings without local definitions, guards or complex lhs.
   17 mkSimpleBind' :: Name dom -> Expr dom -> ValueBind dom
   18 mkSimpleBind' n e = mkSimpleBind (mkVarPat n) (mkUnguardedRhs e) Nothing
   19 
   20 -- | Creates a value binding (@ v = "12" @).
   21 mkSimpleBind :: Pattern dom -> Rhs dom -> Maybe (LocalBinds dom) -> ValueBind dom
   22 mkSimpleBind p r l = mkAnn (child <> child <> child) (USimpleBind p r (mkAnnMaybe opt l))
   23 
   24 -- | Creates a function binding (@ f 0 = 1; f x = x @). All matches must have the same name.
   25 mkFunctionBind :: [Match dom] -> ValueBind dom
   26 mkFunctionBind = mkAnn child . UFunBind . mkAnnList (indented list)
   27 
   28 -- | A simplified function for creating function bindings without local definitions or guards.
   29 mkFunctionBind' :: Name dom -> [([Pattern dom], Expr dom)] -> ValueBind dom
   30 mkFunctionBind' name matches = mkFunctionBind $ map (\(args, rhs) -> mkMatch (mkMatchLhs name args) (mkUnguardedRhs rhs) Nothing) matches
   31 
   32 -- | Creates a clause of function binding
   33 mkMatch :: MatchLhs dom -> Rhs dom -> Maybe (LocalBinds dom) -> Match dom
   34 mkMatch lhs rhs locs
   35   = mkAnn (child <> child <> child)
   36       $ UMatch lhs rhs (mkAnnMaybe (after " " opt) locs)
   37 
   38 -- | Creates a match lhs with the function name and parameter names (@ f a b @)
   39 mkMatchLhs :: Name dom -> [Pattern dom] -> MatchLhs dom
   40 mkMatchLhs n pats = mkAnn (child <> child) $ UNormalLhs n (mkAnnList (after " " $ separatedBy " " list) pats)
   41 
   42 -- | Creates an infix match lhs for an operator (@ a + b @)
   43 mkInfixLhs :: Pattern dom -> Operator dom -> Pattern dom -> [Pattern dom] -> MatchLhs dom
   44 mkInfixLhs lhs op rhs pats
   45   = mkAnn (child <> child <> child <> child) $ UInfixLhs lhs op rhs (mkAnnList (after " " $ separatedBy " " list) pats)
   46 
   47 -- | Local bindings attached to a declaration (@ where x = 42 @)
   48 mkLocalBinds :: [LocalBind dom] -> MaybeLocalBinds dom
   49 mkLocalBinds = mkAnnMaybe (relativeIndented 2 $ after "\nwhere " opt)
   50                      . Just . mkAnn child . ULocalBinds . mkAnnList (indented list)
   51 
   52 mkLocalBinds' :: [LocalBind dom] -> LocalBinds dom
   53 mkLocalBinds' = mkAnn (" where " <> child) . ULocalBinds . mkAnnList (indented list)
   54 
   55 -- | Creates a local binding for a value
   56 mkLocalValBind :: ValueBind dom -> LocalBind dom
   57 mkLocalValBind = mkAnn child . ULocalValBind
   58 
   59 -- | Creates a local type signature
   60 mkLocalTypeSig :: TypeSignature dom -> LocalBind dom
   61 mkLocalTypeSig = mkAnn child . ULocalSignature
   62 
   63 -- | Creates a local fixity declaration
   64 mkLocalFixity :: FixitySignature dom -> LocalBind dom
   65 mkLocalFixity = mkAnn child . ULocalFixity
   66 
   67 -- | Creates a type signature (@ f :: Int -> Int @)
   68 mkTypeSignature :: Name dom -> Type dom -> TypeSignature dom
   69 mkTypeSignature n t = mkAnn (child <> " :: " <> child) (UTypeSignature (mkAnnList (separatedBy ", " list) [n]) t)
   70 
   71 -- | Creates a left-associative fixity declaration (@ infixl 5 +, - @).
   72 mkInfixL :: Int -> Operator dom -> FixitySignature dom
   73 mkInfixL prec op = mkAnn (child <> " " <> child <> " " <> child)
   74                      $ UFixitySignature (mkAnn "infixl" AssocLeft) (mkAnnMaybe opt $ Just $ mkAnn (fromString (show prec)) (Precedence prec)) (mkAnnList (separatedBy ", " list) [op])
   75 
   76 -- | Creates a right-associative fixity declaration (@ infixr 5 +, - @).
   77 mkInfixR :: Int -> Operator dom -> FixitySignature dom
   78 mkInfixR prec op = mkAnn (child <> " " <> child <> " " <> child)
   79                      $ UFixitySignature (mkAnn "infixr" AssocRight) (mkAnnMaybe opt $ Just $ mkAnn (fromString (show prec)) (Precedence prec)) (mkAnnList (separatedBy ", " list) [op])
   80 
   81 -- | Creates a non-associative fixity declaration (@ infix 5 +, - @).
   82 mkInfix :: Int -> Operator dom -> FixitySignature dom
   83 mkInfix prec op = mkAnn (child <> " " <> child <> " " <> child)
   84                     $ UFixitySignature (mkAnn "infix" AssocNone) (mkAnnMaybe opt $ Just $ mkAnn (fromString (show prec)) (Precedence prec)) (mkAnnList (separatedBy ", " list) [op])
   85 
   86 -- | Creates an unguarded right-hand-side (@ = 3 @)
   87 mkUnguardedRhs :: Expr dom -> Rhs dom
   88 mkUnguardedRhs = mkAnn (" = " <> child) . UUnguardedRhs
   89 
   90 -- | Creates an unguarded right-hand-side (@ | x == 1 = 3; | otherwise = 4 @)
   91 mkGuardedRhss :: [GuardedRhs dom] -> Rhs dom
   92 mkGuardedRhss = mkAnn child . UGuardedRhss . mkAnnList (indented list)
   93 
   94 -- | Creates a guarded right-hand side of a value binding (@ | x > 3 = 2 @)
   95 mkGuardedRhs :: [RhsGuard dom] -> Expr dom -> GuardedRhs dom
   96 mkGuardedRhs guards expr = mkAnn ("| " <> child <> " = " <> child) $ UGuardedRhs (mkAnnList (separatedBy ", " list) guards) expr
   97 
   98 -- | Creates a bind statement in a pattern guard (@ Just v <- x @)
   99 mkGuardBind :: Pattern dom -> Expr dom -> RhsGuard dom
  100 mkGuardBind pat expr = mkAnn (child <> " <- " <> child) $ UGuardBind pat expr
  101 
  102 -- | Creates a let statement in a pattern guard (@ let x = 3 @)
  103 mkGuardLet :: [LocalBind dom] -> RhsGuard dom
  104 mkGuardLet = mkAnn ("let " <> child) . UGuardLet . mkAnnList (indented list)
  105 
  106 -- | Creates an expression to check for a pattern guard
  107 mkGuardCheck :: Expr dom -> RhsGuard dom
  108 mkGuardCheck = mkAnn child . UGuardCheck
