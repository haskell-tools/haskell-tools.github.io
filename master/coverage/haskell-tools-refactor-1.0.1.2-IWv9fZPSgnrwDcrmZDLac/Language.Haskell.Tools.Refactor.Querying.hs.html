   27 performQuery :: [QueryChoice] -- ^ The set of available queries
   28                   -> [String] -- ^ The query command
   29                   -> Either FilePath ModuleDom -- ^ The module in which the refactoring is performed
   30                   -> [ModuleDom] -- ^ Other modules
   31                   -> Ghc (Either String Value)
   32 performQuery queries (name:args) mod mods =
   33   case (query, mod, args) of
   34     (Just (LocationQuery _ query), Right mod, sp:_)
   35       -> runExceptT $ query (correctRefactorSpan (snd mod) $ readSrcSpan sp) mod mods
   36     (Just (LocationQuery _ query), _, _)
   37       -> return $ Left $ "The query '" ++ name ++ "' needs one argument: a source range"
   38     (Nothing, _, _)
   39       -> return $ Left $ "Unknown command: " ++ name
   40   where query = find ((== name) . queryName) queries
