   18 -- | Remove a separator from the AST while keeping the textual parts of it that should not be removed (like preprocessor pragmas).
   19 removeSeparator :: ([SourceTemplateTextElem], SrcSpan) -> LocalRefactor ()
   20 removeSeparator (txts, range) = tell [Right (range, intercalate lineEnd staying, lineEnd)]
   21   where staying = catMaybes $ map (\case StayingText str _ -> Just str; _ -> Nothing) txts
   22         lineEnd = head $ (catMaybes $ map (\case StayingText _ lnEnd -> Just lnEnd; _ -> Nothing) txts) ++ [""]
   23 
   24 -- | Remove an element from the AST while keeping the textual parts of it that should not be removed (like preprocessor pragmas).
   25 removeChild :: (SourceInfoTraversal e) => e dom SrcTemplateStage -> LocalRefactor ()
   26 removeChild e = tell $ map Right $ keptText e
   27 
   28 -- | Extracts all text elements that should be kept
   29 keptText :: SourceInfoTraversal e => e dom SrcTemplateStage -> [(SrcSpan,String,String)]
   30 keptText = execWriter . sourceInfoTraverse (SourceInfoTrf
   31   (\ni -> mapM_ writeStaying (mapMaybe (\case (TextElem elems range) -> Just (elems,range)
   32                                               _ -> Nothing)
   33                                        (ni ^? sourceTemplateNodeElems & traversal)) >> return ni)
   34   (\ni -> mapM_ writeStaying (ni ^. srcTmpSeparators) >> return ni)
   35   pure)
   36 
   37 -- | Writes the elements to be kept into a writer monad with ranges and default line ending.
   38 writeStaying :: ([SourceTemplateTextElem], SrcSpan) -> Writer [(SrcSpan,String,String)] ()
   39 writeStaying (txts, range) = tell staying
   40   where staying = mapMaybe (\case StayingText str lnEnd -> Just (range, str, lnEnd)
   41                                   _ -> Nothing) txts
