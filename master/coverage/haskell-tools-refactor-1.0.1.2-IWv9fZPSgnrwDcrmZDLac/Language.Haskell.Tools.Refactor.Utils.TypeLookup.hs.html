   24 -- | Looks up the Type of an entity with an Id of any locality.
   25 -- If the entity being scrutinised is a type variable, it fails.
   26 lookupTypeFromId :: (HasIdInfo' id, GhcMonad m) => id -> MaybeT m GHC.Type
   27 lookupTypeFromId idn
   28   | GHC.isLocalId  . semanticsId $ idn = return . typeOrKindFromId $ idn
   29   | GHC.isGlobalId . semanticsId $ idn = lookupTypeFromGlobalName idn
   30   | otherwise = fail "Couldn't lookup name"
   31 
   32 -- | Looks up the Type or the Kind of an entity that has an Id.
   33 -- Note: In some cases we only get the Kind of the Id (e.g. for type constructors)
   34 typeOrKindFromId :: HasIdInfo' id => id -> GHC.Type
   35 typeOrKindFromId idn = GHC.varType . semanticsId $ idn

   37 -- | Extracts a Type from a TyThing when possible.
   38 typeFromTyThing :: GHC.TyThing -> Maybe GHC.Type
   39 typeFromTyThing (GHC.AnId     idn)  = Just . GHC.varType $ idn
   40 typeFromTyThing (GHC.ATyCon   tc)   = GHC.synTyConRhs_maybe tc
   41 typeFromTyThing (GHC.ACoAxiom _)    = fail "CoAxioms are not supported for type lookup"
   42 typeFromTyThing (GHC.AConLike con)  = handleCon con
   43   where handleCon (GHC.RealDataCon dc) = Just . GHC.dataConUserType $ dc
   44         handleCon (GHC.PatSynCon   pc) = do
   45           (idn,_) <- GHC.patSynBuilder pc
   46           return . GHC.varType $ idn

   48 -- | Looks up a GHC Type from a Haskell Tools Name (given the name is global)
   49 -- For an identifier, it returns its type.
   50 -- For a data constructor, it returns its type.
   51 -- For a pattern synonym, it returns its builder's type.
   52 -- For a type synonym constructor, it returns its right-hand side.
   53 -- For a coaxiom, it fails.
   54 lookupTypeFromGlobalName :: (HasNameInfo' n, GhcMonad m) => n -> MaybeT m GHC.Type
   55 lookupTypeFromGlobalName name = do
   56   sname <- liftMaybe . semanticsName $ name
   57   tt    <- MaybeT    . GHC.lookupName $ sname
   58   liftMaybe . typeFromTyThing $ tt

   61 -- | Looks up the right-hand side (GHC representation)
   62 -- of a Haskell Tools Name corresponding to a type synonym
   63 lookupTypeSynRhs :: (HasNameInfo' n, GhcMonad m) => n -> MaybeT m GHC.Type
   64 lookupTypeSynRhs name = do
   65   sname <- liftMaybe . semanticsName $ name
   66   tt    <- MaybeT    . GHC.lookupName $ sname
   67   tc    <- liftMaybe . tyconFromTyThing $ tt
   68   liftMaybe . GHC.synTyConRhs_maybe $ tc
   69 
   70 -- NOTE: Returns Nothing if it is not a type synonym
   71 lookupSynDef :: GHC.TyThing -> Maybe GHC.TyCon
   72 lookupSynDef syn = do
   73   tycon <- tyconFromTyThing syn
   74   rhs   <- GHC.synTyConRhs_maybe tycon
   75   tyconFromGHCType rhs
   76 
   77 tyconFromTyThing :: GHC.TyThing -> Maybe GHC.TyCon
   78 tyconFromTyThing (GHC.ATyCon tycon) = Just tycon
   79 tyconFromTyThing _ = Nothing
   80 
   81 -- won't bother
   82 tyconFromGHCType :: GHC.Type -> Maybe GHC.TyCon
   83 tyconFromGHCType (GHC.AppTy t1 _) = tyconFromGHCType t1
   84 tyconFromGHCType (GHC.TyConApp tycon _) = Just tycon
   85 tyconFromGHCType _ = Nothing

   88 -- NOTE: Returns false if the type is certainly not a newtype
   89 --       Returns true if it is a newtype or it could not have been looked up
   90 isNewtype :: GhcMonad m => Type -> m Bool
   91 isNewtype t = do
   92   tycon <- runMaybeT . lookupType $ t
   93   return $! maybe True isNewtypeTyCon tycon

   97 lookupType :: GhcMonad m => Type -> MaybeT m GHC.TyThing
   98 lookupType t = do
   99   name  <- liftMaybe . nameFromType $ t
  100   sname <- liftMaybe . semanticsName   $ name
  101   MaybeT . GHC.lookupName $ sname

  103 -- | Looks up a GHC.Class from something that has a type class constructor in it
  104 -- Fails if the argument does not contain a class type constructor
  105 lookupClassWith :: GhcMonad m => (a -> MaybeT m GHC.Name) -> a -> MaybeT m GHC.Class
  106 lookupClassWith getName x = do
  107   sname   <- getName x
  108   tything <- MaybeT . GHC.lookupName $ sname
  109   case tything of
  110     GHC.ATyCon tc | GHC.isClassTyCon tc -> liftMaybe . GHC.tyConClass_maybe $ tc
  111     _ -> fail "TypeLookup.lookupClassWith: Argument does not contain a class type constructor"
  112 
  113 lookupClassFromInstance :: GhcMonad m => InstanceHead -> MaybeT m GHC.Class
  114 lookupClassFromInstance = lookupClassWith instHeadSemName
  115 
  116 lookupClassFromDeclHead :: GhcMonad m => DeclHead -> MaybeT m GHC.Class
  117 lookupClassFromDeclHead = lookupClassWith declHeadSemName

  119 -- | Looks up the right-hand side (GHC representation)
  120 -- of a Haskell Tools Type corresponding to a type synonym
  121 semanticsTypeSynRhs :: GhcMonad m => Type -> MaybeT m GHC.Type
  122 semanticsTypeSynRhs ty = (liftMaybe . nameFromType $ ty) >>= lookupTypeSynRhs
  123 
  124 -- | Converts a global Haskell Tools type to a GHC type
  125 semanticsType :: GhcMonad m => Type -> MaybeT m GHC.Type
  126 semanticsType ty = (liftMaybe . nameFromType $ ty) >>= lookupTypeFromGlobalName
  127 
  128 -- | Extracts the name of a type
  129 -- In case of a type application, it finds the type being applied
  130 nameFromType :: Type -> Maybe Name
  131 nameFromType (TypeApp f _)    = nameFromType f
  132 nameFromType (ParenType x)    = nameFromType x
  133 nameFromType (KindedType t _) = nameFromType t
  134 nameFromType (BangType t)     = nameFromType t
  135 nameFromType (LazyType t)     = nameFromType t
  136 nameFromType (UnpackType t)   = nameFromType t
  137 nameFromType (NoUnpackType t) = nameFromType t
  138 nameFromType (VarType x)      = Just x
  139 nameFromType _                = Nothing
  140 
  141 isNewtypeTyCon :: GHC.TyThing -> Bool
  142 isNewtypeTyCon (GHC.ATyCon tycon) = GHC.isNewTyCon tycon
  143 isNewtypeTyCon _ = False

  145 -- | Decides whether a given name is a standard Haskell98 data constructor.
  146 -- Fails if not given a proper name.
  147 isVanillaDataConNameM :: (HasNameInfo' n, GhcMonad m) => n -> MaybeT m Bool
  148 isVanillaDataConNameM name = do
  149   sname <- liftMaybe . semanticsName  $ name
  150   tt    <- MaybeT    . GHC.lookupName $ sname
  151   dc    <- liftMaybe . extractDataCon $ tt
  152   return . GHC.isVanillaDataCon $ dc
  153   where extractDataCon (GHC.AConLike (GHC.RealDataCon dc)) = Just dc
  154         extractDataCon  _                                  = Nothing
