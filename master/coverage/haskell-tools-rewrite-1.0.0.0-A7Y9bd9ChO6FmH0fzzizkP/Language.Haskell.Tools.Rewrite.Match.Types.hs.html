    1 -- | UPattern matching on type-level AST fragments for refactorings.
    2 {-# LANGUAGE PatternSynonyms #-}
    3 module Language.Haskell.Tools.Rewrite.Match.Types where
    4 
    5 import Language.Haskell.Tools.AST
    6 import Language.Haskell.Tools.Rewrite.ElementTypes
    7 
    8 -- * Types
    9 
   10 -- | Forall types (@ forall x y . type @)
   11 pattern ForallType :: TyVarList -> Type -> Type
   12 pattern ForallType vars t <- Ann _ (UTyForall vars t)
   13 
   14 -- | Type with a context (@ forall x y . type @)
   15 pattern CtxType :: Context -> Type -> Type
   16 pattern CtxType ctx t <- Ann _ (UTyCtx ctx t)
   17 
   18 -- | Function types (@ a -> b @)
   19 pattern FunctionType :: Type -> Type -> Type
   20 pattern FunctionType at rt <- Ann _ (UTyFun at rt)
   21 
   22 -- | Tuple types (@ (a,b) @)
   23 pattern TupleType :: TypeList -> Type
   24 pattern TupleType args <- Ann _ (UTyTuple args)
   25 
   26 -- | Unboxed tuple types (@ (\#a,b\#) @)
   27 pattern UnboxedTupleType :: TypeList -> Type
   28 pattern UnboxedTupleType args <- Ann _ (UTyUnbTuple args)
   29 
   30 -- | List type with special syntax (@ [a] @)
   31 pattern ListType :: Type -> Type
   32 pattern ListType t <- Ann _ (UTyList t)
   33 
   34 -- | Parallel array type (@ [:a:] @)
   35 pattern ParArrayType :: Type -> Type
   36 pattern ParArrayType t <- Ann _ (UTyParArray t)
   37 
   38 -- | Type application (@ F a @)
   39 pattern TypeApp :: Type -> Type -> Type
   40 pattern TypeApp ft at <- Ann _ (UTyApp ft at)
   41 
   42 -- | Infix type constructor (@ (a <: b) @)
   43 pattern InfixTypeApp :: Type -> Operator -> Type -> Type
   44 pattern InfixTypeApp left op right <- Ann _ (UTyInfix left op right)
   45 
   46 -- | Type surrounded by parentheses (@ (T a) @)
   47 pattern ParenType :: Type -> Type
   48 pattern ParenType t <- Ann _ (UTyParen t)
   49 
   50 -- | Type variable or constructor (@ a @)
   51 pattern VarType :: Name -> Type
   52 pattern VarType n <- Ann _ (UTyVar n)
   53 
   54 -- | Type with explicit kind signature (@ a :: * @)
   55 pattern KindedType :: Type -> Kind -> Type
   56 pattern KindedType t k <- Ann _ (UTyKinded t k)
   57 
   58 -- | Strict type marked with @!@.
   59 pattern BangType :: Type -> Type
   60 pattern BangType n <- Ann _ (UTyBang n)
   61 
   62 -- | Lazy type marked with @~@. (Should only be used if @Strict@ or @StrictData@ language extension is used)
   63 pattern LazyType :: Type -> Type
   64 pattern LazyType n <- Ann _ (UTyLazy n)
   65 
   66 -- | Strict type marked with UNPACK pragma. (Usually contains the bang mark.)
   67 pattern UnpackType :: Type -> Type
   68 pattern UnpackType n <- Ann _ (UTyUnpack n)
   69 
   70 -- | Strict type marked with NOUNPACK pragma. (Usually contains the bang mark.)
   71 pattern NoUnpackType :: Type -> Type
   72 pattern NoUnpackType t <- Ann _ (UTyNoUnpack t)
   73 
   74 -- | A wildcard type (@ _ @) with @-XPartialTypeSignatures@
   75 pattern WildcardType :: Type
   76 pattern WildcardType <- Ann _ UTyWildcard
   77 
   78 -- | A named wildcard type (@ _t @) with @-XPartialTypeSignatures@
   79 pattern NamedWildcardType :: Name -> Type
   80 pattern NamedWildcardType n <- Ann _ (UTyNamedWildc n)
   81 
   82 -- | A Template Haskell splice type (@ $(genType) @).
   83 pattern SpliceType :: Splice -> Type
   84 pattern SpliceType spl <- Ann _ (UTySplice spl)
   85 
   86 -- | A Template Haskell splice type (@ $(genType) @).
   87 pattern QuasiQuoteType :: QuasiQuote -> Type
   88 pattern QuasiQuoteType spl <- Ann _ (UTyQuasiQuote spl)
   89 
   90 -- | Numeric value promoted to the type level.
   91 pattern PromotedIntType :: Integer -> Type
   92 pattern PromotedIntType i <- Ann _ (UTyPromoted (Ann _ (UPromotedInt i)))
   93 
   94 -- | String value promoted to the type level.
   95 pattern PromotedStringType :: String -> Type
   96 pattern PromotedStringType s <- Ann _ (UTyPromoted (Ann _ (UPromotedString s)))
   97 
   98 -- | A data constructor value promoted to the type level.
   99 pattern PromotedConType :: Name -> Type
  100 pattern PromotedConType s <- Ann _ (UTyPromoted (Ann _ (UPromotedCon s)))
  101 
  102 -- | A list of elements as a type.
  103 pattern PromotedListType :: TypeList -> Type
  104 pattern PromotedListType elems <- Ann _ (UTyPromoted (Ann _ (UPromotedList elems)))
  105 
  106 -- | A tuple of elements as a type.
  107 pattern PromotedTupleType :: TypeList -> Type
  108 pattern PromotedTupleType elems <- Ann _ (UTyPromoted (Ann _ (UPromotedTuple elems)))
  109 
  110 -- | Kind of the unit value @()@.
  111 pattern PromotedUnitType :: Type
  112 pattern PromotedUnitType <- Ann _ (UTyPromoted (Ann _ UPromotedUnit))
  113 
  114 -- * Type variable
  115 
  116 -- | Type variable declaration
  117 pattern TyVarDecl :: Name -> TyVar
  118 pattern TyVarDecl n <- Ann _ (UTyVarDecl n _)
  119 
  120 -- | Kinded type variable declaration (@ v :: * @)
  121 pattern KindedTyVarDecl :: Name -> Kind -> TyVar
  122 pattern KindedTyVarDecl n k <- Ann _ (UTyVarDecl n (AnnJust (Ann _ (UKindConstraint k))))
  123 
  124 -- * Contexts
  125 
  126 -- | A context of assertions (@ C a => ... @)
  127 pattern Context :: Assertion -> Context
  128 pattern Context n <- Ann _ (UContext n)
  129 
  130 -- * Assertions
  131 
  132 -- | Class assertion (@Cls x@)
  133 pattern ClassAssert :: Name -> TypeList -> Assertion
  134 pattern ClassAssert n args <- Ann _ (UClassAssert n args)
  135 
  136 -- | Infix class assertion, also contains type equations (@ a ~ X y @)
  137 pattern InfixAssert :: Type -> Operator -> Type -> Assertion
  138 pattern InfixAssert left op right <- Ann _ (UInfixAssert left op right)
  139 
  140 -- | Assertion for implicit parameter binding (@ ?cmp :: a -> a -> Bool @)
  141 pattern ImplicitAssert :: Name -> Type -> Assertion
  142 pattern ImplicitAssert n t <- Ann _ (UImplicitAssert n t)
  143 
  144 -- | A list of assertions (@ (Eq a, Show a) @)
  145 pattern TupleAssert :: [Assertion] -> Assertion
  146 pattern TupleAssert ass <- Ann _ (UTupleAssert (AnnList ass))
