    1 -- | Generation of binding-level AST fragments for refactorings.
    2 -- The bindings defined here create a the annotated version of the AST constructor with the same name.
    3 -- For example, @mkMatch@ creates the annotated version of the @UMatch@ constructor.
    4 {-# LANGUAGE MonoLocalBinds, OverloadedStrings #-}
    5 
    6 module Language.Haskell.Tools.Rewrite.Create.Binds where
    7 
    8 import Data.String (IsString(..))
    9 import Language.Haskell.Tools.AST
   10 import Language.Haskell.Tools.PrettyPrint.Prepare
   11 import Language.Haskell.Tools.Rewrite.Create.Patterns (mkVarPat)
   12 import Language.Haskell.Tools.Rewrite.Create.Utils (mkAnn, mkAnnList, mkAnnMaybe)
   13 import Language.Haskell.Tools.Rewrite.ElementTypes

   15 -- | A simplified function to generate simple value bindings without local definitions, guards or complex lhs.
   16 mkSimpleBind' :: Name -> Expr -> ValueBind
   17 mkSimpleBind' n e = mkSimpleBind (mkVarPat n) (mkUnguardedRhs e) Nothing
   18 
   19 -- | Creates a value binding (@ v = "12" @).
   20 mkSimpleBind :: Pattern -> Rhs -> Maybe LocalBinds -> ValueBind
   21 mkSimpleBind p r l = mkAnn (child <> child <> child) (USimpleBind p r (mkAnnMaybe opt l))
   22 
   23 -- | Creates a function binding (@ f 0 = 1; f x = x @). All matches must have the same name.
   24 mkFunctionBind :: [Match] -> ValueBind
   25 mkFunctionBind = mkAnn child . UFunBind . mkAnnList (indented list)
   26 
   27 -- | A simplified function for creating function bindings without local definitions or guards.
   28 mkFunctionBind' :: Name -> [([Pattern], Expr)] -> ValueBind
   29 mkFunctionBind' name matches = mkFunctionBind $ map (\(args, rhs) -> mkMatch (mkMatchLhs name args) (mkUnguardedRhs rhs) Nothing) matches
   30 
   31 -- | Creates a clause of function binding
   32 mkMatch :: MatchLhs -> Rhs -> Maybe LocalBinds -> Match
   33 mkMatch lhs rhs locs
   34   = mkAnn (child <> child <> child)
   35       $ UMatch lhs rhs (mkAnnMaybe (after " " opt) locs)
   36 
   37 -- | Creates a match lhs with the function name and parameter names (@ f a b @)
   38 mkMatchLhs :: Name -> [Pattern] -> MatchLhs
   39 mkMatchLhs n pats = mkAnn (child <> child) $ UNormalLhs n (mkAnnList (after " " $ separatedBy " " list) pats)
   40 
   41 -- | Creates an infix match lhs for an operator (@ a + b @)
   42 mkInfixLhs :: Pattern -> Operator -> Pattern -> [Pattern] -> MatchLhs
   43 mkInfixLhs lhs op rhs pats
   44   = mkAnn (child <> child <> child <> child) $ UInfixLhs lhs op rhs (mkAnnList (after " " $ separatedBy " " list) pats)
   45 
   46 -- | Local bindings attached to a declaration (@ where x = 42 @)
   47 mkLocalBinds :: [LocalBind] -> MaybeLocalBinds
   48 mkLocalBinds = mkAnnMaybe (relativeIndented 2 $ after "\nwhere " opt)
   49                      . Just . mkAnn child . ULocalBinds . mkAnnList (indented list)
   50 
   51 mkLocalBinds' :: [LocalBind] -> LocalBinds
   52 mkLocalBinds' = mkAnn (" where " <> child) . ULocalBinds . mkAnnList (indented list)
   53 
   54 -- | Creates a local binding for a value
   55 mkLocalValBind :: ValueBind -> LocalBind
   56 mkLocalValBind = mkAnn child . ULocalValBind
   57 
   58 -- | Creates a local type signature
   59 mkLocalTypeSig :: TypeSignature -> LocalBind
   60 mkLocalTypeSig = mkAnn child . ULocalSignature
   61 
   62 -- | Creates a local fixity declaration
   63 mkLocalFixity :: FixitySignature -> LocalBind
   64 mkLocalFixity = mkAnn child . ULocalFixity
   65 
   66 -- | Creates a type signature (@ f :: Int -> Int @)
   67 mkTypeSignature :: Name -> Type -> TypeSignature
   68 mkTypeSignature n t = mkAnn (child <> " :: " <> child) (UTypeSignature (mkAnnList (separatedBy ", " list) [n]) t)
   69 
   70 -- | Creates a left-associative fixity declaration (@ infixl 5 +, - @).
   71 mkInfixL :: Int -> Operator -> FixitySignature
   72 mkInfixL prec op = mkAnn (child <> " " <> child <> " " <> child)
   73                      $ UFixitySignature (mkAnn "infixl" AssocLeft) (mkAnnMaybe opt $ Just $ mkAnn (fromString (show prec)) (Precedence prec)) (mkAnnList (separatedBy ", " list) [op])
   74 
   75 -- | Creates a right-associative fixity declaration (@ infixr 5 +, - @).
   76 mkInfixR :: Int -> Operator -> FixitySignature
   77 mkInfixR prec op = mkAnn (child <> " " <> child <> " " <> child)
   78                      $ UFixitySignature (mkAnn "infixr" AssocRight) (mkAnnMaybe opt $ Just $ mkAnn (fromString (show prec)) (Precedence prec)) (mkAnnList (separatedBy ", " list) [op])
   79 
   80 -- | Creates a non-associative fixity declaration (@ infix 5 +, - @).
   81 mkInfix :: Int -> Operator -> FixitySignature
   82 mkInfix prec op = mkAnn (child <> " " <> child <> " " <> child)
   83                     $ UFixitySignature (mkAnn "infix" AssocNone) (mkAnnMaybe opt $ Just $ mkAnn (fromString (show prec)) (Precedence prec)) (mkAnnList (separatedBy ", " list) [op])
   84 
   85 -- | Creates an unguarded right-hand-side (@ = 3 @)
   86 mkUnguardedRhs :: Expr -> Rhs
   87 mkUnguardedRhs = mkAnn (" = " <> child) . UUnguardedRhs
   88 
   89 -- | Creates an unguarded right-hand-side (@ | x == 1 = 3; | otherwise = 4 @)
   90 mkGuardedRhss :: [GuardedRhs] -> Rhs
   91 mkGuardedRhss = mkAnn child . UGuardedRhss . mkAnnList (indented list)
   92 
   93 -- | Creates a guarded right-hand side of a value binding (@ | x > 3 = 2 @)
   94 mkGuardedRhs :: [RhsGuard] -> Expr -> GuardedRhs
   95 mkGuardedRhs guards expr = mkAnn ("| " <> child <> " = " <> child) $ UGuardedRhs (mkAnnList (separatedBy ", " list) guards) expr
   96 
   97 -- | Creates a bind statement in a pattern guard (@ Just v <- x @)
   98 mkGuardBind :: Pattern -> Expr -> RhsGuard
   99 mkGuardBind pat expr = mkAnn (child <> " <- " <> child) $ UGuardBind pat expr
  100 
  101 -- | Creates a let statement in a pattern guard (@ let x = 3 @)
  102 mkGuardLet :: [LocalBind] -> RhsGuard
  103 mkGuardLet = mkAnn ("let " <> child) . UGuardLet . mkAnnList (indented list)
  104 
  105 -- | Creates an expression to check for a pattern guard
  106 mkGuardCheck :: Expr -> RhsGuard
  107 mkGuardCheck = mkAnn child . UGuardCheck
