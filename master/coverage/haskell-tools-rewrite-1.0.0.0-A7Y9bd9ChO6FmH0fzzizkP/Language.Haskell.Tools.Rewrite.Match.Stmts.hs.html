    1 -- | UPattern matching on statement-level AST fragments for refactorings.
    2 {-# LANGUAGE PatternSynonyms #-}
    3 module Language.Haskell.Tools.Rewrite.Match.Stmts where

    8 -- * Do-notation
    9 
   10 -- | Binding statement (@ x <- action @)
   11 pattern BindStmt :: Pattern -> Expr -> Stmt
   12 pattern BindStmt bound expr <- Ann _ (UBindStmt bound expr)
   13 
   14 -- | Non-binding statement (@ action @)
   15 pattern ExprStmt :: Expr -> Stmt
   16 pattern ExprStmt expr <- Ann _ (UExprStmt expr)
   17 
   18 -- | Let statement (@ let x = 3; y = 4 @)
   19 pattern LetStmt :: LocalBindList -> Stmt
   20 pattern LetStmt binds <- Ann _ (ULetStmt binds)
   21 
   22 -- | A recursive binding statement with (@ rec b <- f a c; c <- f b a @)
   23 pattern RecStmt :: StmtList -> Stmt
   24 pattern RecStmt stmts <- Ann _ (URecStmt stmts)
   25 
   26 pattern DoKeyword :: DoKind
   27 pattern DoKeyword <- Ann _ UDoKeyword
   28 
   29 pattern MDoKeyword :: DoKind
   30 pattern MDoKeyword <- Ann _ UMDoKeyword
   31 
   32 -- * List comprehensions
   33 
   34 -- | Body of a list comprehension: (@ | x <- [1..10] @)
   35 pattern ListCompBody :: CompStmtList -> ListCompBody
   36 pattern ListCompBody stmts <- Ann _ (UListCompBody stmts)
   37 
   38 -- | Normal monadic statement of a list comprehension
   39 pattern CompStmt :: Stmt -> CompStmt
   40 pattern CompStmt stmt <- Ann _ (UCompStmt stmt)
   41 
   42 -- | Then statements by @TransformListComp@ (@ then sortWith by (x + y) @)
   43 pattern ThenStmt :: Expr -> MaybeExpr -> CompStmt
   44 pattern ThenStmt then_ by <- Ann _ (UThenStmt then_ by)
   45 
   46 -- | Grouping statements by @TransformListComp@ (@ then group by (x + y) using groupWith @) 
   47 pattern GroupStmt :: MaybeExpr -> MaybeExpr -> CompStmt
   48 pattern GroupStmt by using <- Ann _ (UGroupStmt by using)
   49

   56 -- | Non-binding statement command (@ action @)

   60 -- | Let statement command (@ let x = 3; y = 4 @)
   61 pattern LetStmtCmd :: LocalBindList -> CmdStmt
   62 pattern LetStmtCmd binds <- Ann _ (ULetStmt binds)

   65 pattern RecStmtCmd :: CmdStmtList -> CmdStmt
