    8 -- | Non-function binding (@ v = "12" @)
    9 pattern SimpleBind :: Pattern -> Rhs -> MaybeLocalBinds -> ValueBind
   10 pattern SimpleBind p r l <- Ann _ (USimpleBind p r l)
   11 
   12 -- | Function binding (@ f 0 = 1; f x = x @). All matches must have the same name.
   13 pattern FunctionBind :: MatchList -> ValueBind
   14 pattern FunctionBind matches <- Ann _ (UFunBind matches)
   15 
   16 -- | Clause of function binding
   17 pattern Match :: MatchLhs -> Rhs -> MaybeLocalBinds -> Match
   18 pattern Match lhs rhs locs <- Ann _ (UMatch lhs rhs locs)
   19 
   20 -- | A match lhs with the function name and parameter names (@ f a b @)
   21 pattern MatchLhs :: Name -> PatternList -> MatchLhs
   22 pattern MatchLhs n pats <- Ann _ (UNormalLhs n pats)
   23 
   24 -- | An infix match lhs for an operator (@ a + b @)
   25 pattern InfixLhs :: Pattern -> Operator -> Pattern -> PatternList -> MatchLhs
   26 pattern InfixLhs lhs op rhs pats <- Ann _ (UInfixLhs lhs op rhs pats)
   27 
   28 -- | Local bindings attached to a declaration (@ where x = 42 @)
   29 pattern LocalBinds :: LocalBindList -> LocalBinds
   30 pattern LocalBinds binds <- Ann _ (ULocalBinds binds)
   31 
   32 -- | A local binding for a value
   33 pattern LocalValBind :: ValueBind -> LocalBind
   34 pattern LocalValBind bind <- Ann _ (ULocalValBind bind)
   35 
   36 -- | A local type signature
   37 pattern LocalTypeSig :: TypeSignature -> LocalBind
   38 pattern LocalTypeSig typeSig <- Ann _ (ULocalSignature typeSig)
   39 
   40 -- | A local fixity declaration
   41 pattern LocalFixity :: FixitySignature -> LocalBind
   42 pattern LocalFixity fixity <- Ann _ (ULocalFixity fixity)
   43 
   44 -- | A type signature (@ f :: Int -> Int @)
   45 pattern TypeSignature :: NameList -> Type -> TypeSignature
   46 pattern TypeSignature n t <- Ann _ (UTypeSignature n t)
   47 
   48 -- | A left-associative fixity declaration (@ infixl 5 +, - @).
   49 pattern InfixL :: OperatorList -> FixitySignature
   50 pattern InfixL op <- Ann _ (UFixitySignature (Ann _ AssocLeft) _ op)
   51 
   52 -- | A right-associative fixity declaration (@ infixr 5 +, - @).
   53 pattern InfixR :: OperatorList -> FixitySignature
   54 pattern InfixR op <- Ann _ (UFixitySignature (Ann _ AssocRight) _ op)
   55 
   56 -- | A non-associative fixity declaration (@ infix 5 +, - @).
   57 pattern Infix :: OperatorList -> FixitySignature
   58 pattern Infix op <- Ann _ (UFixitySignature (Ann _ AssocNone) _ op)
   59 
   60 -- | An unguarded right-hand-side (@ = 3 @)
   61 pattern UnguardedRhs :: Expr -> Rhs
   62 pattern UnguardedRhs expr <- Ann _ (UUnguardedRhs expr)
   63 
   64 -- | An unguarded right-hand-side (@ | x == 1 = 3; | otherwise = 4 @)
   65 pattern GuardedRhss :: GuardedRhsList -> Rhs
   66 pattern GuardedRhss rhss <- Ann _ (UGuardedRhss rhss)
   67 
   68 -- | A guarded right-hand side of a value binding (@ | x > 3 = 2 @)
   69 pattern GuardedRhs :: RhsGuardList -> Expr -> GuardedRhs
   70 pattern GuardedRhs guards expr <- Ann _ (UGuardedRhs guards expr)
   71 
   72 -- | A bind statement in a pattern guard (@ Just v <- x @)
   73 pattern GuardBind :: Pattern -> Expr -> RhsGuard
   74 pattern GuardBind pat expr <- Ann _ (UGuardBind pat expr)
   75 
   76 -- | A let statement in a pattern guard (@ let x = 3 @)
   77 pattern GuardLet :: LocalBindList -> RhsGuard
   78 pattern GuardLet binds <- Ann _ (UGuardLet binds)
   79 
   80 -- | An expression to check for a pattern guard
   81 pattern GuardCheck :: Expr -> RhsGuard
