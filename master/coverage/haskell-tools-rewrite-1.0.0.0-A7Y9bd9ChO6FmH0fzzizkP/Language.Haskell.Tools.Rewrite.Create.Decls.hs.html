   44 -- | Creates a data or newtype declaration.

   46 mkDataDecl keyw ctx dh cons derivs
   47   = mkAnn (child <> " " <> child <> child <> child <> child)

   51 -- | Creates a GADT-style data or newtype declaration.

   53                     -> [GadtConDecl] -> [Deriving] -> Decl
   54 mkGADTDataDecl keyw ctx dh kind cons derivs
   55   = mkAnn (child <> " " <> child <> child <> child <> child <> child)
   56       $ UGDataDecl keyw (mkAnnMaybe (after " " opt) ctx) dh
   57                   (mkAnnMaybe (after " " opt) kind) (mkAnnList (after " = " $ separatedBy " | " list) cons)

   60 -- | Creates a GADT constructor declaration (@ D1 :: Int -> T String @)
   61 mkGadtConDecl :: [Name] -> Type -> GadtConDecl
   62 mkGadtConDecl names typ
   63   = mkAnn (child <> " :: " <> child <> child <> child)
   64       $ UGadtConDecl (mkAnnList (separatedBy ", " list) names) emptyList noth (mkAnn child $ UGadtNormalType typ)
   65 
   66 -- | Creates a GADT constructor declaration with record syntax (@ D1 :: { val :: Int } -> T String @)
   67 mkGadtRecordConDecl :: [Name] -> [FieldDecl] -> Type -> GadtConDecl
   68 mkGadtRecordConDecl names flds typ
   69   = mkAnn (child <> " :: " <> child <> child <> child) $ UGadtConDecl (mkAnnList (separatedBy ", " list) names) emptyList noth
   70       $ mkAnn (child <> " -> " <> child)
   71       $ UGadtRecordType (mkAnnList (after "{ " $ separatedBy ", " $ followedBy " }" list) flds) typ

  105 -- | Creates a type class declaration (@ class X a where f = ... @)
  106 mkClassDecl :: Maybe Context -> DeclHead -> [FunDep] -> Maybe ClassBody -> Decl
  107 mkClassDecl ctx dh funDeps body
  108   = let fdeps = case funDeps of [] -> Nothing
  109                                 _ -> Just $ mkAnn child $ UFunDeps $ mkAnnList (separatedBy ", " list) funDeps
  110      in mkAnn ("class " <> child <> child <> child <> child)
  111           $ UClassDecl (mkAnnMaybe (followedBy " " opt) ctx) dh (mkAnnMaybe (after " | " opt) fdeps) (mkAnnMaybe opt body)
  112 
  113 -- | Creates the list of declarations that can appear in a typeclass
  114 mkClassBody :: [ClassElement] -> ClassBody
  115 mkClassBody = mkAnn (" where " <> child) . UClassBody . mkAnnList (indented list)

  443       $ mkAnn child $ UInlinablePragma (mkAnnMaybe (followedBy " " opt) phase) name

  445 -- | A pragma for maintaining line numbers in generated sources (@ {-\# LINE 123 "somefile" \#-} @)
  446 mkLinePragma :: Int -> Maybe (StringNode) -> TopLevelPragma
  447 mkLinePragma line filename
  448   = mkAnn ("{-# LINE " <> child <> child <> " #-}")
  449      $ ULinePragma (mkAnn child $ LineNumber line) (mkAnnMaybe (after " " opt) filename)
  450 
  451 -- | A pragma that tells the compiler that a polymorph function should be optimized for a given type (@ {-\# SPECIALISE f :: Int -> b -> b \#-} @)
  452 mkSpecializePragma :: Maybe PhaseControl -> Name -> [Type] -> TopLevelPragma
  453 mkSpecializePragma phase def specTypes
  454   = mkAnn child (USpecializeDecl
  455                   $ mkAnn ("{-# SPECIALIZE " <> child <> child <> " " <> child <> " #-}")
  456                     $ USpecializePragma (mkAnnMaybe (after " " opt) phase) def $ mkAnnList (separatedBy ", " list) specTypes)
  457 
  458 -- | Marks that the pragma should be applied from a given compile phase (@ [2] @)
  459 mkPhaseControlFrom :: Integer -> PhaseControl
  460 mkPhaseControlFrom phaseNum
  461   = mkAnn ("[" <> child <> child <> "]") $ UPhaseControl (mkAnnMaybe opt Nothing) (mkAnnMaybe opt $ Just $ mkAnn child $ PhaseNumber phaseNum)
  462 
  463 -- | Marks that the pragma should be applied until a given compile phase (@ [~2] @)
  464 mkPhaseControlUntil :: Integer -> PhaseControl
  465 mkPhaseControlUntil phaseNum
  466   = mkAnn ("[" <> child <> child <> "]") $ UPhaseControl (mkAnnMaybe opt $ Just $ mkAnn "~" PhaseInvert)
  467                                                          (mkAnnMaybe opt $ Just $ mkAnn child $ PhaseNumber phaseNum)
  468 
  469 -- | A rewrite rule (@ "map/map" forall f g xs. map f (map g xs) = map (f.g) xs @)
  470 mkRewriteRule :: String -> Maybe PhaseControl -> [RuleVar] -> Expr -> Expr -> Rule
  471 mkRewriteRule name phase vars lhs rhs
  472   = mkAnn (child <> " " <> child <> child <> child <> " = " <> child)
  473       $ URule (mkAnn ("\"" <> child <> "\"") $ UStringNode name) (mkAnnMaybe (followedBy " " opt) phase)
  474               (mkAnnList (after "forall " $ separatedBy " " $ followedBy ". " list) (vars)) lhs rhs
  475 
  476 mkRuleVar :: Name -> RuleVar
  477 mkRuleVar name = mkAnn child (URuleVar name)
  478 
  479 -- | The definition with the given name is annotated
  480 mkNameAnnotation :: Name -> AnnotationSubject
  481 mkNameAnnotation name = mkAnn child $ UNameAnnotation name
  482 
  483 -- | A type with the given name is annotated
  484 mkTypeAnnotation :: Name -> AnnotationSubject
  485 mkTypeAnnotation name = mkAnn ("type " <> child) $ UTypeAnnotation name

  488 mkModuleAnnotation :: AnnotationSubject
  489 mkModuleAnnotation = mkAnn "module" UModuleAnnotation

  491 -- | A @CONLIKE@ modifier for an @INLINE@ pragma.
  492 mkConlikeAnnotation :: ConlikeAnnot
