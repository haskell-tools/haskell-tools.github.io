    1 -- | Creating a dependency graph of the modules loaded into a session.
    2 -- Code copied from GHC because it is not public in GhcMake module

   15 type NodeKey   = (ModuleName, IsBoot)
   16 type NodeMap a = Map.Map NodeKey a
   17 type SummaryNode = Node Int ModSummary
   18 
   19 getModFromNode :: SummaryNode -> ModSummary
   20 getModFromNode = summaryNodeSummary
   21 
   22 summaryNodeSummary :: SummaryNode -> ModSummary
   23 summaryNodeSummary = node_payload
   24 
   25 -- Creates the dependency graph of modules currently loaded. Used for checking which modules need
   26 -- to be reloaded after a recompilation.
   27 moduleGraphNodes drop_hs_boot_nodes summaries =
   28   (graphFromEdgedVerticesUniq nodes, lookup_node)
   29   where
   30     numbered_summaries = zip summaries [1..]
   31 
   32     lookup_node :: HscSource -> ModuleName -> Maybe SummaryNode
   33     lookup_node hs_src mod = Map.lookup (mod, hscSourceToIsBoot hs_src) node_map
   34 
   35     lookup_key :: HscSource -> ModuleName -> Maybe Int
   36     lookup_key hs_src mod = fmap summaryNodeKey (lookup_node hs_src mod)
   37 
   38     node_map :: NodeMap SummaryNode
   39     node_map = Map.fromList [ ((moduleName (ms_mod s),
   40                                 hscSourceToIsBoot (ms_hsc_src s)), node)
   41                             | node <- nodes
   42                             , let s = summaryNodeSummary node ]
   43 
   44     -- We use integers as the keys for the SCC algorithm
   45     nodes :: [SummaryNode]
   46     nodes = [ DigraphNode s key out_keys
   47             | (s, key) <- numbered_summaries
   48              -- Drop the hi-boot ones if told to do so
   49             , not (isBootSummary s && drop_hs_boot_nodes)
   50             , let out_keys = out_edge_keys hs_boot_key (map unLoc (ms_home_srcimps s)) ++
   51                              out_edge_keys HsSrcFile   (map unLoc (ms_home_imps s)) ++
   52                              (-- see [boot-edges] below
   53                               if drop_hs_boot_nodes || ms_hsc_src s == HsBootFile
   54                               then []
   55                               else case lookup_key HsBootFile (ms_mod_name s) of
   56                                     Nothing -> []
   57                                     Just k  -> [k]) ]
   58 
   59     -- [boot-edges] if this is a .hs and there is an equivalent
   60     -- .hs-boot, add a link from the former to the latter.  This
   61     -- has the effect of detecting bogus cases where the .hs-boot
   62     -- depends on the .hs, by introducing a cycle.  Additionally,
   63     -- it ensures that we will always process the .hs-boot before
   64     -- the .hs, and so the HomePackageTable will always have the
   65     -- most up to date information.
   66 
   67     -- Drop hs-boot nodes by using HsSrcFile as the key
   68     hs_boot_key | drop_hs_boot_nodes = HsSrcFile
   69                 | otherwise          = HsBootFile
   70 
   71     out_edge_keys :: HscSource -> [ModuleName] -> [Int]
   72     out_edge_keys hi_boot ms = mapMaybe (lookup_key hi_boot) ms
   73         -- If we want keep_hi_boot_nodes, then we do lookup_key with
   74         -- IsBoot; else NotBoot
   75 
   76 -- | 'Bool' indicating if a module is a boot module or not.  We need to treat
   77 -- boot modules specially when building compilation graphs, since they break
   78 -- cycles.  Regular source files and signature files are treated equivalently.
   79 data IsBoot = IsBoot | NotBoot
   80     deriving (Ord, Eq, Show, Read)
   81 
   82 -- | Tests if an 'HscSource' is a boot file, primarily for constructing
   83 -- elements of 'BuildModule'.
   84 hscSourceToIsBoot :: HscSource -> IsBoot
   85 hscSourceToIsBoot HsBootFile = IsBoot
   86 hscSourceToIsBoot _ = NotBoot
   87 
   88 summaryNodeKey :: SummaryNode -> Int
   89 summaryNodeKey = node_key
   90 
   91 ms_home_imps :: ModSummary -> [Located ModuleName]
   92 ms_home_imps = home_imps . ms_imps
   93 
   94 ms_home_srcimps :: ModSummary -> [Located ModuleName]
   95 ms_home_srcimps = home_imps . ms_srcimps
   96 
   97 home_imps :: [(Maybe FastString, Located ModuleName)] -> [Located ModuleName]
   98 home_imps imps = [ lmodname |  (mb_pkg, lmodname) <- imps,
   99                                   isLocal mb_pkg ]
  100   where isLocal Nothing = True
  101         isLocal (Just pkg) | pkg == fsLit "this" = True -- "this" is special
  102         isLocal _ = False
  103 
  104 supportingModules :: (ModSummary -> Ghc Bool) -> Ghc [ModSummary]
  105 supportingModules = reachedModules False
  106 
  107 dependentModules :: (ModSummary -> Ghc Bool) -> Ghc [ModSummary]
  108 dependentModules = reachedModules True
  109 
  110 reachedModules :: Bool -> (ModSummary -> Ghc Bool) -> Ghc [ModSummary]
  111 reachedModules dependent pred = do
  112   let op = if dependent then transposeG else id
  113   allMods <- mgModSummaries <$> getModuleGraph
  114   selected <- filterM pred allMods
  115   let (allModsGraph, lookup) = moduleGraphNodes False allMods
  116       selectedMods = catMaybes $ map (\ms -> lookup (ms_hsc_src ms) (moduleName $ ms_mod ms)) selected
  117       recompMods = map (moduleName . ms_mod . getModFromNode) $ reachablesG (op allModsGraph) selectedMods -- TODO: compare on file name
  118       sortedMods = map getModFromNode $ reverse $ topologicalSortG allModsGraph
  119       sortedSelectedMods = filter ((`elem` recompMods) . moduleName . ms_mod) sortedMods -- TODO: compare on file name
  120   return sortedSelectedMods
