   64 chkOperatorForTypeEq' :: CheckNode Operator
   65 chkOperatorForTypeEq' op
   66   | Just name <- semanticsName (op ^. operatorName)
   67   , name == eqTyConName
   68   = addRelation (TypeFamilies `lOr` GADTs) op
   69   | otherwise = return op

   71 -- | Checks whether a given name's has a type equality operator in it.
   72 -- If the type lookup fails, it returns Nothing. If given a type variable,
   73 -- it returns False.
   74 chkQNameForTyEqn :: QualifiedName -> MaybeT ExtMonad Bool
   75 chkQNameForTyEqn name =
   76   if not . isId . semanticsId $ name
   77     then return False
   78     else do
   79       ty <- lookupTypeFromId name
   80       let ty'    = GHC.expandTypeSynonyms ty
   81           tycons = universeBi ty' :: [GHC.TyCon]
   82       if any isEqTyCon tycons then return True
   83                               else return False
   84       where isEqTyCon tc = tc `hasKey` eqTyConKey
   85                         || tc `hasKey` heqTyConKey
   86                         || tc `hasKey` eqPrimTyConKey
   87                         || tc `hasKey` eqReprPrimTyConKey
   88                         || tc `hasKey` eqPhantPrimTyConKey
   89 
   90 gblChkQNamesForTypeEq' :: CheckNode Module
   91 gblChkQNamesForTypeEq' m = do
   92   -- Names appearing on the rhs of bindings are just hints,
   93   -- they don't necessarily require TypeFamilies.
   94   let allNames   = universeBi (m ^. modDecl) :: [QualifiedName]
   95       rhs        = universeBi (m ^. modDecl) :: [Rhs]
   96       hints      = universeBi rhs            :: [QualifiedName]
   97 
   98       -- Separating hints from evidence,
   99       -- and grouping them together based on their GHC.Name
  100       evidence  = filter (isJust . semanticsName) (allNames \\ hints)
  101       hints'    = filter (isJust . semanticsName) hints
  102       groupedEs = equivalenceGroupsBy semanticsName evidence
  103       groupedHs = equivalenceGroupsBy semanticsName hints'
  104 
  105       -- names failed at the semanticsName lookup
  106       nmFailedNames = filter (isNothing . semanticsName) allNames
  107 
  108   -- Checking the representative elements, whether they need FC,
  109   -- if they do, add occurences for every node in their group.
  110   -- If chkQNameForTyEqn fails, we add MissingInformation.
  111   let hasTypeEq    = fromMaybeT False . chkQNameForTyEqn
  112 
  113   es <- filterM (hasTypeEq . fst) groupedEs
  114   hs <- filterM (hasTypeEq . fst) groupedHs
  115   mapM_ (addRelation     (TypeFamilies `lOr` GADTs)) (concatMap snd es)
  116   mapM_ (addRelationHint (TypeFamilies `lOr` GADTs)) (concatMap snd hs)
  117 
  118   mapM_ (addRelationMI (TypeFamilies `lOr` GADTs)) nmFailedNames
  119 
  120   -- NOTE: this would result in many false positive results
  121   -- names failed at the chkQNameForTyEqn (lookupTypeFromId) stage
  122   -- idFailedNames <- filterM failedLookup allNames
  123   -- mapM_ (addRelationMI (TypeFamilies `lOr` GADTs)) idFailedNames
  124 
  125   return m
