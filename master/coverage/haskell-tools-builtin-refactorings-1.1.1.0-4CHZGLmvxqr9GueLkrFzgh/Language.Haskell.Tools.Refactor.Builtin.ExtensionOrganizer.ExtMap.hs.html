   41 prettyPrintFormula :: Show a => LogicalRelation a -> String
   42 prettyPrintFormula (Var x) = show x
   43 prettyPrintFormula (x :||: y) = prettyPrintFormula x ++ " OR " ++ prettyPrintFormula y
   44 prettyPrintFormula (x :&&: y) = prettyPrintFormula x ++ " AND " ++ prettyPrintFormula y
   45 prettyPrintFormula (x :++: y) = prettyPrintFormula x ++ " XOR " ++ prettyPrintFormula y
   46 prettyPrintFormula (x :->: y) = prettyPrintFormula x ++ " => " ++ prettyPrintFormula y
   47 prettyPrintFormula (x :<->: y) = prettyPrintFormula x ++ " <=> " ++ prettyPrintFormula y
   48 prettyPrintFormula (All xs) = "All [" ++ (intercalate ", " . map prettyPrintFormula $ xs) ++ "]"
   49 prettyPrintFormula (Some xs) = "Some [" ++ (intercalate ", " . map prettyPrintFormula $ xs) ++ "]"
   50 prettyPrintFormula (None xs) = "None [" ++ (intercalate ", " . map prettyPrintFormula $ xs) ++ "]"
   51 prettyPrintFormula (ExactlyOne xs) = "ExactlyOne [" ++ (intercalate ", " . map prettyPrintFormula $ xs) ++ "]"
   52 prettyPrintFormula (AtMostOne xs) = "AtMostOne [" ++ (intercalate ", " . map prettyPrintFormula $ xs) ++ "]"
   53 prettyPrintFormula (Not x) = "Not " ++ prettyPrintFormula x
   54 prettyPrintFormula Yes = "True"
   55 prettyPrintFormula No = "False"
   56 prettyPrintFormula Let{} = "Let ..."
   57 prettyPrintFormula Bound{} = error "Bound is for internal use only"

   84 complexity :: Extension -> Int
   85 complexity = length . expandExtension'
   86 
   87 -- | Completely expand extension
   88 expandExtension' :: Extension -> [Extension]
   89 expandExtension' e = findFixedPoint (nub . expand) [e]
   90   where expand = concatMap expandExtension
   91 
   92 -- | Terminating variant of Control.Monad.Fix.fix
   93 -- It tries to find the fixpoint of a function with a given starting value.
   94 -- It terminates if a value repeats itself.
   95 findFixedPoint :: Eq a => (a -> a) -> a -> a
   96 findFixedPoint f x = findFix' f (f x) x
   97   where findFix' f cur prev = if cur == prev then cur else findFix' f (f cur) cur
   98 
   99 determineExtensions :: SMap.Map (LogicalRelation Extension) v -> [Extension]
  100 {- NOTE:
  101  We calculate all possible extension sets that satisfy the logical relation,
  102  then for each extension we remove all the extensions implied by it, (mergeImplied)
  103  finally we select the minimal extension set.
  104 
  105  An extension set is minimal if:
  106   - it contains the least amount of extensions
  107   - it contains extension with minimal complexity
  108   - it is lexicographically the first one (based on the Ord instance of Extension)
  109 
  110   TODO: Comparing on length might be unnecessary
  111 -}
  112 determineExtensions x = minimal . map toExts $ solution
  113   where solution = solve_all . All . LMap.keys $ x
  114         toExts   = mergeImplied . map fst . filter snd . LMap.toList
  115         minimal  = minimumBy ((compare `on` length) <> (compare `on` (sum . map complexity)) <> (compare `on` sort))
  116 
  117 rmImplied :: Extension -> [Extension] -> [Extension]
  118 rmImplied e = flip (\\) implied
  119   where implied = delete e . expandExtension' $ e
  120 
  121 mergeImplied :: [Extension] -> [Extension]
  122 mergeImplied exts = foldl (flip rmImplied) exts exts
