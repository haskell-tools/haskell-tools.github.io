   12 instance SourceInfo SrcTemplateStage where
   13   data SpanInfo SrcTemplateStage
   14          = SourceTemplateNode { _sourceTemplateNodeRange :: SrcSpan -- ^ The (original) range of the given element
   15                               , _sourceTemplateNodeElems :: [SourceTemplateElem] -- ^ The children of the given node, could be text or child nodes
   16                               , _srcTmpMinInd :: Int -- ^ Minimum indentation for the element
   17                               , _srcTmpRelPos :: Maybe Int -- ^ Relative indentation for newly created elements
   18                               }
   19     deriving (Eq, Ord, Data)
   20   data ListInfo SrcTemplateStage
   21          = SourceTemplateList { _sourceTemplateListRange :: SrcSpan -- ^ The (original) range of the given element
   22                               , _srcTmpListBefore :: String -- ^ Text that should be put before the first element if the list becomes populated
   23                               , _srcTmpListAfter :: String -- ^ Text that should be put after the last element if the list becomes populated
   24                               , _srcTmpDefaultSeparator :: String -- ^ The default separator if the list were empty
   25                               , _srcTmpIndented :: Maybe [Bool] -- ^ False for elements that should be not aligned
   26                               , _srcTmpSeparators :: [([SourceTemplateTextElem], SrcSpan)] -- ^ The actual separators that were found in the source code
   27                               , _srcTmpListMinInd :: Int -- ^ Minimum indentation for the element
   28                               , _srcTmpListRelPos :: Maybe Int -- ^ Relative indentation for newly created elements
   29                               }
   30     deriving (Eq, Ord, Data)
   31   data OptionalInfo SrcTemplateStage
   32          = SourceTemplateOpt { _sourceTemplateOptRange :: SrcSpan -- ^ The (original) range of the given element
   33                              , _srcTmpOptBefore :: String -- ^ Text that should be put before the element if it appears
   34                              , _srcTmpOptAfter :: String -- ^ Text that should be put after the element if it appears
   35                              , _srcTmpOptMinInd :: Int -- ^ Minimum indentation for the element
   36                              , _srcTmpOptRelPos :: Maybe Int -- ^ Relative indentation for newly created elements
   37                              }
   38     deriving (Eq, Ord, Data)

   98 -- | An element of a source template for a singleton AST node.
   99 data SourceTemplateElem
  100   = TextElem { _sourceTemplateTextElem :: [SourceTemplateTextElem]
  101              , _sourceTemplateTextRange :: SrcSpan
  102              } -- ^ Source text belonging to the current node
  103   | ChildElem -- ^ Placeholder for the next children of the node
  104      deriving (Eq, Ord, Data)
  105 
  106 isStayingText :: SourceTemplateTextElem -> Bool
  107 isStayingText StayingText{} = True
  108 isStayingText _ = False
  109 
  110 data SourceTemplateTextElem
  111   = NormalText { _sourceTemplateText :: String }
  112   | StayingText { _sourceTemplateText :: String, _lineEndings :: String }
  113      deriving (Eq, Ord, Data)
  114

  118 instance HasRange (SpanInfo SrcTemplateStage) where

  130 instance Show (SpanInfo SrcTemplateStage) where
  131   show (SourceTemplateNode _ sp _ _) = concatMap show sp
  132 instance Show (ListInfo SrcTemplateStage) where
  133   show SourceTemplateList{..} = "<*" ++ show _srcTmpListBefore ++ " " ++ show _srcTmpDefaultSeparator ++ " " ++ show _srcTmpListAfter ++ " " ++ show _srcTmpSeparators ++ "*>"
  134 instance Show (OptionalInfo SrcTemplateStage) where
  135   show SourceTemplateOpt{..} = "<?" ++ show _srcTmpOptBefore ++ " " ++ show _srcTmpOptAfter ++ "?>"
  136 
  137 instance Show SourceTemplateElem where
  138   show (TextElem s _) = show s
  139   show ChildElem = "<.>"
  140 
  141 instance Show SourceTemplateTextElem where
  142   show (NormalText s) = show s
  143   show (StayingText s _) = "|" ++ show s ++ "|"
