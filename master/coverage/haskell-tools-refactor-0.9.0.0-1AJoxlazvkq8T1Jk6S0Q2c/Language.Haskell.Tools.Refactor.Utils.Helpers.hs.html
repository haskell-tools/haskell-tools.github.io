   20 replaceWithJust :: Ann e dom SrcTemplateStage -> AnnMaybe e dom -> AnnMaybe e dom
   21 replaceWithJust e = annMaybe .= Just e
   22 
   23 replaceWithNothing :: AnnMaybe e dom -> AnnMaybe e dom
   24 replaceWithNothing = annMaybe .= Nothing

   26 -- | Remove the container (where or let) when the last binding is removed.
   27 removeEmptyBnds :: Simple Traversal (Module dom) (ValueBind dom)
   28                      -> Simple Traversal (Module dom) (Expr dom)
   29                      -> AST.Module dom -> AST.Module dom
   30 removeEmptyBnds binds exprs = (binds .- removeEmptyBindsAndGuards) . (exprs .- removeEmptyLetsAndStmts)
   31   where removeEmptyBindsAndGuards sb@(SimpleBind _ _ _)
   32           = (valBindLocals .- removeIfEmpty) . (valBindRhs .- removeEmptyGuards) $ sb
   33         removeEmptyBindsAndGuards fb@(FunctionBind _)
   34           = (funBindMatches & annList & matchBinds .- removeIfEmpty) . (funBindMatches & annList & matchRhs .- removeEmptyGuards) $ fb
   35 
   36         removeEmptyGuards rhs = rhsGuards & annList & guardStmts .- filterList (\case GuardLet (AnnList []) -> False; _ -> True) $ rhs
   37 
   38         removeIfEmpty mb@(AnnJust (LocalBinds (AnnList []))) = annMaybe .= Nothing $ mb
   39         removeIfEmpty mb = mb
   40 
   41         removeEmptyLetsAndStmts (Let (AnnList []) e) = e
   42         removeEmptyLetsAndStmts e = exprStmts .- removeEmptyStmts $ e
   43 
   44         removeEmptyStmts ls = (annList & cmdStmtBinds .- removeEmptyStmts)
   45                                 . filterList (\case LetStmt (AnnList []) -> False; _ -> True) $ ls
   46 
   47 -- | Puts the elements in the orginal order and remove duplicates (elements with the same source range)
   48 normalizeElements :: [Ann e dom SrcTemplateStage] -> [Ann e dom SrcTemplateStage]
   49 normalizeElements elems = nubBy ((==) `on` getRange) $ sortBy (compare `on` srcSpanStart . getRange) elems
