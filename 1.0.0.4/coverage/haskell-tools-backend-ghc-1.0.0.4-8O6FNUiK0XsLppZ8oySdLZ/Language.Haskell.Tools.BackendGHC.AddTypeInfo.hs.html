   35 addTypeInfos :: LHsBinds Id -> Ann AST.UModule (Dom GHC.Name) RangeStage -> Ghc (Ann AST.UModule IdDom RangeStage)
   36 addTypeInfos bnds mod = do
   37   ut <- liftIO mkUnknownType
   38   let getType = getType' ut
   39   fixities <- getFixities
   40   -- evaluates fixities to normal form before going on, otherwise the reference to session would be kept
   41   liftIO $ evaluate (forceElements fixities)
   42   let createCName sc def id = mkCNameInfo sc def id fixity
   43         where fixity = if any (any ((getOccName id ==) . getOccName . (^. _1))) (drop 1 sc)
   44                           then Nothing
   45                           else fmap (snd . snd) $ List.find (\(mod,(occ,_)) -> Just mod == (nameModule_maybe $ varName id) && occ == getOccName id) fixities
   46   evalStateT (semaTraverse
   47     (AST.SemaTrf
   48       (\ni -> case (AST.semanticsSourceInfo ni, AST.semanticsName ni) of
   49                 (_, Just name) -> lift $ createCName (AST.semanticsScope ni) (AST.semanticsDefining ni) <$> getType name
   50                 (Just l@(RealSrcSpan loc), _)
   51                   -> case Map.lookup l locMapping of
   52                             Just id -> return $ createCName (AST.semanticsScope ni) (AST.semanticsDefining ni) id
   53                             _ -> do (none,rest) <- gets (break ((\case (RealSrcSpan sp) -> sp `containsSpan` loc) . fst))
   54                                     case rest of [] -> throw $ ConvertionProblem (RealSrcSpan loc) "Ambiguous or implicit name missing"
   55                                                  ((_,id):more) -> do put (none ++ more)
   56                                                                      return $ createCName (AST.semanticsScope ni) (AST.semanticsDefining ni) id
   57                 _ -> convProblem "addTypeInfos: Cannot access a the semantics of a name.")
   58       pure (traverse (lift . getType)) (traverse (lift . getType)) pure
   59         pure) mod) (extractSigIds bnds ++ extractSigBindIds bnds)
   60   where locMapping = Map.fromList $ map (\(L l id) -> (l, id)) $ extractExprIds bnds
   61         getType' ut name = fromMaybe (mkVanillaGlobal name ut) <$> ((<|> Map.lookup name ids) <$> getTopLevelId name)
   62         ids = Map.fromList $ map (\id -> (getName id, id)) $ extractTypes bnds
   63         extractTypes :: LHsBinds Id -> [Id]
   64         extractTypes = concatMap universeBi . bagToList
   65 
   66         mkUnknownType :: IO Type
   67         mkUnknownType = do
   68           tUnique <- mkSplitUniqSupply 'x'
   69           return $ mkTyVarTy $ mkVanillaGlobal (mkSystemName (uniqFromSupply tUnique) (mkDataOcc "TypeNotFound")) (mkTyConTy starKindTyCon)
   70 
   71         getFixities :: Ghc [(Module, (OccName, GHC.Fixity))]
   72         getFixities = do env <- getSession
   73                          pit <- liftIO $ eps_PIT <$> hscEPS env
   74                          let hpt = hsc_HPT env
   75                              ifaces = moduleEnvElts pit ++ map hm_iface (eltsUDFM hpt)
   76                          return $ concatMap (\mi -> map (mi_module mi, ) $ mi_fixities mi) ifaces
   77 
   78 extractExprIds :: LHsBinds Id -> [Located Id]
   79         -- expressions like HsRecFld are removed from the typechecked representation, they are replaced by HsVar
   80 extractExprIds = catMaybes . map (\case L l (HsVar (L _ n)) -> Just (L l n)
   81                                         L l (HsWrap _ (HsVar (L _ n))) -> Just (L l n)
   82                                         _ -> Nothing
   83                                  ) . concatMap universeBi . bagToList
   84 
   85 extractSigIds :: LHsBinds Id -> [(SrcSpan,Id)]
   86 extractSigIds = filter (isGoodSrcSpan . fst)
   87                   . concat
   88                   . map (\case L l bs@(AbsBindsSig {} :: HsBind Id) -> map (l,) $ getImplVars (abs_sig_ev_bind bs)
   89                                _                                    -> [] )
   90                   . concatMap universeBi . bagToList
   91   where getImplVars (EvBinds evbnds) = catMaybes $ map getEvVar $ bagToList evbnds
   92         getImplVars _                = []
   93         getEvVar (EvBind lhs _ False) = Just lhs
   94         getEvVar _                    = Nothing
   95 
   96 extractSigBindIds :: LHsBinds Id -> [(SrcSpan,Id)]
   97 extractSigBindIds = filter (isGoodSrcSpan . fst)
   98                       . catMaybes
   99                       . map (\case L l (IPBind (Right id) _) -> Just (l,id)
  100                                    _                         -> Nothing )
  101                       . concatMap universeBi . bagToList
