   26 trfPattern :: TransformName n r => Located (Pat n) -> Trf (Ann AST.UPattern (Dom r) RangeStage)
   27 -- field wildcards are not directly represented in GHC AST
   28 trfPattern (L l (ConPatIn name (RecCon (HsRecFields flds _)))) | any ((l ==) . getLoc) flds
   29   = focusOn l $ do
   30       let (fromWC, notWC) = partition ((l ==) . getLoc) flds
   31       normalFields <- mapM (trfLocNoSema trfPatternField') notWC
   32       wildc <- annLocNoSema (tokenLocBack AnnDotdot) (AST.UFieldWildcardPattern <$> annCont (createImplicitFldInfo (unLoc . (\(VarPat n) -> n) . unLoc) (map unLoc fromWC)) (pure AST.FldWildcard))
   33       annLocNoSema (pure l) (AST.URecPat <$> trfName name <*> makeNonemptyList ", " (pure (normalFields ++ [wildc])))
   34 trfPattern p = trfLocNoSema trfPattern' (correctPatternLoc p)
   35 
   36 -- | Locations for right-associative infix patterns are incorrect in GHC AST
   37 correctPatternLoc :: Located (Pat n) -> Located (Pat n)
   38 correctPatternLoc (L _ p@(ConPatIn _ (InfixCon left right)))
   39   = L (getLoc (correctPatternLoc left) `combineSrcSpans` getLoc (correctPatternLoc right)) p
   40 correctPatternLoc p = p

   42 trfPattern' :: TransformName n r => Pat n -> Trf (AST.UPattern (Dom r) RangeStage)
   43 trfPattern' (WildPat _) = pure AST.UWildPat
   44 trfPattern' (VarPat name) = define $ AST.UVarPat <$> trfName name
   45 trfPattern' (LazyPat pat) = AST.UIrrefutablePat <$> trfPattern pat
   46 trfPattern' (AsPat name pat) = AST.UAsPat <$> define (trfName name) <*> trfPattern pat
   47 trfPattern' (ParPat pat) = AST.UParenPat <$> trfPattern pat
   48 trfPattern' (BangPat pat) = AST.UBangPat <$> trfPattern pat
   49 trfPattern' (ListPat pats _ _) = AST.UListPat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   50 trfPattern' (TuplePat pats Boxed _) = AST.UTuplePat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   51 trfPattern' (TuplePat pats Unboxed _) = AST.UUnboxTuplePat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   52 trfPattern' (PArrPat pats _) = AST.UParArrPat <$> makeList ", " atTheEnd (mapM trfPattern pats)
   53 trfPattern' (ConPatIn name (PrefixCon args)) = AST.UAppPat <$> trfName name <*> makeList " " atTheEnd (mapM trfPattern args)
   54 trfPattern' (ConPatIn name (RecCon (HsRecFields flds _))) = AST.URecPat <$> trfName name <*> trfAnnList ", " trfPatternField' flds
   55 trfPattern' (ConPatIn name (InfixCon left right)) = AST.UInfixAppPat <$> trfPattern left <*> trfOperator name <*> trfPattern right
   56 trfPattern' (ViewPat expr pat _) = AST.UViewPat <$> trfExpr expr <*> trfPattern pat
   57 trfPattern' (SplicePat qq@(HsQuasiQuote {})) = AST.UQuasiQuotePat <$> annContNoSema (trfQuasiQuotation' qq)
   58 trfPattern' (SplicePat splice) = AST.USplicePat <$> trfSplice splice
   59 trfPattern' (LitPat lit) = AST.ULitPat <$> annContNoSema (trfLiteral' lit)
   60 trfPattern' (SigPatIn pat (hsib_body . hswc_body -> typ)) = AST.UTypeSigPat <$> trfPattern pat <*> trfType typ
   61 trfPattern' (NPat (ol_val . unLoc -> lit) _ _ _) = AST.ULitPat <$> annContNoSema (trfOverloadedLit lit)
   62 trfPattern' (NPlusKPat id (L l lit) _ _ _ _) = AST.UNPlusKPat <$> define (trfName id) <*> annLocNoSema (pure l) (trfOverloadedLit (ol_val lit))
   63 trfPattern' (CoPat _ pat _) = trfPattern' pat -- coercion pattern introduced by GHC
   64 trfPattern' (SumPat pat tag arity _)
   65   = do sepsBefore <- focusBeforeLoc (srcSpanStart (getLoc pat)) (eachTokenLoc (AnnOpen : replicate (tag - 1) AnnVbar))
   66        sepsAfter <- focusAfterLoc (srcSpanEnd (getLoc pat)) (eachTokenLoc (replicate (arity - tag) AnnVbar))
   67        let locsBefore = map srcSpanEnd $ init sepsBefore
   68            locsAfter = map srcSpanEnd sepsAfter
   69        AST.UUnboxedSumPat <$> makeList " | " (after AnnOpen) (mapM makePlaceholder locsBefore)
   70                           <*> trfPattern pat
   71                           <*> makeList " | " (before AnnClose) (mapM makePlaceholder locsAfter)
   72   where makePlaceholder l = annLocNoSema (pure (srcLocSpan l)) (pure AST.UUnboxedSumPlaceHolder)
   73 trfPattern' p = unhandledElement "pattern" p
   74 
   75 trfPatternField' :: TransformName n r => HsRecField n (LPat n) -> Trf (AST.UPatternField (Dom r) RangeStage)
   76 trfPatternField' (HsRecField id arg False) = AST.UNormalFieldPattern <$> trfName (getFieldOccName id) <*> trfPattern arg
   77 trfPatternField' (HsRecField id _ True) = AST.UFieldPunPattern <$> trfName (getFieldOccName id)
