    9 -- * Expressions
   10 
   11 -- | An expression for a variable or a data constructor (@ a @)
   12 pattern Var :: Name -> Expr
   13 pattern Var name <- Ann _ (UVar name)
   14 
   15 -- | A literal expression (@ 42 @)
   16 pattern Lit :: Literal -> Expr
   17 pattern Lit lit <- Ann _ (ULit lit)
   18 
   19 -- | An infix operator application (@ a + b @)
   20 pattern InfixApp :: Expr -> Operator -> Expr -> Expr
   21 pattern InfixApp lhs op rhs <- Ann _ (UInfixApp lhs op rhs)
   22 
   23 -- | Prefix operator application (@ -x @)
   24 pattern PrefixApp :: Operator -> Expr -> Expr
   25 pattern PrefixApp op rhs <- Ann _ (UPrefixApp op rhs)
   26 
   27 -- | Function application (@ f 4 @)
   28 pattern App :: Expr -> Expr -> Expr
   29 pattern App f e <- Ann _ (UApp f e)
   30 
   31 -- | Lambda expression (@ \\a b -> a + b @)
   32 pattern Lambda :: PatternList -> Expr -> Expr
   33 pattern Lambda pats rhs <- Ann _ (ULambda pats rhs)
   34 
   35 -- | Local binding (@ let x = 2; y = 3 in e x y @)
   36 pattern Let :: LocalBindList -> Expr -> Expr
   37 pattern Let pats expr <- Ann _ (ULet pats expr)
   38 
   39 -- | If expression (@ if a then b else c @)
   40 pattern If :: Expr -> Expr -> Expr -> Expr
   41 pattern If cond then_ else_ <- Ann _ (UIf cond then_ else_)
   42 
   43 -- | Multi way if expressions with @MultiWayIf@ extension (@ if | guard1 -> expr1; guard2 -> expr2 @)
   44 pattern MultiIf :: GuardedCaseRhsList -> Expr
   45 pattern MultiIf cases <- Ann _ (UMultiIf cases)
   46 
   47 -- | Pattern matching expression (@ case expr of pat1 -> expr1; pat2 -> expr2 @)
   48 pattern Case :: Expr -> AltList -> Expr
   49 pattern Case expr cases <- Ann _ (UCase expr cases)

   51 -- | Do-notation expressions (@ do x <- act1; act2 @)

   55 pattern ParArrayComp :: Expr -> ListCompBodyList -> Expr
   56 pattern ParArrayComp expr stmts <- Ann _ (UParArrayComp expr stmts)
   57

   59 pattern Tuple :: ExprList -> Expr
