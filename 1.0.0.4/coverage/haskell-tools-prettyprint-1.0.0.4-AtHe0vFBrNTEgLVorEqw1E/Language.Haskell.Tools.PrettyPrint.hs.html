   20 -- | Pretty prints an AST by using source templates stored as node info
   21 prettyPrint :: (SourceInfoTraversal node) => node dom SrcTemplateStage -> String
   22 prettyPrint = toList . printRose . toRoseTree

   24 printRose :: RoseTree SrcTemplateStage -> Seq Char
   25 printRose rt = evalState (printRose' startLoc rt) startLoc
   26   where startLoc = mkRealSrcLoc (fsLit "") 1 1

   30 -- | Pretty prints a rose tree according to the source templates remaining from the original AST
   31 printRose' :: RealSrcLoc -> RoseTree SrcTemplateStage -> PPState (Seq Char)
   32 -- simple implementation could be optimized a bit
   33 -- warning: the length of the file should not exceed maxbound::Int
   34 printRose' parent (RoseTree (RoseSpan (SourceTemplateNode rng elems minInd relInd)) children)
   35   = do slide <- calculateSlide rng
   36        let printTemplateElems :: [SourceTemplateElem] -> [RoseTree SrcTemplateStage] -> PPState (Seq Char)
   37            printTemplateElems (TextElem txtElems _ : rest) children = putString slide min txt >+< printTemplateElems rest children
   38              where txt = concatMap (^. sourceTemplateText) txtElems
   39            printTemplateElems (ChildElem : rest) (child : children) = printRose' parent child >+< printTemplateElems rest children
   40            printTemplateElems [] [] = return empty
   41            printTemplateElems _ []
   42              = pprProblem $ "More child elem in template than actual children (elems: " ++ show elems
   43                                ++ ", children: " ++ show children
   44                                ++ "). Check that the default templates for AST elements are correct."
   45            printTemplateElems [] _
   46              = pprProblem $ "Not all children are used to pretty printing. (elems: "
   47                               ++ show elems ++ ", children: " ++ show children
   48                               ++ "). Check that the default templates for AST elements are correct."
   49 
   50            min = minInd `max` getPosByRelative parent relInd
   51 
   52        printTemplateElems elems children
   53 
   54 printRose' _ (RoseTree (RoseList (SourceTemplateList {})) []) = return empty
   55 printRose' parent (RoseTree (RoseList (SourceTemplateList rng bef aft defSep indented seps minInd relInd)) children)
   56     = do slide <- calculateSlide rng
   57          actRng <- get
   58          let min = minInd `max` getPosByRelative parent relInd
   59          putString slide min bef
   60            >+< (maybe printListWithSeps printListWithSepsIndented indented) actRng slide min actualSeps children
   61            >+< putString slide min aft
   62   where stringSeps :: [String]
   63         stringSeps = map (concatMap (^. sourceTemplateText)) (map fst seps)
   64         actualSeps = case stringSeps of [] -> repeat defSep
   65                                         _  -> stringSeps ++ repeat (last stringSeps)
   66 
   67 printRose' _ (RoseTree (RoseOptional (SourceTemplateOpt {})) []) = return empty
   68 printRose' parent (RoseTree (RoseOptional (SourceTemplateOpt rng bef aft minInd relInd)) [child])
   69   = do slide <- calculateSlide rng
   70        actRng <- get
   71        let min = minInd `max` getPosByRelative parent relInd
   72        putString slide min bef >+< printRose' actRng child >+< putString slide min aft
   73 printRose' _ (RoseTree (RoseOptional _) _) = pprProblem "More than one child element in an optional node."

   75 getPosByRelative :: RealSrcLoc -> Maybe Int -> Int
   76 getPosByRelative sp (Just i) = srcLocCol sp + i - 1
   77 getPosByRelative _ _ = 0
   78 
   79 calculateSlide :: SrcSpan -> PPState Int
   80 calculateSlide (RealSrcSpan originalSpan) = do
   81   actualSpan <- get
   82   return $ srcLocCol actualSpan - srcLocCol (realSrcSpanStart originalSpan)
   83 calculateSlide _ = return 0

   85 putString :: Int -> Int -> String -> PPState (Seq Char)
   86 putString slide minInd s
   87   = do modify $ advanceStr newStr
   88        return (fromList newStr)
   89   where start:rest = splitOn "\n" s
   90         newStr = concat $ List.intersperse ("\n" ++ replicate slide ' ') (start : map (extendToNSpaces minInd) rest)
   91         extendToNSpaces n str = replicate n ' ' ++ (List.dropWhile (== ' ') $ List.take n str) ++ List.drop n str
   92 
   93 advanceStr :: String -> RealSrcLoc -> RealSrcLoc
   94 advanceStr s loc = foldl advanceSrcLoc loc s
   95 
   96 untilReaches :: String -> RealSrcLoc -> RealSrcLoc -> (String, Int)
   97 untilReaches s start end
   98   = let ls = splitOn "\n" s
   99      in case ls of _:_:_ -> (unlines (init ls) ++)
  100                               `mapFst` untilReaches' (last ls) (advanceSrcLoc start '\n') end
  101                    _ -> (s, srcLocCol $ foldl advanceSrcLoc start s)
  102   where
  103     untilReaches' [] curr _ = ([], srcLocCol curr)
  104     untilReaches' (c:rest) curr until | srcLocCol advancedLoc <= srcLocCol until
  105       = (c:) `mapFst` untilReaches' rest advancedLoc until
  106       where advancedLoc = advanceSrcLoc curr c
  107     untilReaches' _ curr _ = ([], srcLocCol curr)
  108 
  109 mapFst :: (a -> b) -> (a, x) -> (b, x)
  110 mapFst f (a, x) = (f a, x)
  111 
  112 (>+<) :: PPState (Seq Char) -> PPState (Seq Char) -> PPState (Seq Char)
  113 (>+<) = liftM2 (><)
  114 
  115 printListWithSeps :: RealSrcLoc -> Int -> Int -> [String] -> [RoseTree SrcTemplateStage] -> PPState (Seq Char)
  116 printListWithSeps = printListWithSeps' (const putString) 0

  118 -- | Prints the elements of a list where the elements must be printed in the same line (do stmts, case alts, let binds, ...)
  119 printListWithSepsIndented :: [Bool] -> RealSrcLoc -> Int -> Int -> [String] -> [RoseTree SrcTemplateStage] -> PPState (Seq Char)
  120 printListWithSepsIndented indentedChildren parent slide minInd seps children
  121   = do base <- get
  122        let putCorrectSep i _ min s | isIndented i
  123               = do curr <- get
  124                    let (shortened, currCol) = untilReaches s curr base
  125                    putString 0 min $ shortened ++ replicate (srcLocCol base - currCol) ' '
  126            putCorrectSep _ slide minInd s = putString slide minInd s
  127        printListWithSeps' putCorrectSep 0 parent slide minInd seps children
  128   where -- the ith separator is before the ith element
  129         isIndented i = case List.drop i indentedChildren of False:_ -> False; _ -> True
  130 
  131 printListWithSeps' :: (Int -> Int -> Int -> String -> PPState (Seq Char)) -> Int -> RealSrcLoc -> Int -> Int -> [String] -> [RoseTree SrcTemplateStage] -> PPState (Seq Char)
  132 printListWithSeps' _ _ _ _ _ _ [] = return empty
  133 printListWithSeps' _ _ parent _ _ _ [child] = printRose' parent child
  134 printListWithSeps' putCorrectSep i parent slide minInd (sep:seps) (child:children)
  135   = printRose' parent child >+< putCorrectSep i slide minInd sep >+< printListWithSeps' putCorrectSep (i+1) parent slide minInd seps children
  136 printListWithSeps' _ _ _ _ _ [] _ = pprProblem "printListWithSeps': the number of elements and separators does not match"
